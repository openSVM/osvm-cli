21impl Config {
22    /// Validate configuration values to ensure they are reasonable and safe
23    ///
24    /// # Validation Checks (Bug #12)
25    /// - RPC URL must be a valid HTTP/HTTPS URL
26    /// - Verbose level must be 0-4 (reasonable debug levels)
27    /// - Theme name if provided must match known themes
28    ///
29    /// # Returns
30    /// * `Ok(())` if all validation passes
31    /// * `Err()` if any validation fails
32    fn validate(&self) -> Result<(), Box<dyn std::error::Error>> {
33        // Validate RPC URL format
34        if !self.json_rpc_url.starts_with("http://") && !self.json_rpc_url.starts_with("https://") {
35            return Err(format!(
36                "Invalid RPC URL: must start with http:// or https://, got: {}",
37                self.json_rpc_url
38            ).into());
39        }
40
41        // Validate verbose level (0-4 is reasonable)
42        if self.verbose > 4 {
43            return Err(format!(
44                "Verbose level {} is too high (max 4). Using default.",
45                self.verbose
46            ).into());
47        }
48
49        // Validate theme name if provided
50        if let Some(ref theme) = self.theme_name {
51            let valid_themes = vec!["light", "dark", "nord", "dracula", "monokai", "solarized"];
52            if !valid_themes.contains(&theme.as_str()) {
53                eprintln!(
54                    "Warning: Unknown theme '{}'. Valid themes: {:?}",
55                    theme, valid_themes
56                );
57                // Don't fail, just warn - use default theme
58            }
59        }
60
61        Ok(())
62    }

64    pub async fn load(
65        app_matches: &clap::ArgMatches, // For global flags like no_color, verbose
66        sub_matches: &clap::ArgMatches, // For command-specific or overridable flags
67    ) -> Result<Self, Box<dyn std::error::Error>> {
68        // Determine no_color from global matches
69        let no_color = app_matches.contains_id("no_color") || env::var("NO_COLOR").is_ok();
70        if no_color {
71            colored::control::set_override(false);
72        }
73
74        // Determine verbosity. Assuming 'verbose' is a global flag, get it from app_matches.
75        // If it can be per-subcommand, clparse.rs and arg definitions would need to confirm.
76        // Original main.rs used sub_matches.get_count("verbose") for the Config struct.
77        // For now, let's assume it could be on app_matches or sub_matches.
78        // If 'verbose' is defined as global in clparse, app_matches.get_count is safer.
79        // Let's take it from app_matches, as it's a common global flag.
80        let verbose = app_matches.get_count("verbose");
81
82        let cli_config_path = sub_matches
83            .get_one::<String>("config_file")
84            .map(|s| s.as_str())
85            .unwrap_or("~/.config/osvm/config.yml");
86
87        let cli_config = solana_cli_config::Config::load(cli_config_path).unwrap_or_default();
88
89        // OSVM uses its own keypair directory, NOT ~/.config/solana/
90        // This prevents accidentally touching user's main Solana keypairs
91        let default_osvm_keypair = dirs::home_dir()
92            .unwrap_or_else(|| std::path::PathBuf::from("."))
93            .join(".config/osvm/keypair.json")
94            .to_string_lossy()
95            .to_string();
96
97        let keypair_path = sub_matches
98            .get_one::<String>("keypair")
99            .map(|s| s.to_string())
100            .unwrap_or(default_osvm_keypair);
101
102        let signer = match read_keypair_with_repair(&keypair_path).await {
103            Ok(signer) => signer,
104            Err(err) => {
105                return Err(format!("Error reading keypair file {}: {}", keypair_path, err).into());
106            }
107        };
108
109        // Load theme configuration
110        let theme_name = sub_matches
111            .get_one::<String>("theme")
112            .map(|s| s.to_string());
113
114        // get_flag() returns bool, not Option<bool>, so no need for or_else/unwrap
115        let auto_theme_switching = sub_matches.get_flag("auto_theme");
116
117        let config = Self {
118            json_rpc_url: normalize_to_url_if_moniker(
119                sub_matches
120                    .get_one::<String>("json_rpc_url")
121                    .map(|s| s.as_str())
122                    .unwrap_or(&cli_config.json_rpc_url),
123            ),
124            default_signer: Box::new(signer), // signer is Keypair, needs to be Box<dyn Signer>
125            verbose,
126            no_color,
127            commitment_config: CommitmentConfig::confirmed(),
128            theme_name,
129            auto_theme_switching,
130        };
131
132        // Validate configuration values (Bug #12)
133        config.validate()?;
134
135        Ok(config)
136    }

138    pub fn setup_logging_and_display_info(&self) -> Result<(), Box<dyn std::error::Error>> {
139        match self.verbose {
140            0 => solana_logger::setup_with_default("solana=info"),
141            1 => solana_logger::setup_with_default("solana=debug"),
142            2 => solana_logger::setup_with_default("solana=debug,program=trace"),
143            _ => solana_logger::setup_with_default("solana=trace,program=trace"),
144        }
145
146        if self.verbose > 0 {
147            println!("JSON RPC URL: {}", self.json_rpc_url);
148            if self.verbose >= 2 {
149                println!("Using keypair: {}", self.default_signer.pubkey());
150                println!("Commitment level: {:?}", self.commitment_config.commitment);
151                if self.verbose >= 3 {
152                    // This RpcClient is created just for display purposes.
153                    // The main RpcClient is created later in main.rs.
154                    let rpc_client = RpcClient::new(self.json_rpc_url.clone());
155                    let balance = rpc_client.get_balance_with_commitment(
156                        &self.default_signer.pubkey(),
157                        self.commitment_config,
158                    )?;
159                    println!("Wallet balance: {} SOL", Sol(balance.value));
160                }
161            }
162        }
163        Ok(())
164    }
