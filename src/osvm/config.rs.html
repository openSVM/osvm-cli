21impl Config {
22    /// Validate configuration values to ensure they are reasonable and safe
23    ///
24    /// # Validation Checks (Bug #12)
25    /// - RPC URL must be a valid HTTP/HTTPS URL
26    /// - Verbose level must be 0-4 (reasonable debug levels)
27    /// - Theme name if provided must match known themes
28    ///
29    /// # Returns
30    /// * `Ok(())` if all validation passes
31    /// * `Err()` if any validation fails
32    fn validate(&self) -> Result<(), Box<dyn std::error::Error>> {
33        // Validate RPC URL format
34        if !self.json_rpc_url.starts_with("http://") && !self.json_rpc_url.starts_with("https://") {
35            return Err(format!(
36                "Invalid RPC URL: must start with http:// or https://, got: {}",
37                self.json_rpc_url
38            )
39            .into());
40        }
41
42        // Validate verbose level (0-4 is reasonable)
43        if self.verbose > 4 {
44            return Err(format!(
45                "Verbose level {} is too high (max 4). Using default.",
46                self.verbose
47            )
48            .into());
49        }
50
51        // Validate theme name if provided
52        if let Some(ref theme) = self.theme_name {
53            let valid_themes = vec!["light", "dark", "nord", "dracula", "monokai", "solarized"];
54            if !valid_themes.contains(&theme.as_str()) {
55                eprintln!(
56                    "Warning: Unknown theme '{}'. Valid themes: {:?}",
57                    theme, valid_themes
58                );
59                // Don't fail, just warn - use default theme
60            }
61        }
62
63        Ok(())
64    }

66    pub async fn load(
67        app_matches: &clap::ArgMatches, // For global flags like no_color, verbose
68        sub_matches: &clap::ArgMatches, // For command-specific or overridable flags
69    ) -> Result<Self, Box<dyn std::error::Error>> {
70        // Determine no_color from global matches
71        let no_color = app_matches.contains_id("no_color") || env::var("NO_COLOR").is_ok();
72        if no_color {
73            colored::control::set_override(false);
74        }
75
76        // Determine verbosity. Assuming 'verbose' is a global flag, get it from app_matches.
77        // If it can be per-subcommand, clparse.rs and arg definitions would need to confirm.
78        // Original main.rs used sub_matches.get_count("verbose") for the Config struct.
79        // For now, let's assume it could be on app_matches or sub_matches.
80        // If 'verbose' is defined as global in clparse, app_matches.get_count is safer.
81        // Let's take it from app_matches, as it's a common global flag.
82        let verbose = app_matches.get_count("verbose");
83
84        let cli_config_path = sub_matches
85            .get_one::<String>("config_file")
86            .map(|s| s.as_str())
87            .unwrap_or("~/.config/osvm/config.yml");
88
89        let cli_config = solana_cli_config::Config::load(cli_config_path).unwrap_or_default();
90
91        // OSVM uses its own keypair directory, NOT ~/.config/solana/
92        // This prevents accidentally touching user's main Solana keypairs
93        let default_osvm_keypair = dirs::home_dir()
94            .unwrap_or_else(|| std::path::PathBuf::from("."))
95            .join(".config/osvm/keypair.json")
96            .to_string_lossy()
97            .to_string();
98
99        let keypair_path = sub_matches
100            .get_one::<String>("keypair")
101            .map(|s| s.to_string())
102            .unwrap_or(default_osvm_keypair);
103
104        let signer = match read_keypair_with_repair(&keypair_path).await {
105            Ok(signer) => signer,
106            Err(err) => {
107                return Err(format!("Error reading keypair file {}: {}", keypair_path, err).into());
108            }
109        };
110
111        // Load theme configuration
112        let theme_name = sub_matches
113            .get_one::<String>("theme")
114            .map(|s| s.to_string());
115
116        // get_flag() returns bool, not Option<bool>, so no need for or_else/unwrap
117        let auto_theme_switching = sub_matches.get_flag("auto_theme");
118
119        let config = Self {
120            json_rpc_url: normalize_to_url_if_moniker(
121                sub_matches
122                    .get_one::<String>("json_rpc_url")
123                    .map(|s| s.as_str())
124                    .unwrap_or(&cli_config.json_rpc_url),
125            ),
126            default_signer: Box::new(signer), // signer is Keypair, needs to be Box<dyn Signer>
127            verbose,
128            no_color,
129            commitment_config: CommitmentConfig::confirmed(),
130            theme_name,
131            auto_theme_switching,
132        };
133
134        // Validate configuration values (Bug #12)
135        config.validate()?;
136
137        Ok(config)
138    }

140    pub fn setup_logging_and_display_info(&self) -> Result<(), Box<dyn std::error::Error>> {
141        match self.verbose {
142            0 => solana_logger::setup_with_default("solana=info"),
143            1 => solana_logger::setup_with_default("solana=debug"),
144            2 => solana_logger::setup_with_default("solana=debug,program=trace"),
145            _ => solana_logger::setup_with_default("solana=trace,program=trace"),
146        }
147
148        if self.verbose > 0 {
149            println!("JSON RPC URL: {}", self.json_rpc_url);
150            if self.verbose >= 2 {
151                println!("Using keypair: {}", self.default_signer.pubkey());
152                println!("Commitment level: {:?}", self.commitment_config.commitment);
153                if self.verbose >= 3 {
154                    // This RpcClient is created just for display purposes.
155                    // The main RpcClient is created later in main.rs.
156                    let rpc_client = RpcClient::new(self.json_rpc_url.clone());
157                    let balance = rpc_client.get_balance_with_commitment(
158                        &self.default_signer.pubkey(),
159                        self.commitment_config,
160                    )?;
161                    println!("Wallet balance: {} SOL", Sol(balance.value));
162                }
163            }
164        }
165        Ok(())
166    }
