1//! Ephemeral MicroVM launcher for running MCP tools in isolated instances
2//!
3//! This module provides functionality to launch ephemeral microVMs for each tool call,
4//! ensuring complete isolation and automatic cleanup after execution.

21/// Default timeout for tool execution in seconds
22const DEFAULT_TOOL_TIMEOUT_SECS: u64 = 30;
23
24/// Default memory allocation for ephemeral VMs (MB)
25const DEFAULT_EPHEMERAL_MEMORY_MB: usize = 512;
26
27/// Default CPU count for ephemeral VMs
28const DEFAULT_EPHEMERAL_CPUS: u8 = 1;
29
30/// Maximum concurrent ephemeral VMs
31const MAX_CONCURRENT_VMS: usize = 50;
32
33/// TCP port for tool communication (must match guest/mcp_vsock_wrapper)
34const TCP_TOOL_PORT: u16 = 8080;

36/// Guest VM IP address (fixed by network configuration)
37const GUEST_VM_IP: &str = "172.16.0.2";

39/// Next available vsock CID (starts at 3, reserves 0-2) - kept for VM ID generation
40static NEXT_VSOCK_CID: AtomicU32 = AtomicU32::new(3);

42/// Ephemeral MicroVM configuration

48    pub tool_name: String,
49    /// Server ID this tool belongs to
50    pub server_id: String,

52    pub server_command: String,
53    /// Memory allocation in MB
54    pub memory_mb: usize,
55    /// Number of CPUs
56    pub cpus: u8,
57    /// Timeout for tool execution
58    pub timeout_secs: u64,
59    /// Kernel image path

61    /// Rootfs image path

63    /// Additional environment variables
64    pub env_vars: HashMap<String, String>,
65    /// vsock CID for this VM
66    pub vsock_cid: u32,
67    /// Whether to enable debug logging
68    pub debug: bool,
69    /// TAP device name for networking

71}

73impl Default for EphemeralVmConfig {
74    fn default() -> Self {
75        let cid = NEXT_VSOCK_CID.fetch_add(1, Ordering::SeqCst);
76        Self {
77            id: Uuid::new_v4().to_string(),
78            tool_name: String::new(),
79            server_id: String::new(),
80            server_command: String::new(),
81            memory_mb: DEFAULT_EPHEMERAL_MEMORY_MB,
82            cpus: DEFAULT_EPHEMERAL_CPUS,
83            timeout_secs: DEFAULT_TOOL_TIMEOUT_SECS,
84            kernel_path: dirs::home_dir()
85                .map(|h| h.join(".osvm/kernel/vmlinux.bin"))
86                .unwrap_or_else(|| PathBuf::from("/usr/share/osvm/vmlinux")),
87            rootfs_path: dirs::home_dir()
88                .map(|h| h.join(".osvm/rootfs/osvm-runtime.ext4"))
89                .unwrap_or_else(|| PathBuf::from("/usr/share/osvm/rootfs.ext4")),
90            env_vars: HashMap::new(),
91            vsock_cid: cid,
92            debug: false,
93            tap_device: format!("osvm-tap{}", cid),
94        }
95    }

98/// Handle to a running ephemeral microVM
99pub struct EphemeralVmHandle {
100    /// VM configuration
101    pub config: EphemeralVmConfig,
102    /// Firecracker process
103    firecracker_process: Child,
104    /// Socket path for API
105    api_socket: PathBuf,
106    /// Temporary directory for VM files
107    temp_dir: PathBuf,
108    /// Start time for tracking
109    start_time: SystemTime,
110}
111
112impl EphemeralVmHandle {
113    /// Get the VM ID
114    pub fn id(&self) -> &str {
115        &self.config.id
116    }
117
118    /// Get the vsock CID
119    pub fn vsock_cid(&self) -> u32 {
120        self.config.vsock_cid
121    }
122
123    /// Kill the VM and cleanup resources
124    pub async fn terminate(mut self) -> Result<()> {
125        debug!("Terminating ephemeral VM: {}", self.config.id);
126
127        // Kill the Firecracker process
128        if let Err(e) = self.firecracker_process.kill().await {
129            warn!("Failed to kill Firecracker process: {}", e);
130        }
131
132        // Cleanup temporary directory
133        if self.temp_dir.exists() {
134            if let Err(e) = tokio::fs::remove_dir_all(&self.temp_dir).await {
135                warn!("Failed to cleanup temp directory: {}", e);
136            }
137        }
138
139        let duration = SystemTime::now()
140            .duration_since(self.start_time)
141            .unwrap_or_default();
142        debug!("Ephemeral VM {} lived for {:?}", self.config.id, duration);
143
144        Ok(())
145    }
146}

158impl EphemeralVmManager {
159    /// Create a new ephemeral VM manager

162            active_vms: Arc::new(RwLock::new(HashMap::new())),

168    /// Ensure Firecracker binary has necessary network capabilities
169    async fn ensure_firecracker_capabilities(&self) -> Result<()> {
170        // Check if firecracker binary exists
171        let firecracker_path = which::which("firecracker")
172            .context("Firecracker binary not found in PATH")?;
173
174        debug!("Granting CAP_NET_ADMIN capability to: {:?}", firecracker_path);
175
176        // Grant CAP_NET_ADMIN capability to Firecracker binary
177        let output = tokio::process::Command::new("sudo")
178            .args(&[
179                "setcap",
180                "cap_net_admin+ep",
181                firecracker_path.to_str().unwrap(),
182            ])
183            .output()
184            .await
185            .context("Failed to set capabilities on Firecracker binary")?;
186
187        if !output.status.success() {
188            warn!(
189                "Failed to set capabilities on Firecracker: {}",
190                String::from_utf8_lossy(&output.stderr)
191            );
192        }
193
194        Ok(())
195    }

197    /// Setup TAP device for VM networking
198    async fn setup_tap_device(&self, tap_name: &str) -> Result<()> {
199        debug!("Setting up TAP device: {}", tap_name);
200
201        // Ensure Firecracker has network capabilities
202        self.ensure_firecracker_capabilities().await?;
203
204        // Create TAP device
205        tokio::process::Command::new("sudo")
206            .args(&["ip", "tuntap", "add", tap_name, "mode", "tap"])
207            .output()
208            .await
209            .context("Failed to create TAP device")?;
210
211        // Set TAP device up
212        tokio::process::Command::new("sudo")
213            .args(&["ip", "link", "set", tap_name, "up"])
214            .output()
215            .await
216            .context("Failed to bring TAP device up")?;
217
218        // Assign IP to TAP device (host side)
219        tokio::process::Command::new("sudo")
220            .args(&["ip", "addr", "add", "172.16.0.1/24", "dev", tap_name])
221            .output()
222            .await
223            .ok(); // Ignore if already exists
224
225        // Change ownership to current user so Firecracker can access it
226        let current_user = std::env::var("USER").unwrap_or_else(|_| "root".to_string());
227        tokio::process::Command::new("sudo")
228            .args(&["chown", &current_user, &format!("/dev/net/tun")])
229            .output()
230            .await
231            .ok();
232
233        // Enable IP forwarding
234        tokio::process::Command::new("sudo")
235            .args(&["sysctl", "-w", "net.ipv4.ip_forward=1"])
236            .output()
237            .await
238            .context("Failed to enable IP forwarding")?;
239
240        // Setup NAT (masquerade)
241        tokio::process::Command::new("sudo")
242            .args(&["iptables", "-t", "nat", "-A", "POSTROUTING", "-o", "eth0", "-j", "MASQUERADE"])
243            .output()
244            .await
245            .ok(); // Ignore if rule already exists
246
247        Ok(())
248    }

250    /// Cleanup TAP device

252        debug!("Cleaning up TAP device: {}", tap_name);
253
254        tokio::process::Command::new("sudo")
255            .args(&["ip", "link", "delete", tap_name])
256            .output()
257            .await

263    /// Launch an ephemeral VM for tool execution
264    pub async fn launch_tool_vm(
265        &self,
266        server_id: &str,
267        server_command: &str,
268        tool_name: &str,
269        tool_args: Option<serde_json::Value>,
270    ) -> Result<serde_json::Value> {
271        // Acquire semaphore permit to limit concurrent VMs
272        let _permit = self
273            .launch_semaphore
274            .acquire()
275            .await
276            .context("Failed to acquire VM launch permit")?;
277
278        info!(
279            "Launching ephemeral VM for tool: {}/{}",
280            server_id, tool_name
281        );
282
283        // Create configuration for this VM
284        let config = EphemeralVmConfig {
285            tool_name: tool_name.to_string(),
286            server_id: server_id.to_string(),
287            server_command: server_command.to_string(),
288            debug: self.debug_mode,
289            ..Default::default()
290        };
291
292        // Launch the VM
293        let handle = self.spawn_ephemeral_vm(&config).await?;
294        let vm_id = handle.config.id.clone();
295        let vsock_cid = handle.vsock_cid();
296
297        // Store handle
298        {
299            let mut vms = self.active_vms.write().await;
300            vms.insert(vm_id.clone(), Arc::new(Mutex::new(handle)));
301        }
302
303        // Execute the tool in the VM
304        let result = self
305            .execute_tool_in_vm(vsock_cid, tool_name, tool_args, config.timeout_secs)
306            .await;
307
308        // Cleanup VM regardless of result
309        self.cleanup_vm(&vm_id).await?;
310
311        result
312    }

314    /// Spawn an ephemeral microVM
315    async fn spawn_ephemeral_vm(&self, config: &EphemeralVmConfig) -> Result<EphemeralVmHandle> {
316        // Create temporary directory for this VM
317        let temp_dir = PathBuf::from("/tmp/osvm-ephemeral").join(&config.id);
318        tokio::fs::create_dir_all(&temp_dir)
319            .await
320            .context("Failed to create temp directory")?;
321
322        // Copy rootfs CPIO to temp directory
323        let rootfs_dest = temp_dir.join("rootfs.cpio.gz");
324        let rootfs_source = config
325            .rootfs_path
326            .parent()
327            .unwrap()
328            .join("mcp-server.cpio.gz");
329
330        debug!(
331            "Copying rootfs from {:?} to {:?}",
332            rootfs_source, rootfs_dest
333        );
334
335        let copy_result = Command::new("cp")
336            .arg(&rootfs_source)
337            .arg(&rootfs_dest)
338            .output()
339            .await
340            .context("Failed to copy rootfs")?;
341
342        if !copy_result.status.success() {
343            return Err(anyhow!(
344                "Failed to copy rootfs: {}",
345                String::from_utf8_lossy(&copy_result.stderr)
346            ));
347        }
348
349        // Prepare Firecracker configuration
350        let api_socket = temp_dir.join("firecracker.sock");
351        let vm_config = temp_dir.join("vm_config.json");
352
353        // Create VM configuration file
354        let vm_config_json = self.create_vm_config(config, &temp_dir)?;
355        tokio::fs::write(&vm_config, serde_json::to_string_pretty(&vm_config_json)?)
356            .await
357            .context("Failed to write VM config")?;
358
359        // Setup TAP device for networking
360        self.setup_tap_device(&config.tap_device).await?;
361
362        // Launch Firecracker with sudo to access TAP devices
363        // TODO: In production, use Firecracker jailer instead of sudo
364        let mut firecracker_cmd = Command::new("sudo");
365        firecracker_cmd
366            .arg("firecracker")
367            .arg("--api-sock")
368            .arg(&api_socket)
369            .arg("--config-file")
370            .arg(&vm_config)
371            .stdout(if self.debug_mode {
372                std::process::Stdio::inherit()
373            } else {
374                std::process::Stdio::null()
375            })
376            .stderr(if self.debug_mode {
377                std::process::Stdio::inherit()
378            } else {
379                std::process::Stdio::null()
380            })
381            .kill_on_drop(true);
382
383        let firecracker_process = firecracker_cmd
384            .spawn()
385            .context("Failed to spawn Firecracker process")?;
386
387        // Wait for VM to be ready
388        self.wait_for_vm_ready(&api_socket, config.vsock_cid)
389            .await?;
390
391        Ok(EphemeralVmHandle {
392            config: config.clone(),
393            firecracker_process,
394            api_socket,
395            temp_dir,
396            start_time: SystemTime::now(),
397        })
398    }

438    /// Wait for VM to be ready
439    async fn wait_for_vm_ready(&self, api_socket: &Path, vsock_cid: u32) -> Result<()> {
440        let start = SystemTime::now();
441        let timeout_duration = Duration::from_secs(30);
442        let guest_addr = format!("{}:{}", GUEST_VM_IP, TCP_TOOL_PORT);
443
444        while SystemTime::now().duration_since(start)? < timeout_duration {
445            // Check if API socket exists
446            if api_socket.exists() {
447                // Try to connect via TCP
448                if let Ok(stream) = timeout(
449                    Duration::from_secs(1),
450                    TcpStream::connect(&guest_addr),
451                )
452                .await
453                {
454                    if stream.is_ok() {
455                        debug!("VM {} is ready (connected to {})", vsock_cid, guest_addr);
456                        return Ok(());
457                    }
458                }
459            }
460            sleep(Duration::from_millis(100)).await;
461        }
462
463        Err(anyhow!("Timeout waiting for VM to be ready"))
464    }

466    /// Execute tool in the VM via TCP
467    async fn execute_tool_in_vm(
468        &self,
469        vsock_cid: u32,
470        tool_name: &str,
471        tool_args: Option<serde_json::Value>,
472        timeout_secs: u64,
473    ) -> Result<serde_json::Value> {
474        let guest_addr = format!("{}:{}", GUEST_VM_IP, TCP_TOOL_PORT);
475        debug!("Executing tool {} in VM (CID: {}, addr: {})", tool_name, vsock_cid, guest_addr);
476
477        // Connect to VM via TCP
478        let mut stream = timeout(
479            Duration::from_secs(15),
480            TcpStream::connect(&guest_addr),
481        )
482        .await
483        .context("Timeout connecting to ephemeral VM")?
484        .context("Failed to connect to ephemeral VM")?;
485
486        // Send tool execution request with length prefix
487        let request = serde_json::json!({
488            "tool": tool_name,
489            "args": tool_args,
490        });
491
492        let request_bytes = serde_json::to_vec(&request)?;
493
494        // Send 4-byte length prefix (little-endian to match guest)
495        let length = request_bytes.len() as u32;
496        stream
497            .write_all(&length.to_le_bytes())
498            .await
499            .context("Failed to send length prefix")?;
500
501        // Send JSON payload
502        stream
503            .write_all(&request_bytes)
504            .await
505            .context("Failed to send tool request")?;
506        stream.flush().await?;
507
508        // Read response with timeout
509        let response = timeout(
510            Duration::from_secs(timeout_secs),
511            self.read_tool_response(&mut stream),
512        )
513        .await
514        .context("Tool execution timeout")?
515        .context("Failed to read tool response")?;
516
517        Ok(response)
518    }
519
520    /// Read tool response from TCP stream (length-prefixed)
521    async fn read_tool_response(
522        &self,
523        stream: &mut TcpStream,
524    ) -> Result<serde_json::Value> {
525        use tokio::io::AsyncReadExt;
526
527        // Read 4-byte length prefix (little-endian to match guest)
528        let mut len_buf = [0u8; 4];
529        stream
530            .read_exact(&mut len_buf)
531            .await
532            .context("Failed to read response length prefix")?;
533
534        let msg_len = u32::from_le_bytes(len_buf) as usize;
535
536        if msg_len > 10 * 1024 * 1024 {
537            return Err(anyhow!("Response too large: {} bytes", msg_len));
538        }
539
540        // Read the JSON payload
541        let mut buffer = vec![0u8; msg_len];
542        stream
543            .read_exact(&mut buffer)
544            .await
545            .context("Failed to read response payload")?;
546
547        serde_json::from_slice(&buffer).context("Failed to parse tool response")
548    }

550    /// Cleanup a VM
551    async fn cleanup_vm(&self, vm_id: &str) -> Result<()> {
552        let handle = {
553            let mut vms = self.active_vms.write().await;
554            vms.remove(vm_id)
555        };
556
557        if let Some(handle) = handle {
558            let handle = Arc::try_unwrap(handle)
559                .map_err(|_| anyhow!("VM handle still in use"))?
560                .into_inner();
561            handle.terminate().await?;
562            info!("Cleaned up ephemeral VM: {}", vm_id);
563        }
564
565        Ok(())
566    }
567
568    /// Cleanup all active VMs (for shutdown)
569    pub async fn cleanup_all(&self) -> Result<()> {
570        let vms = {
571            let mut vms = self.active_vms.write().await;
572            std::mem::take(&mut *vms)
573        };
574
575        for (id, handle) in vms {
576            if let Ok(handle) = Arc::try_unwrap(handle) {
577                if let Err(e) = handle.into_inner().terminate().await {
578                    error!("Failed to cleanup VM {}: {}", id, e);
579                }
580            }
581        }
582
583        Ok(())
584    }

587/// Main chat microVM orchestrator
588pub struct ChatVmOrchestrator {
589    /// Handle to the persistent chat VM
590    chat_vm: Option<Arc<Mutex<EphemeralVmHandle>>>,
591    /// Ephemeral VM manager for tools
592    ephemeral_manager: Arc<EphemeralVmManager>,
593    /// Debug mode flag
594    debug_mode: bool,
595}
596
597impl ChatVmOrchestrator {
598    /// Create a new chat VM orchestrator
599    pub fn new(debug_mode: bool) -> Self {
600        Self {
601            chat_vm: None,
602            ephemeral_manager: Arc::new(EphemeralVmManager::new(debug_mode)),
603            debug_mode,
604        }
605    }
606
607    /// Start the main chat VM
608    pub async fn start_chat_vm(&mut self) -> Result<()> {
609        info!("Starting main chat microVM");
610
611        let config = EphemeralVmConfig {
612            id: format!("chat-{}", Uuid::new_v4()),
613            tool_name: "main-chat".to_string(),
614            server_id: "osvm-chat".to_string(),
615            memory_mb: 512,     // More memory for chat VM
616            cpus: 2,            // More CPUs for chat VM
617            timeout_secs: 3600, // 1 hour timeout for chat
618            debug: self.debug_mode,
619            ..Default::default()
620        };
621
622        let handle = self.ephemeral_manager.spawn_ephemeral_vm(&config).await?;
623        self.chat_vm = Some(Arc::new(Mutex::new(handle)));
624
625        Ok(())
626    }
627
628    /// Execute a tool in an ephemeral VM
629    pub async fn execute_tool(
630        &self,
631        server_id: &str,
632        tool_name: &str,
633        tool_args: Option<serde_json::Value>,
634    ) -> Result<serde_json::Value> {
635        self.ephemeral_manager
636            .launch_tool_vm(server_id, "", tool_name, tool_args)
637            .await
638    }
639
640    /// Stop the chat VM and cleanup
641    pub async fn stop(&mut self) -> Result<()> {
642        // Stop chat VM
643        if let Some(chat_vm) = self.chat_vm.take() {
644            if let Ok(handle) = Arc::try_unwrap(chat_vm) {
645                handle.into_inner().terminate().await?;
646            }
647        }
648
649        // Cleanup all ephemeral VMs
650        self.ephemeral_manager.cleanup_all().await?;
651
652        Ok(())
653    }
654}
