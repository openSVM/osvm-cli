1//! Ephemeral MicroVM launcher for running MCP tools in isolated instances
2//!
3//! This module provides functionality to launch ephemeral microVMs for each tool call,
4//! ensuring complete isolation and automatic cleanup after execution.

21/// Default timeout for tool execution in seconds
22const DEFAULT_TOOL_TIMEOUT_SECS: u64 = 30;
23
24/// Default memory allocation for ephemeral VMs (MB)
25const DEFAULT_EPHEMERAL_MEMORY_MB: usize = 512;
26
27/// Default CPU count for ephemeral VMs
28const DEFAULT_EPHEMERAL_CPUS: u8 = 1;
29
30/// Maximum concurrent ephemeral VMs
31const MAX_CONCURRENT_VMS: usize = 50;
32
33/// TCP port for tool communication (must match guest/mcp_vsock_wrapper)
34const TCP_TOOL_PORT: u16 = 8080;

36/// Guest VM IP address (fixed by network configuration)
37const GUEST_VM_IP: &str = "172.16.0.2";

39/// Next available vsock CID (starts at 3, reserves 0-2) - kept for VM ID generation
40static NEXT_VSOCK_CID: AtomicU32 = AtomicU32::new(3);

42/// Ephemeral MicroVM configuration
43#[derive(Debug, Clone, Serialize, Deserialize)]
44pub struct EphemeralVmConfig {
45    /// Unique ID for this ephemeral VM
46    pub id: String,
47    /// Tool name being executed
48    pub tool_name: String,
49    /// Server ID this tool belongs to
50    pub server_id: String,
51    /// MCP server command to execute
52    pub server_command: String,
53    /// Memory allocation in MB
54    pub memory_mb: usize,
55    /// Number of CPUs
56    pub cpus: u8,
57    /// Timeout for tool execution
58    pub timeout_secs: u64,
59    /// Kernel image path
60    pub kernel_path: PathBuf,
61    /// Rootfs image path
62    pub rootfs_path: PathBuf,
63    /// Additional environment variables
64    pub env_vars: HashMap<String, String>,
65    /// vsock CID for this VM
66    pub vsock_cid: u32,
67    /// Whether to enable debug logging
68    pub debug: bool,
69    /// TAP device name for networking
70    pub tap_device: String,
71}

73impl Default for EphemeralVmConfig {
74    fn default() -> Self {
75        let cid = NEXT_VSOCK_CID.fetch_add(1, Ordering::SeqCst);
76        Self {
77            id: Uuid::new_v4().to_string(),
78            tool_name: String::new(),
79            server_id: String::new(),
80            server_command: String::new(),
81            memory_mb: DEFAULT_EPHEMERAL_MEMORY_MB,
82            cpus: DEFAULT_EPHEMERAL_CPUS,
83            timeout_secs: DEFAULT_TOOL_TIMEOUT_SECS,
84            kernel_path: dirs::home_dir()
85                .map(|h| h.join(".osvm/kernel/vmlinux.bin"))
86                .unwrap_or_else(|| PathBuf::from("/usr/share/osvm/vmlinux")),
87            rootfs_path: dirs::home_dir()
88                .map(|h| h.join(".osvm/rootfs/osvm-runtime.ext4"))
89                .unwrap_or_else(|| PathBuf::from("/usr/share/osvm/rootfs.ext4")),
90            env_vars: HashMap::new(),
91            vsock_cid: cid,
92            debug: false,
93            tap_device: format!("osvm-tap{}", cid),
94        }
95    }

98/// Handle to a running ephemeral microVM
99pub struct EphemeralVmHandle {
100    /// VM configuration
101    pub config: EphemeralVmConfig,
102    /// Firecracker process
103    firecracker_process: Child,
104    /// Socket path for API
105    api_socket: PathBuf,
106    /// Temporary directory for VM files
107    temp_dir: PathBuf,
108    /// Start time for tracking
109    start_time: SystemTime,
110}
111
112impl EphemeralVmHandle {
113    /// Get the VM ID
114    pub fn id(&self) -> &str {
115        &self.config.id
116    }
117
118    /// Get the vsock CID
119    pub fn vsock_cid(&self) -> u32 {
120        self.config.vsock_cid
121    }
122
123    /// Kill the VM and cleanup resources
124    pub async fn terminate(mut self) -> Result<()> {
125        debug!("Terminating ephemeral VM: {}", self.config.id);
126
127        // Kill the Firecracker process
128        if let Err(e) = self.firecracker_process.kill().await {
129            warn!("Failed to kill Firecracker process: {}", e);
130        }
131
132        // Cleanup temporary directory
133        if self.temp_dir.exists() {
134            if let Err(e) = tokio::fs::remove_dir_all(&self.temp_dir).await {
135                warn!("Failed to cleanup temp directory: {}", e);
136            }
137        }
138
139        let duration = SystemTime::now()
140            .duration_since(self.start_time)
141            .unwrap_or_default();
142        debug!("Ephemeral VM {} lived for {:?}", self.config.id, duration);
143
144        Ok(())
145    }
146}

168    /// Ensure Firecracker binary has necessary network capabilities
169    async fn ensure_firecracker_capabilities(&self) -> Result<()> {
170        // Check if firecracker binary exists
171        let firecracker_path =
172            which::which("firecracker").context("Firecracker binary not found in PATH")?;
173
174        debug!(
175            "Granting CAP_NET_ADMIN capability to: {:?}",
176            firecracker_path
177        );
178
179        // Grant CAP_NET_ADMIN capability to Firecracker binary
180        let output = tokio::process::Command::new("sudo")
181            .args(&[
182                "setcap",
183                "cap_net_admin+ep",
184                firecracker_path.to_str().unwrap(),
185            ])
186            .output()
187            .await
188            .context("Failed to set capabilities on Firecracker binary")?;
189
190        if !output.status.success() {
191            warn!(
192                "Failed to set capabilities on Firecracker: {}",
193                String::from_utf8_lossy(&output.stderr)
194            );
195        }
196
197        Ok(())
198    }

200    /// Setup TAP device for VM networking
201    async fn setup_tap_device(&self, tap_name: &str) -> Result<()> {
202        debug!("Setting up TAP device: {}", tap_name);
203
204        // Ensure Firecracker has network capabilities
205        self.ensure_firecracker_capabilities().await?;
206
207        // Create TAP device
208        tokio::process::Command::new("sudo")
209            .args(&["ip", "tuntap", "add", tap_name, "mode", "tap"])
210            .output()
211            .await
212            .context("Failed to create TAP device")?;
213
214        // Set TAP device up
215        tokio::process::Command::new("sudo")
216            .args(&["ip", "link", "set", tap_name, "up"])
217            .output()
218            .await
219            .context("Failed to bring TAP device up")?;
220
221        // Assign IP to TAP device (host side)
222        tokio::process::Command::new("sudo")
223            .args(&["ip", "addr", "add", "172.16.0.1/24", "dev", tap_name])
224            .output()
225            .await
226            .ok(); // Ignore if already exists
227
228        // Change ownership to current user so Firecracker can access it
229        let current_user = std::env::var("USER").unwrap_or_else(|_| "root".to_string());
230        tokio::process::Command::new("sudo")
231            .args(&["chown", &current_user, &format!("/dev/net/tun")])
232            .output()
233            .await
234            .ok();
235
236        // Enable IP forwarding
237        tokio::process::Command::new("sudo")
238            .args(&["sysctl", "-w", "net.ipv4.ip_forward=1"])
239            .output()
240            .await
241            .context("Failed to enable IP forwarding")?;
242
243        // Setup NAT (masquerade)
244        tokio::process::Command::new("sudo")
245            .args(&[
246                "iptables",
247                "-t",
248                "nat",
249                "-A",
250                "POSTROUTING",
251                "-o",
252                "eth0",
253                "-j",
254                "MASQUERADE",
255            ])
256            .output()
257            .await
258            .ok(); // Ignore if rule already exists
259
260        Ok(())
261    }

268            .args(&["ip", "link", "delete", tap_name])

276    /// Launch an ephemeral VM for tool execution
277    pub async fn launch_tool_vm(
278        &self,
279        server_id: &str,
280        server_command: &str,
281        tool_name: &str,
282        tool_args: Option<serde_json::Value>,
283    ) -> Result<serde_json::Value> {
284        // Acquire semaphore permit to limit concurrent VMs
285        let _permit = self
286            .launch_semaphore
287            .acquire()
288            .await
289            .context("Failed to acquire VM launch permit")?;
290
291        info!(
292            "Launching ephemeral VM for tool: {}/{}",
293            server_id, tool_name
294        );
295
296        // Create configuration for this VM
297        let config = EphemeralVmConfig {
298            tool_name: tool_name.to_string(),
299            server_id: server_id.to_string(),
300            server_command: server_command.to_string(),
301            debug: self.debug_mode,
302            ..Default::default()
303        };
304
305        // Launch the VM
306        let handle = self.spawn_ephemeral_vm(&config).await?;
307        let vm_id = handle.config.id.clone();
308        let vsock_cid = handle.vsock_cid();
309
310        // Store handle
311        {
312            let mut vms = self.active_vms.write().await;
313            vms.insert(vm_id.clone(), Arc::new(Mutex::new(handle)));
314        }
315
316        // Execute the tool in the VM
317        let result = self
318            .execute_tool_in_vm(vsock_cid, tool_name, tool_args, config.timeout_secs)
319            .await;
320
321        // Cleanup VM regardless of result
322        self.cleanup_vm(&vm_id).await?;
323
324        result
325    }

327    /// Spawn an ephemeral microVM
328    async fn spawn_ephemeral_vm(&self, config: &EphemeralVmConfig) -> Result<EphemeralVmHandle> {
329        // Create temporary directory for this VM
330        let temp_dir = PathBuf::from("/tmp/osvm-ephemeral").join(&config.id);
331        tokio::fs::create_dir_all(&temp_dir)
332            .await
333            .context("Failed to create temp directory")?;
334
335        // Copy rootfs CPIO to temp directory
336        let rootfs_dest = temp_dir.join("rootfs.cpio.gz");
337        let rootfs_source = config
338            .rootfs_path
339            .parent()
340            .unwrap()
341            .join("mcp-server.cpio.gz");
342
343        debug!(
344            "Copying rootfs from {:?} to {:?}",
345            rootfs_source, rootfs_dest
346        );
347
348        let copy_result = Command::new("cp")
349            .arg(&rootfs_source)
350            .arg(&rootfs_dest)
351            .output()
352            .await
353            .context("Failed to copy rootfs")?;
354
355        if !copy_result.status.success() {
356            return Err(anyhow!(
357                "Failed to copy rootfs: {}",
358                String::from_utf8_lossy(&copy_result.stderr)
359            ));
360        }
361
362        // Prepare Firecracker configuration
363        let api_socket = temp_dir.join("firecracker.sock");
364        let vm_config = temp_dir.join("vm_config.json");
365
366        // Create VM configuration file
367        let vm_config_json = self.create_vm_config(config, &temp_dir)?;
368        tokio::fs::write(&vm_config, serde_json::to_string_pretty(&vm_config_json)?)
369            .await
370            .context("Failed to write VM config")?;
371
372        // Setup TAP device for networking
373        self.setup_tap_device(&config.tap_device).await?;
374
375        // Launch Firecracker with sudo to access TAP devices
376        // TODO: In production, use Firecracker jailer instead of sudo
377        let mut firecracker_cmd = Command::new("sudo");
378        firecracker_cmd
379            .arg("firecracker")
380            .arg("--api-sock")
381            .arg(&api_socket)
382            .arg("--config-file")
383            .arg(&vm_config)
384            .stdout(if self.debug_mode {
385                std::process::Stdio::inherit()
386            } else {
387                std::process::Stdio::null()
388            })
389            .stderr(if self.debug_mode {
390                std::process::Stdio::inherit()
391            } else {
392                std::process::Stdio::null()
393            })
394            .kill_on_drop(true);
395
396        let firecracker_process = firecracker_cmd
397            .spawn()
398            .context("Failed to spawn Firecracker process")?;
399
400        // Wait for VM to be ready
401        self.wait_for_vm_ready(&api_socket, config.vsock_cid)
402            .await?;
403
404        Ok(EphemeralVmHandle {
405            config: config.clone(),
406            firecracker_process,
407            api_socket,
408            temp_dir,
409            start_time: SystemTime::now(),
410        })
411    }

451    /// Wait for VM to be ready
452    async fn wait_for_vm_ready(&self, api_socket: &Path, vsock_cid: u32) -> Result<()> {
453        let start = SystemTime::now();
454        let timeout_duration = Duration::from_secs(30);
455        let guest_addr = format!("{}:{}", GUEST_VM_IP, TCP_TOOL_PORT);
456
457        while SystemTime::now().duration_since(start)? < timeout_duration {
458            // Check if API socket exists
459            if api_socket.exists() {
460                // Try to connect via TCP
461                if let Ok(stream) =
462                    timeout(Duration::from_secs(1), TcpStream::connect(&guest_addr)).await
463                {
464                    if stream.is_ok() {
465                        debug!("VM {} is ready (connected to {})", vsock_cid, guest_addr);
466                        return Ok(());
467                    }
468                }
469            }
470            sleep(Duration::from_millis(100)).await;
471        }
472
473        Err(anyhow!("Timeout waiting for VM to be ready"))
474    }

476    /// Execute tool in the VM via TCP
477    async fn execute_tool_in_vm(
478        &self,
479        vsock_cid: u32,
480        tool_name: &str,
481        tool_args: Option<serde_json::Value>,
482        timeout_secs: u64,
483    ) -> Result<serde_json::Value> {
484        let guest_addr = format!("{}:{}", GUEST_VM_IP, TCP_TOOL_PORT);
485        debug!(
486            "Executing tool {} in VM (CID: {}, addr: {})",
487            tool_name, vsock_cid, guest_addr
488        );
489
490        // Connect to VM via TCP
491        let mut stream = timeout(Duration::from_secs(15), TcpStream::connect(&guest_addr))
492            .await
493            .context("Timeout connecting to ephemeral VM")?
494            .context("Failed to connect to ephemeral VM")?;
495
496        // Send tool execution request with length prefix
497        let request = serde_json::json!({
498            "tool": tool_name,
499            "args": tool_args,
500        });
501
502        let request_bytes = serde_json::to_vec(&request)?;
503
504        // Send 4-byte length prefix (little-endian to match guest)
505        let length = request_bytes.len() as u32;
506        stream
507            .write_all(&length.to_le_bytes())
508            .await
509            .context("Failed to send length prefix")?;
510
511        // Send JSON payload
512        stream
513            .write_all(&request_bytes)
514            .await
515            .context("Failed to send tool request")?;
516        stream.flush().await?;
517
518        // Read response with timeout
519        let response = timeout(
520            Duration::from_secs(timeout_secs),
521            self.read_tool_response(&mut stream),
522        )
523        .await
524        .context("Tool execution timeout")?
525        .context("Failed to read tool response")?;
526
527        Ok(response)
528    }

530    /// Read tool response from TCP stream (length-prefixed)
531    async fn read_tool_response(&self, stream: &mut TcpStream) -> Result<serde_json::Value> {
532        use tokio::io::AsyncReadExt;
533
534        // Read 4-byte length prefix (little-endian to match guest)
535        let mut len_buf = [0u8; 4];
536        stream
537            .read_exact(&mut len_buf)
538            .await
539            .context("Failed to read response length prefix")?;
540
541        let msg_len = u32::from_le_bytes(len_buf) as usize;
542
543        if msg_len > 10 * 1024 * 1024 {
544            return Err(anyhow!("Response too large: {} bytes", msg_len));
545        }
546
547        // Read the JSON payload
548        let mut buffer = vec![0u8; msg_len];
549        stream
550            .read_exact(&mut buffer)
551            .await
552            .context("Failed to read response payload")?;
553
554        serde_json::from_slice(&buffer).context("Failed to parse tool response")
555    }

557    /// Cleanup a VM
558    async fn cleanup_vm(&self, vm_id: &str) -> Result<()> {
559        let handle = {
560            let mut vms = self.active_vms.write().await;
561            vms.remove(vm_id)
562        };
563
564        if let Some(handle) = handle {
565            let handle = Arc::try_unwrap(handle)
566                .map_err(|_| anyhow!("VM handle still in use"))?
567                .into_inner();
568            handle.terminate().await?;
569            info!("Cleaned up ephemeral VM: {}", vm_id);
570        }
571
572        Ok(())
573    }
574
575    /// Cleanup all active VMs (for shutdown)
576    pub async fn cleanup_all(&self) -> Result<()> {
577        let vms = {
578            let mut vms = self.active_vms.write().await;
579            std::mem::take(&mut *vms)
580        };
581
582        for (id, handle) in vms {
583            if let Ok(handle) = Arc::try_unwrap(handle) {
584                if let Err(e) = handle.into_inner().terminate().await {
585                    error!("Failed to cleanup VM {}: {}", id, e);
586                }
587            }
588        }
589
590        Ok(())
591    }

594/// Main chat microVM orchestrator
595pub struct ChatVmOrchestrator {
596    /// Handle to the persistent chat VM
597    chat_vm: Option<Arc<Mutex<EphemeralVmHandle>>>,
598    /// Ephemeral VM manager for tools
599    ephemeral_manager: Arc<EphemeralVmManager>,
600    /// Debug mode flag
601    debug_mode: bool,
602}
603
604impl ChatVmOrchestrator {
605    /// Create a new chat VM orchestrator
606    pub fn new(debug_mode: bool) -> Self {
607        Self {
608            chat_vm: None,
609            ephemeral_manager: Arc::new(EphemeralVmManager::new(debug_mode)),
610            debug_mode,
611        }
612    }
613
614    /// Start the main chat VM
615    pub async fn start_chat_vm(&mut self) -> Result<()> {
616        info!("Starting main chat microVM");
617
618        let config = EphemeralVmConfig {
619            id: format!("chat-{}", Uuid::new_v4()),
620            tool_name: "main-chat".to_string(),
621            server_id: "osvm-chat".to_string(),
622            memory_mb: 512,     // More memory for chat VM
623            cpus: 2,            // More CPUs for chat VM
624            timeout_secs: 3600, // 1 hour timeout for chat
625            debug: self.debug_mode,
626            ..Default::default()
627        };
628
629        let handle = self.ephemeral_manager.spawn_ephemeral_vm(&config).await?;
630        self.chat_vm = Some(Arc::new(Mutex::new(handle)));
631
632        Ok(())
633    }
634
635    /// Execute a tool in an ephemeral VM
636    pub async fn execute_tool(
637        &self,
638        server_id: &str,
639        tool_name: &str,
640        tool_args: Option<serde_json::Value>,
641    ) -> Result<serde_json::Value> {
642        self.ephemeral_manager
643            .launch_tool_vm(server_id, "", tool_name, tool_args)
644            .await
645    }
646
647    /// Stop the chat VM and cleanup
648    pub async fn stop(&mut self) -> Result<()> {
649        // Stop chat VM
650        if let Some(chat_vm) = self.chat_vm.take() {
651            if let Ok(handle) = Arc::try_unwrap(chat_vm) {
652                handle.into_inner().terminate().await?;
653            }
654        }
655
656        // Cleanup all ephemeral VMs
657        self.ephemeral_manager.cleanup_all().await?;
658
659        Ok(())
660    }
661}
