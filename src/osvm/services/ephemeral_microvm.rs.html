1//! Ephemeral MicroVM launcher for running MCP tools in isolated instances
2//!
3//! This module provides functionality to launch ephemeral microVMs for each tool call,
4//! ensuring complete isolation and automatic cleanup after execution.

20/// Default timeout for tool execution in seconds
21const DEFAULT_TOOL_TIMEOUT_SECS: u64 = 30;
22
23/// Default memory allocation for ephemeral VMs (MB)
24const DEFAULT_EPHEMERAL_MEMORY_MB: usize = 256;
25
26/// Default CPU count for ephemeral VMs
27const DEFAULT_EPHEMERAL_CPUS: u8 = 1;
28
29/// Maximum concurrent ephemeral VMs
30const MAX_CONCURRENT_VMS: usize = 50;
31
32/// vsock port for tool communication
33const VSOCK_TOOL_PORT: u32 = 5353;
34
35/// Next available vsock CID (starts at 3, reserves 0-2)
36static NEXT_VSOCK_CID: AtomicU32 = AtomicU32::new(3);

38/// Ephemeral MicroVM configuration
39#[derive(Debug, Clone, Serialize, Deserialize)]
40pub struct EphemeralVmConfig {
41    /// Unique ID for this ephemeral VM
42    pub id: String,
43    /// Tool name being executed
44    pub tool_name: String,
45    /// Server ID this tool belongs to
46    pub server_id: String,
47    /// Memory allocation in MB
48    pub memory_mb: usize,
49    /// Number of CPUs
50    pub cpus: u8,
51    /// Timeout for tool execution
52    pub timeout_secs: u64,
53    /// Kernel image path
54    pub kernel_path: PathBuf,
55    /// Rootfs image path
56    pub rootfs_path: PathBuf,
57    /// Additional environment variables
58    pub env_vars: HashMap<String, String>,
59    /// vsock CID for this VM
60    pub vsock_cid: u32,
61    /// Whether to enable debug logging
62    pub debug: bool,
63}

67        Self {
68            id: Uuid::new_v4().to_string(),
69            tool_name: String::new(),

71            memory_mb: DEFAULT_EPHEMERAL_MEMORY_MB,
72            cpus: DEFAULT_EPHEMERAL_CPUS,
73            timeout_secs: DEFAULT_TOOL_TIMEOUT_SECS,
74            kernel_path: PathBuf::from("/usr/share/osvm/vmlinux"),

78            debug: false,
79        }

83/// Handle to a running ephemeral microVM
84pub struct EphemeralVmHandle {
85    /// VM configuration
86    pub config: EphemeralVmConfig,
87    /// Firecracker process
88    firecracker_process: Child,
89    /// Socket path for API
90    api_socket: PathBuf,
91    /// Temporary directory for VM files
92    temp_dir: PathBuf,
93    /// Start time for tracking
94    start_time: SystemTime,
95}
96
97impl EphemeralVmHandle {
98    /// Get the VM ID
99    pub fn id(&self) -> &str {
100        &self.config.id
101    }
102
103    /// Get the vsock CID
104    pub fn vsock_cid(&self) -> u32 {
105        self.config.vsock_cid
106    }
107
108    /// Kill the VM and cleanup resources
109    pub async fn terminate(mut self) -> Result<()> {
110        debug!("Terminating ephemeral VM: {}", self.config.id);
111
112        // Kill the Firecracker process
113        if let Err(e) = self.firecracker_process.kill().await {
114            warn!("Failed to kill Firecracker process: {}", e);
115        }
116
117        // Cleanup temporary directory
118        if self.temp_dir.exists() {
119            if let Err(e) = tokio::fs::remove_dir_all(&self.temp_dir).await {
120                warn!("Failed to cleanup temp directory: {}", e);
121            }
122        }
123
124        let duration = SystemTime::now()
125            .duration_since(self.start_time)
126            .unwrap_or_default();
127        debug!("Ephemeral VM {} lived for {:?}", self.config.id, duration);
128
129        Ok(())
130    }
131}

133/// Ephemeral MicroVM Manager
134pub struct EphemeralVmManager {
135    /// Active ephemeral VMs
136    active_vms: Arc<RwLock<HashMap<String, Arc<Mutex<EphemeralVmHandle>>>>>,
137    /// VM launch semaphore to limit concurrent VMs
138    launch_semaphore: Arc<tokio::sync::Semaphore>,
139    /// Debug mode flag
140    debug_mode: bool,
141}
142
143impl EphemeralVmManager {
144    /// Create a new ephemeral VM manager
145    pub fn new(debug_mode: bool) -> Self {
146        Self {
147            active_vms: Arc::new(RwLock::new(HashMap::new())),
148            launch_semaphore: Arc::new(tokio::sync::Semaphore::new(MAX_CONCURRENT_VMS)),
149            debug_mode,
150        }
151    }
152
153    /// Launch an ephemeral VM for tool execution
154    pub async fn launch_tool_vm(
155        &self,
156        server_id: &str,
157        tool_name: &str,
158        tool_args: Option<serde_json::Value>,
159    ) -> Result<serde_json::Value> {
160        // Acquire semaphore permit to limit concurrent VMs
161        let _permit = self
162            .launch_semaphore
163            .acquire()
164            .await
165            .context("Failed to acquire VM launch permit")?;
166
167        info!(
168            "Launching ephemeral VM for tool: {}/{}",
169            server_id, tool_name
170        );
171
172        // Create configuration for this VM
173        let mut config = EphemeralVmConfig::default();
174        config.tool_name = tool_name.to_string();
175        config.server_id = server_id.to_string();
176        config.debug = self.debug_mode;
177
178        // Launch the VM
179        let handle = self.spawn_ephemeral_vm(&config).await?;
180        let vm_id = handle.config.id.clone();
181        let vsock_cid = handle.vsock_cid();
182
183        // Store handle
184        {
185            let mut vms = self.active_vms.write().await;
186            vms.insert(vm_id.clone(), Arc::new(Mutex::new(handle)));
187        }
188
189        // Execute the tool in the VM
190        let result = self
191            .execute_tool_in_vm(vsock_cid, tool_name, tool_args, config.timeout_secs)
192            .await;
193
194        // Cleanup VM regardless of result
195        self.cleanup_vm(&vm_id).await?;
196
197        result
198    }
199
200    /// Spawn an ephemeral microVM
201    async fn spawn_ephemeral_vm(&self, config: &EphemeralVmConfig) -> Result<EphemeralVmHandle> {
202        // Create temporary directory for this VM
203        let temp_dir = PathBuf::from("/tmp/osvm-ephemeral").join(&config.id);
204        tokio::fs::create_dir_all(&temp_dir)
205            .await
206            .context("Failed to create temp directory")?;
207
208        // Prepare Firecracker configuration
209        let api_socket = temp_dir.join("firecracker.sock");
210        let vm_config = temp_dir.join("vm_config.json");
211
212        // Create VM configuration file
213        let vm_config_json = self.create_vm_config(config, &temp_dir)?;
214        tokio::fs::write(&vm_config, serde_json::to_string_pretty(&vm_config_json)?)
215            .await
216            .context("Failed to write VM config")?;
217
218        // Launch Firecracker
219        let mut firecracker_cmd = Command::new("firecracker");
220        firecracker_cmd
221            .arg("--api-sock")
222            .arg(&api_socket)
223            .arg("--config-file")
224            .arg(&vm_config)
225            .stdout(if self.debug_mode {
226                std::process::Stdio::inherit()
227            } else {
228                std::process::Stdio::null()
229            })
230            .stderr(if self.debug_mode {
231                std::process::Stdio::inherit()
232            } else {
233                std::process::Stdio::null()
234            })
235            .kill_on_drop(true);
236
237        let firecracker_process = firecracker_cmd
238            .spawn()
239            .context("Failed to spawn Firecracker process")?;
240
241        // Wait for VM to be ready
242        self.wait_for_vm_ready(&api_socket, config.vsock_cid)
243            .await?;
244
245        Ok(EphemeralVmHandle {
246            config: config.clone(),
247            firecracker_process,
248            api_socket,
249            temp_dir,
250            start_time: SystemTime::now(),
251        })
252    }

254    /// Create Firecracker VM configuration
255    fn create_vm_config(
256        &self,
257        config: &EphemeralVmConfig,
258        temp_dir: &Path,
259    ) -> Result<serde_json::Value> {
260        let rootfs_path = temp_dir.join("rootfs.ext4");
261
262        Ok(serde_json::json!({
263            "boot-source": {
264                "kernel_image_path": config.kernel_path.to_string_lossy(),
265                "boot_args": format!(
266                    "console=ttyS0 reboot=k panic=1 pci=off init=/usr/bin/osvm-tool-executor tool={} server={}",
267                    config.tool_name,
268                    config.server_id
269                ),
270            },
271            "drives": [{
272                "drive_id": "rootfs",
273                "path_on_host": rootfs_path.to_string_lossy(),
274                "is_root_device": true,
275                "is_read_only": false,
276            }],
277            "machine-config": {
278                "vcpu_count": config.cpus,
279                "mem_size_mib": config.memory_mb,
280                "smt": false,
281            },
282            "network-interfaces": [],
283            "vsock": {
284                "guest_cid": config.vsock_cid,
285                "uds_path": format!("/tmp/osvm-vsock-{}.sock", config.id),
286            },
287            "logger": if self.debug_mode {
288                serde_json::json!({
289                    "level": "Debug",
290                    "log_path": temp_dir.join("firecracker.log").to_string_lossy(),
291                })
292            } else {
293                serde_json::json!(null)
294            },
295            "metrics": serde_json::json!(null),
296        }))
297    }

299    /// Wait for VM to be ready
300    async fn wait_for_vm_ready(&self, api_socket: &Path, vsock_cid: u32) -> Result<()> {
301        let start = SystemTime::now();
302        let timeout_duration = Duration::from_secs(10);
303
304        while SystemTime::now().duration_since(start)? < timeout_duration {
305            // Check if API socket exists
306            if api_socket.exists() {
307                // Try to connect via vsock
308                if let Ok(stream) = timeout(
309                    Duration::from_secs(1),
310                    tokio_vsock::VsockStream::connect(vsock_cid, VSOCK_TOOL_PORT),
311                )
312                .await
313                {
314                    if stream.is_ok() {
315                        debug!("VM {} is ready", vsock_cid);
316                        return Ok(());
317                    }
318                }
319            }
320            sleep(Duration::from_millis(100)).await;
321        }
322
323        Err(anyhow!("Timeout waiting for VM to be ready"))
324    }

326    /// Execute tool in the VM via vsock
327    async fn execute_tool_in_vm(
328        &self,
329        vsock_cid: u32,
330        tool_name: &str,
331        tool_args: Option<serde_json::Value>,
332        timeout_secs: u64,
333    ) -> Result<serde_json::Value> {
334        debug!("Executing tool {} in VM (CID: {})", tool_name, vsock_cid);
335
336        // Connect to VM via vsock
337        let mut stream = timeout(
338            Duration::from_secs(5),
339            tokio_vsock::VsockStream::connect(vsock_cid, VSOCK_TOOL_PORT),
340        )
341        .await
342        .context("Timeout connecting to ephemeral VM")?
343        .context("Failed to connect to ephemeral VM")?;
344
345        // Send tool execution request
346        let request = serde_json::json!({
347            "tool": tool_name,
348            "args": tool_args,
349        });
350
351        let request_bytes = serde_json::to_vec(&request)?;
352        stream
353            .write_all(&request_bytes)
354            .await
355            .context("Failed to send tool request")?;
356        stream.flush().await?;
357
358        // Read response with timeout
359        let response = timeout(
360            Duration::from_secs(timeout_secs),
361            self.read_tool_response(&mut stream),
362        )
363        .await
364        .context("Tool execution timeout")?
365        .context("Failed to read tool response")?;
366
367        Ok(response)
368    }
369
370    /// Read tool response from vsock stream
371    async fn read_tool_response(
372        &self,
373        stream: &mut tokio_vsock::VsockStream,
374    ) -> Result<serde_json::Value> {
375        let mut buffer = Vec::new();
376        let mut temp_buf = [0u8; 4096];
377
378        loop {
379            match stream.read(&mut temp_buf).await {
380                Ok(0) => break, // EOF
381                Ok(n) => {
382                    buffer.extend_from_slice(&temp_buf[..n]);
383                    if buffer.len() > 10 * 1024 * 1024 {
384                        return Err(anyhow!("Response too large"));
385                    }
386                }
387                Err(e) => return Err(anyhow!("Failed to read response: {}", e)),
388            }
389        }
390
391        serde_json::from_slice(&buffer).context("Failed to parse tool response")
392    }

394    /// Cleanup a VM
395    async fn cleanup_vm(&self, vm_id: &str) -> Result<()> {
396        let handle = {
397            let mut vms = self.active_vms.write().await;
398            vms.remove(vm_id)
399        };
400
401        if let Some(handle) = handle {
402            let handle = Arc::try_unwrap(handle)
403                .map_err(|_| anyhow!("VM handle still in use"))?
404                .into_inner();
405            handle.terminate().await?;
406            info!("Cleaned up ephemeral VM: {}", vm_id);
407        }
408
409        Ok(())
410    }
411
412    /// Cleanup all active VMs (for shutdown)
413    pub async fn cleanup_all(&self) -> Result<()> {
414        let vms = {
415            let mut vms = self.active_vms.write().await;
416            std::mem::take(&mut *vms)
417        };
418
419        for (id, handle) in vms {
420            if let Ok(handle) = Arc::try_unwrap(handle) {
421                if let Err(e) = handle.into_inner().terminate().await {
422                    error!("Failed to cleanup VM {}: {}", id, e);
423                }
424            }
425        }
426
427        Ok(())
428    }

431/// Main chat microVM orchestrator
432pub struct ChatVmOrchestrator {
433    /// Handle to the persistent chat VM
434    chat_vm: Option<Arc<Mutex<EphemeralVmHandle>>>,
435    /// Ephemeral VM manager for tools
436    ephemeral_manager: Arc<EphemeralVmManager>,
437    /// Debug mode flag
438    debug_mode: bool,
439}
440
441impl ChatVmOrchestrator {
442    /// Create a new chat VM orchestrator
443    pub fn new(debug_mode: bool) -> Self {
444        Self {
445            chat_vm: None,
446            ephemeral_manager: Arc::new(EphemeralVmManager::new(debug_mode)),
447            debug_mode,
448        }
449    }
450
451    /// Start the main chat VM
452    pub async fn start_chat_vm(&mut self) -> Result<()> {
453        info!("Starting main chat microVM");
454
455        let mut config = EphemeralVmConfig::default();
456        config.id = format!("chat-{}", Uuid::new_v4());
457        config.tool_name = "main-chat".to_string();
458        config.server_id = "osvm-chat".to_string();
459        config.memory_mb = 512; // More memory for chat VM
460        config.cpus = 2; // More CPUs for chat VM
461        config.timeout_secs = 3600; // 1 hour timeout for chat
462        config.debug = self.debug_mode;
463
464        let handle = self.ephemeral_manager.spawn_ephemeral_vm(&config).await?;
465        self.chat_vm = Some(Arc::new(Mutex::new(handle)));
466
467        Ok(())
468    }
469
470    /// Execute a tool in an ephemeral VM
471    pub async fn execute_tool(
472        &self,
473        server_id: &str,
474        tool_name: &str,
475        tool_args: Option<serde_json::Value>,
476    ) -> Result<serde_json::Value> {
477        self.ephemeral_manager
478            .launch_tool_vm(server_id, tool_name, tool_args)
479            .await
480    }
481
482    /// Stop the chat VM and cleanup
483    pub async fn stop(&mut self) -> Result<()> {
484        // Stop chat VM
485        if let Some(chat_vm) = self.chat_vm.take() {
486            if let Ok(handle) = Arc::try_unwrap(chat_vm) {
487                handle.into_inner().terminate().await?;
488            }
489        }
490
491        // Cleanup all ephemeral VMs
492        self.ephemeral_manager.cleanup_all().await?;
493
494        Ok(())
495    }
496}
