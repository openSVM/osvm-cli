142    /// Analyze terminal screen layout
143    pub fn analyze_layout(&self, screen_content: &str, width: u16, height: u16) -> Result<LayoutAnalysis> {
144        let components = self.detect_components(screen_content, width, height)?;
145        let borders = self.detect_borders(screen_content, width, height)?;
146        let text_regions = self.detect_text_regions(screen_content, width, height)?;
147        let layout_metrics = self.calculate_metrics(&components, &borders, width, height);
148
149        Ok(LayoutAnalysis {
150            width,
151            height,
152            components,
153            borders,
154            text_regions,
155            layout_metrics,
156        })
157    }

159    /// Detect UI components in screen content
160    fn detect_components(&self, content: &str, width: u16, height: u16) -> Result<Vec<DetectedComponent>> {
161        let mut components = Vec::new();
162        let lines: Vec<&str> = content.lines().collect();
163
164        // Detect title bar (usually top line with centered text)
165        if let Some(first_line) = lines.first() {
166            if self.looks_like_title_bar(first_line) {
167                components.push(DetectedComponent {
168                    component_type: ComponentType::TitleBar,
169                    position: Position { row: 0, col: 0 },
170                    size: Size { width, height: 1 },
171                    content: first_line.trim().to_string(),
172                });
173            }
174        }
175
176        // Detect status bar (usually bottom line)
177        if let Some(last_line) = lines.last() {
178            if self.looks_like_status_bar(last_line) {
179                components.push(DetectedComponent {
180                    component_type: ComponentType::StatusBar,
181                    position: Position { row: height - 1, col: 0 },
182                    size: Size { width, height: 1 },
183                    content: last_line.trim().to_string(),
184                });
185            }
186        }
187
188        // Detect input fields (lines with prompts like ">" or ":")
189        for (idx, line) in lines.iter().enumerate() {
190            if self.looks_like_input_field(line) {
191                components.push(DetectedComponent {
192                    component_type: ComponentType::InputField,
193                    position: Position { row: idx as u16, col: 0 },
194                    size: Size { width, height: 1 },
195                    content: line.trim().to_string(),
196                });
197            }
198        }
199
200        // Detect help dialogs (boxes with "Help" or "?" in content)
201        if content.contains("Help") || content.contains("Commands") {
202            components.push(DetectedComponent {
203                component_type: ComponentType::HelpDialog,
204                position: Position { row: 0, col: 0 },
205                size: Size { width: 0, height: 0 }, // Will be calculated
206                content: "Help dialog detected".to_string(),
207            });
208        }
209
210        // Detect session list (FAR-style left panel)
211        if content.contains("Sessions") || content.contains("Session") {
212            components.push(DetectedComponent {
213                component_type: ComponentType::SessionList,
214                position: Position { row: 0, col: 0 },
215                size: Size { width: 0, height: 0 },
216                content: "Session list detected".to_string(),
217            });
218        }
219
220        Ok(components)
221    }

223    /// Detect borders (box drawing characters)
224    fn detect_borders(&self, content: &str, width: u16, height: u16) -> Result<Vec<DetectedBorder>> {
225        let mut borders = Vec::new();
226        let lines: Vec<&str> = content.lines().collect();
227
228        for (row, line) in lines.iter().enumerate() {
229            let mut current_border: Option<(usize, BorderStyle)> = None;
230
231            for (col, ch) in line.chars().enumerate() {
232                if self.is_border_char(ch) {
233                    let style = self.detect_border_style(ch);
234
235                    if let Some((start_col, existing_style)) = current_border {
236                        if style != existing_style {
237                            // Border style changed, save previous border
238                            borders.push(DetectedBorder {
239                                style: existing_style,
240                                position: Position { row: row as u16, col: start_col as u16 },
241                                size: Size { width: (col - start_col) as u16, height: 1 },
242                            });
243                            current_border = Some((col, style));
244                        }
245                    } else {
246                        current_border = Some((col, style));
247                    }
248                } else if let Some((start_col, style)) = current_border {
249                    // Border ended
250                    borders.push(DetectedBorder {
251                        style,
252                        position: Position { row: row as u16, col: start_col as u16 },
253                        size: Size { width: (col - start_col) as u16, height: 1 },
254                    });
255                    current_border = None;
256                }
257            }
258
259            // Handle border that extends to end of line
260            if let Some((start_col, style)) = current_border {
261                borders.push(DetectedBorder {
262                    style,
263                    position: Position { row: row as u16, col: start_col as u16 },
264                    size: Size { width: (line.len() - start_col) as u16, height: 1 },
265                });
266            }
267        }
268
269        Ok(borders)
270    }

272    /// Detect text regions
273    fn detect_text_regions(&self, content: &str, width: u16, height: u16) -> Result<Vec<TextRegion>> {
274        let mut regions = Vec::new();
275        let lines: Vec<&str> = content.lines().collect();
276
277        for (row, line) in lines.iter().enumerate() {
278            let trimmed = line.trim();
279            if !trimmed.is_empty() && !self.is_all_borders(line) {
280                regions.push(TextRegion {
281                    position: Position { row: row as u16, col: 0 },
282                    size: Size { width: trimmed.len() as u16, height: 1 },
283                    content: trimmed.to_string(),
284                    is_emphasized: self.has_emphasis(line),
285                });
286            }
287        }
288
289        Ok(regions)
290    }

292    /// Calculate layout metrics
293    fn calculate_metrics(
294        &self,
295        components: &[DetectedComponent],
296        borders: &[DetectedBorder],
297        width: u16,
298        height: u16,
299    ) -> LayoutMetrics {
300        let total_borders = borders.len();
301        let total_text_regions = components.len();
302
303        // Simple heuristic for nesting depth (count overlapping borders)
304        let max_nesting_depth = self.calculate_nesting_depth(borders);
305
306        // Calculate symmetry (how balanced the layout is)
307        let symmetry_score = self.calculate_symmetry(components, width, height);
308
309        // Calculate alignment (how well components align)
310        let alignment_score = self.calculate_alignment(components);
311
312        LayoutMetrics {
313            total_borders,
314            total_text_regions,
315            max_nesting_depth,
316            symmetry_score,
317            alignment_score,
318        }
319    }

321    /// Compare two screenshots for visual regression
322    pub fn compare_screenshots(&self, baseline: &str, current: &str, name: &str) -> Result<RegressionResult> {
323        let baseline_path = self.baseline_dir.join(format!("{}_baseline.txt", name));
324        let current_path = self.baseline_dir.join(format!("{}_current.txt", name));
325
326        // Save current for comparison
327        std::fs::write(&current_path, current)?;
328
329        // Check if baseline exists
330        if !baseline_path.exists() {
331            // Create baseline
332            std::fs::write(&baseline_path, baseline)?;
333            return Ok(RegressionResult {
334                passed: true,
335                similarity_score: 1.0,
336                differences: vec![],
337                baseline_path,
338                current_path,
339            });
340        }
341
342        // Load baseline
343        let baseline_content = std::fs::read_to_string(&baseline_path)?;
344
345        // Calculate similarity
346        let similarity = self.calculate_similarity(&baseline_content, current);
347        let passed = similarity >= self.tolerance;
348
349        // Find differences
350        let differences = self.find_differences(&baseline_content, current);
351
352        Ok(RegressionResult {
353            passed,
354            similarity_score: similarity,
355            differences,
356            baseline_path,
357            current_path,
358        })
359    }

361    /// Calculate similarity between two screenshots (0.0 - 1.0)
362    fn calculate_similarity(&self, baseline: &str, current: &str) -> f32 {
363        let baseline_lines: Vec<&str> = baseline.lines().collect();
364        let current_lines: Vec<&str> = current.lines().collect();
365
366        if baseline_lines.is_empty() && current_lines.is_empty() {
367            return 1.0;
368        }
369
370        let max_lines = baseline_lines.len().max(current_lines.len());
371        if max_lines == 0 {
372            return 1.0;
373        }
374
375        let mut matching_lines = 0;
376        let mut partial_matches = 0.0;
377
378        for i in 0..max_lines {
379            match (baseline_lines.get(i), current_lines.get(i)) {
380                (Some(baseline_line), Some(current_line)) => {
381                    if baseline_line == current_line {
382                        matching_lines += 1;
383                    } else {
384                        // Calculate line similarity
385                        let line_similarity = self.calculate_line_similarity(baseline_line, current_line);
386                        partial_matches += line_similarity;
387                    }
388                }
389                _ => {
390                    // One line is missing
391                }
392            }
393        }
394
395        (matching_lines as f32 + partial_matches) / max_lines as f32
396    }

398    /// Calculate similarity between two lines
399    fn calculate_line_similarity(&self, baseline: &str, current: &str) -> f32 {
400        let max_len = baseline.len().max(current.len());
401        if max_len == 0 {
402            return 1.0;
403        }
404
405        let matching_chars: usize = baseline
406            .chars()
407            .zip(current.chars())
408            .filter(|(a, b)| a == b)
409            .count();
410
411        matching_chars as f32 / max_len as f32
412    }

414    /// Find specific differences between screenshots
415    fn find_differences(&self, baseline: &str, current: &str) -> Vec<VisualDifference> {
416        let mut differences = Vec::new();
417        let baseline_lines: Vec<&str> = baseline.lines().collect();
418        let current_lines: Vec<&str> = current.lines().collect();
419
420        let max_lines = baseline_lines.len().max(current_lines.len());
421
422        for i in 0..max_lines {
423            match (baseline_lines.get(i), current_lines.get(i)) {
424                (Some(baseline_line), Some(current_line)) => {
425                    if baseline_line != current_line {
426                        differences.push(VisualDifference {
427                            diff_type: DifferenceType::ContentChanged,
428                            position: Position { row: i as u16, col: 0 },
429                            description: format!("Line {} changed", i + 1),
430                        });
431                    }
432                }
433                (Some(_), None) => {
434                    differences.push(VisualDifference {
435                        diff_type: DifferenceType::MissingComponent,
436                        position: Position { row: i as u16, col: 0 },
437                        description: format!("Line {} missing in current", i + 1),
438                    });
439                }
440                (None, Some(_)) => {
441                    differences.push(VisualDifference {
442                        diff_type: DifferenceType::ExtraComponent,
443                        position: Position { row: i as u16, col: 0 },
444                        description: format!("Extra line {} in current", i + 1),
445                    });
446                }
447                (None, None) => unreachable!(),
448            }
449        }
450
451        differences
452    }

456    fn looks_like_title_bar(&self, line: &str) -> bool {
457        let trimmed = line.trim();
458        // Title bars usually have some centering and are not too long
459        trimmed.len() > 5 && trimmed.len() < 60 && !self.is_all_borders(line)
460    }

462    fn looks_like_status_bar(&self, line: &str) -> bool {
463        // Status bars often contain key hints or status info
464        line.contains("F10") || line.contains("Help") || line.contains("Quit") || line.contains("Session")
465    }
466
467    fn looks_like_input_field(&self, line: &str) -> bool {
468        let trimmed = line.trim();
469        trimmed.starts_with('>') || trimmed.starts_with(':') || trimmed.contains("Type a message")
470    }

472    fn is_border_char(&self, ch: char) -> bool {
473        matches!(ch,
474            '─' | '│' | '┌' | '┐' | '└' | '┘' |  // Single
475            '═' | '║' | '╔' | '╗' | '╚' | '╝' |  // Double
476            '━' | '┃' | '┏' | '┓' | '┗' | '┛' |  // Heavy
477            '╭' | '╮' | '╰' | '╯' |              // Rounded
478            '├' | '┤' | '┬' | '┴' | '┼' |        // Connectors
479            '╠' | '╣' | '╦' | '╩' | '╬'          // Double connectors
480        )
481    }
482
483    fn detect_border_style(&self, ch: char) -> BorderStyle {
484        match ch {
485            '─' | '│' | '┌' | '┐' | '└' | '┘' | '├' | '┤' | '┬' | '┴' | '┼' => BorderStyle::Single,
486            '═' | '║' | '╔' | '╗' | '╚' | '╝' | '╠' | '╣' | '╦' | '╩' | '╬' => BorderStyle::Double,
487            '━' | '┃' | '┏' | '┓' | '┗' | '┛' => BorderStyle::Heavy,
488            '╭' | '╮' | '╰' | '╯' => BorderStyle::Rounded,
489            _ => BorderStyle::Mixed,
490        }
491    }
492
493    fn is_all_borders(&self, line: &str) -> bool {
494        !line.is_empty() && line.chars().all(|ch| self.is_border_char(ch) || ch.is_whitespace())
495    }

497    fn has_emphasis(&self, line: &str) -> bool {
498        // Check for ANSI codes indicating emphasis
499        line.contains("\x1b[1m") || line.contains("\x1b[4m") || line.contains("\x1b[7m")
500    }

502    fn calculate_nesting_depth(&self, borders: &[DetectedBorder]) -> usize {
503        // Simple heuristic: count distinct rows with borders
504        let mut rows = std::collections::HashSet::new();
505        for border in borders {
506            rows.insert(border.position.row);
507        }
508        rows.len().min(5) // Cap at 5 for sanity
509    }
510
511    fn calculate_symmetry(&self, components: &[DetectedComponent], width: u16, height: u16) -> f32 {
512        if components.is_empty() {
513            return 1.0;
514        }
515
516        // Calculate center of mass
517        let center_col = width / 2;
518        let mut left_weight = 0.0;
519        let mut right_weight = 0.0;
520
521        for component in components {
522            let component_center = component.position.col + component.size.width / 2;
523            if component_center < center_col {
524                left_weight += component.size.width as f32;
525            } else {
526                right_weight += component.size.width as f32;
527            }
528        }
529
530        let total_weight = left_weight + right_weight;
531        if total_weight == 0.0 {
532            return 1.0;
533        }
534
535        1.0 - ((left_weight - right_weight).abs() / total_weight)
536    }
537
538    fn calculate_alignment(&self, components: &[DetectedComponent]) -> f32 {
539        if components.len() < 2 {
540            return 1.0;
541        }
542
543        // Check how many components share the same column positions
544        let mut col_counts: HashMap<u16, usize> = HashMap::new();
545        for component in components {
546            *col_counts.entry(component.position.col).or_insert(0) += 1;
547        }
548
549        let max_aligned = col_counts.values().max().copied().unwrap_or(1);
550        max_aligned as f32 / components.len() as f32
551    }
