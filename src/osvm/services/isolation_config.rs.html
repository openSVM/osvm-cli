1//! Isolation configuration for MCP tools and unikernel execution
2//!
3//! This module defines the configuration structure for controlling how MCP tools
4//! are executed - either directly in microVMs or in isolated unikernels.
5
6use anyhow::{Context, Result};
7use serde::{Deserialize, Serialize};
8use std::collections::HashMap;
9use std::path::PathBuf;

11/// Execution mode for MCP tools
12#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
13#[serde(rename_all = "lowercase")]
14#[derive(Default)]
15pub enum ExecutionMode {
16    /// Execute tool in ephemeral unikernel (high security, ~100ms overhead)
17    Unikernel,
18    /// Execute tool directly in microVM (Firecracker - preferred approach)
19    #[default]
20    MicroVM,
21}
22
23/// Mount configuration for filesystem access
24#[derive(Debug, Clone, Serialize, Deserialize)]
25pub struct MountConfig {
26    /// Path on the host system
27    pub host_path: String,
28    /// Path inside the VM/unikernel
29    pub vm_path: String,
30    /// Whether the mount is read-only
31    #[serde(default)]
32    pub readonly: bool,
33}
34
35/// Configuration for a specific MCP tool
36#[derive(Debug, Clone, Serialize, Deserialize)]
37pub struct ToolConfig {
38    /// How this tool should be executed
39    #[serde(default)]
40    pub execution_mode: ExecutionMode,
41    /// Path to the unikernel image (if execution_mode is Unikernel)
42    pub unikernel_image: Option<String>,
43    /// Filesystem mounts available to this tool
44    #[serde(default)]
45    pub mounts: Vec<MountConfig>,
46    /// Memory limit in MB for unikernel
47    #[serde(default = "default_memory_mb")]
48    pub memory_mb: u32,
49    /// Number of vCPUs for unikernel
50    #[serde(default = "default_vcpus")]
51    pub vcpus: u32,
52}
53
54fn default_memory_mb() -> u32 {
55    128
56}
57
58fn default_vcpus() -> u32 {
59    1
60}
61
62impl Default for ToolConfig {
63    fn default() -> Self {
64        Self {
65            execution_mode: ExecutionMode::MicroVM,  // Use Firecracker MicroVM by default
66            unikernel_image: None,
67            mounts: Vec::new(),
68            memory_mb: default_memory_mb(),
69            vcpus: default_vcpus(),
70        }
71    }
72}

102/// Configuration for an MCP server
103#[derive(Debug, Clone, Serialize, Deserialize)]
104pub struct ServerConfig {
105    /// MicroVM ID where this server runs
106    pub microvm_id: String,
107    /// Whether to run this server in a dedicated microVM (Phase 3.2)
108    #[serde(default)]
109    pub use_microvm: bool,
110    /// Resource configuration for the server's microVM (if use_microvm is true)
111    #[serde(default)]
112    pub microvm_config: MicroVmResourceConfig,
113    /// Command to run the MCP server (used when use_microvm is true)
114    #[serde(default)]
115    pub server_command: Option<String>,
116    /// Mounts available to the microVM (server level)
117    #[serde(default)]
118    pub microvm_mounts: Vec<MountConfig>,
119    /// Per-tool configuration
120    #[serde(default)]
121    pub tools: HashMap<String, ToolConfig>,
122}

124/// Root isolation configuration
125#[derive(Debug, Clone, Serialize, Deserialize)]
126pub struct IsolationConfig {
127    /// Default execution mode for tools without specific config
128    #[serde(default)]
129    pub default_execution_mode: ExecutionMode,
130    /// Directory where unikernel images are stored
131    #[serde(default = "default_unikernel_dir")]
132    pub unikernel_dir: String,
133    /// Per-server configuration
134    #[serde(default)]
135    pub mcp_servers: HashMap<String, ServerConfig>,

138fn default_unikernel_dir() -> String {
139    "/var/osvm/unikernels".to_string()
140}

152impl IsolationConfig {
153    /// Load isolation configuration from file
154    pub fn load() -> Result<Self> {
155        let config_path = Self::get_config_path()?;
156
157        if !config_path.exists() {
158            // Create default config if it doesn't exist
159            let default_config = Self::create_default_config();
160            default_config.save()?;
161            return Ok(default_config);
162        }
163
164        let content = std::fs::read_to_string(&config_path)
165            .context("Failed to read isolation config file")?;
166
167        let config: IsolationConfig =
168            serde_json::from_str(&content).context("Failed to parse isolation config")?;
169
170        Ok(config)
171    }
172
173    /// Save isolation configuration to file
174    pub fn save(&self) -> Result<()> {
175        let config_path = Self::get_config_path()?;
176
177        // Create parent directory if it doesn't exist
178        if let Some(parent) = config_path.parent() {
179            std::fs::create_dir_all(parent).context("Failed to create config directory")?;
180        }
181
182        let json =
183            serde_json::to_string_pretty(self).context("Failed to serialize isolation config")?;
184
185        std::fs::write(&config_path, json).context("Failed to write isolation config")?;
186
187        Ok(())
188    }
189
190    /// Get the path to the isolation config file
191    fn get_config_path() -> Result<PathBuf> {
192        let home = std::env::var("HOME").context("HOME environment variable not set")?;
193
194        Ok(PathBuf::from(home)
195            .join(".config")
196            .join("osvm")
197            .join("isolation_config.json"))
198    }

294    /// Get configuration for a specific tool
295    pub fn get_tool_config(&self, server_id: &str, tool_name: &str) -> ToolConfig {
296        self.mcp_servers
297            .get(server_id)
298            .and_then(|server| server.tools.get(tool_name))
299            .cloned()
300            .unwrap_or_else(|| ToolConfig {
301                execution_mode: self.default_execution_mode.clone(),
302                ..Default::default()
303            })
304    }
305
306    /// Get server configuration
307    pub fn get_server_config(&self, server_id: &str) -> Option<&ServerConfig> {
308        self.mcp_servers.get(server_id)
309    }
310
311    /// Check if a tool should use unikernel execution
312    pub fn should_use_unikernel(&self, server_id: &str, tool_name: &str) -> bool {
313        let tool_config = self.get_tool_config(server_id, tool_name);
314        tool_config.execution_mode == ExecutionMode::Unikernel
315    }
