101    pub async fn read_accounts(
102        &self,
103        output_config: OutputConfig,
104        filter_options: FilterOptions,
105        limit: Option<usize>,
106        offset: usize,
107    ) -> Result<()> {
108        let accounts_dir = self.snapshot_dir.join("accounts");
109        if !accounts_dir.exists() {
110            return Err(anyhow!("Accounts directory not found in snapshot"));
111        }
112
113        // Read all account files
114        let mut entries: Vec<_> = fs::read_dir(&accounts_dir)?
115            .filter_map(|e| e.ok())
116            .collect();
117
118        entries.sort_by_key(|e| e.file_name());
119
120        let total_accounts = entries.len();
121        let entries_to_process: Vec<_> = entries
122            .into_iter()
123            .skip(offset)
124            .take(limit.unwrap_or(usize::MAX))
125            .collect();
126
127        if !output_config.quiet {
128            println!("=== Solana Snapshot Account Stream ===");
129            println!("Snapshot Directory: {:?}", self.snapshot_dir);
130            println!("Total accounts: {}", total_accounts);
131            println!("Processing: {} accounts", entries_to_process.len());
132            println!("{}", "=".repeat(80));
133            println!();
134        }
135
136        let progress = if output_config.show_progress {
137            let pb = ProgressBar::new(entries_to_process.len() as u64);
138            pb.set_style(
139                ProgressStyle::default_bar()
140                    .template("[{elapsed_precise}] {bar:40.cyan/blue} {pos}/{len} {msg}")
141                    .unwrap()
142                    .progress_chars("=>-"),
143            );
144            Some(pb)
145        } else {
146            None
147        };
148
149        let results = Arc::new(Mutex::new(Vec::new()));
150
151        if let Some(threads) = self.parallel_threads {
152            // Parallel processing
153            let pool = rayon::ThreadPoolBuilder::new()
154                .num_threads(threads)
155                .build()?;
156
157            pool.install(|| {
158                entries_to_process.par_iter().for_each(|entry| {
159                    let path = entry.path();
160                    match read_account_file(&path) {
161                        Ok(info) => {
162                            if filter_matches(&info, &filter_options) {
163                                results.lock().unwrap().push((entry.file_name(), info));
164                            }
165                        }
166                        Err(e) => {
167                            eprintln!("Error reading {:?}: {}", path, e);
168                        }
169                    }
170                    if let Some(ref pb) = progress {
171                        pb.inc(1);
172                    }
173                });
174            });
175        } else {
176            // Sequential processing
177            for entry in entries_to_process {
178                let path = entry.path();
179                match read_account_file(&path) {
180                    Ok(info) => {
181                        if filter_matches(&info, &filter_options) {
182                            results.lock().unwrap().push((entry.file_name(), info));
183                        }
184                    }
185                    Err(e) => {
186                        eprintln!("Error reading {:?}: {}", path, e);
187                    }
188                }
189                if let Some(ref pb) = progress {
190                    pb.inc(1);
191                }
192            }
193        }
194
195        if let Some(pb) = progress {
196            pb.finish_with_message("Complete");
197        }
198
199        // Display results
200        let results = results.lock().unwrap();
201        for (idx, (filename, info)) in results.iter().enumerate() {
202            display_account(idx + 1, &filename.to_string_lossy(), info, &output_config)?;
203        }
204
205        if !output_config.quiet {
206            println!("{}", "=".repeat(80));
207            println!("Processed {} accounts", results.len());
208        }
209
210        Ok(())
211    }
212}

231    pub async fn compute_statistics(&self) -> Result<Statistics> {
232        let accounts_dir = self.snapshot_dir.join("accounts");
233        let entries: Vec<_> = fs::read_dir(&accounts_dir)?
234            .filter_map(|e| e.ok())
235            .collect();
236
237        let mut total_lamports = 0u64;
238        let mut total_data_size = 0u64;
239        let mut balances = Vec::new();
240        let mut owner_counts: HashMap<String, usize> = HashMap::new();
241        let mut executable_count = 0;
242        let mut rent_exempt_count = 0;
243        let mut account_balances: Vec<(String, u64)> = Vec::new();
244
245        for entry in &entries {
246            if let Ok(info) = read_account_file(&entry.path()) {
247                total_lamports += info.lamports;
248                total_data_size += info.data_len;
249                balances.push(info.lamports);
250
251                let owner_str = info.owner.to_string();
252                *owner_counts.entry(owner_str).or_insert(0) += 1;
253
254                if info.executable {
255                    executable_count += 1;
256                }
257
258                // Approximate rent exemption (> 2 years rent)
259                if info.rent_epoch == u64::MAX {
260                    rent_exempt_count += 1;
261                }
262
263                if let Some(pubkey) = info.pubkey {
264                    account_balances.push((pubkey.to_string(), info.lamports));
265                }
266            }
267        }
268
269        balances.sort_unstable();
270        let median_balance = if !balances.is_empty() {
271            balances[balances.len() / 2]
272        } else {
273            0
274        };
275
276        let average_balance = if !entries.is_empty() {
277            total_lamports / entries.len() as u64
278        } else {
279            0
280        };
281
282        let average_data_size = if !entries.is_empty() {
283            total_data_size / entries.len() as u64
284        } else {
285            0
286        };
287
288        let mut accounts_by_owner: Vec<_> = owner_counts.into_iter().collect();
289        accounts_by_owner.sort_by(|a, b| b.1.cmp(&a.1));
290
291        account_balances.sort_by(|a, b| b.1.cmp(&a.1));
292        let largest_accounts = account_balances.into_iter().take(10).collect();
293
294        Ok(Statistics {
295            total_accounts: entries.len(),
296            total_lamports,
297            average_balance,
298            median_balance,
299            total_data_size,
300            average_data_size,
301            accounts_by_owner,
302            largest_accounts,
303            executable_count,
304            rent_exempt_count,
305        })
306    }

308    pub async fn export(
309        &self,
310        output_file: PathBuf,
311        format: ExportFormat,
312        filter_options: FilterOptions,
313    ) -> Result<()> {
314        let accounts_dir = self.snapshot_dir.join("accounts");
315        let entries: Vec<_> = fs::read_dir(&accounts_dir)?
316            .filter_map(|e| e.ok())
317            .collect();
318
319        let mut accounts = Vec::new();
320        for entry in entries {
321            if let Ok(info) = read_account_file(&entry.path()) {
322                if filter_matches(&info, &filter_options) {
323                    accounts.push(info);
324                }
325            }
326        }
327
328        match format {
329            ExportFormat::Json => {
330                let json = serde_json::to_string_pretty(&accounts)?;
331                fs::write(output_file, json)?;
332            }
333            ExportFormat::Csv => {
334                let mut wtr = csv::Writer::from_path(output_file)?;
335                wtr.write_record([
336                    "pubkey",
337                    "lamports",
338                    "data_len",
339                    "owner",
340                    "executable",
341                    "rent_epoch",
342                ])?;
343                for account in accounts {
344                    wtr.write_record(&[
345                        account.pubkey.map(|p| p.to_string()).unwrap_or_default(),
346                        account.lamports.to_string(),
347                        account.data_len.to_string(),
348                        account.owner.to_string(),
349                        account.executable.to_string(),
350                        account.rent_epoch.to_string(),
351                    ])?;
352                }
353                wtr.flush()?;
354            }
355            ExportFormat::Parquet => {
356                return Err(anyhow!("Parquet export not yet implemented"));
357            }
358            ExportFormat::MessagePack => {
359                let bytes = rmp_serde::to_vec(&accounts)?;
360                fs::write(output_file, bytes)?;
361            }
362        }
363
364        Ok(())
365    }

367    pub async fn compare_with(&self, other_snapshot: &Path) -> Result<serde_json::Value> {
368        let accounts_dir1 = self.snapshot_dir.join("accounts");
369        let accounts_dir2 = other_snapshot.join("accounts");
370
371        if !accounts_dir2.exists() {
372            return Err(anyhow!("Second snapshot accounts directory not found"));
373        }
374
375        // Read accounts from both snapshots
376        let entries1: Vec<_> = fs::read_dir(&accounts_dir1)?
377            .filter_map(|e| e.ok())
378            .collect();
379        let entries2: Vec<_> = fs::read_dir(&accounts_dir2)?
380            .filter_map(|e| e.ok())
381            .collect();
382
383        let mut accounts1: HashMap<String, AccountInfo> = HashMap::new();
384        let mut accounts2: HashMap<String, AccountInfo> = HashMap::new();
385
386        // Load first snapshot accounts
387        for entry in entries1 {
388            if let Ok(info) = read_account_file(&entry.path()) {
389                let key = entry.file_name().to_string_lossy().to_string();
390                accounts1.insert(key, info);
391            }
392        }
393
394        // Load second snapshot accounts
395        for entry in entries2 {
396            if let Ok(info) = read_account_file(&entry.path()) {
397                let key = entry.file_name().to_string_lossy().to_string();
398                accounts2.insert(key, info);
399            }
400        }
401
402        // Compare snapshots
403        let mut new_accounts = Vec::new();
404        let mut deleted_accounts = Vec::new();
405        let mut modified_accounts = Vec::new();
406
407        // Find new and modified accounts
408        for (key, info2) in &accounts2 {
409            if let Some(info1) = accounts1.get(key) {
410                // Account exists in both - check if modified
411                if info1.lamports != info2.lamports
412                    || info1.data_len != info2.data_len
413                    || info1.owner != info2.owner
414                {
415                    modified_accounts.push(serde_json::json!({
416                        "account": key,
417                        "changes": {
418                            "lamports": {
419                                "old": info1.lamports,
420                                "new": info2.lamports,
421                                "diff": info2.lamports as i128 - info1.lamports as i128
422                            },
423                            "data_len": {
424                                "old": info1.data_len,
425                                "new": info2.data_len,
426                                "diff": info2.data_len as i128 - info1.data_len as i128
427                            },
428                            "owner_changed": info1.owner != info2.owner
429                        }
430                    }));
431                }
432            } else {
433                // New account in snapshot2
434                new_accounts.push(serde_json::json!({
435                    "account": key,
436                    "lamports": info2.lamports,
437                    "data_len": info2.data_len,
438                    "owner": info2.owner.to_string()
439                }));
440            }
441        }
442
443        // Find deleted accounts
444        for (key, info1) in &accounts1 {
445            if !accounts2.contains_key(key) {
446                deleted_accounts.push(serde_json::json!({
447                    "account": key,
448                    "lamports": info1.lamports,
449                    "data_len": info1.data_len,
450                    "owner": info1.owner.to_string()
451                }));
452            }
453        }
454
455        Ok(serde_json::json!({
456            "snapshot1": self.snapshot_dir.to_string_lossy(),
457            "snapshot2": other_snapshot.to_string_lossy(),
458            "summary": {
459                "total_accounts_snapshot1": accounts1.len(),
460                "total_accounts_snapshot2": accounts2.len(),
461                "new_accounts": new_accounts.len(),
462                "deleted_accounts": deleted_accounts.len(),
463                "modified_accounts": modified_accounts.len(),
464                "unchanged_accounts": accounts1.len() - deleted_accounts.len() - modified_accounts.len()
465            },
466            "new_accounts": new_accounts,
467            "deleted_accounts": deleted_accounts,
468            "modified_accounts": modified_accounts
469        }))
470    }

472    pub async fn validate(&self) -> Result<serde_json::Value> {
473        let accounts_dir = self.snapshot_dir.join("accounts");
474
475        if !accounts_dir.exists() {
476            return Ok(serde_json::json!({
477                "valid": false,
478                "errors": ["Accounts directory not found"],
479                "warnings": [],
480                "info": []
481            }));
482        }
483
484        let entries: Vec<_> = fs::read_dir(&accounts_dir)?
485            .filter_map(|e| e.ok())
486            .collect();
487
488        let mut errors = Vec::new();
489        let mut warnings = Vec::new();
490        let mut info = Vec::new();
491
492        let mut successful_reads = 0;
493        let mut failed_reads = 0;
494        let mut total_lamports = 0u64;
495        let mut suspicious_balances = 0;
496        let mut zero_balance_accounts = 0;
497
498        for entry in &entries {
499            match read_account_file(&entry.path()) {
500                Ok(account_info) => {
501                    successful_reads += 1;
502                    total_lamports += account_info.lamports;
503
504                    // Check for suspicious patterns
505                    if account_info.lamports == 0 && account_info.data_len > 0 {
506                        zero_balance_accounts += 1;
507                    }
508
509                    // Check for extremely large balances (potential corruption)
510                    if account_info.lamports > 1_000_000_000_000_000_000 {
511                        suspicious_balances += 1;
512                        warnings.push(format!(
513                            "Account {} has suspiciously large balance: {} lamports",
514                            entry.file_name().to_string_lossy(),
515                            account_info.lamports
516                        ));
517                    }
518
519                    // Validate data length consistency
520                    if account_info.data_len as usize != account_info.data.len() {
521                        warnings.push(format!(
522                            "Account {}: data_len mismatch (expected: {}, actual: {})",
523                            entry.file_name().to_string_lossy(),
524                            account_info.data_len,
525                            account_info.data.len()
526                        ));
527                    }
528                }
529                Err(e) => {
530                    failed_reads += 1;
531                    errors.push(format!(
532                        "Failed to read {}: {}",
533                        entry.file_name().to_string_lossy(),
534                        e
535                    ));
536                }
537            }
538        }
539
540        // Add informational messages
541        info.push(format!("Total account files: {}", entries.len()));
542        info.push(format!("Successfully parsed: {}", successful_reads));
543        info.push(format!("Failed to parse: {}", failed_reads));
544        info.push(format!("Total lamports: {}", total_lamports));
545        info.push(format!(
546            "Zero-balance accounts with data: {}",
547            zero_balance_accounts
548        ));
549
550        if suspicious_balances > 0 {
551            warnings.push(format!(
552                "Found {} accounts with suspiciously large balances",
553                suspicious_balances
554            ));
555        }
556
557        // Validate snapshot structure
558        if !self.snapshot_dir.join("version").exists() {
559            warnings.push("Version file not found".to_string());
560        }
561
562        if !self.snapshot_dir.join("snapshots").exists() {
563            warnings.push("Snapshots directory not found".to_string());
564        }
565
566        let is_valid = errors.is_empty() && successful_reads > 0;
567
568        Ok(serde_json::json!({
569            "valid": is_valid,
570            "summary": {
571                "total_files": entries.len(),
572                "successful_reads": successful_reads,
573                "failed_reads": failed_reads,
574                "total_lamports": total_lamports,
575                "zero_balance_with_data": zero_balance_accounts,
576                "suspicious_balances": suspicious_balances
577            },
578            "errors": errors,
579            "warnings": warnings,
580            "info": info
581        }))
582    }

584    pub async fn find_account(&self, pubkey: &str) -> Result<serde_json::Value> {
585        let accounts_dir = self.snapshot_dir.join("accounts");
586        let entries: Vec<_> = fs::read_dir(&accounts_dir)?
587            .filter_map(|e| e.ok())
588            .collect();
589
590        let total_files = entries.len();
591
592        // Search through all accounts
593        for entry in entries {
594            if let Ok(info) = read_account_file(&entry.path()) {
595                // Check if this is the account we're looking for
596                if let Some(account_pubkey) = info.pubkey {
597                    if account_pubkey.to_string() == pubkey {
598                        return Ok(serde_json::json!({
599                            "found": true,
600                            "file": entry.file_name().to_string_lossy(),
601                            "account": {
602                                "pubkey": account_pubkey.to_string(),
603                                "lamports": info.lamports,
604                                "data_len": info.data_len,
605                                "owner": info.owner.to_string(),
606                                "executable": info.executable,
607                                "rent_epoch": info.rent_epoch,
608                                "data_preview": format!("{:?}", &info.data[..info.data.len().min(64)])
609                            }
610                        }));
611                    }
612                }
613            }
614        }
615
616        Ok(serde_json::json!({
617            "found": false,
618            "message": format!("Account {} not found in snapshot", pubkey),
619            "searched_files": total_files
620        }))
621    }

645fn read_account_file(path: &Path) -> Result<AccountInfo> {
646    let mut file = fs::File::open(path)?;
647    let mut buffer = Vec::new();
648    file.read_to_end(&mut buffer)?;
649
650    if buffer.len() < 100 {
651        return Err(anyhow!("File too small to be a valid account"));
652    }
653
654    // Parse the account storage format
655    // This is a simplified parser - in production should use solana-runtime's AppendVec
656    let data_len = u64::from_le_bytes(buffer[8..16].try_into()?);
657    let owner_bytes: [u8; 32] = buffer[17..49].try_into()?;
658    let owner = Pubkey::new_from_array(owner_bytes);
659    let lamports = u64::from_le_bytes(buffer[49..57].try_into()?);
660    let rent_epoch = u64::from_le_bytes(buffer[57..65].try_into()?);
661
662    // Try to extract pubkey if available
663    let pubkey = if buffer.len() > 81 {
664        if let Ok(pubkey_bytes) = buffer[65..97].try_into() {
665            Some(Pubkey::new_from_array(pubkey_bytes))
666        } else {
667            None
668        }
669    } else {
670        None
671    };
672
673    let executable = false; // Would need proper parsing
674
675    let data_start = 130;
676    let data = if buffer.len() > data_start {
677        buffer[data_start..].to_vec()
678    } else {
679        Vec::new()
680    };
681
682    Ok(AccountInfo {
683        pubkey,
684        lamports,
685        data_len,
686        owner,
687        executable,
688        rent_epoch,
689        data,
690    })
691}

693fn filter_matches(account: &AccountInfo, filter: &FilterOptions) -> bool {
694    if let Some(ref owner_filter) = filter.owner {
695        let owner_str = account.owner.to_string();
696        if !owner_str.contains(owner_filter) {
697            return false;
698        }
699    }
700
701    if let Some(min) = filter.min_balance {
702        if account.lamports < min {
703            return false;
704        }
705    }
706
707    if let Some(max) = filter.max_balance {
708        if account.lamports > max {
709            return false;
710        }
711    }
712
713    if let Some(min) = filter.min_size {
714        if account.data_len < min {
715            return false;
716        }
717    }
718
719    if let Some(max) = filter.max_size {
720        if account.data_len > max {
721            return false;
722        }
723    }
724
725    if let Some(exec) = filter.executable {
726        if account.executable != exec {
727            return false;
728        }
729    }
730
731    if let Some(rent_exempt) = filter.rent_exempt {
732        let is_rent_exempt = account.rent_epoch == u64::MAX;
733        if is_rent_exempt != rent_exempt {
734            return false;
735        }
736    }
737
738    true
739}

741fn display_account(
742    index: usize,
743    filename: &str,
744    account: &AccountInfo,
745    config: &OutputConfig,
746) -> Result<()> {
747    if config.format == OutputFormat::Json {
748        println!("{}", serde_json::to_string_pretty(account)?);
749        return Ok(());
750    }
751
752    // Try to decode account data
753    let decoder_registry = DecoderRegistry::new();
754    let decoded = decoder_registry.decode(&account.owner, &account.data).ok();
755    let program_name = decoder_registry.get_program_name(&account.owner);
756
757    if config.colorized {
758        println!(
759            "{} ({})",
760            format!("Account #{}", index).bold().cyan(),
761            filename.bright_black()
762        );
763
764        // Show program type if known
765        if let Some(name) = program_name {
766            println!("  {}: {}", "Type".bold(), name.bright_magenta());
767        }
768
769        println!(
770            "  {}: {}",
771            "Lamports".bold(),
772            format_lamports(account.lamports).green()
773        );
774        println!("  {}: {} bytes", "Data Length".bold(), account.data_len);
775        println!(
776            "  {}: {}",
777            "Owner".bold(),
778            account.owner.to_string().bright_blue()
779        );
780        println!("  {}: {}", "Executable".bold(), account.executable);
781        println!("  {}: {}", "Rent Epoch".bold(), account.rent_epoch);
782
783        // Display decoded data if available
784        if let Some(DecodedAccount::Unknown) = decoded {
785            // Don't show anything for unknown types
786        } else if let Some(ref decoded_data) = decoded {
787            let formatted = format_decoded_account(decoded_data);
788            if !formatted.is_empty() {
789                println!();
790                println!("  {}", "Decoded Data:".bold().yellow());
791                for line in formatted.lines() {
792                    println!("  {}", line);
793                }
794            }
795        }
796    } else {
797        println!("Account #{} ({})", index, filename);
798
799        if let Some(name) = program_name {
800            println!("  Type: {}", name);
801        }
802
803        println!("  Lamports: {}", account.lamports);
804        println!("  Data Length: {} bytes", account.data_len);
805        println!("  Owner: {}", account.owner);
806        println!("  Executable: {}", account.executable);
807        println!("  Rent Epoch: {}", account.rent_epoch);
808
809        // Display decoded data
810        if let Some(DecodedAccount::Unknown) = decoded {
811            // Don't show anything for unknown types
812        } else if let Some(ref decoded_data) = decoded {
813            let formatted = format_decoded_account(decoded_data);
814            if !formatted.is_empty() {
815                println!();
816                println!("  Decoded Data:");
817                for line in formatted.lines() {
818                    println!("  {}", line);
819                }
820            }
821        }
822    }
823
824    if !account.data.is_empty() && !config.quiet {
825        println!("  Data (first {} bytes):", account.data.len().min(128));
826        display_hex_dump(&account.data[..account.data.len().min(128)]);
827        if account.data.len() > 128 {
828            println!("    ... ({} more bytes)", account.data.len() - 128);
829        }
830    }
831
832    println!();
833    Ok(())
834}

836fn display_hex_dump(data: &[u8]) {
837    for (i, chunk) in data.chunks(16).enumerate() {
838        print!("    {:04x}: ", i * 16);
839        for byte in chunk {
840            print!("{:02x} ", byte);
841        }
842        for _ in 0..(16 - chunk.len()) {
843            print!("   ");
844        }
845        print!(" |");
846        for byte in chunk {
847            let c = if byte.is_ascii_graphic() || *byte == b' ' {
848                *byte as char
849            } else {
850                '.'
851            };
852            print!("{}", c);
853        }
854        println!("|");
855    }
856}

858fn format_lamports(lamports: u64) -> String {
859    if lamports >= 1_000_000_000 {
860        format!("{:.4} SOL", lamports as f64 / 1_000_000_000.0)
861    } else {
862        format!("{} lamports", lamports)
863    }
864}
