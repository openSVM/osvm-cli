1use clap::{Arg, ArgAction, Command};
2
3/// Build the mount command
4pub fn build_mount_command() -> Command {
5    Command::new("mount")
6        .about("Manage folder mounts for OSVM microVMs and MCP tools")
7        .arg_required_else_help(true)
8        .subcommand(
9            Command::new("add")
10                .about("Mount a folder to OSVM microVM (auto-detected VM path)")
11                .arg(
12                    Arg::new("host_path")
13                        .help("Host folder path to mount (e.g., ~/Documents)")
14                        .required(true)
15                        .index(1),
16                )
17                .arg(
18                    Arg::new("readonly")
19                        .long("readonly")
20                        .short('r')
21                        .action(ArgAction::SetTrue)
22                        .help("Mount as read-only"),
23                ),
24        )
25        .subcommand(
26            Command::new("remove")
27                .about("Unmount a folder from OSVM microVM")
28                .arg(
29                    Arg::new("host_path")
30                        .help("Host folder path to unmount")
31                        .required(true)
32                        .index(1),
33                ),
34        )
35        .subcommand(Command::new("list").about("List all OSVM microVM mounts"))
36}
