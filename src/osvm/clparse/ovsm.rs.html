1//! OVSM (Open Versatile Seeker Mind) language interpreter command definition
2
3use clap::{Arg, ArgAction, Command};
4
5pub fn build_ovsm_command() -> Command {
6    Command::new("ovsm")
7        .about("OVSM language interpreter for blockchain automation scripting")
8        .long_about("Execute and manage OVSM (Open Versatile Seeker Mind) scripts for blockchain automation.\n\
9                   \n\
10                   OVSM is a production-ready scripting language (97.3% test coverage) designed for\n\
11                   blockchain operations with full control flow, loops, and rich data types.\n\
12                   \n\
13                   Features:\n\
14                   • Control Flow: IF/THEN/ELSE, FOR, WHILE, BREAK, CONTINUE\n\
15                   • Data Types: Int, Float, String, Bool, Arrays, Objects, Ranges\n\
16                   • OSVM Integration: Direct access to blockchain operations\n\
17                   • MCP Tools: Call MCP servers from scripts\n\
18                   • MicroVM Isolation: Run untrusted scripts safely\n\
19                   \n\
20                   Quick Examples:\n\
21                   • osvm ovsm run script.ovsm              # Execute script file\n\
22                   • osvm ovsm repl                         # Interactive REPL\n\
23                   • osvm ovsm eval '$x = 10; RETURN $x'    # Inline evaluation\n\
24                   • osvm ovsm examples                     # Show example scripts\n\
25                   • osvm ovsm generate \"monitor validators\" # AI script generation")
26        .arg_required_else_help(true)
27        .subcommand(
28            Command::new("run")
29                .about("Execute an OVSM script file")
30                .arg(
31                    Arg::new("script")
32                        .value_name("SCRIPT")
33                        .help("Path to OVSM script file (.ovsm)")
34                        .required(true)
35                        .index(1)
36                )
37                .arg(
38                    Arg::new("args")
39                        .value_name("ARGS")
40                        .help("Arguments to pass to the script (as JSON array)")
41                        .long("args")
42                )
43                .arg(
44                    Arg::new("isolated")
45                        .long("isolated")
46                        .action(ArgAction::SetTrue)
47                        .help("Run script in isolated microVM for security")
48                )
49                .arg(
50                    Arg::new("verbose")
51                        .long("verbose")
52                        .short('v')
53                        .action(ArgAction::Count)
54                        .help("Show detailed execution information")
55                )
56                .arg(
57                    Arg::new("debug")
58                        .long("debug")
59                        .action(ArgAction::SetTrue)
60                        .help("Enable debug mode with step-by-step execution")
61                )
62                .arg(
63                    Arg::new("json")
64                        .long("json")
65                        .action(ArgAction::SetTrue)
66                        .help("Output results in JSON format")
67                )
68        )
69        .subcommand(
70            Command::new("repl")
71                .about("Launch interactive OVSM REPL (Read-Eval-Print Loop)")
72                .long_about("Start an interactive OVSM shell for experimentation and learning.\n\
73                           \n\
74                           REPL Features:\n\
75                           • Line editing with history\n\
76                           • Multi-line input support\n\
77                           • Access to all OSVM tools\n\
78                           • Session recording\n\
79                           • Help and documentation lookup")
80                .arg(
81                    Arg::new("history")
82                        .long("history")
83                        .value_name("FILE")
84                        .help("History file location (default: ~/.osvm/ovsm_history)")
85                )
86        )
87        .subcommand(
88            Command::new("eval")
89                .about("Evaluate inline OVSM code")
90                .arg(
91                    Arg::new("code")
92                        .value_name("CODE")
93                        .help("OVSM code to evaluate")
94                        .required(true)
95                        .index(1)
96                )
97                .arg(
98                    Arg::new("json")
99                        .long("json")
100                        .action(ArgAction::SetTrue)
101                        .help("Output result in JSON format")
102                )
103        )
104        .subcommand(
105            Command::new("check")
106                .about("Check OVSM script syntax without executing")
107                .arg(
108                    Arg::new("script")
109                        .value_name("SCRIPT")
110                        .help("Path to OVSM script file")
111                        .required(true)
112                        .index(1)
113                )
114                .arg(
115                    Arg::new("strict")
116                        .long("strict")
117                        .action(ArgAction::SetTrue)
118                        .help("Enable strict syntax checking")
119                )
120        )
121        .subcommand(
122            Command::new("examples")
123                .about("Show OVSM example scripts and tutorials")
124                .arg(
125                    Arg::new("category")
126                        .long("category")
127                        .short('c')
128                        .value_name("CATEGORY")
129                        .value_parser(clap::builder::PossibleValuesParser::new([
130                            "basics",
131                            "blockchain",
132                            "automation",
133                            "mcp",
134                            "advanced"
135                        ]))
136                        .help("Filter examples by category")
137                )
138                .arg(
139                    Arg::new("list")
140                        .long("list")
141                        .action(ArgAction::SetTrue)
142                        .help("List all available examples")
143                )
144                .arg(
145                    Arg::new("show")
146                        .long("show")
147                        .value_name("NAME")
148                        .help("Show specific example script")
149                )
150        )
151        .subcommand(
152            Command::new("generate")
153                .about("Generate OVSM script from natural language description (AI-powered)")
154                .arg(
155                    Arg::new("description")
156                        .value_name("DESCRIPTION")
157                        .help("Natural language description of desired script")
158                        .required(true)
159                        .index(1)
160                )
161                .arg(
162                    Arg::new("output")
163                        .long("output")
164                        .short('o')
165                        .value_name("FILE")
166                        .help("Save generated script to file")
167                )
168                .arg(
169                    Arg::new("interactive")
170                        .long("interactive")
171                        .short('i')
172                        .action(ArgAction::SetTrue)
173                        .help("Interactive mode with refinement")
174                )
175        )
176        .subcommand(
177            Command::new("library")
178                .about("Manage OVSM script library")
179                .arg_required_else_help(true)
180                .subcommand(
181                    Command::new("list")
182                        .about("List installed scripts in library")
183                )
184                .subcommand(
185                    Command::new("install")
186                        .about("Install script from URL or GitHub")
187                        .arg(
188                            Arg::new("source")
189                                .value_name("URL")
190                                .help("GitHub URL or script URL")
191                                .required(true)
192                                .index(1)
193                        )
194                        .arg(
195                            Arg::new("name")
196                                .long("name")
197                                .value_name("NAME")
198                                .help("Custom name for the script")
199                        )
200                )
201                .subcommand(
202                    Command::new("remove")
203                        .about("Remove script from library")
204                        .arg(
205                            Arg::new("name")
206                                .value_name("NAME")
207                                .help("Script name to remove")
208                                .required(true)
209                                .index(1)
210                        )
211                )
212                .subcommand(
213                    Command::new("run")
214                        .about("Run installed library script")
215                        .arg(
216                            Arg::new("name")
217                                .value_name("NAME")
218                                .help("Script name to run")
219                                .required(true)
220                                .index(1)
221                        )
222                        .arg(
223                            Arg::new("args")
224                                .value_name("ARGS")
225                                .help("Arguments to pass (JSON array)")
226                                .long("args")
227                        )
228                )
229                .subcommand(
230                    Command::new("update")
231                        .about("Update installed scripts")
232                        .arg(
233                            Arg::new("name")
234                                .value_name("NAME")
235                                .help("Script name to update (updates all if not specified)")
236                                .index(1)
237                        )
238                )
239        )
