5pub fn build_snapshot_command() -> Command {

7        .about("Analyze and manage Solana snapshots")
8        .arg_required_else_help(true)
9        .subcommand(
10            Command::new("read")
11                .about("Read and display snapshot accounts")
12                .arg(
13                    Arg::new("snapshot-dir")

17                )
18                .arg(
19                    Arg::new("limit")
20                        .long("limit")

26                    Arg::new("offset")
27                        .long("offset")
28                        .value_name("COUNT")
29                        .default_value("0")
30                        .help("Skip first N accounts"),
31                )
32                .arg(
33                    Arg::new("parallel")
34                        .long("parallel")
35                        .action(ArgAction::SetTrue)
36                        .help("Enable parallel processing"),
37                )
38                .arg(
39                    Arg::new("threads")
40                        .long("threads")
41                        .value_name("COUNT")
42                        .help("Number of threads for parallel processing (default: CPU count)"),
43                )
44                .arg(
45                    Arg::new("filter-owner")
46                        .long("filter-owner")
47                        .value_name("PUBKEY")
48                        .help("Filter accounts by owner program"),
49                )
50                .arg(
51                    Arg::new("filter-min-balance")
52                        .long("filter-min-balance")
53                        .value_name("LAMPORTS")
54                        .help("Filter accounts with balance >= LAMPORTS"),
55                )
56                .arg(
57                    Arg::new("filter-max-balance")
58                        .long("filter-max-balance")
59                        .value_name("LAMPORTS")
60                        .help("Filter accounts with balance <= LAMPORTS"),
61                )
62                .arg(
63                    Arg::new("filter-min-size")
64                        .long("filter-min-size")
65                        .value_name("BYTES")
66                        .help("Filter accounts with data size >= BYTES"),
67                )
68                .arg(
69                    Arg::new("filter-max-size")
70                        .long("filter-max-size")
71                        .value_name("BYTES")
72                        .help("Filter accounts with data size <= BYTES"),
73                )
74                .arg(
75                    Arg::new("filter-executable")
76                        .long("filter-executable")
77                        .action(ArgAction::SetTrue)
78                        .help("Filter only executable accounts"),
79                )
80                .arg(
81                    Arg::new("filter-rent-exempt")
82                        .long("filter-rent-exempt")
83                        .action(ArgAction::SetTrue)
84                        .help("Filter only rent-exempt accounts"),
85                )
86                .arg(
87                    Arg::new("json")
88                        .long("json")
89                        .action(ArgAction::SetTrue)
90                        .help("Output in JSON format"),
91                )
92                .arg(
93                    Arg::new("quiet")
94                        .long("quiet")
95                        .short('q')
96                        .action(ArgAction::SetTrue)
97                        .help("Suppress non-essential output"),
98                ),
99        )
100        .subcommand(
101            Command::new("stats")
102                .about("Show comprehensive snapshot statistics")
103                .arg(
104                    Arg::new("snapshot-dir")
105                        .long("snapshot-dir")
106                        .value_name("PATH")
107                        .help("Path to snapshot directory"),
108                )
109                .arg(
110                    Arg::new("json")
111                        .long("json")
112                        .action(ArgAction::SetTrue)
113                        .help("Output in JSON format"),
114                ),
115        )
116        .subcommand(
117            Command::new("export")
118                .about("Export snapshot to various formats")
119                .arg(
120                    Arg::new("snapshot-dir")
121                        .long("snapshot-dir")
122                        .value_name("PATH")
123                        .help("Path to snapshot directory"),
124                )
125                .arg(
126                    Arg::new("output")
127                        .long("output")
128                        .short('o')
129                        .value_name("FILE")
130                        .required(true)
131                        .help("Output file path"),
132                )
133                .arg(
134                    Arg::new("format")
135                        .long("format")
136                        .short('f')
137                        .value_name("FORMAT")
138                        .value_parser(clap::builder::PossibleValuesParser::new([
139                            "json", "csv", "parquet", "msgpack",
140                        ]))
141                        .default_value("json")
142                        .help("Export format"),
143                )
144                .arg(
145                    Arg::new("filter-owner")
146                        .long("filter-owner")
147                        .value_name("PUBKEY")
148                        .help("Filter accounts by owner program"),
149                )
150                .arg(
151                    Arg::new("filter-min-balance")
152                        .long("filter-min-balance")
153                        .value_name("LAMPORTS")
154                        .help("Filter accounts with balance >= LAMPORTS"),
155                )
156                .arg(
157                    Arg::new("filter-max-balance")
158                        .long("filter-max-balance")
159                        .value_name("LAMPORTS")
160                        .help("Filter accounts with balance <= LAMPORTS"),
161                ),
162        )
163        .subcommand(
164            Command::new("compare")
165                .about("Compare two snapshots and show differences")
166                .arg(
167                    Arg::new("snapshot1")
168                        .value_name("SNAPSHOT1")
169                        .required(true)
170                        .index(1)
171                        .help("First snapshot directory"),
172                )
173                .arg(
174                    Arg::new("snapshot2")
175                        .value_name("SNAPSHOT2")
176                        .required(true)
177                        .index(2)
178                        .help("Second snapshot directory"),
179                )
180                .arg(
181                    Arg::new("json")
182                        .long("json")
183                        .action(ArgAction::SetTrue)
184                        .help("Output in JSON format"),
185                ),
186        )
187        .subcommand(
188            Command::new("validate")
189                .about("Validate snapshot integrity")
190                .arg(
191                    Arg::new("snapshot-dir")
192                        .long("snapshot-dir")
193                        .value_name("PATH")
194                        .help("Path to snapshot directory"),
195                )
196                .arg(
197                    Arg::new("json")
198                        .long("json")
199                        .action(ArgAction::SetTrue)
200                        .help("Output in JSON format"),
201                ),
202        )
203        .subcommand(
204            Command::new("find")
205                .about("Find specific account by pubkey")
206                .arg(
207                    Arg::new("pubkey")
208                        .value_name("PUBKEY")
209                        .required(true)
210                        .index(1)
211                        .help("Account public key to search for"),
212                )
213                .arg(
214                    Arg::new("snapshot-dir")
215                        .long("snapshot-dir")
216                        .value_name("PATH")
217                        .help("Path to snapshot directory"),
218                )
219                .arg(
220                    Arg::new("json")
221                        .long("json")
222                        .action(ArgAction::SetTrue)
223                        .help("Output in JSON format"),
224                ),
225        )
226        .subcommand(
227            Command::new("interactive")
228                .about("Launch interactive TUI for snapshot exploration")
229                .arg(
230                    Arg::new("snapshot-dir")
231                        .long("snapshot-dir")
232                        .value_name("PATH")
233                        .help("Path to snapshot directory"),
234                ),
235        )
