3/// Build the agent command
4pub fn build_agent_command() -> Command {
5    Command::new("agent")
6        .about("Execute agent commands with AI planning and MCP tool execution")
7        .long_about(
8            "Execute a single agent command with AI-powered planning and tool execution.\n\
9                           \n\
10                           The agent will:\n\
11                           • Analyze your request using AI\n\
12                           • Create an execution plan with available MCP tools\n\
13                           • Execute the tools in sequence\n\
14                           • Provide a contextual response\n\
15                           \n\
16                           Examples:\n\
17                           • osvm agent \"What's my wallet balance?\"\n\
18                           • osvm agent \"Show recent transactions\"\n\
19                           • osvm agent \"Deploy a validator node\"",
20        )
21        .arg(
22            Arg::new("prompt")
23                .value_name("PROMPT")
24                .help("The prompt or command for the agent to execute")
25                .required(true)
26                .num_args(1..)
27                .index(1),
28        )
29        .arg(
30            Arg::new("json")
31                .long("json")
32                .action(ArgAction::SetTrue)
33                .help("Output results in JSON format"),
34        )
35        .arg(
36            Arg::new("verbose")
37                .long("verbose")
38                .short('v')
39                .action(ArgAction::Count)
40                .help("Show detailed execution steps"),
41        )
42        .arg(
43            Arg::new("no-tools")
44                .long("no-tools")
45                .action(ArgAction::SetTrue)
46                .help("Disable MCP tool execution (AI response only)"),
47        )
48        .arg(
49            Arg::new("timeout")
50                .long("timeout")
51                .value_name("SECONDS")
52                .default_value("30")
53                .help("Maximum execution time in seconds"),
54        )
55}
