3//! This module handles checking and installing system-level dependencies

12/// System dependency error
13#[derive(Debug)]
14pub enum SystemDepsError {
15    CommandFailed(String),
16    NotInstalled(String),
17    VersionCheckFailed(String),
18    InstallationFailed(String),
19    PackageManagerError(PackageManagerError),
20}
21
22impl fmt::Display for SystemDepsError {
23    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
24        match self {
25            SystemDepsError::CommandFailed(msg) => write!(f, "Command failed: {}", msg),
26            SystemDepsError::NotInstalled(pkg) => write!(f, "Package not installed: {}", pkg),
27            SystemDepsError::VersionCheckFailed(msg) => write!(f, "Version check failed: {}", msg),
28            SystemDepsError::InstallationFailed(msg) => write!(f, "Installation failed: {}", msg),
29            SystemDepsError::PackageManagerError(e) => write!(f, "Package manager error: {}", e),
30        }
31    }
32}
33
34impl StdError for SystemDepsError {}
35
36impl From<PackageManagerError> for SystemDepsError {
37    fn from(err: PackageManagerError) -> Self {
38        SystemDepsError::PackageManagerError(err)
39    }
40}

67    /// Check all system dependencies
68    pub async fn check_all_dependencies(&self) -> Result<Vec<DependencyInfo>, SystemDepsError> {
69        let mut dependencies = Vec::new();
70
71        // Check Rust toolchain
72        dependencies.push(self.check_rust_toolchain().await?);
73
74        // Check build tools
75        let build_tools = super::package_managers::get_build_dependencies();
76        for tool in build_tools {
77            dependencies.push(self.check_system_package(tool).await?);
78        }
79
80        // Check additional tools
81        dependencies.push(self.check_system_package("git").await?);
82        dependencies.push(self.check_system_package("curl").await?);
83
84        Ok(dependencies)
85    }

87    /// Check Rust toolchain status
88    pub async fn check_rust_toolchain(&self) -> Result<DependencyInfo, SystemDepsError> {
89        let installed = self.is_rust_installed().await?;
90
91        if !installed {
92            return Ok(DependencyInfo {
93                name: "rust".to_string(),
94                installed: false,
95                version: None,
96                required_version: Some("1.70.0+".to_string()),
97                update_available: false,
98            });
99        }
100
101        let version = self.get_rust_version().await?;
102        let update_available = self.check_rust_updates().await?;
103
104        Ok(DependencyInfo {
105            name: "rust".to_string(),
106            installed: true,
107            version: Some(version),
108            required_version: Some("1.70.0+".to_string()),
109            update_available,
110        })
111    }

113    /// Check if Rust is installed
114    pub async fn is_rust_installed(&self) -> Result<bool, SystemDepsError> {
115        match Command::new("rustc").arg("--version").output() {
116            Ok(output) => Ok(output.status.success()),
117            Err(_) => Ok(false),
118        }
119    }

121    /// Get current Rust version
122    async fn get_rust_version(&self) -> Result<String, SystemDepsError> {
123        let output = Command::new("rustc")
124            .arg("--version")
125            .output()
126            .map_err(|e| SystemDepsError::CommandFailed(e.to_string()))?;
127
128        if output.status.success() {
129            let version_str = String::from_utf8_lossy(&output.stdout);
130            // Extract version from "rustc 1.76.0 (07dca489a 2024-02-04)"
131            if let Some(version) = version_str.split_whitespace().nth(1) {
132                Ok(version.to_string())
133            } else {
134                Err(SystemDepsError::VersionCheckFailed(
135                    "Could not parse Rust version".to_string(),
136                ))
137            }
138        } else {
139            Err(SystemDepsError::CommandFailed(
140                "Failed to get Rust version".to_string(),
141            ))
142        }
143    }

145    /// Check for Rust updates
146    async fn check_rust_updates(&self) -> Result<bool, SystemDepsError> {
147        match Command::new("rustup").arg("check").output() {
148            Ok(output) => {
149                let output_str = String::from_utf8_lossy(&output.stdout);
150                Ok(output_str.contains("Update available"))
151            }
152            Err(_) => Ok(false), // If rustup is not available, assume no updates
153        }
154    }

156    /// Install or update Rust toolchain
157    pub async fn install_rust_toolchain(&self) -> Result<String, SystemDepsError> {
158        // Check if rustup is installed
159        if !self.is_rustup_installed().await? {
160            // Install rustup
161            let install_script = if cfg!(unix) {
162                "curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh -s -- -y"
163            } else {
164                return Err(SystemDepsError::InstallationFailed(
165                    "Please install Rust manually on Windows".to_string(),
166                ));
167            };
168
169            let output = Command::new("sh")
170                .arg("-c")
171                .arg(install_script)
172                .output()
173                .map_err(|e| SystemDepsError::InstallationFailed(e.to_string()))?;
174
175            if !output.status.success() {
176                return Err(SystemDepsError::InstallationFailed(
177                    String::from_utf8_lossy(&output.stderr).to_string(),
178                ));
179            }
180
181            // Source the cargo environment
182            let source_cmd = "source ~/.cargo/env";
183            let _ = Command::new("sh").arg("-c").arg(source_cmd).output();
184
185            Ok("Rust toolchain installed successfully".to_string())
186        } else {
187            // Update existing installation
188            let output = Command::new("rustup")
189                .arg("update")
190                .output()
191                .map_err(|e| SystemDepsError::InstallationFailed(e.to_string()))?;
192
193            if output.status.success() {
194                Ok("Rust toolchain updated successfully".to_string())
195            } else {
196                Err(SystemDepsError::InstallationFailed(
197                    String::from_utf8_lossy(&output.stderr).to_string(),
198                ))
199            }
200        }
201    }

211    /// Check system package status
212    pub async fn check_system_package(
213        &self,
214        package: &str,
215    ) -> Result<DependencyInfo, SystemDepsError> {
216        let mapped_package =
217            super::package_managers::map_package_name(package, &self.package_manager);
218        let installed = self.package_manager.is_package_installed(mapped_package)?;
219
220        Ok(DependencyInfo {
221            name: package.to_string(),
222            installed,
223            version: if installed {
224                Some("installed".to_string())
225            } else {
226                None
227            },
228            required_version: None,
229            update_available: false, // Note: Update checking not yet implemented
230        })
231    }

244    /// Install system build dependencies
245    pub async fn install_build_dependencies(&self) -> Result<String, SystemDepsError> {
246        let build_deps = super::package_managers::get_build_dependencies();
247        let mut missing_deps = Vec::new();
248
249        // Check which dependencies are missing
250        for dep in &build_deps {
251            let dep_info = self.check_system_package(dep).await?;
252            if !dep_info.installed {
253                missing_deps.push(*dep);
254            }
255        }
256
257        if missing_deps.is_empty() {
258            return Ok("All build dependencies are already installed".to_string());
259        }
260
261        println!("üì¶ Installing build dependencies: {:?}", missing_deps);
262
263        // Map package names for the current package manager
264        let mapped_deps: Vec<&str> = missing_deps
265            .iter()
266            .map(|dep| super::package_managers::map_package_name(dep, &self.package_manager))
267            .collect();
268
269        let result = self.package_manager.install_packages(&mapped_deps)?;
270        Ok(result)
271    }

284        // Map package names for the current package manager

286            .iter()
287            .map(|dep| super::package_managers::map_package_name(dep, &self.package_manager))
288            .collect();
289
290        let result = self.package_manager.install_packages(&mapped_deps)?;
291        Ok(result)
292    }

314/// Get Solana CLI version if installed
315pub async fn get_solana_version() -> Result<Option<String>, SystemDepsError> {
316    match Command::new("solana").arg("--version").output() {
317        Ok(output) if output.status.success() => {
318            let version_str = String::from_utf8_lossy(&output.stdout);
319            // Extract version from "solana-cli 1.18.23 (src:devbuild; feat:4215500972)"
320            if let Some(version) = version_str.split_whitespace().nth(1) {
321                Ok(Some(version.to_string()))
322            } else {
323                Ok(Some("unknown".to_string()))
324            }
325        }
326        _ => Ok(None),
327    }

330/// Install Solana CLI
331pub async fn install_solana_cli() -> Result<String, SystemDepsError> {
332    println!("üåê Installing Solana CLI...");
333
334    let install_script =
335        "curl --proto '=https' --tlsv1.2 -sSfL https://solana-install.solana.workers.dev | bash";
336
337    let output = Command::new("sh")
338        .arg("-c")
339        .arg(install_script)
340        .output()
341        .map_err(|e| SystemDepsError::InstallationFailed(e.to_string()))?;
342
343    if output.status.success() {
344        // Add Solana to PATH for current session
345        let path_cmd = "export PATH=\"$HOME/.local/share/solana/install/active_release/bin:$PATH\"";
346        let _ = Command::new("sh").arg("-c").arg(path_cmd).output();
347
348        Ok("Solana CLI installed successfully".to_string())
349    } else {
350        Err(SystemDepsError::InstallationFailed(
351            String::from_utf8_lossy(&output.stderr).to_string(),
352        ))
353    }
