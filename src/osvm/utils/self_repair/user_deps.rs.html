14/// User dependency error
15#[derive(Debug)]
16pub enum UserDepsError {
17    FileSystemError(String),
18    CommandFailed(String),
19    ConfigError(String),
20    KeypairError(String),
21    PermissionError(String),
22}
23
24impl fmt::Display for UserDepsError {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        match self {
27            UserDepsError::FileSystemError(msg) => write!(f, "File system error: {}", msg),
28            UserDepsError::CommandFailed(msg) => write!(f, "Command failed: {}", msg),
29            UserDepsError::ConfigError(msg) => write!(f, "Configuration error: {}", msg),
30            UserDepsError::KeypairError(msg) => write!(f, "Keypair error: {}", msg),
31            UserDepsError::PermissionError(msg) => write!(f, "Permission error: {}", msg),
32        }
33    }
34}
35
36impl StdError for UserDepsError {}

66    /// Check all user-level dependencies
67    pub async fn check_all_user_dependencies(&self) -> Result<SolanaConfigInfo, UserDepsError> {
68        let cli_installed = self.is_solana_cli_installed().await?;
69        let cli_version = if cli_installed {
70            self.get_solana_cli_version().await?
71        } else {
72            None
73        };
74
75        let config_dir = self.get_solana_config_dir();
76        let config_dir_exists = config_dir.exists();
77
78        let config_file = config_dir.join("cli").join("config.yml");
79        let config_file_exists = config_file.exists();
80
81        let default_keypair_path = config_dir.join("id.json");
82        let keypair_path = if config_file_exists {
83            self.get_keypair_path_from_config(&config_file)
84                .await
85                .unwrap_or_else(|_| default_keypair_path.to_string_lossy().to_string())
86        } else {
87            default_keypair_path.to_string_lossy().to_string()
88        };
89
90        let keypair_exists = Path::new(&keypair_path).exists();
91        let current_network = if cli_installed {
92            self.get_current_network().await?
93        } else {
94            None
95        };
96
97        Ok(SolanaConfigInfo {
98            cli_installed,
99            cli_version,
100            config_dir_exists,
101            config_file_exists,
102            keypair_exists,
103            keypair_path: Some(keypair_path),
104            current_network,
105        })
106    }

120                let version_str = String::from_utf8_lossy(&output.stdout);

126            }
127            _ => Ok(None),

149    /// Create Solana configuration directory
150    pub async fn create_solana_config_dir(&self) -> Result<String, UserDepsError> {
151        let config_dir = self.get_solana_config_dir();
152
153        if config_dir.exists() {
154            return Ok("Solana config directory already exists".to_string());
155        }
156
157        fs::create_dir_all(&config_dir).map_err(|e| {
158            UserDepsError::FileSystemError(format!(
159                "Failed to create config directory {}: {}",
160                config_dir.display(),
161                e
162            ))
163        })?;
164
165        // Also create the CLI subdirectory
166        let cli_dir = config_dir.join("cli");
167        fs::create_dir_all(&cli_dir).map_err(|e| {
168            UserDepsError::FileSystemError(format!(
169                "Failed to create CLI config directory {}: {}",
170                cli_dir.display(),
171                e
172            ))
173        })?;
174
175        Ok(format!(
176            "Created Solana config directory: {}",
177            config_dir.display()
178        ))
179    }

181    /// Generate a new Solana keypair
182    pub async fn generate_keypair(
183        &self,
184        output_path: Option<&str>,
185    ) -> Result<String, UserDepsError> {
186        let keypair_path = if let Some(path) = output_path {
187            PathBuf::from(path)
188        } else {
189            self.get_solana_config_dir().join("id.json")
190        };
191
192        // Ensure the parent directory exists
193        if let Some(parent) = keypair_path.parent() {
194            if !parent.exists() {
195                fs::create_dir_all(parent).map_err(|e| {
196                    UserDepsError::FileSystemError(format!(
197                        "Failed to create keypair directory {}: {}",
198                        parent.display(),
199                        e
200                    ))
201                })?;
202            }
203        }
204
205        // Check if keypair already exists
206        if keypair_path.exists() {
207            return Err(UserDepsError::KeypairError(format!(
208                "Keypair already exists at {}",
209                keypair_path.display()
210            )));
211        }
212
213        // Generate the keypair
214        let output = Command::new("solana-keygen")
215            .arg("new")
216            .arg("-o")
217            .arg(&keypair_path)
218            .arg("--no-passphrase")
219            .output()
220            .map_err(|e| {
221                UserDepsError::CommandFailed(format!("Failed to execute solana-keygen: {}", e))
222            })?;
223
224        if output.status.success() {
225            let pubkey = self
226                .get_pubkey_from_keypair(&keypair_path)
227                .await
228                .unwrap_or_else(|_| "unknown".to_string());
229
230            Ok(format!(
231                "Generated new keypair: {} (pubkey: {})",
232                keypair_path.display(),
233                pubkey
234            ))
235        } else {
236            let error_msg = String::from_utf8_lossy(&output.stderr);
237            Err(UserDepsError::KeypairError(format!(
238                "Failed to generate keypair: {}",
239                error_msg
240            )))
241        }
242    }

244    /// Get public key from a keypair file
245    async fn get_pubkey_from_keypair(&self, keypair_path: &Path) -> Result<String, UserDepsError> {
246        let output = Command::new("solana-keygen")
247            .arg("pubkey")
248            .arg(keypair_path)
249            .output()
250            .map_err(|e| {
251                UserDepsError::CommandFailed(format!("Failed to get pubkey from keypair: {}", e))
252            })?;
253
254        if output.status.success() {
255            let pubkey = String::from_utf8_lossy(&output.stdout);
256            Ok(pubkey.trim().to_string())
257        } else {
258            Err(UserDepsError::KeypairError(
259                "Failed to extract public key from keypair".to_string(),
260            ))
261        }
262    }

264    /// Set Solana configuration
265    pub async fn configure_solana_network(&self, network: &str) -> Result<String, UserDepsError> {
266        let rpc_url = match network.to_lowercase().as_str() {
267            "mainnet" | "mainnet-beta" => "https://api.mainnet-beta.solana.com",
268            "testnet" => "https://api.testnet.solana.com",
269            "devnet" => "https://api.devnet.solana.com",
270            _ => {
271                return Err(UserDepsError::ConfigError(format!(
272                    "Unknown network: {}",
273                    network
274                )))
275            }
276        };
277
278        let output = Command::new("solana")
279            .arg("config")
280            .arg("set")
281            .arg("--url")
282            .arg(rpc_url)
283            .output()
284            .map_err(|e| {
285                UserDepsError::CommandFailed(format!("Failed to set Solana network: {}", e))
286            })?;
287
288        if output.status.success() {
289            Ok(format!("Set Solana network to: {}", network))
290        } else {
291            let error_msg = String::from_utf8_lossy(&output.stderr);
292            Err(UserDepsError::ConfigError(format!(
293                "Failed to set network: {}",
294                error_msg
295            )))
296        }
297    }

299    /// Get current Solana network configuration
300    async fn get_current_network(&self) -> Result<Option<String>, UserDepsError> {
301        match Command::new("solana").arg("config").arg("get").output() {
302            Ok(output) if output.status.success() => {
303                let config_str = String::from_utf8_lossy(&output.stdout);
304                // Parse the config output to extract the RPC URL
305                for line in config_str.lines() {
306                    if line.contains("RPC URL:") {
307                        if line.contains("mainnet-beta") {
308                            return Ok(Some("mainnet".to_string()));
309                        } else if line.contains("testnet") {
310                            return Ok(Some("testnet".to_string()));
311                        } else if line.contains("devnet") {
312                            return Ok(Some("devnet".to_string()));
313                        }
314                    }
315                }
316                Ok(Some("unknown".to_string()))
317            }
318            _ => Ok(None),
319        }
320    }

322    /// Get keypair path from Solana config file
323    async fn get_keypair_path_from_config(
324        &self,
325        config_file: &Path,
326    ) -> Result<String, UserDepsError> {
327        let config_content = fs::read_to_string(config_file).map_err(|e| {
328            UserDepsError::ConfigError(format!(
329                "Failed to read config file {}: {}",
330                config_file.display(),
331                e
332            ))
333        })?;
334
335        // Parse YAML config to extract keypair path
336        if let Ok(config) = serde_yaml::from_str::<serde_yaml::Value>(&config_content) {
337            if let Some(keypair_path) = config.get("keypair_path").and_then(|v| v.as_str()) {
338                return Ok(keypair_path.to_string());
339            }
340        }
341
342        // Fallback to default path
343        Ok(self
344            .get_solana_config_dir()
345            .join("id.json")
346            .to_string_lossy()
347            .to_string())
348    }

373    /// Install Solana CLI using the official installer
374    pub async fn install_solana_cli(&self) -> Result<String, UserDepsError> {
375        println!("🌐 Installing Solana CLI...");
376
377        let install_script = "curl --proto '=https' --tlsv1.2 -sSfL https://solana-install.solana.workers.dev | bash";
378
379        let output = Command::new("sh")
380            .arg("-c")
381            .arg(install_script)
382            .output()
383            .map_err(|e| {
384                UserDepsError::CommandFailed(format!("Failed to download Solana installer: {}", e))
385            })?;
386
387        if !output.status.success() {
388            let error_msg = String::from_utf8_lossy(&output.stderr);
389            return Err(UserDepsError::CommandFailed(format!(
390                "Solana installation failed: {}",
391                error_msg
392            )));
393        }
394
395        // Update PATH for current session
396        let path_update = format!(
397            "export PATH=\"{}/.local/share/solana/install/active_release/bin:$PATH\"",
398            home_dir().unwrap_or_default().display()
399        );
400
401        let _ = Command::new("sh").arg("-c").arg(&path_update).output();
402
403        // Verify installation
404        tokio::time::sleep(tokio::time::Duration::from_secs(2)).await;
405
406        if self.is_solana_cli_installed().await? {
407            Ok("Solana CLI installed successfully".to_string())
408        } else {
409            Err(UserDepsError::CommandFailed(
410                "Solana CLI installation verification failed. Please restart your terminal."
411                    .to_string(),
412            ))
413        }
414    }

416    /// Create a complete Solana setup (config directory + keypair + network config)
417    pub async fn setup_solana_complete(
418        &self,
419        network: Option<&str>,
420    ) -> Result<Vec<String>, UserDepsError> {
421        let mut results = Vec::new();
422
423        // 1. Create config directory
424        results.push(self.create_solana_config_dir().await?);
425
426        // 2. Generate keypair if it doesn't exist
427        let config_info = self.check_all_user_dependencies().await?;
428        if !config_info.keypair_exists {
429            results.push(self.generate_keypair(None).await?);
430        }
431
432        // 3. Set network configuration
433        let target_network = network.unwrap_or("mainnet");
434        results.push(self.configure_solana_network(target_network).await?);
435
436        // 4. Set keypair path in config
437        if let Some(keypair_path) = config_info.keypair_path {
438            results.push(self.set_keypair_path(&keypair_path).await?);
439        }
440
441        Ok(results)
442    }

444    /// Validate existing Solana configuration
445    pub async fn validate_solana_config(&self) -> Result<Vec<String>, UserDepsError> {
446        let mut issues = Vec::new();
447        let config_info = self.check_all_user_dependencies().await?;
448
449        if !config_info.cli_installed {
450            issues.push("Solana CLI is not installed".to_string());
451        }
452
453        if !config_info.config_dir_exists {
454            issues.push("Solana configuration directory does not exist".to_string());
455        }
456
457        if !config_info.keypair_exists {
458            issues.push("Solana keypair file does not exist".to_string());
459        }
460
461        if config_info.current_network.is_none() {
462            issues.push("Solana network is not configured".to_string());
463        }
464
465        Ok(issues)
466    }
