1//! Certificate management for mTLS authentication
2//!
3//! This module provides certificate authority functionality using step-ca.
4//! step-ca is a modern, automated certificate authority that provides:
5//! - Automatic certificate issuance
6//! - Certificate rotation
7//! - Revocation support
8//! - ACME protocol support
9//!
10//! # Architecture
11//!
12//! ```text
13//! ┌─────────────────────────────────────────┐
14//! │  OSVM Core                              │
15//! │  ┌────────────────────────────────────┐ │
16//! │  │  CertificateAuthority              │ │
17//! │  │  - Issues certs for components     │ │
18//! │  │  - Manages CA keys                 │ │
19//! │  │  - Handles revocation              │ │
20//! │  └────────────────────────────────────┘ │
21//! └─────────────────────────────────────────┘
22//!          │
23//!          ▼ (HTTPS API)
24//! ┌─────────────────────────────────────────┐
25//! │  step-ca Server                         │
26//! │  - Root CA                              │
27//! │  - Intermediate CA                      │
28//! │  - CRL (Certificate Revocation List)    │
29//! └─────────────────────────────────────────┘
30//!          │
31//!          ▼ (Issues)
32//! ┌─────────────────────────────────────────┐
33//! │  Component Certificates                 │
34//! │  - Validator cert                       │
35//! │  - RPC cert                             │
36//! │  - MCP cert (per server)                │
37//! └─────────────────────────────────────────┘
38//! ```

99    pub fn new(
100        ca_url: String,
101        ca_root_cert_path: PathBuf,
102        provisioner: String,
103        config: CertificateAuthorityConfig,
104    ) -> Result<Self> {
105        // Verify CA root cert exists
106        if !ca_root_cert_path.exists() {
107            return Err(anyhow!(
108                "CA root certificate not found at {:?}. Run: step ca bootstrap --ca-url {} --fingerprint <fingerprint>",
109                ca_root_cert_path,

111            ));
112        }
113
114        // Create cert directory if it doesn't exist
115        if !config.cert_dir.exists() {
116            fs::create_dir_all(&config.cert_dir)
117                .context("Failed to create certificate directory")?;
118        }
119
120        Ok(Self {
121            ca_url,
122            ca_root_cert_path,
123            provisioner,
124            provisioner_password: None,
125            config,
126        })
127    }

152    /// Issue a certificate for a component
153    pub async fn issue_certificate(
154        &self,
155        identity: &str,
156        validity_days: u32,
157    ) -> Result<Certificate> {
158        log::info!("Issuing certificate for identity: {}", identity);
159
160        // Validate validity period
161        let validity_days = validity_days.min(self.config.max_validity_days);
162
163        // Generate paths
164        let cert_path = self.config.cert_dir.join(format!("{}.crt", identity));
165        let key_path = self.config.cert_dir.join(format!("{}.key", identity));
166
167        // Use step CLI to issue certificate
168        let mut cmd = Command::new("step");
169        cmd.arg("ca")
170            .arg("certificate")
171            .arg(identity) // Subject/CN
172            .arg(&cert_path) // Output cert path
173            .arg(&key_path) // Output key path
174            .arg("--ca-url")
175            .arg(&self.ca_url)
176            .arg("--root")
177            .arg(&self.ca_root_cert_path)
178            .arg("--provisioner")
179            .arg(&self.provisioner)
180            .arg("--not-after")
181            .arg(format!("{}h", validity_days * 24));
182
183        // Add password if configured
184        if let Some(password) = &self.provisioner_password {
185            cmd.arg("--provisioner-password-file").arg("-"); // Read from stdin
186            cmd.env("STEP_CA_PASSWORD", password);
187        } else {
188            cmd.arg("--force"); // Non-interactive
189        }
190
191        // Execute command
192        let output = cmd
193            .output()
194            .context("Failed to execute step ca certificate")?;
195
196        if !output.status.success() {
197            let stderr = String::from_utf8_lossy(&output.stderr);
198            return Err(anyhow!("Certificate issuance failed: {}", stderr));
199        }
200
201        // Read the generated certificate
202        let cert_pem =
203            fs::read_to_string(&cert_path).context("Failed to read generated certificate")?;
204
205        // Parse certificate to extract details
206        let cert = self.parse_certificate(&cert_pem)?;
207
208        log::info!(
209            "Certificate issued for {} (valid until {})",
210            identity,
211            cert.not_after
212        );
213
214        Ok(cert)
215    }

301    /// Check if certificate needs renewal
302    pub fn needs_renewal(&self, cert: &Certificate) -> bool {
303        if !self.config.auto_renew {
304            return false;
305        }
306
307        let threshold = Utc::now() + Duration::days(self.config.renewal_threshold_days as i64);
308        cert.not_after < threshold

320    /// Parse certificate from PEM
321    fn parse_certificate(&self, pem: &str) -> Result<Certificate> {
322        // Parse certificate details using step CLI
323        // NOTE: Full implementation should use openssl or x509-parser crate
324        //
325        // Production implementation with x509-parser:
326        // ```
327        // use x509_parser::prelude::*;
328        // let pem_data = pem::parse(pem)?;
329        // let cert = X509Certificate::from_der(&pem_data.contents)?;
330        //
331        // let subject = cert.subject().to_string();
332        // let serial = cert.serial.to_string();
333        // let not_before = convert_asn1_time(cert.validity().not_before)?;
334        // let not_after = convert_asn1_time(cert.validity().not_after)?;
335        // ```
336
337        log::debug!("Parsing certificate from PEM");
338
339        // Use step CLI to inspect certificate
340        let mut child = Command::new("step")
341            .arg("certificate")
342            .arg("inspect")
343            .arg("--format")
344            .arg("json")
345            .stdin(std::process::Stdio::piped())
346            .stdout(std::process::Stdio::piped())
347            .stderr(std::process::Stdio::piped())
348            .spawn()
349            .context("Failed to spawn step certificate inspect")?;
350
351        // Write PEM to stdin
352        use std::io::Write;
353        if let Some(ref mut stdin) = child.stdin {
354            stdin
355                .write_all(pem.as_bytes())
356                .context("Failed to write PEM to stdin")?;
357            // Close stdin to signal EOF
358            drop(child.stdin.take());
359        }
360
361        let result = child
362            .wait_with_output()
363            .context("Failed to wait for step certificate inspect")?;
364
365        if !result.status.success() {
366            log::warn!("step certificate inspect failed, using fallback parsing");
367            // Fallback to basic parsing
368            return Ok(Certificate {
369                subject: "unknown-subject".to_string(),
370                serial_number: "unknown-serial".to_string(),
371                not_before: Utc::now(),
372                not_after: Utc::now() + Duration::days(self.config.default_validity_days as i64),
373                pem: pem.to_string(),
374            });
375        }
376
377        // Parse JSON output
378        let json_str = String::from_utf8_lossy(&result.stdout);
379        let json: serde_json::Value =
380            serde_json::from_str(&json_str).context("Failed to parse JSON from step")?;
381
382        // Extract certificate details
383        let subject = json
384            .get("subject")
385            .and_then(|v| v.get("common_name"))
386            .and_then(|v| v.as_str())
387            .unwrap_or("unknown-subject")
388            .to_string();
389
390        let serial_number = json
391            .get("serial_number")
392            .and_then(|v| v.as_str())
393            .unwrap_or("unknown-serial")
394            .to_string();
395
396        let not_before_str = json
397            .get("validity")
398            .and_then(|v| v.get("start"))
399            .and_then(|v| v.as_str())
400            .context("Missing validity start")?;
401
402        let not_after_str = json
403            .get("validity")
404            .and_then(|v| v.get("end"))
405            .and_then(|v| v.as_str())
406            .context("Missing validity end")?;
407
408        // Parse dates (step outputs RFC3339 format)
409        let not_before = DateTime::parse_from_rfc3339(not_before_str)
410            .context("Failed to parse not_before date")?
411            .with_timezone(&Utc);
412
413        let not_after = DateTime::parse_from_rfc3339(not_after_str)
414            .context("Failed to parse not_after date")?
415            .with_timezone(&Utc);
416
417        Ok(Certificate {
418            subject,
419            serial_number,
420            not_before,
421            not_after,
422            pem: pem.to_string(),
423        })
424    }

454/// Certificate manager for a component
455///
456/// Manages the lifecycle of a component's certificate including:
457/// - Loading certificate and private key
458/// - Automatic renewal when nearing expiry
459/// - Secure storage
460pub struct CertificateManager {
461    /// Path to certificate file
462    cert_path: PathBuf,
463
464    /// Path to private key file
465    key_path: PathBuf,
466
467    /// Cached certificate
468    cached_cert: Option<Certificate>,
469
470    /// Certificate authority for renewals
471    ca: Option<CertificateAuthority>,
472}
473
474impl CertificateManager {
475    /// Create a new certificate manager
476    pub fn new(cert_path: PathBuf, key_path: PathBuf) -> Self {
477        Self {
478            cert_path,
479            key_path,
480            cached_cert: None,
481            ca: None,
482        }
483    }
484
485    /// Create with CA for automatic renewal
486    pub fn with_ca(mut self, ca: CertificateAuthority) -> Self {
487        self.ca = Some(ca);
488        self
489    }
490
491    /// Load certificate from disk
492    pub fn load_certificate(&mut self) -> Result<Certificate> {
493        log::debug!("Loading certificate from {:?}", self.cert_path);
494
495        // Check if cert file exists
496        if !self.cert_path.exists() {
497            return Err(IsolationError::CertificateError(format!(
498                "Certificate not found at {:?}",
499                self.cert_path
500            ))
501            .into());
502        }
503
504        // Read PEM file
505        let pem = fs::read_to_string(&self.cert_path).context("Failed to read certificate file")?;
506
507        // Parse certificate
508        let cert = self.parse_certificate_pem(&pem)?;
509
510        // Cache it
511        self.cached_cert = Some(cert.clone());
512
513        Ok(cert)
514    }
515
516    /// Get certificate (load if not cached)
517    pub fn get_certificate(&mut self) -> Result<Certificate> {
518        if let Some(cert) = &self.cached_cert {
519            // Check if needs renewal
520            if let Some(ca) = &self.ca {
521                if ca.needs_renewal(cert) {
522                    log::info!("Certificate needs renewal");
523                    // Note: Can't await in non-async function
524                    // Caller should use renew_if_needed() separately
525                }
526            }
527
528            return Ok(cert.clone());
529        }
530
531        // Load from disk
532        self.load_certificate()
533    }
534
535    /// Check if certificate needs renewal
536    pub fn needs_renewal(&self) -> bool {
537        if let (Some(cert), Some(ca)) = (&self.cached_cert, &self.ca) {
538            ca.needs_renewal(cert)
539        } else {
540            false
541        }
542    }
543
544    /// Renew certificate if needed
545    pub async fn renew_if_needed(&mut self, identity: &str) -> Result<bool> {
546        if !self.needs_renewal() {
547            return Ok(false);
548        }
549
550        let ca = self
551            .ca
552            .as_ref()
553            .ok_or_else(|| anyhow!("No CA configured for renewal"))?;
554
555        log::info!("Renewing certificate for {}", identity);
556        let new_cert = ca.renew_certificate(identity).await?;
557
558        // Update cached cert
559        self.cached_cert = Some(new_cert);
560
561        // Reload from disk (step-ca wrote new files)
562        self.load_certificate()?;
563
564        Ok(true)
565    }
566
567    /// Save certificate to disk
568    pub fn save_certificate(&self, cert: &Certificate) -> Result<()> {
569        log::debug!("Saving certificate to {:?}", self.cert_path);
570
571        // Write PEM to file
572        fs::write(&self.cert_path, &cert.pem).context("Failed to write certificate file")?;
573
574        // Set appropriate permissions (readable only by owner)
575        #[cfg(unix)]
576        {
577            use std::os::unix::fs::PermissionsExt;
578            let mut perms = fs::metadata(&self.cert_path)?.permissions();
579            perms.set_mode(0o600); // rw-------
580            fs::set_permissions(&self.cert_path, perms)?;
581        }
582
583        Ok(())
584    }

599    }

697/// A certificate
698#[derive(Debug, Clone, Serialize, Deserialize)]
699pub struct Certificate {
700    pub subject: String,
701    pub serial_number: String,
702    pub not_before: DateTime<Utc>,
703    pub not_after: DateTime<Utc>,
704    pub pem: String,
705}
706
707impl Certificate {
708    /// Check if certificate is expired
709    pub fn is_expired(&self) -> bool {
710        Utc::now() > self.not_after
711    }
712
713    /// Check if certificate is not yet valid
714    pub fn is_not_yet_valid(&self) -> bool {
715        Utc::now() < self.not_before
716    }
717
718    /// Check if certificate is currently valid
719    pub fn is_valid(&self) -> bool {
720        !self.is_expired() && !self.is_not_yet_valid()
721    }
722}
