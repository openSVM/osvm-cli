16/// Runtime abstraction for running components in isolated environments
17#[async_trait]
18pub trait Runtime: Send + Sync {
19    /// Get runtime name
20    fn name(&self) -> &str;
21
22    /// Check if runtime is available on this system
23    fn is_available(&self) -> bool;
24
25    /// Start a component
26    async fn start_component(&self, component: &mut Component) -> Result<()>;
27
28    /// Stop a component
29    async fn stop_component(&self, component_id: ComponentId) -> Result<()>;
30
31    /// Restart a component
32    async fn restart_component(&self, component_id: ComponentId) -> Result<()>;
33
34    /// Get component status
35    async fn get_status(&self, component_id: ComponentId) -> Result<super::ComponentStatus>;
36
37    /// Execute command in component
38    async fn exec(&self, component_id: ComponentId, command: Vec<String>) -> Result<String>;
39}

41/// Runtime manager that selects appropriate runtime based on configuration
42pub struct RuntimeManager {
43    runtimes: Vec<Arc<dyn Runtime>>,
44}
45
46impl RuntimeManager {
47    /// Create a new runtime manager
48    pub fn new() -> Self {
49        Self { runtimes: vec![] }
50    }
51
52    /// Create a runtime manager with default runtimes
53    pub fn with_defaults() -> Self {
54        let mut manager = Self::new();
55
56        // Register process runtime (always available)
57        manager.register_runtime(Arc::new(ProcessRuntime::new()));
58
59        // Register Firecracker runtime if available
60        let firecracker_config = firecracker::FirecrackerConfig::default();
61        if let Ok(firecracker) = FirecrackerRuntime::new(firecracker_config) {
62            if firecracker.is_available() {
63                manager.register_runtime(Arc::new(firecracker));
64            }
65        }
66
67        // Register HermitCore runtime if available
68        let hermit_config = hermit::HermitConfig::default();
69        if let Ok(hermit) = HermitRuntime::new(hermit_config) {
70            if hermit.is_available() {
71                manager.register_runtime(Arc::new(hermit));
72            }
73        }
74
75        manager
76    }
77
78    /// Register a runtime
79    pub fn register_runtime(&mut self, runtime: Arc<dyn Runtime>) {
80        self.runtimes.push(runtime);
81    }
82
83    /// Get runtime for isolation config
84    pub fn get_runtime(&self, config: &IsolationConfig) -> Result<Arc<dyn Runtime>> {
85        // Select runtime based on isolation type
86        let preferred_runtime = match &config.isolation_type {
87            IsolationType::None => "Process",
88            IsolationType::ProcessSandbox { .. } => "Process",
89            IsolationType::Container { .. } => {
90                // Container runtime implementation
91                // NOTE: Requires implementing a Docker/Podman runtime
92                // See runtime/docker.rs for stub implementation
93                log::warn!("Container isolation requested but not fully implemented - using Process runtime");
94                "Docker"
95            }
96            IsolationType::MicroVM { .. } => {
97                // MicroVM runtime (Firecracker)
98                // NOTE: Firecracker runtime is partially implemented in runtime/firecracker.rs
99                // Full production use requires:
100                // - Firecracker binary installed
101                // - KVM support enabled
102                // - Proper networking configuration
103                "Firecracker"
104            }
105            IsolationType::Unikernel { .. } => "HermitCore",
106            IsolationType::TEE { .. } => {
107                // TEE (Trusted Execution Environment) runtime
108                // NOTE: Requires implementing SGX or AMD SEV support
109                // This is complex and requires:
110                // - Hardware TEE support (Intel SGX or AMD SEV)
111                // - Attestation service
112                // - Enclave/confidential VM runtime
113                log::warn!("TEE isolation requested but not implemented - using Process runtime");
114                "SGX"
115            }
116        };
117
118        // Find runtime by name
119        self.runtimes
120            .iter()
121            .find(|r| r.name() == preferred_runtime && r.is_available())
122            .cloned()
123            // Fallback to first available runtime
124            .or_else(|| self.runtimes.iter().find(|r| r.is_available()).cloned())
125            .ok_or_else(|| {
126                IsolationError::RuntimeNotAvailable(format!(
127                    "No runtime available for isolation type: {:?}",
128                    config.isolation_type
129                ))
130                .into()
131            })
132    }
133
134    /// List available runtimes
135    pub fn list_available_runtimes(&self) -> Vec<String> {
136        self.runtimes
137            .iter()
138            .filter(|r| r.is_available())
139            .map(|r| r.name().to_string())
140            .collect()
141    }
142
143    /// Check if a specific runtime is available
144    pub fn is_runtime_available(&self, name: &str) -> bool {
145        self.runtimes
146            .iter()
147            .any(|r| r.name() == name && r.is_available())
148    }
149}
150
151impl Default for RuntimeManager {
152    fn default() -> Self {
153        Self::with_defaults()
154    }
155}

157#[cfg(test)]
158mod tests {
159    use super::*;
160
161    #[test]
162    fn test_runtime_manager_creation() {
163        let manager = RuntimeManager::new();
164        assert_eq!(manager.list_available_runtimes().len(), 0);
165    }
166
167    #[test]
168    fn test_runtime_manager_with_defaults() {
169        let manager = RuntimeManager::with_defaults();
170        let available = manager.list_available_runtimes();
171
172        // Process runtime should always be available
173        assert!(available.contains(&"Process".to_string()));
174
175        println!("Available runtimes: {:?}", available);
176    }
177
178    #[test]
179    fn test_process_runtime_always_available() {
180        let manager = RuntimeManager::with_defaults();
181        assert!(manager.is_runtime_available("Process"));
182    }
183
184    #[test]
185    fn test_runtime_manager_get_runtime_for_isolation_types() {
186        use crate::utils::isolation::config::{IsolationConfig, IsolationType};
187
188        let manager = RuntimeManager::with_defaults();
189
190        // Process isolation should always work
191        let process_config = IsolationConfig {
192            isolation_type: IsolationType::None,
193            ..Default::default()
194        };
195        let runtime = manager.get_runtime(&process_config);
196        assert!(runtime.is_ok());
197        assert_eq!(runtime.unwrap().name(), "Process");
198
199        // Process sandbox should use Process runtime
200        let sandbox_config = IsolationConfig {
201            isolation_type: IsolationType::ProcessSandbox {
202                seccomp_profile: None,
203                apparmor_profile: None,
204            },
205            ..Default::default()
206        };
207        let runtime = manager.get_runtime(&sandbox_config);
208        assert!(runtime.is_ok());
209    }
210
211    #[test]
212    fn test_runtime_manager_multiple_runtimes() {
213        let manager = RuntimeManager::with_defaults();
214        let available = manager.list_available_runtimes();
215
216        // Should have at least Process runtime
217        assert!(!available.is_empty());
218        assert!(available.contains(&"Process".to_string()));
219    }
220
221    #[test]
222    fn test_runtime_selection_fallback() {
223        use crate::utils::isolation::config::{HypervisorType, IsolationConfig, IsolationType};
224
225        let manager = RuntimeManager::with_defaults();
226
227        // Request MicroVM but it might not be available
228        // Should fallback to Process runtime
229        let microvm_config = IsolationConfig {
230            isolation_type: IsolationType::MicroVM {
231                hypervisor: HypervisorType::Firecracker,
232                kernel_path: None,
233                rootfs_path: None,
234            },
235            ..Default::default()
236        };
237
238        let runtime = manager.get_runtime(&microvm_config);
239        // Should get some runtime (likely Process as fallback)
240        assert!(runtime.is_ok());
241    }
