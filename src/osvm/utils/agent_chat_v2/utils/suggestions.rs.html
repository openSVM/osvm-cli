1//! AI-powered suggestion generation and handling
2
3use anyhow::{anyhow, Result};
4use log::warn;
5use uuid::Uuid;
6
7use super::super::state::AdvancedChatState;
8use super::super::types::ChatMessage;
9
10impl AdvancedChatState {
11    pub async fn generate_reply_suggestions(&self, session_id: Uuid) -> Result<()> {
12        // Get the chat context
13        let session = self
14            .get_session_by_id(session_id)
15            .ok_or_else(|| anyhow!("Session not found"))?;
16
17        // Build context from recent messages (last 10 messages)
18        let recent_messages = session
19            .messages
20            .iter()
21            .rev()
22            .take(10)
23            .rev()
24            .map(|msg| match msg {
25                ChatMessage::User(text) => format!("User: {}", text),
26                ChatMessage::Agent(text) => format!("Agent: {}", text),
27                ChatMessage::ToolResult {
28                    tool_name, result, ..
29                } => {
30                    format!(
31                        "Tool Result [{}]: {}",
32                        tool_name,
33                        serde_json::to_string(result).unwrap_or_default()
34                    )
35                }
36                _ => String::new(),
37            })
38            .filter(|s| !s.is_empty())
39            .collect::<Vec<_>>()
40            .join("\n");
41
42        if recent_messages.is_empty() {
43            return Ok(());
44        }
45
46        // Create prompt for AI to generate suggestions
47        let prompt = format!(
48            "Based on this conversation context:\n{}\n\n\
49            Generate exactly 5 short, practical follow-up questions or commands. \
50            Focus on blockchain operations, wallet actions, or clarifying questions. \
51            Format: Just list them as:\n\
52            1. Check wallet balance\n\
53            2. View recent transactions\n\
54            3. Get network status\n\
55            4. Show stake accounts\n\
56            5. Export transaction history\n\n\
57            Keep each suggestion under 6 words and directly actionable.",
58            recent_messages
59        );
60
61        // Call AI service to generate suggestions
62        match self.ai_service.query_with_debug(&prompt, false).await {
63            Ok(response) => {
64                // Parse suggestions from response (expecting numbered list)
65                let suggestions: Vec<String> = response
66                    .lines()
67                    .filter_map(|line| {
68                        let line = line.trim();
69                        // Match lines like "1. suggestion" or "1) suggestion" or just starting with a number
70                        if let Some(dot_pos) = line.find('.') {
71                            if line[..dot_pos].trim().parse::<u32>().is_ok() {
72                                return Some(line[dot_pos + 1..].trim().to_string());
73                            }
74                        }
75                        if let Some(paren_pos) = line.find(')') {
76                            if line[..paren_pos].trim().parse::<u32>().is_ok() {
77                                return Some(line[paren_pos + 1..].trim().to_string());
78                            }
79                        }
80                        // Also accept lines starting with digits
81                        if line.len() > 2 && line.chars().next()?.is_ascii_digit() {
82                            if let Some(rest) = line.get(2..) {
83                                return Some(rest.trim().to_string());
84                            }
85                        }
86                        None
87                    })
88                    .take(5)
89                    .collect();
90
91                if !suggestions.is_empty() {
92                    // Store suggestions
93                    if let Ok(mut sugg_guard) = self.current_suggestions.write() {
94                        *sugg_guard = suggestions;
95                    }
96                    if let Ok(mut vis_guard) = self.suggestions_visible.write() {
97                        *vis_guard = true;
98                    }
99
100                    // Add a system message to show suggestions are available
101                    self.add_message_to_session(
102                        session_id,
103                        ChatMessage::System(
104                            "Reply suggestions available (press 1-5 to insert)".to_string(),
105                        ),
106                    )?;
107
108                    // Note: UI refresh needs to happen from the main thread
109                }
110            }
111            Err(e) => {
112                warn!("Failed to generate suggestions: {}", e);
113            }
114        }
115
116        Ok(())
117    }
118}
