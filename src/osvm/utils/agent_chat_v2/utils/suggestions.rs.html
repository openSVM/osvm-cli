1//! AI-powered suggestion generation and handling
2
3use anyhow::{anyhow, Result};
4use log::warn;
5use uuid::Uuid;
6
7use super::super::state::AdvancedChatState;
8use super::super::types::ChatMessage;
9
10impl AdvancedChatState {
11    pub async fn generate_reply_suggestions(&self, session_id: Uuid) -> Result<()> {
12        // Get the chat context
13        let session = self
14            .get_session_by_id(session_id)
15            .ok_or_else(|| anyhow!("Session not found"))?;
16
17        // Build context from recent messages (last 10 messages)
18        let recent_messages = session
19            .messages
20            .iter()
21            .rev()
22            .take(10)
23            .rev()
24            .map(|msg| match msg {
25                ChatMessage::User(text) => format!("User: {}", text),
26                ChatMessage::Agent(text) => format!("Agent: {}", text),
27                ChatMessage::ToolResult {
28                    tool_name, result, ..
29                } => {
30                    format!(
31                        "Tool Result [{}]: {}",
32                        tool_name,
33                        serde_json::to_string(result).unwrap_or_default()
34                    )
35                }
36                _ => String::new(),
37            })
38            .filter(|s| !s.is_empty())
39            .collect::<Vec<_>>()
40            .join("\n");
41
42        if recent_messages.is_empty() {
43            return Ok(());
44        }
45
46        // Create prompt for AI to generate suggestions
47        let prompt = format!(
48            "Based on this conversation context:\n{}\n\n\
49            Generate exactly 5 short, practical follow-up questions or commands. \
50            Focus on blockchain operations, wallet actions, or clarifying questions. \
51            Format: Just list them as:\n\
52            1. Check wallet balance\n\
53            2. View recent transactions\n\
54            3. Get network status\n\
55            4. Show stake accounts\n\
56            5. Export transaction history\n\n\
57            Keep each suggestion under 6 words and directly actionable.",
58            recent_messages
59        );
60
61        // Call AI service to generate suggestions
62        match self.ai_service.query_with_debug(&prompt, false).await {
63            Ok(response) => {
64                // Parse suggestions from response (expecting numbered list)
65                // BUG-2010 fix: Use char-safe string operations for UTF-8 safety
66                // Previously used byte-based indexing which could panic on emoji/multibyte chars
67                let suggestions: Vec<String> = response
68                    .lines()
69                    .filter_map(|line| {
70                        let line = line.trim();
71                        // Match lines like "1. suggestion" or "1) suggestion" or just starting with a number
72                        if let Some(dot_pos) = line.find('.') {
73                            // Use char-safe slicing instead of byte-based
74                            let prefix: String = line.chars().take(dot_pos).collect();
75                            if prefix.trim().parse::<u32>().is_ok() {
76                                // Get everything after the dot using char-safe operations
77                                let suffix: String = line.chars().skip(dot_pos + 1).collect();
78                                return Some(suffix.trim().to_string());
79                            }
80                        }
81                        if let Some(paren_pos) = line.find(')') {
82                            // Use char-safe slicing instead of byte-based
83                            let prefix: String = line.chars().take(paren_pos).collect();
84                            if prefix.trim().parse::<u32>().is_ok() {
85                                // Get everything after the paren using char-safe operations
86                                let suffix: String = line.chars().skip(paren_pos + 1).collect();
87                                return Some(suffix.trim().to_string());
88                            }
89                        }
90                        // Also accept lines starting with digits
91                        if line.len() > 2 && line.chars().next()?.is_ascii_digit() {
92                            // Skip first 2 chars safely
93                            let rest: String = line.chars().skip(2).collect();
94                            return Some(rest.trim().to_string());
95                        }
96                        None
97                    })
98                    .take(5)
99                    .collect();
100
101                if !suggestions.is_empty() {
102                    // Store suggestions
103                    if let Ok(mut sugg_guard) = self.current_suggestions.write() {
104                        *sugg_guard = suggestions;
105                    }
106                    if let Ok(mut vis_guard) = self.suggestions_visible.write() {
107                        *vis_guard = true;
108                    }
109
110                    // Add a system message to show suggestions are available
111                    self.add_message_to_session(
112                        session_id,
113                        ChatMessage::System(
114                            "Reply suggestions available (press 1-5 to insert)".to_string(),
115                        ),
116                    )?;
117
118                    // Note: UI refresh needs to happen from the main thread
119                }
120            }
121            Err(e) => {
122                warn!("Failed to generate suggestions: {}", e);
123            }
124        }
125
126        Ok(())
127    }
128}
