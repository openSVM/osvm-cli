6/// Sanitize text for safe display in TUI
7pub fn sanitize_text_for_ui(text: &str) -> String {
8    let mut sanitized = text.to_string();
9
10    // Remove common ANSI escape sequences (colors, cursor movement) which break TUI layout
11    if let Ok(ansi_re) = Regex::new(r"\x1B\[[0-9;?]*[ -/]*[@-~]") {
12        sanitized = ansi_re.replace_all(&sanitized, "").to_string();
13    }
14
15    // Convert tabs to 4 spaces to keep columns aligned
16    sanitized = sanitized.replace('\t', "    ");
17
18    // Normalize line endings
19    sanitized = sanitized.replace("\r\n", "\n").replace('\r', "\n");
20
21    // Remove other control characters (except newline) that can corrupt the TUI
22    sanitized = sanitized
23        .chars()
24        .filter(|c| *c == '\n' || !c.is_control())
25        .collect();
26
27    // Redact potential private keys (base58-like patterns)
28    if let Ok(key_pattern) = Regex::new(r"\b[1-9A-HJ-NP-Za-km-z]{32,44}\b") {
29        sanitized = key_pattern
30            .replace_all(&sanitized, "[REDACTED_KEY]")
31            .to_string();
32    }
33
34    // Collapse long runs of whitespace/newlines to avoid blowups
35    if let Ok(multi_nl) = Regex::new(r"\n{3,}") {
36        sanitized = multi_nl.replace_all(&sanitized, "\n\n").to_string();
37    }
38
39    // Limit length to prevent display overflow
40    if sanitized.len() > 2000 {
41        sanitized.truncate(1997);
42        sanitized.push_str("...");
43    }
44
45    sanitized
46}

48/// Sanitize JSON for safe display in TUI
49pub fn sanitize_json_for_ui(value: &Value) -> String {
50    match serde_json::to_string_pretty(value) {
51        Ok(json_str) => {
52            let sanitized = sanitize_text_for_ui(&json_str);
53            if sanitized.len() > 1000 {
54                let take = 997.min(sanitized.len());
55                format!("{}...", &sanitized[..take])
56            } else {
57                sanitized
58            }
59        }
60        Err(_) => "[Invalid JSON]".to_string(),
61    }
62}

64/// Enhanced text sanitization with emoji removal (alias for backward compatibility)
65pub fn sanitize_text(text: &str) -> String {
66    sanitize_text_for_ui(text)
67}

69/// Enhanced JSON sanitization (alias for backward compatibility)
70pub fn sanitize_json(value: &Value) -> String {
