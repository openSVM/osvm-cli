23    pub async fn process_input_async(&self, session_id: Uuid, input: String) -> Result<()> {
24        // Set agent state to thinking
25        self.set_agent_state(session_id, AgentState::Thinking);
26        let _ = self.add_message_to_session(session_id, ChatMessage::User(input.clone()));
27        let _ = self.add_message_to_session(
28            session_id,
29            ChatMessage::Processing {
30                message: "Analyzing your request...".to_string(),
31                spinner_index: self.spinner_state.load(Ordering::Relaxed),
32            },
33        );
34
35        // First, refresh available tools to ensure we have the latest
36        if let Err(e) = self.refresh_tools_from_mcp().await {
37            warn!("Failed to refresh tools: {}", e);
38        }
39
40        self.set_agent_state(session_id, AgentState::Planning);
41        let _ = self.add_message_to_session(
42            session_id,
43            ChatMessage::Processing {
44                message: "Creating execution plan...".to_string(),
45                spinner_index: self.spinner_state.load(Ordering::Relaxed),
46            },
47        );
48
49        // Get available tools
50        let available_tools = match self.available_tools.read() {
51            Ok(tools) => tools.clone(),
52            Err(e) => {
53                error!("Failed to read available tools: {}", e);
54                HashMap::new() // Use empty map as fallback
55            }
56        };
57
58        // Use the proper AI service method to create tool plan with validation
59        let tool_plan_result = tokio::time::timeout(
60            Duration::from_secs(120), // Extended timeout for comprehensive AI responses
61            self.ai_service
62                .create_validated_tool_plan(&input, &available_tools, 3),
63        )
64        .await;
65
66        // Helper closure to build heuristic tools based on input
67        let build_heuristic_plan = |text: &str| -> Vec<PlannedTool> {
68            let mut tools = Vec::new();
69            let lc = text.to_lowercase();
70
71            // Batch validator analysis
72            if (lc.contains("validator") || lc.contains("validators"))
73                && (lc.contains("100") || lc.contains("top") || lc.contains("analyze"))
74            {
75                tools.push(PlannedTool {
76                    server_id: "local_sim".into(),
77                    tool_name: "analyze_batch_validators".into(),
78                    args: serde_json::json!({
79                        "count": 100,
80                        "metrics": ["stake", "commission", "uptime", "vote_credits", "delinquency"]
81                    }),
82                    reason: "Batch analyze validators with simulated data".into(),
83                });
84            }
85
86            // Batch token analysis
87            if (lc.contains("token") || lc.contains("tokens"))
88                && (lc.contains("50") || lc.contains("top") || lc.contains("portfolio"))
89            {
90                tools.push(PlannedTool {
91                    server_id: "local_sim".into(),
92                    tool_name: "analyze_batch_tokens".into(),
93                    args: serde_json::json!({
94                        "count": 50,
95                        "timeframe": "30d",
96                        "metrics": ["price", "volume", "market_cap"]
97                    }),
98                    reason: "Batch analyze tokens with market data".into(),
99                });
100            }
101
102            // Batch account analysis
103            if (lc.contains("account") || lc.contains("accounts"))
104                && (lc.contains("transaction") || lc.contains("pattern"))
105            {
106                tools.push(PlannedTool {
107                    server_id: "local_sim".into(),
108                    tool_name: "analyze_batch_accounts".into(),
109                    args: serde_json::json!({
110                        "account_count": 50,
111                        "transaction_limit": 100
112                    }),
113                    reason: "Batch analyze account transaction patterns".into(),
114                });
115            }
116
117            // Single balance query
118            if lc.contains("balance") && !lc.contains("validator") && !lc.contains("token") {
119                tools.push(PlannedTool {
120                    server_id: "local_sim".into(),
121                    tool_name: "get_balance".into(),
122                    args: serde_json::json!({}),
123                    reason: "Heuristic: user asked about balance".into(),
124                });
125            }
126
127            // Single transaction query
128            if (lc.contains("transaction") || lc.contains("transactions") || lc.contains("tx"))
129                && !lc.contains("account")
130                && !lc.contains("validator")
131            {
132                tools.push(PlannedTool {
133                    server_id: "local_sim".into(),
134                    tool_name: "get_transactions".into(),
135                    args: serde_json::json!({}),
136                    reason: "Heuristic: user asked about transactions".into(),
137                });
138            }
139
140            tools
141        };
142
143        // Determine if any MCP servers/tools are configured
144        let no_configured_tools = available_tools.is_empty();
145
146        match tool_plan_result {
147            Err(_) => {
148                warn!("AI planning timed out");
149                let _ = self.add_message_to_session(
150                    session_id,
151                    ChatMessage::Error(
152                        "AI planning timed out. Using heuristic fallback.".to_string(),
153                    ),
154                );
155                if no_configured_tools {
156                    let _ = self
157                        .run_heuristic_fallback(session_id, &input, build_heuristic_plan(&input))
158                        .await;
159                } else {
160                    let _ = self.simple_response(session_id, &input).await;
161                }
162            }
163            Ok(Err(e)) => {
164                error!("AI service failed: {}", e);
165                let _ = self.add_message_to_session(
166                    session_id,
167                    ChatMessage::Error(
168                        "AI planning service failed. Using heuristic fallback.".to_string(),
169                    ),
170                );
171                if no_configured_tools {
172                    let _ = self
173                        .run_heuristic_fallback(session_id, &input, build_heuristic_plan(&input))
174                        .await;
175                } else {
176                    let _ = self.simple_response(session_id, &input).await;
177                }
178            }
179            Ok(Ok(tool_plan)) => {
180                // Successfully got tool plan from AI service
181                let _ = self.add_message_to_session(
182                    session_id,
183                    ChatMessage::AgentPlan(tool_plan.osvm_tools_to_use.clone()),
184                );
185
186                if tool_plan.osvm_tools_to_use.is_empty() {
187                    // No tools needed according to AI
188                    if no_configured_tools {
189                        // Try heuristic plan instead
190                        let heur = build_heuristic_plan(&input);
191                        if !heur.is_empty() {
192                            let _ = self.add_message_to_session(
193                                session_id,
194                                ChatMessage::AgentPlan(heur.clone()),
195                            );
196                            let _ = self.run_heuristic_fallback(session_id, &input, heur).await;
197                        } else {
198                            let _ = self.simple_response(session_id, &input).await;
199                        }
200                    } else {
201                        let _ = self.simple_response(session_id, &input).await;
202                    }
203                } else {
204                    // Phase 2: Check if we have raw OVSM plan for execution engine
205                    if let Some(ref raw_ovsm) = tool_plan.raw_ovsm_plan {
206                        // Execute using OVSM engine with runtime branching
207                        debug!("Phase 2: Executing plan using OVSM execution engine");
208                        let _ = self.execute_ovsm_plan(session_id, raw_ovsm, &input).await;
209                    } else {
210                        // Fallback: Execute tools iteratively (Phase 1 behavior)
211                        debug!("No raw OVSM plan available, using iterative execution");
212                        let _ = self
213                            .execute_tools_iteratively(
214                                session_id,
215                                tool_plan.osvm_tools_to_use,
216                                &input,
217                                &available_tools,
218                            )
219                            .await;
220                    }
221
222                    // Generate final response with all executed tools
223                    let _ = self
224                        .generate_final_response(session_id, &input, &tool_plan.expected_outcome)
225                        .await;
226                }
227            }
228        }
229
230        // ALWAYS cleanup processing messages, even if errors occurred above
231        // This ensures the spinner stops regardless of success/failure
232        self.cleanup_processing_messages(session_id);
233
234        self.set_agent_state(session_id, AgentState::Idle);
235
236        // Generate suggestions after agent completes
237        let _ = self.generate_reply_suggestions(session_id).await;
238
239        Ok(())
240    }

242    /// Helper to ensure cleanup always happens
243    fn cleanup_processing_messages(&self, session_id: Uuid) {
244        // Remove ALL processing messages from the session
245        loop {
246            match self.remove_last_processing_message(session_id) {
247                Ok(()) => {
248                    // Check if there are more processing messages
249                    if let Some(session) = self.get_session_by_id(session_id) {
250                        let has_more_processing = session
251                            .messages
252                            .iter()
253                            .any(|msg| matches!(msg, ChatMessage::Processing { .. }));
254                        if !has_more_processing {
255                            break; // No more processing messages
256                        }
257                    } else {
258                        break; // Session not found
259                    }
260                }
261                Err(_) => break, // Error or no processing message found
262            }
263        }
264    }

266    // Heuristic fallback execution simulating basic tools so user sees end-to-end flow
267    async fn run_heuristic_fallback(
268        &self,
269        session_id: Uuid,
270        original_input: &str,
271        tools: Vec<PlannedTool>,
272    ) -> Result<()> {
273        if tools.is_empty() {
274            // Nothing heuristic matched; fall back to simple
275            return self.simple_response(session_id, original_input).await;
276        }
277        self.add_message_to_session(session_id, ChatMessage::AgentPlan(tools.clone()))?;
278        for planned_tool in tools {
279            self.execute_planned_tool(session_id, planned_tool).await?;
280        }
281        self.generate_final_response(session_id, original_input, "Heuristic simulated execution")
282            .await?;
283        Ok(())
284    }

286    async fn execute_planned_tool(
287        &self,
288        session_id: Uuid,
289        planned_tool: PlannedTool,
290    ) -> Result<()> {
291        let execution_id = Uuid::new_v4().to_string();
292
293        self.set_agent_state(
294            session_id,
295            AgentState::ExecutingTool(planned_tool.tool_name.clone()),
296        );
297        self.add_message_to_session(
298            session_id,
299            ChatMessage::ToolCall {
300                tool_name: planned_tool.tool_name.clone(),
301                description: planned_tool.reason.clone(),
302                args: Some(planned_tool.args.clone()),
303                execution_id: execution_id.clone(),
304            },
305        )?;
306
307        // Execute the tool using MCP service
308        match self.call_mcp_tool(&planned_tool).await {
309            Ok(result) => {
310                self.add_message_to_session(
311                    session_id,
312                    ChatMessage::ToolResult {
313                        tool_name: planned_tool.tool_name,
314                        result,
315                        execution_id,
316                    },
317                )?;
318            }
319            Err(e) => {
320                error!("Tool execution failed: {}", e);
321                self.add_message_to_session(
322                    session_id,
323                    ChatMessage::Error(format!("Tool {} failed: {}", planned_tool.tool_name, e)),
324                )?;
325            }
326        }
327
328        Ok(())
329    }

331    async fn call_mcp_tool(&self, planned_tool: &PlannedTool) -> Result<Value> {
332        // Load isolation configuration
333        let isolation_config = match IsolationConfig::load() {
334            Ok(config) => config,
335            Err(e) => {
336                warn!(
337                    "Failed to load isolation config: {}, using direct MCP execution",
338                    e
339                );
340                return self.call_mcp_tool_direct(planned_tool).await;
341            }
342        };
343
344        // Check if this tool should be executed in a unikernel
345        if isolation_config.should_use_unikernel(&planned_tool.server_id, &planned_tool.tool_name) {
346            // Attempt unikernel execution with fallback
347            match self
348                .execute_tool_in_unikernel(planned_tool, &isolation_config)
349                .await
350            {
351                Ok(result) => return Ok(result),
352                Err(e) => {
353                    warn!(
354                        "Unikernel execution failed: {}, falling back to direct MCP",
355                        e
356                    );
357                    return self.call_mcp_tool_direct(planned_tool).await;
358                }
359            }
360        }
361
362        // Use direct MCP execution (MicroVM mode or no isolation config)
363        self.call_mcp_tool_direct(planned_tool).await
364    }

366    /// Execute tool in ephemeral unikernel for maximum isolation
367    async fn execute_tool_in_unikernel(
368        &self,
369        planned_tool: &PlannedTool,
370        isolation_config: &IsolationConfig,
371    ) -> Result<Value> {
372        info!(
373            "Executing tool '{}' in ephemeral unikernel for server '{}'",
374            planned_tool.tool_name, planned_tool.server_id
375        );
376
377        // Get tool-specific configuration
378        let tool_config =
379            isolation_config.get_tool_config(&planned_tool.server_id, &planned_tool.tool_name);
380
381        // Validate unikernel image is configured
382        let unikernel_image = match &tool_config.unikernel_image {
383            Some(img) => img.clone(),
384            None => {
385                warn!(
386                    "No unikernel image configured for tool '{}' on server '{}', falling back",
387                    planned_tool.tool_name, planned_tool.server_id
388                );
389                return Err(anyhow!("No unikernel image configured"));
390            }
391        };
392
393        // Create unikernel runtime
394        let unikernel_runtime =
395            UnikernelRuntime::new(PathBuf::from(&isolation_config.unikernel_dir))
396                .context("Failed to create unikernel runtime")?;
397
398        // Build unikernel configuration
399        use crate::services::unikernel_runtime::UnikernelLauncher;
400
401        let unikernel_config = UnikernelConfig {
402            image_path: PathBuf::from(unikernel_image),
403            mounts: tool_config.mounts.clone(),
404            memory_mb: tool_config.memory_mb,
405            vcpus: tool_config.vcpus,
406            tool_name: planned_tool.tool_name.clone(),
407            server_id: planned_tool.server_id.clone(),
408            launcher: UnikernelLauncher::Unikraft,
409            kraft_config: None,
410            vsock_cid: None, // Will be auto-allocated
411        };
412
413        // Spawn the unikernel (~100ms overhead)
414        let handle = unikernel_runtime
415            .spawn_unikernel(unikernel_config)
416            .await
417            .context("Failed to spawn unikernel")?;
418
419        // Execute the tool in the isolated unikernel (pass runtime for vsock communication)
420        let result = handle
421            .execute_tool(
422                &planned_tool.tool_name,
423                Some(planned_tool.args.clone()),
424                &unikernel_runtime,
425            )
426            .await;
427
428        // Always terminate the ephemeral unikernel
429        handle.terminate();
430
431        match result {
432            Ok(value) => {
433                info!(
434                    "Successfully executed tool '{}' in unikernel",
435                    planned_tool.tool_name
436                );
437                Ok(value)
438            }
439            Err(e) => {
440                error!("Unikernel tool execution failed: {}", e);
441                Err(e)
442            }
443        }
444    }

761    /// Phase 2: Execute OVSM plan using the execution engine with runtime branching
762    async fn execute_ovsm_plan(
763        &self,
764        session_id: Uuid,
765        raw_ovsm_plan: &str,
766        _original_input: &str,
767    ) -> Result<()> {
768        info!("Phase 2: Executing OVSM plan with execution engine");
769
770        // Get all available tools to register them with the executor
771        let available_tools = match self.available_tools.read() {
772            Ok(tools) => tools.clone(),
773            Err(e) => {
774                error!("Failed to read available tools: {}", e);
775                return Err(anyhow!("Failed to read available tools: {}", e));
776            }
777        };
778
779        // Register all available tools with OVSM executor
780        let executor = self.ovsm_executor.lock().await;
781        for (server_id, tool_list) in available_tools.iter() {
782            for tool in tool_list {
783                let tool_wrapper = McpToolWrapper {
784                    state: self.clone(),
785                    session_id,
786                    server_id: server_id.clone(),
787                    tool_name: tool.name.clone(),
788                };
789                executor
790                    .register_tool(tool.name.clone(), Box::new(tool_wrapper))
791                    .await?;
792            }
793        }
794
795        // Execute the plan
796        match executor.execute_plan(raw_ovsm_plan).await {
797            Ok(result) => {
798                info!(
799                    "Phase 2: OVSM execution completed successfully (confidence: {:.2})",
800                    result.confidence
801                );
802                debug!(
803                    "OVSM execution metadata: {} tools called, {} branches taken, {} warnings, execution time: {}ms",
804                    result.tools_called.len(),
805                    result.branches_taken.len(),
806                    result.warnings.len(),
807                    result.execution_time_ms
808                );
809
810                // Log execution details
811                if !result.branches_taken.is_empty() {
812                    debug!("Branch decisions: {:?}", result.branches_taken);
813                }
814                if !result.warnings.is_empty() {
815                    warn!("OVSM execution warnings: {:?}", result.warnings);
816                }
817
818                // Add execution metadata as a chat message (optional - for debugging)
819                if log::log_enabled!(log::Level::Debug) {
820                    let metadata_msg = format!(
821                        " Execution Stats: {} tools, {} branches, {}ms",
822                        result.tools_called.len(),
823                        result.branches_taken.len(),
824                        result.execution_time_ms
825                    );
826                    let _ = self.add_message_to_session(
827                        session_id,
828                        ChatMessage::AgentThinking(metadata_msg),
829                    );
830                }
831
832                Ok(())
833            }
834            Err(e) => {
835                error!("Phase 2: OVSM execution failed: {}", e);
836                let _ = self.add_message_to_session(
837                    session_id,
838                    ChatMessage::Error(format!("Plan execution failed: {}", e)),
839                );
840                Err(e)
841            }
842        }
843    }

845    /// Phase 1 fallback: Execute tools iteratively without OVSM engine
846    async fn execute_tools_iteratively(
847        &self,
848        session_id: Uuid,
849        tools: Vec<PlannedTool>,
850        input: &str,
851        available_tools: &HashMap<String, Vec<McpTool>>,
852    ) -> Result<()> {
853        let mut executed_tools = Vec::new();
854        let mut iteration_count = 0;
855        let max_iterations = 3; // Reduced to prevent excessive loops
856
857        let mut current_tools = tools;
858        let mut executed_tool_signatures = std::collections::HashSet::new();
859
860        while !current_tools.is_empty() && iteration_count < max_iterations {
861            iteration_count += 1;
862
863            // Filter out duplicate tools to prevent infinite loops
864            current_tools.retain(|tool| {
865                let signature = format!("{}:{:?}", tool.tool_name, tool.args);
866                !executed_tool_signatures.contains(&signature)
867            });
868
869            if current_tools.is_empty() {
870                warn!("All remaining tools are duplicates, breaking execution loop");
871                break;
872            }
873
874            // Execute current batch of tools
875            for planned_tool in &current_tools {
876                let signature = format!("{}:{:?}", planned_tool.tool_name, planned_tool.args);
877                executed_tool_signatures.insert(signature);
878
879                let _ = self
880                    .execute_planned_tool(session_id, planned_tool.clone())
881                    .await;
882                executed_tools.push(planned_tool.clone());
883            }
884
885            // Check if we need follow-up actions based on results
886            let session = match self.get_session_by_id(session_id) {
887                Some(s) => s,
888                None => break, // Session not found, exit loop
889            };
890
891            let recent_results: Vec<(String, Value)> = session
892                .messages
893                .iter()
894                .rev()
895                .filter_map(|msg| match msg {
896                    ChatMessage::ToolResult {
897                        tool_name, result, ..
898                    } => Some((tool_name.clone(), result.clone())),
899                    _ => None,
900                })
901                .take(current_tools.len()) // Get results for current batch of tools
902                .collect();
903
904            // Ask AI if we need follow-up tools based on results
905            if iteration_count < max_iterations && !recent_results.is_empty() {
906                match self
907                    .check_for_follow_up_actions(input, &recent_results, available_tools)
908                    .await
909                {
910                    Ok(mut follow_up_tools) if !follow_up_tools.is_empty() => {
911                        // Filter follow-up tools to remove duplicates
912                        follow_up_tools.retain(|tool| {
913                            let signature = format!("{}:{:?}", tool.tool_name, tool.args);
914                            !executed_tool_signatures.contains(&signature)
915                        });
916
917                        if !follow_up_tools.is_empty() {
918                            let _ = self.add_message_to_session(
919                                session_id,
920                                ChatMessage::AgentThinking(format!(
921                                    "Executing {} follow-up actions...",
922                                    follow_up_tools.len()
923                                )),
924                            );
925                            current_tools = follow_up_tools;
926                        } else {
927                            warn!("All follow-up tools are duplicates, breaking execution loop");
928                            break;
929                        }
930                    }
931                    _ => break, // No more tools needed
932                }
933            } else {
934                break;
935            }
936        }
937
938        Ok(())
939    }
940}

942/// Phase 2: Wrapper that implements McpToolExecutor for OVSM executor
943struct McpToolWrapper {
944    state: AdvancedChatState,
945    session_id: Uuid,
946    server_id: String,
947    tool_name: String,
948}
949
950impl crate::services::ovsm_executor::McpToolExecutor for McpToolWrapper {
951    fn execute(&self, args: &serde_json::Value) -> Result<serde_json::Value> {
952        // We need to run async code in a sync context
953        // Use tokio's block_on from a new runtime
954        let state = self.state.clone();
955        let session_id = self.session_id;
956        let server_id = self.server_id.clone();
957        let tool_name = self.tool_name.clone();
958        let args = args.clone();
959
960        // Create PlannedTool from OVSM execution context
961        let planned_tool = PlannedTool {
962            server_id: server_id.clone(),
963            tool_name: tool_name.clone(),
964            args: args.clone(),
965            reason: format!("OVSM plan execution: {}", tool_name),
966        };
967
968        // We need to use a runtime to execute async code
969        // Try to get the current runtime, otherwise create one
970        let result = match tokio::runtime::Handle::try_current() {
971            Ok(handle) => {
972                // We're already in a tokio runtime
973                handle.block_on(async {
974                    execute_tool_async(&state, session_id, &planned_tool, &tool_name).await
975                })
976            }
977            Err(_) => {
978                // No runtime, create a new one
979                tokio::runtime::Runtime::new()
980                    .context("Failed to create Tokio runtime")?
981                    .block_on(async {
982                        execute_tool_async(&state, session_id, &planned_tool, &tool_name).await
983                    })
984            }
985        };
986
987        result
988    }
989}

991// Helper function to keep async logic separate
992async fn execute_tool_async(
993    state: &AdvancedChatState,
994    session_id: Uuid,
995    planned_tool: &PlannedTool,
996    tool_name: &str,
997) -> Result<serde_json::Value> {
998    // Add tool call message to chat
999    let execution_id = Uuid::new_v4().to_string();
1000    state.set_agent_state(session_id, AgentState::ExecutingTool(tool_name.to_string()));
1001    let _ = state.add_message_to_session(
1002        session_id,
1003        ChatMessage::ToolCall {
1004            tool_name: tool_name.to_string(),
1005            description: format!("Executing: {}", tool_name),
1006            args: Some(planned_tool.args.clone()),
1007            execution_id: execution_id.clone(),
1008        },
1009    );
1010
1011    // Execute the tool
1012    match state.call_mcp_tool(planned_tool).await {
1013        Ok(result) => {
1014            // Add result message to chat
1015            let _ = state.add_message_to_session(
1016                session_id,
1017                ChatMessage::ToolResult {
1018                    tool_name: tool_name.to_string(),
1019                    result: result.clone(),
1020                    execution_id,
1021                },
1022            );
1023            Ok(result)
1024        }
1025        Err(e) => {
1026            error!("MCP tool execution failed: {}", e);
1027            let _ = state.add_message_to_session(
1028                session_id,
1029                ChatMessage::Error(format!("Tool {} failed: {}", tool_name, e)),
1030            );
1031            Err(e)
1032        }
1033    }
