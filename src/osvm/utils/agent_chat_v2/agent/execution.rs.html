23    pub async fn process_input_async(&self, session_id: Uuid, input: String) -> Result<()> {
24        // Set agent state to thinking
25        self.set_agent_state(session_id, AgentState::Thinking);
26        let _ = self.add_message_to_session(session_id, ChatMessage::User(input.clone()));
27        let _ = self.add_message_to_session(
28            session_id,
29            ChatMessage::Processing {
30                message: "Analyzing your request...".to_string(),
31                spinner_index: self.spinner_state.load(Ordering::Relaxed),
32            },
33        );
34
35        // First, refresh available tools to ensure we have the latest
36        if let Err(e) = self.refresh_tools_from_mcp().await {
37            warn!("Failed to refresh tools: {}", e);
38        }
39
40        self.set_agent_state(session_id, AgentState::Planning);
41        let _ = self.add_message_to_session(
42            session_id,
43            ChatMessage::Processing {
44                message: "Creating execution plan...".to_string(),
45                spinner_index: self.spinner_state.load(Ordering::Relaxed),
46            },
47        );
48
49        // Get available tools
50        let available_tools = match self.available_tools.read() {
51            Ok(tools) => tools.clone(),
52            Err(e) => {
53                error!("Failed to read available tools: {}", e);
54                HashMap::new() // Use empty map as fallback
55            }
56        };
57
58        // Use the proper AI service method to create tool plan
59        let tool_plan_result = tokio::time::timeout(
60            Duration::from_secs(120), // Extended timeout for comprehensive AI responses
61            self.ai_service.create_tool_plan(&input, &available_tools),
62        )
63        .await;
64
65        // Helper closure to build heuristic tools based on input
66        let build_heuristic_plan = |text: &str| -> Vec<PlannedTool> {
67            let mut tools = Vec::new();
68            let lc = text.to_lowercase();
69
70            // Batch validator analysis
71            if (lc.contains("validator") || lc.contains("validators"))
72                && (lc.contains("100") || lc.contains("top") || lc.contains("analyze")) {
73                tools.push(PlannedTool {
74                    server_id: "local_sim".into(),
75                    tool_name: "analyze_batch_validators".into(),
76                    args: serde_json::json!({
77                        "count": 100,
78                        "metrics": ["stake", "commission", "uptime", "vote_credits", "delinquency"]
79                    }),
80                    reason: "Batch analyze validators with simulated data".into(),
81                });
82            }
83
84            // Batch token analysis
85            if (lc.contains("token") || lc.contains("tokens"))
86                && (lc.contains("50") || lc.contains("top") || lc.contains("portfolio")) {
87                tools.push(PlannedTool {
88                    server_id: "local_sim".into(),
89                    tool_name: "analyze_batch_tokens".into(),
90                    args: serde_json::json!({
91                        "count": 50,
92                        "timeframe": "30d",
93                        "metrics": ["price", "volume", "market_cap"]
94                    }),
95                    reason: "Batch analyze tokens with market data".into(),
96                });
97            }
98
99            // Batch account analysis
100            if (lc.contains("account") || lc.contains("accounts"))
101                && (lc.contains("transaction") || lc.contains("pattern")) {
102                tools.push(PlannedTool {
103                    server_id: "local_sim".into(),
104                    tool_name: "analyze_batch_accounts".into(),
105                    args: serde_json::json!({
106                        "account_count": 50,
107                        "transaction_limit": 100
108                    }),
109                    reason: "Batch analyze account transaction patterns".into(),
110                });
111            }
112
113            // Single balance query
114            if lc.contains("balance") && !lc.contains("validator") && !lc.contains("token") {
115                tools.push(PlannedTool {
116                    server_id: "local_sim".into(),
117                    tool_name: "get_balance".into(),
118                    args: serde_json::json!({}),
119                    reason: "Heuristic: user asked about balance".into(),
120                });
121            }
122
123            // Single transaction query
124            if (lc.contains("transaction") || lc.contains("transactions") || lc.contains("tx"))
125                && !lc.contains("account") && !lc.contains("validator") {
126                tools.push(PlannedTool {
127                    server_id: "local_sim".into(),
128                    tool_name: "get_transactions".into(),
129                    args: serde_json::json!({}),
130                    reason: "Heuristic: user asked about transactions".into(),
131                });
132            }
133
134            tools
135        };
136
137        // Determine if any MCP servers/tools are configured
138        let no_configured_tools = available_tools.is_empty();
139
140        match tool_plan_result {
141            Err(_) => {
142                warn!("AI planning timed out");
143                let _ = self.add_message_to_session(
144                    session_id,
145                    ChatMessage::Error(
146                        "AI planning timed out. Using heuristic fallback.".to_string(),
147                    ),
148                );
149                if no_configured_tools {
150                    let _ = self
151                        .run_heuristic_fallback(session_id, &input, build_heuristic_plan(&input))
152                        .await;
153                } else {
154                    let _ = self.simple_response(session_id, &input).await;
155                }
156            }
157            Ok(Err(e)) => {
158                error!("AI service failed: {}", e);
159                let _ = self.add_message_to_session(
160                    session_id,
161                    ChatMessage::Error(
162                        "AI planning service failed. Using heuristic fallback.".to_string(),
163                    ),
164                );
165                if no_configured_tools {
166                    let _ = self
167                        .run_heuristic_fallback(session_id, &input, build_heuristic_plan(&input))
168                        .await;
169                } else {
170                    let _ = self.simple_response(session_id, &input).await;
171                }
172            }
173            Ok(Ok(tool_plan)) => {
174                // Successfully got tool plan from AI service
175                let _ = self.add_message_to_session(
176                    session_id,
177                    ChatMessage::AgentPlan(tool_plan.osvm_tools_to_use.clone()),
178                );
179
180                if tool_plan.osvm_tools_to_use.is_empty() {
181                    // No tools needed according to AI
182                    if no_configured_tools {
183                        // Try heuristic plan instead
184                        let heur = build_heuristic_plan(&input);
185                        if !heur.is_empty() {
186                            let _ = self.add_message_to_session(
187                                session_id,
188                                ChatMessage::AgentPlan(heur.clone()),
189                            );
190                            let _ = self.run_heuristic_fallback(session_id, &input, heur).await;
191                        } else {
192                            let _ = self.simple_response(session_id, &input).await;
193                        }
194                    } else {
195                        let _ = self.simple_response(session_id, &input).await;
196                    }
197                } else {
198                    // Phase 2: Check if we have raw OVSM plan for execution engine
199                    if let Some(ref raw_ovsm) = tool_plan.raw_ovsm_plan {
200                        // Execute using OVSM engine with runtime branching
201                        debug!("Phase 2: Executing plan using OVSM execution engine");
202                        let _ = self.execute_ovsm_plan(session_id, raw_ovsm, &input).await;
203                    } else {
204                        // Fallback: Execute tools iteratively (Phase 1 behavior)
205                        debug!("No raw OVSM plan available, using iterative execution");
206                        let _ = self.execute_tools_iteratively(session_id, tool_plan.osvm_tools_to_use, &input, &available_tools).await;
207                    }
208
209                    // Generate final response with all executed tools
210                    let _ = self
211                        .generate_final_response(session_id, &input, &tool_plan.expected_outcome)
212                        .await;
213                }
214            }
215        }
216
217        // ALWAYS cleanup processing messages, even if errors occurred above
218        // This ensures the spinner stops regardless of success/failure
219        self.cleanup_processing_messages(session_id);
220
221        self.set_agent_state(session_id, AgentState::Idle);
222
223        // Generate suggestions after agent completes
224        let _ = self.generate_reply_suggestions(session_id).await;
225
226        Ok(())
227    }

229    /// Helper to ensure cleanup always happens
230    fn cleanup_processing_messages(&self, session_id: Uuid) {
231        // Remove ALL processing messages from the session
232        loop {
233            match self.remove_last_processing_message(session_id) {
234                Ok(()) => {
235                    // Check if there are more processing messages
236                    if let Some(session) = self.get_session_by_id(session_id) {
237                        let has_more_processing = session
238                            .messages
239                            .iter()
240                            .any(|msg| matches!(msg, ChatMessage::Processing { .. }));
241                        if !has_more_processing {
242                            break; // No more processing messages
243                        }
244                    } else {
245                        break; // Session not found
246                    }
247                }
248                Err(_) => break, // Error or no processing message found
249            }
250        }
251    }

253    // Heuristic fallback execution simulating basic tools so user sees end-to-end flow
254    async fn run_heuristic_fallback(
255        &self,
256        session_id: Uuid,
257        original_input: &str,
258        tools: Vec<PlannedTool>,
259    ) -> Result<()> {
260        if tools.is_empty() {
261            // Nothing heuristic matched; fall back to simple
262            return self.simple_response(session_id, original_input).await;
263        }
264        self.add_message_to_session(session_id, ChatMessage::AgentPlan(tools.clone()))?;
265        for planned_tool in tools {
266            self.execute_planned_tool(session_id, planned_tool).await?;
267        }
268        self.generate_final_response(session_id, original_input, "Heuristic simulated execution")
269            .await?;
270        Ok(())
271    }

273    async fn execute_planned_tool(
274        &self,
275        session_id: Uuid,
276        planned_tool: PlannedTool,
277    ) -> Result<()> {
278        let execution_id = Uuid::new_v4().to_string();
279
280        self.set_agent_state(
281            session_id,
282            AgentState::ExecutingTool(planned_tool.tool_name.clone()),
283        );
284        self.add_message_to_session(
285            session_id,
286            ChatMessage::ToolCall {
287                tool_name: planned_tool.tool_name.clone(),
288                description: planned_tool.reason.clone(),
289                args: Some(planned_tool.args.clone()),
290                execution_id: execution_id.clone(),
291            },
292        )?;
293
294        // Execute the tool using MCP service
295        match self.call_mcp_tool(&planned_tool).await {
296            Ok(result) => {
297                self.add_message_to_session(
298                    session_id,
299                    ChatMessage::ToolResult {
300                        tool_name: planned_tool.tool_name,
301                        result,
302                        execution_id,
303                    },
304                )?;
305            }
306            Err(e) => {
307                error!("Tool execution failed: {}", e);
308                self.add_message_to_session(
309                    session_id,
310                    ChatMessage::Error(format!("Tool {} failed: {}", planned_tool.tool_name, e)),
311                )?;
312            }
313        }
314
315        Ok(())
316    }

318    async fn call_mcp_tool(&self, planned_tool: &PlannedTool) -> Result<Value> {
319        // Load isolation configuration
320        let isolation_config = match IsolationConfig::load() {
321            Ok(config) => config,
322            Err(e) => {
323                warn!(
324                    "Failed to load isolation config: {}, using direct MCP execution",
325                    e
326                );
327                return self.call_mcp_tool_direct(planned_tool).await;
328            }
329        };
330
331        // Check if this tool should be executed in a unikernel
332        if isolation_config.should_use_unikernel(&planned_tool.server_id, &planned_tool.tool_name) {
333            // Attempt unikernel execution with fallback
334            match self
335                .execute_tool_in_unikernel(planned_tool, &isolation_config)
336                .await
337            {
338                Ok(result) => return Ok(result),
339                Err(e) => {
340                    warn!(
341                        "Unikernel execution failed: {}, falling back to direct MCP",
342                        e
343                    );
344                    return self.call_mcp_tool_direct(planned_tool).await;
345                }
346            }
347        }
348
349        // Use direct MCP execution (MicroVM mode or no isolation config)
350        self.call_mcp_tool_direct(planned_tool).await
351    }

353    /// Execute tool in ephemeral unikernel for maximum isolation
354    async fn execute_tool_in_unikernel(
355        &self,
356        planned_tool: &PlannedTool,
357        isolation_config: &IsolationConfig,
358    ) -> Result<Value> {
359        info!(
360            "Executing tool '{}' in ephemeral unikernel for server '{}'",
361            planned_tool.tool_name, planned_tool.server_id
362        );
363
364        // Get tool-specific configuration
365        let tool_config =
366            isolation_config.get_tool_config(&planned_tool.server_id, &planned_tool.tool_name);
367
368        // Validate unikernel image is configured
369        let unikernel_image = match &tool_config.unikernel_image {
370            Some(img) => img.clone(),
371            None => {
372                warn!(
373                    "No unikernel image configured for tool '{}' on server '{}', falling back",
374                    planned_tool.tool_name, planned_tool.server_id
375                );
376                return Err(anyhow!("No unikernel image configured"));
377            }
378        };
379
380        // Create unikernel runtime
381        let unikernel_runtime =
382            UnikernelRuntime::new(PathBuf::from(&isolation_config.unikernel_dir))
383                .context("Failed to create unikernel runtime")?;
384
385        // Build unikernel configuration
386        use crate::services::unikernel_runtime::UnikernelLauncher;
387
388        let unikernel_config = UnikernelConfig {
389            image_path: PathBuf::from(unikernel_image),
390            mounts: tool_config.mounts.clone(),
391            memory_mb: tool_config.memory_mb,
392            vcpus: tool_config.vcpus,
393            tool_name: planned_tool.tool_name.clone(),
394            server_id: planned_tool.server_id.clone(),
395            launcher: UnikernelLauncher::Unikraft,
396            kraft_config: None,
397            vsock_cid: None, // Will be auto-allocated
398        };
399
400        // Spawn the unikernel (~100ms overhead)
401        let handle = unikernel_runtime
402            .spawn_unikernel(unikernel_config)
403            .await
404            .context("Failed to spawn unikernel")?;
405
406        // Execute the tool in the isolated unikernel (pass runtime for vsock communication)
407        let result = handle
408            .execute_tool(
409                &planned_tool.tool_name,
410                Some(planned_tool.args.clone()),
411                &unikernel_runtime,
412            )
413            .await;
414
415        // Always terminate the ephemeral unikernel
416        handle.terminate();
417
418        match result {
419            Ok(value) => {
420                info!(
421                    "Successfully executed tool '{}' in unikernel",
422                    planned_tool.tool_name
423                );
424                Ok(value)
425            }
426            Err(e) => {
427                error!("Unikernel tool execution failed: {}", e);
428                Err(e)
429            }
430        }
431    }

638    async fn generate_final_response(
639        &self,
640        session_id: Uuid,
641        original_input: &str,
642        expected_outcome: &str,
643    ) -> Result<()> {

743    /// Phase 2: Execute OVSM plan using the execution engine with runtime branching
744    async fn execute_ovsm_plan(
745        &self,
746        session_id: Uuid,
747        raw_ovsm_plan: &str,
748        _original_input: &str,
749    ) -> Result<()> {
750        info!("Phase 2: Executing OVSM plan with execution engine");
751
752        // Get all available tools to register them with the executor
753        let available_tools = match self.available_tools.read() {
754            Ok(tools) => tools.clone(),
755            Err(e) => {
756                error!("Failed to read available tools: {}", e);
757                return Err(anyhow!("Failed to read available tools: {}", e));
758            }
759        };
760
761        // Register all available tools with OVSM executor
762        let executor = self.ovsm_executor.lock().await;
763        for (server_id, tool_list) in available_tools.iter() {
764            for tool in tool_list {
765                let tool_wrapper = McpToolWrapper {
766                    state: self.clone(),
767                    session_id,
768                    server_id: server_id.clone(),
769                    tool_name: tool.name.clone(),
770                };
771                executor
772                    .register_tool(tool.name.clone(), Box::new(tool_wrapper))
773                    .await?;
774            }
775        }
776
777        // Execute the plan
778        match executor.execute_plan(raw_ovsm_plan).await {
779            Ok(result) => {
780                info!(
781                    "Phase 2: OVSM execution completed successfully (confidence: {:.2})",
782                    result.confidence
783                );
784                debug!(
785                    "OVSM execution metadata: {} tools called, {} branches taken, {} warnings, execution time: {}ms",
786                    result.tools_called.len(),
787                    result.branches_taken.len(),
788                    result.warnings.len(),
789                    result.execution_time_ms
790                );
791
792                // Log execution details
793                if !result.branches_taken.is_empty() {
794                    debug!("Branch decisions: {:?}", result.branches_taken);
795                }
796                if !result.warnings.is_empty() {
797                    warn!("OVSM execution warnings: {:?}", result.warnings);
798                }
799
800                // Add execution metadata as a chat message (optional - for debugging)
801                if log::log_enabled!(log::Level::Debug) {
802                    let metadata_msg = format!(
803                        "📊 Execution Stats: {} tools, {} branches, {}ms",
804                        result.tools_called.len(),
805                        result.branches_taken.len(),
806                        result.execution_time_ms
807                    );
808                    let _ = self.add_message_to_session(
809                        session_id,
810                        ChatMessage::AgentThinking(metadata_msg),
811                    );
812                }
813
814                Ok(())
815            }
816            Err(e) => {
817                error!("Phase 2: OVSM execution failed: {}", e);
818                let _ = self.add_message_to_session(
819                    session_id,
820                    ChatMessage::Error(format!("Plan execution failed: {}", e)),
821                );
822                Err(e)
823            }
824        }
825    }

827    /// Phase 1 fallback: Execute tools iteratively without OVSM engine
828    async fn execute_tools_iteratively(
829        &self,
830        session_id: Uuid,
831        tools: Vec<PlannedTool>,
832        input: &str,
833        available_tools: &HashMap<String, Vec<McpTool>>,
834    ) -> Result<()> {
835        let mut executed_tools = Vec::new();
836        let mut iteration_count = 0;
837        let max_iterations = 3; // Reduced to prevent excessive loops
838
839        let mut current_tools = tools;
840        let mut executed_tool_signatures = std::collections::HashSet::new();
841
842        while !current_tools.is_empty() && iteration_count < max_iterations {
843            iteration_count += 1;
844
845            // Filter out duplicate tools to prevent infinite loops
846            current_tools.retain(|tool| {
847                let signature = format!("{}:{:?}", tool.tool_name, tool.args);
848                !executed_tool_signatures.contains(&signature)
849            });
850
851            if current_tools.is_empty() {
852                warn!("All remaining tools are duplicates, breaking execution loop");
853                break;
854            }
855
856            // Execute current batch of tools
857            for planned_tool in &current_tools {
858                let signature = format!("{}:{:?}", planned_tool.tool_name, planned_tool.args);
859                executed_tool_signatures.insert(signature);
860
861                let _ = self
862                    .execute_planned_tool(session_id, planned_tool.clone())
863                    .await;
864                executed_tools.push(planned_tool.clone());
865            }
866
867            // Check if we need follow-up actions based on results
868            let session = match self.get_session_by_id(session_id) {
869                Some(s) => s,
870                None => break, // Session not found, exit loop
871            };
872
873            let recent_results: Vec<(String, Value)> = session
874                .messages
875                .iter()
876                .rev()
877                .filter_map(|msg| match msg {
878                    ChatMessage::ToolResult {
879                        tool_name, result, ..
880                    } => Some((tool_name.clone(), result.clone())),
881                    _ => None,
882                })
883                .take(current_tools.len()) // Get results for current batch of tools
884                .collect();
885
886            // Ask AI if we need follow-up tools based on results
887            if iteration_count < max_iterations && !recent_results.is_empty() {
888                match self
889                    .check_for_follow_up_actions(input, &recent_results, available_tools)
890                    .await
891                {
892                    Ok(mut follow_up_tools) if !follow_up_tools.is_empty() => {
893                        // Filter follow-up tools to remove duplicates
894                        follow_up_tools.retain(|tool| {
895                            let signature = format!("{}:{:?}", tool.tool_name, tool.args);
896                            !executed_tool_signatures.contains(&signature)
897                        });
898
899                        if !follow_up_tools.is_empty() {
900                            let _ = self.add_message_to_session(
901                                session_id,
902                                ChatMessage::AgentThinking(format!(
903                                    "Executing {} follow-up actions...",
904                                    follow_up_tools.len()
905                                )),
906                            );
907                            current_tools = follow_up_tools;
908                        } else {
909                            warn!("All follow-up tools are duplicates, breaking execution loop");
910                            break;
911                        }
912                    }
913                    _ => break, // No more tools needed
914                }
915            } else {
916                break;
917            }
918        }
919
920        Ok(())
921    }
922}

924/// Phase 2: Wrapper that implements McpToolExecutor for OVSM executor
925struct McpToolWrapper {
926    state: AdvancedChatState,
927    session_id: Uuid,
928    server_id: String,
929    tool_name: String,
930}
931
932impl crate::services::ovsm_executor::McpToolExecutor for McpToolWrapper {
933    fn execute(&self, args: &serde_json::Value) -> Result<serde_json::Value> {
934        // We need to run async code in a sync context
935        // Use tokio's block_on from a new runtime
936        let state = self.state.clone();
937        let session_id = self.session_id;
938        let server_id = self.server_id.clone();
939        let tool_name = self.tool_name.clone();
940        let args = args.clone();
941
942        // Create PlannedTool from OVSM execution context
943        let planned_tool = PlannedTool {
944            server_id: server_id.clone(),
945            tool_name: tool_name.clone(),
946            args: args.clone(),
947            reason: format!("OVSM plan execution: {}", tool_name),
948        };
949
950        // We need to use a runtime to execute async code
951        // Try to get the current runtime, otherwise create one
952        let result = match tokio::runtime::Handle::try_current() {
953            Ok(handle) => {
954                // We're already in a tokio runtime
955                handle.block_on(async {
956                    execute_tool_async(&state, session_id, &planned_tool, &tool_name).await
957                })
958            }
959            Err(_) => {
960                // No runtime, create a new one
961                tokio::runtime::Runtime::new()
962                    .context("Failed to create Tokio runtime")?
963                    .block_on(async {
964                        execute_tool_async(&state, session_id, &planned_tool, &tool_name).await
965                    })
966            }
967        };
968
969        result
970    }
971}

973// Helper function to keep async logic separate
974async fn execute_tool_async(
975    state: &AdvancedChatState,
976    session_id: Uuid,
977    planned_tool: &PlannedTool,
978    tool_name: &str,
979) -> Result<serde_json::Value> {
980    // Add tool call message to chat
981    let execution_id = Uuid::new_v4().to_string();
982    state.set_agent_state(
983        session_id,
984        AgentState::ExecutingTool(tool_name.to_string()),
985    );
986    let _ = state.add_message_to_session(
987        session_id,
988        ChatMessage::ToolCall {
989            tool_name: tool_name.to_string(),
990            description: format!("Executing: {}", tool_name),
991            args: Some(planned_tool.args.clone()),
992            execution_id: execution_id.clone(),
993        },
994    );
995
996    // Execute the tool
997    match state.call_mcp_tool(planned_tool).await {
998        Ok(result) => {
999            // Add result message to chat
1000            let _ = state.add_message_to_session(
1001                session_id,
1002                ChatMessage::ToolResult {
1003                    tool_name: tool_name.to_string(),
1004                    result: result.clone(),
1005                    execution_id,
1006                },
1007            );
1008            Ok(result)
1009        }
1010        Err(e) => {
1011            error!("MCP tool execution failed: {}", e);
1012            let _ = state.add_message_to_session(
1013                session_id,
1014                ChatMessage::Error(format!("Tool {} failed: {}", tool_name, e)),
1015            );
1016            Err(e)
1017        }
1018    }
