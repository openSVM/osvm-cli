158    /// Show error dialog with recovery suggestions
159    pub fn show_error_dialog(&self, siv: &mut Cursive) {
160        let message = self.message();
161        let suggestions = self.recovery_suggestions();
162
163        let mut full_message = message.clone();
164        full_message.push_str("\n\n");
165        full_message.push_str("‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê\n");
166        full_message.push_str("üí° Recovery Suggestions:\n");
167        full_message.push_str("‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê\n");
168        for (i, suggestion) in suggestions.iter().enumerate() {
169            full_message.push_str(&format!("\n{}. {}", i + 1, suggestion));
170        }
171
172        siv.add_layer(
173            Dialog::text(full_message)
174                .title("‚ö†Ô∏è Error")
175                .button("OK", |s| {
176                    s.pop_layer();
177                })
178                .button("Show Help", |s| {
179                    s.pop_layer();
180                    super::handlers::show_advanced_help(s);
181                })
182                .max_width(80),
183        );
184    }
