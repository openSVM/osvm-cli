29pub fn retry_last_message(siv: &mut Cursive, state: AdvancedChatState) {
30    if let Some(session) = state.get_active_session() {
31        // Find the last user message
32        if let Some(last_user_msg) = session.messages.iter().rev().find_map(|msg| {
33            if let ChatMessage::User(text) = msg {
34                Some(text.clone())
35            } else {
36                None
37            }
38        }) {
39            // Re-send the message
40            handle_user_input(siv, &last_user_msg, state);
41        }
42    }
43}

45pub fn copy_last_message(siv: &mut Cursive, state: AdvancedChatState) {
46    if let Some(session) = state.get_active_session() {
47        if let Some(last_msg) = session.messages.last() {
48            let text_to_copy = match last_msg {
49                ChatMessage::User(text) | ChatMessage::Agent(text) => text.clone(),
50                _ => String::new(),
51            };
52
53            if !text_to_copy.is_empty() {
54                // Copy to actual system clipboard
55                match arboard::Clipboard::new() {
56                    Ok(mut clipboard) => {
57                        if let Err(e) = clipboard.set_text(&text_to_copy) {
58                            siv.add_layer(
59                                Dialog::info(format!("Failed to copy to clipboard: {}", e))
60                                    .title("Error"),
61                            );
62                        } else {
63                            siv.add_layer(
64                                Dialog::info("Message copied to clipboard!")
65                                    .title("Success")
66                                    .button("OK", |s| {
67                                        s.pop_layer();
68                                    }),
69                            );
70                        }
71                    }
72                    Err(e) => {
73                        siv.add_layer(
74                            Dialog::info(format!(
75                                "Clipboard not available: {}\nText:\n{}",
76                                e, text_to_copy
77                            ))
78                            .title("Clipboard Error")
79                            .button("OK", |s| {
80                                s.pop_layer();
81                            }),
82                        );
83                    }
84                }
85            }
86        }
87    }
88}

90pub fn delete_last_message(siv: &mut Cursive, state: AdvancedChatState) {
91    if let Some(session) = state.get_active_session() {
92        let session_id = session.id;
93
94        // First, clear any processing messages that might be stuck
95        if let Err(e) = state.remove_last_processing_message(session_id) {
96            error!("Failed to remove processing message: {}", e);
97        }
98
99        // Then delete the last actual message
100        if let Ok(mut sessions) = state.sessions.write() {
101            if let Some(session) = sessions.get_mut(&session_id) {
102                if !session.messages.is_empty() {
103                    session.messages.pop();
104                }
105            }
106        }
107
108        // Update UI after all changes
109        update_ui_displays(siv);
110    }
111}

113pub fn fork_conversation(siv: &mut Cursive, state: AdvancedChatState) {
114    if let Some(current_session) = state.get_active_session() {
115        // Create a new session with the same messages
116        let forked_name = format!("{} (Fork)", current_session.name);
117
118        if let Ok(new_session_id) = state.create_session(forked_name) {
119            // Copy messages to the new session
120            if let Ok(mut sessions) = state.sessions.write() {
121                if let Some(new_session) = sessions.get_mut(&new_session_id) {
122                    new_session.messages = current_session.messages.clone();
123                }
124            }
125
126            // Switch to the forked session
127            let _ = state.set_active_session(new_session_id);
128            update_ui_displays(siv);
129        }
130    }
131}

133pub fn insert_suggestion_at_cursor(siv: &mut Cursive, index: usize, state: AdvancedChatState) {
134    // Get the suggestion
135    let suggestion = {
136        let suggestions = state
137            .current_suggestions
138            .read()
139            .map(|s| s.clone())
140            .unwrap_or_default();
141
142        if index < suggestions.len() {
143            suggestions[index].clone()
144        } else {
145            return;
146        }
147    };
148
149    // Insert into the input field at cursor position
150    if let Some(mut input) = siv.find_name::<EditView>("input") {
151        let current_content = input.get_content();
152        let cursor_pos = input.get_cursor();
153
154        // Insert suggestion at cursor position
155        let mut new_content = String::new();
156        new_content.push_str(&current_content[..cursor_pos]);
157        new_content.push_str(&suggestion);
158        if cursor_pos < current_content.len() {
159            new_content.push_str(&current_content[cursor_pos..]);
160        }
161
162        input.set_content(new_content);
163        // Move cursor to end of inserted text
164        input.set_cursor(cursor_pos + suggestion.len());
165    }
166
167    // Hide suggestions after insertion
168    if let Ok(mut vis) = state.suggestions_visible.write() {
169        *vis = false;
170    }
171}

173pub fn handle_user_input(siv: &mut Cursive, text: &str, state: AdvancedChatState) {
174    if text.trim().is_empty() {
175        return;
176    }
177
178    let user_message = text.to_string();
179
180    // Hide suggestions when sending a message
181    if let Ok(mut vis) = state.suggestions_visible.write() {
182        *vis = false;
183    }
184
185    // Clear input
186    if let Some(mut input) = siv.find_name::<EditView>("input") {
187        input.set_content("");
188    }
189
190    // Immediately add user message to session and update display
191    if let Some(session) = state.get_active_session() {
192        // Add user message to the session
193        let _ = state.add_message_to_session(session.id, ChatMessage::User(user_message.clone()));
194
195        // Add processing indicator with animated spinner
196        let _ = state.add_message_to_session(
197            session.id,
198            ChatMessage::Processing {
199                message: "🤖 Processing your request...".to_string(),
200                spinner_index: 0,
201            },
202        );
203
204        // Update the display immediately
205        update_ui_displays(siv);
206
207        // Start animated processing with live updates
208        start_live_processing(siv, session.id, user_message, state);
209    }
210}

383pub fn export_chat(siv: &mut Cursive) {
384    siv.with_user_data(|state: &mut AdvancedChatState| {
385        if let Some(session) = state.get_active_session() {
386            let filename = format!(
387                "osvm_chat_export_{}_{}.json",
388                session.name.replace(' ', "_"),
389                chrono::Utc::now().format("%Y%m%d_%H%M%S")
390            );
391
392            match serde_json::to_string_pretty(&session) {
393                Ok(json_content) => {
394                    match std::fs::write(&filename, json_content) {
395                        Ok(_) => {
396                            info!("Chat exported to {}", filename);
397                            // Add success message to current session
398                            if let Ok(mut sessions) = state.sessions.write() {
399                                if let Some(current_session) = sessions.get_mut(&session.id) {
400                                    current_session.add_message(ChatMessage::System(format!(
401                                        "Chat exported to {}",
402                                        filename
403                                    )));
404                                }
405                            }
406                        }
407                        Err(e) => {
408                            error!("Failed to write export file: {}", e);
409                        }
410                    }
411                }
412                Err(e) => {
413                    error!("Failed to serialize session: {}", e);
414                }
415            }
416        }
417    });
418
419    update_ui_displays(siv);
420}

650fn add_mcp_server_from_form(siv: &mut Cursive) {
651    let name = siv
652        .find_name::<EditView>("server_name")
653        .map(|v| v.get_content().to_string())
654        .unwrap_or_default();
655
656    let url = siv
657        .find_name::<EditView>("server_url")
658        .map(|v| v.get_content().to_string())
659        .unwrap_or_default();
660
661    let server_type = siv
662        .find_name::<SelectView<String>>("server_type")
663        .and_then(|v| v.selection())
664        .map(|s| s.as_str().to_string())
665        .unwrap_or_else(|| "http".to_string());
666
667    if name.is_empty() || url.is_empty() {
668        siv.add_layer(
669            Dialog::text("Please fill in all required fields.")
670                .title("Error")
671                .button("OK", |s| {
672                    s.pop_layer();
673                }),
674        );
675        return;
676    }
677

1339// Execute AI plan with XML parsing and tool execution
1340async fn execute_ai_plan(
1341    ai_response: String,
1342    session_id: uuid::Uuid,
1343    state: AdvancedChatState,
1344) -> String {
1345    use regex::Regex;
1346
1347    // Extract plan and response sections with better regex
1348    let plan_regex = Regex::new(r"(?s)<plan>(.*?)</plan>").unwrap();
1349    let response_regex = Regex::new(r"(?s)<response>(.*?)</response>").unwrap();
1350
1351    let plan_content = plan_regex
1352        .captures(&ai_response)
1353        .and_then(|cap| cap.get(1))
1354        .map(|m| m.as_str().trim())
1355        .unwrap_or("");
1356
1357    let response_content = response_regex
1358        .captures(&ai_response)
1359        .and_then(|cap| cap.get(1))
1360        .map(|m| m.as_str().trim())
1361        .unwrap_or(&ai_response); // If no proper XML format, return the whole response
1362
1363    // If plan is empty, return the response directly
1364    if plan_content.is_empty() {
1365        return response_content.to_string();
1366    }
1367
1368    // Parse tool calls from plan
1369    let tool_regex =
1370        Regex::new(r#"(?s)<tool name="([^"]+)" server="([^"]+)">(.*?)</tool>"#).unwrap();
1371    let param_regex = Regex::new(r#"<param name="([^"]+)">([^<]*)</param>"#).unwrap();
1372
1373    let mut execution_results = Vec::new();
1374    let mut final_response = response_content.to_string();
1375
1376    let tool_matches: Vec<_> = tool_regex.captures_iter(&plan_content).collect();
1377
1378    for captures in tool_matches {
1379        let tool_name = captures.get(1).unwrap().as_str();
1380        let server_id = captures.get(2).unwrap().as_str();
1381        let params_section = captures.get(3).unwrap().as_str();
1382
1383        // Extract parameters
1384        let mut params = std::collections::HashMap::new();
1385        for param_match in param_regex.captures_iter(params_section) {
1386            let param_name = param_match.get(1).unwrap().as_str();
1387            let param_value = param_match.get(2).unwrap().as_str();
1388            params.insert(
1389                param_name.to_string(),
1390                serde_json::Value::String(param_value.to_string()),
1391            );
1392        }
1393
1394        let execution_id = uuid::Uuid::new_v4().to_string();
1395
1396        // Add tool execution message
1397        let _ = state.add_message_to_session(
1398            session_id,
1399            ChatMessage::ToolCall {
1400                tool_name: tool_name.to_string(),
1401                description: format!("Executing {} on server {}", tool_name, server_id),
1402                args: Some(serde_json::Value::Object(
1403                    params.clone().into_iter().collect(),
1404                )),
1405                execution_id: execution_id.clone(),
1406            },
1407        );
1408
1409        // Execute the tool
1410        let tool_result = execute_mcp_tool(server_id, tool_name, params, state.clone()).await;
1411
1412        // Add tool result message
1413        let _ = state.add_message_to_session(
1414            session_id,
1415            ChatMessage::ToolResult {
1416                tool_name: tool_name.to_string(),
1417                result: serde_json::Value::String(tool_result.clone()),
1418                execution_id,
1419            },
1420        );
1421
1422        execution_results.push(format!("{}: {}", tool_name, tool_result));
1423    }
1424
1425    // Combine response with tool results
1426    if !execution_results.is_empty() {
1427        final_response.push_str("\n\nTool Execution Results:\n");
1428        for result in execution_results {
1429            final_response.push_str(&format!("• {}\n", result));
1430        }
1431    }
1432
1433    final_response
1434}
