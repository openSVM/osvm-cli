51pub fn copy_last_message(siv: &mut Cursive, state: AdvancedChatState) {
52    if let Some(session) = state.get_active_session() {
53        if let Some(last_msg) = session.messages.last() {
54            let text_to_copy = match last_msg {
55                ChatMessage::User(text) | ChatMessage::Agent(text) => text.clone(),
56                _ => String::new(),
57            };
58
59            if !text_to_copy.is_empty() {
60                // Copy to actual system clipboard
61                match arboard::Clipboard::new() {
62                    Ok(mut clipboard) => {
63                        if let Err(e) = clipboard.set_text(&text_to_copy) {
64                            siv.add_layer(
65                                Dialog::info(format!("Failed to copy to clipboard: {}", e))

136pub fn fork_conversation(siv: &mut Cursive, state: AdvancedChatState) {
137    if let Some(current_session) = state.get_active_session() {
138        // Create a new session with the same messages
139        let forked_name = format!("{} (Fork)", current_session.name);
140
141        if let Ok(new_session_id) = state.create_session(forked_name) {
142            // Copy messages to the new session
143            if let Ok(mut sessions) = state.sessions.write() {
144                if let Some(new_session) = sessions.get_mut(&new_session_id) {
145                    new_session.messages = current_session.messages.clone();
146                }
147            }
148
149            // Switch to the forked session
150            if let Err(e) = state.set_active_session(new_session_id) {
151                error!("Failed to set active session after fork: {}", e);
152            }
153            update_ui_displays(siv);
154        }
155    }

234pub fn insert_suggestion_at_cursor(siv: &mut Cursive, index: usize, state: AdvancedChatState) {
235    // Get the suggestion
236    let suggestion = {
237        let suggestions = state

244            suggestions[index].clone()
245        } else {
246            return;
247        }
248    };
249
250    // Insert into the input field at cursor position
251    if let Some(mut input) = siv.find_name::<SendableTextArea>("input") {
252        let current_content = input.get_content().to_string();
253
254        // For TextArea, append suggestion at the end for now
255        // FUTURE: Cursor position insertion (blocked by TextArea API limitations)
256        let mut new_content = current_content;
257        if !new_content.is_empty() && !new_content.ends_with('\n') {
258            new_content.push(' ');
259        }
260        new_content.push_str(&suggestion);
261
262        input.set_content(new_content);
263    }
264
265    // Hide suggestions after insertion
266    if let Ok(mut vis) = state.suggestions_visible.write() {
267        *vis = false;
268    }

271pub fn handle_user_input(siv: &mut Cursive, text: &str, state: AdvancedChatState) {
272    use super::input_validation::{validate_input, ValidationResult, contains_sensitive_pattern, get_sensitive_warning};
273
274    // Validate input first
275    match validate_input(text) {
276        ValidationResult::Empty | ValidationResult::OnlyWhitespace => {
277            // Silently ignore empty input
278            return;
279        }
280        ValidationResult::TooLong { text: truncated, max_length } => {
281            siv.add_layer(
282                Dialog::text(format!(
283                    "Message Too Long\n\n\
284                    Your message is too long ({} characters).\n\
285                    Maximum allowed: {} characters.\n\n\
286                    The message has been truncated. Please shorten it.",
287                    text.len(),
288                    max_length
289                ))
290                .title("Input Validation")
291                .button("OK", |s| {
292                    s.pop_layer();
293                })
294            );
295            return;
296        }
297        ValidationResult::TooManyNewlines { max_lines, .. } => {
298            siv.add_layer(
299                Dialog::text(format!(
300                    "Too Many Line Breaks\n\n\
301                    Your message has too many line breaks.\n\
302                    Maximum allowed: {} lines.\n\n\
303                    Please reduce the number of line breaks.",
304                    max_lines
305                ))
306                .title("Input Validation")
307                .button("OK", |s| {
308                    s.pop_layer();
309                })
310            );
311            return;
312        }
313        ValidationResult::ContainsBinaryData => {
314            siv.add_layer(
315                Dialog::text(
316                    "Invalid Characters\n\n\
317                    Your message contains invalid or binary characters.\n\
318                    Please use only text characters."
319                )
320                .title("Input Validation")
321                .button("OK", |s| {
322                    s.pop_layer();
323                })
324            );
325            return;
326        }
327        ValidationResult::Valid(validated_text) => {
328            // Check for sensitive data
329            if contains_sensitive_pattern(&validated_text) {
330                let text_clone = validated_text.clone();
331                let state_clone = state.clone();
332                siv.add_layer(
333                    Dialog::text(get_sensitive_warning())
334                        .title("Sensitive Data Warning")
335                        .button("Cancel", |s| {
336                            s.pop_layer();
337                        })
338                        .button("Send Anyway", move |s| {
339                            s.pop_layer();
340                            process_validated_input(s, &text_clone, state_clone.clone());
341                        })
342                );
343                return;
344            }
345
346            // Process validated input
347            process_validated_input(siv, &validated_text, state);
348        }
349    }

352fn process_validated_input(siv: &mut Cursive, text: &str, state: AdvancedChatState) {
353    let user_message = text.to_string();
354
355    // Add to history before processing
356    state.add_to_history(user_message.clone());
357
358    // Hide suggestions when sending a message
359    if let Ok(mut vis) = state.suggestions_visible.write() {
360        *vis = false;
361    }
362
363    // Clear input
364    if let Some(mut input) = siv.find_name::<SendableTextArea>("input") {
365        input.set_content("");
366    }
367
368    // Check for theme commands first
369    if user_message.trim().starts_with("/theme") {
370        handle_theme_command(siv, &user_message, state);
371        return;
372    }
373
374    // Check for duplicate input to prevent processing the same message twice
375    if let Some(session) = state.get_active_session() {
376        // Check if the last user message is the same as the current one
377        // Look through messages in reverse to find the last User message
378        if let Some(last_user_msg) = session.messages.iter().rev().find_map(|msg| {
379            if let ChatMessage::User(text) = msg {
380                Some(text)
381            } else {
382                None
383            }
384        }) {
385            if last_user_msg == &user_message {
386                warn!("Duplicate user input detected, ignoring: {}", user_message);
387                return;
388            }
389        }
390
391        // Add user message to the session
392        if let Err(e) = state.add_message_to_session(session.id, ChatMessage::User(user_message.clone())) {
393            error!("Failed to add user message to session: {}", e);
394        }
395
396        // Add processing indicator with animated spinner
397        if let Err(e) = state.add_message_to_session(
398            session.id,
399            ChatMessage::Processing {
400                message: "Processing your request...".to_string(),
401                spinner_index: 0,
402            },
403        ) {
404            error!("Failed to add processing message to session: {}", e);
405        }
406
407        // Update the display immediately
408        update_ui_displays(siv);
409
410        // Start animated processing with live updates
411        start_live_processing(siv, session.id, user_message, state);
412    }
413}

1206        }

1534pub fn start_live_processing(
1535    siv: &mut Cursive,
1536    session_id: uuid::Uuid,
1537    user_input: String,
1538    state: AdvancedChatState,
1539) {
1540    use std::sync::atomic::{AtomicUsize, Ordering};
1541    use std::sync::Arc;
1542    use std::thread;
1543    use std::time::Duration;
1544
1545    let spinner_counter = Arc::new(AtomicUsize::new(0));
1546
1547    // Get a callback sink to update UI from background thread
1548    let cb_sink = siv.cb_sink().clone();
1549
1550    // Clone for background thread
1551    let state_clone = state.clone();
1552    let spinner_clone = spinner_counter.clone();
1553
1554    // Start spinner animation in background thread
1555    thread::spawn(move || {
1556        let stages = [
1557            "Analyzing your request...",
1558            "Searching for relevant information...",
1559            "Processing with AI...",
1560            "Generating response...",
1561            "Finalizing answer...",
1562            "Almost done...",
1563            "Wrapping up...",
1564            "Just a moment...",
1565            "Completing your request...",
1566            "Preparing response...",
1567            "Finalizing response...",
1568            "Almost there...",
1569            "Just a moment longer...",
1570            "Completing the process...",
1571            "Preparing the final response...",
1572            "Finishing up...",
1573            "Polishing response...",
1574            "Nearly complete...",
1575            "Final touches...",
1576            "Quality check...",
1577            "Optimizing results...",
1578            "Validating response...",
1579            "Last steps...",
1580            "Wrapping up processing...",
1581        ];
1582
1583        let mut stage_index = 0;
1584        let mut update_counter = 0;
1585        let start_time = std::time::Instant::now();
1586        const MAX_SPINNER_DURATION: std::time::Duration = std::time::Duration::from_secs(45);
1587
1588        loop {
1589            // Check if processing message still exists in session BEFORE updating
1590            let still_processing = state_clone
1591                .get_session_by_id(session_id)
1592                .map(|session| {
1593                    session
1594                        .messages
1595                        .iter()
1596                        .any(|msg| matches!(msg, ChatMessage::Processing { .. }))
1597                })
1598                .unwrap_or(false);
1599
1600            if !still_processing {
1601                break; // Exit loop when processing message is removed
1602            }
1603
1604            // Safety timeout: force exit after 30 seconds
1605            if start_time.elapsed() > MAX_SPINNER_DURATION {
1606                // Force remove processing message and exit
1607                if let Err(e) = state_clone.remove_last_processing_message(session_id) {
1608                    error!("Failed to remove processing message after timeout: {}", e);
1609                }
1610                break;
1611            }
1612
1613            let spinner_index = spinner_clone.fetch_add(1, Ordering::Relaxed);
1614
1615            // Change message every 30 updates (roughly 3 seconds)
1616            if update_counter % 30 == 0 {
1617                stage_index = (stage_index + 1) % stages.len();
1618            }
1619            update_counter += 1;
1620
1621            let message = stages[stage_index].to_string();
1622
1623            // Only update if processing message still exists
1624            // This check prevents re-adding the message after agent removes it
1625            if still_processing {
1626                // Double-check: only update if message still exists (race condition guard)
1627                let still_processing_after_check = state_clone
1628                    .get_session_by_id(session_id)
1629                    .map(|session| {
1630                        session
1631                            .messages
1632                            .iter()
1633                            .any(|msg| matches!(msg, ChatMessage::Processing { .. }))
1634                    })
1635                    .unwrap_or(false);
1636
1637                if still_processing_after_check {
1638                    if let Err(e) = state_clone.update_processing_message(
1639                        session_id,
1640                        message.clone(),
1641                        spinner_index % 10,
1642                    ) {
1643                        error!("Failed to update processing message: {}", e);
1644                    }
1645                }
1646            }
1647
1648            // Request UI update
1649            cb_sink
1650                .send(Box::new(move |siv| {
1651                    update_ui_displays(siv);
1652                }))
1653                .ok();
1654
1655            thread::sleep(Duration::from_millis(100));
1656        }
1657    });
1658
1659    // Use the proper agent worker flow which calls AI service first
1660    // Send to agent worker instead of handling inline
1661    let command = AgentCommand::ProcessInput {
1662        session_id,
1663        input: user_input,
1664    };
1665
1666    // Send command through the proper channel
1667    state.send_agent_command_sync(command);
1668
1669    // NOTE: We don't stop the spinner here - it will keep running and be cleaned up
1670    // when the agent worker completes and removes the processing message
1671}
