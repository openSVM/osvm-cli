18    pub fn create_chat_list_panel(&self) -> impl View {
19        let mut chat_list_layout = LinearLayout::vertical();
20
21        // Header with gradient effect and icon
22        let header_text = format!("{} Chat Sessions {}", Icons::CHAT, Icons::SPARKLES);
23        chat_list_layout.add_child(
24            Panel::new(TextView::new(StyledText::gradient(&header_text)))
25                .title(Decorations::section_divider("Sessions")),
26        );
27
28        // Chat list - use ListView instead for more reliable navigation
29        let chat_list = ListView::new().with_name("chat_list");
30
31        // We'll populate this manually in the update function
32
33        // Add ListView directly inside a Panel - responsive height
34        chat_list_layout.add_child(
35            Panel::new(
36                ScrollView::new(chat_list)
37                    .scroll_strategy(cursive::view::scroll::ScrollStrategy::KeepRow),
38            )
39            .title("Sessions")
40            .min_height(5)
41            .max_height(20), // Allow more flexibility
42        );
43
44        // New chat button with icon
45        chat_list_layout.add_child(DummyView.fixed_height(1));
46        chat_list_layout.add_child(Button::new(format!("{} New Chat", Icons::NEW), |siv| {
47            create_new_chat_dialog(siv);
48        }));
49
50        // Session controls with icons - responsive spacing
51        chat_list_layout.add_child(DummyView.min_height(1));
52        chat_list_layout.add_child(
53            LinearLayout::horizontal()
54                .child(Button::new(format!("{} Run", Icons::EXECUTING), |siv| {
55                    resume_agent(siv)
56                }))
57                .child(DummyView.min_width(1))
58                .child(Button::new(format!("{} Pause", Icons::PAUSED), |siv| {
59                    pause_agent(siv)
60                }))
61                .child(DummyView.min_width(1))
62                .child(Button::new(format!("{} Stop", Icons::STOP), |siv| {
63                    stop_agent(siv)
64                })),
65        );
66
67        // Recording controls with icons - responsive spacing
68        chat_list_layout.add_child(DummyView.min_height(1));
69        chat_list_layout.add_child(
70            LinearLayout::horizontal()
71                .child(Button::new(format!("{} Record", Icons::RECORD), |siv| {
72                    start_recording(siv)
73                }))
74                .child(DummyView.min_width(1))
75                .child(Button::new(format!("{} Stop Rec", Icons::STOP), |siv| {
76                    stop_recording(siv)
77                })),
78        );
79
80        chat_list_layout.add_child(DummyView.min_height(1));
81
82        // MCP Tools Panel - responsive height with selectable tools
83        let mcp_tools_view = SelectView::<(String, String)>::new()
84            .on_submit(|siv, (server_id, tool_name)| {
85                show_tool_details(siv, server_id.clone(), tool_name.clone());
86            })
87            .with_name("mcp_tools_list");
88        let mcp_panel = Panel::new(ScrollView::new(mcp_tools_view))
89            .title(format!("{} MCP Tools (Enter for details)", Icons::TOOL))
90            .min_height(3)
91            .max_height(15); // Allow more flexibility
92        chat_list_layout.add_child(mcp_panel);
93
94        chat_list_layout
95    }

97    pub fn create_chat_panel(&self) -> impl View {
98        let mut chat_layout = LinearLayout::vertical();
99
100        // Chat history area
101        let chat_view = ScrollView::new(TextView::new("").with_name("chat_display").full_width())
102            .scroll_strategy(cursive::view::scroll::ScrollStrategy::StickToBottom)
103            .with_name("chat_scroll");
104
105        let chat_panel = Panel::new(chat_view)
106            .title(format!("{} Chat History", Icons::CHAT))
107            .title_position(cursive::align::HAlign::Left);
108
109        chat_layout.add_child(chat_panel.full_height());
110
111        // Suggestions area (shown when available) - no frame, just colored text
112        let suggestions_view = LinearLayout::vertical().with_name("suggestions_container");
113        chat_layout.add_child(suggestions_view);
114
115        // Input area
116        let input_layout = LinearLayout::horizontal()
117            .child(TextView::new("You: "))
118            .child(
119                EditView::new()
120                    .on_submit({
121                        let state = self.state.clone();
122                        move |siv, text| {
123                            handle_user_input(siv, text, state.clone());
124                        }
125                    })
126                    .with_name("input")
127                    .full_width(),
128            );
129
130        chat_layout.add_child(Panel::new(input_layout).title("Input"));
131
132        // Agent status bar with live updates and icon
133        let agent_status = TextView::new(format!("{} Agent: Initializing...", Icons::LIGHTNING))
134            .with_name("agent_status");
135        chat_layout
136            .add_child(Panel::new(agent_status).title(format!("{} Agent Status", Icons::ROCKET)));
137
138        // System status bar showing microVM/unikernel statuses and mounts with icon
139        let system_status = TextView::new(format!("{} OSVM: Initializing...", Icons::STAR))
140            .with_name("system_status_bar");
141        chat_layout
142            .add_child(Panel::new(system_status).title(format!("{} System Status", Icons::INFO)));
143
144        // Control buttons
145        let button_layout = LinearLayout::horizontal()
146            .child(Button::new("Clear Chat", |siv| clear_current_chat(siv)))
147            .child(Button::new("Export Chat", |siv| export_chat(siv)))
148            .child(Button::new("Settings", |siv| show_settings(siv)))
149            .child(Button::new("Help", show_advanced_help))
150            .child(Button::new("Quit", |siv| siv.quit()));
151
152        chat_layout.add_child(button_layout);
153
154        chat_layout
155    }
