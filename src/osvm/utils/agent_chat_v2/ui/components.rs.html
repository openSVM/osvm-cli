77    pub fn create_chat_panel(&self) -> impl View {
78        let mut chat_layout = LinearLayout::vertical();
79
80        // Chat history area
81        let chat_view = ScrollView::new(TextView::new("").with_name("chat_display").full_width())
82            .scroll_strategy(cursive::view::scroll::ScrollStrategy::StickToBottom)
83            .with_name("chat_scroll");
84
85        let chat_panel = Panel::new(chat_view)
86            .title("Chat History")
87            .title_position(cursive::align::HAlign::Left);
88
89        chat_layout.add_child(chat_panel.full_height());
90
91        // Suggestions area (shown when available) - no frame, just colored text
92        let suggestions_view = LinearLayout::vertical().with_name("suggestions_container");
93        chat_layout.add_child(suggestions_view);
94
95        // Input area
96        let input_layout = LinearLayout::horizontal()
97            .child(TextView::new("You: "))
98            .child(
99                EditView::new()
100                    .on_submit({
101                        let state = self.state.clone();
102                        move |siv, text| {
103                            handle_user_input(siv, text, state.clone());
104                        }
105                    })
106                    .with_name("input")
107                    .full_width(),
108            );
109
110        chat_layout.add_child(Panel::new(input_layout).title("Input"));
111
112        // Agent status bar with live updates
113        let agent_status = TextView::new("🤖 Agent: Initializing...").with_name("agent_status");
114        chat_layout.add_child(Panel::new(agent_status).title("🤖 Agent Status"));
115
116        // Control buttons
117        let button_layout = LinearLayout::horizontal()
118            .child(Button::new("Clear Chat", |siv| clear_current_chat(siv)))
119            .child(Button::new("Export Chat", |siv| export_chat(siv)))
120            .child(Button::new("Settings", |siv| show_settings(siv)))
121            .child(Button::new("Help", show_advanced_help))
122            .child(Button::new("Quit", |siv| siv.quit()));
123
124        chat_layout.add_child(button_layout);
125
126        chat_layout
127    }
