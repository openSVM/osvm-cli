290    /// Create collapsible MCP tools view
291    pub fn create_collapsible_mcp_tools(&self) -> impl View {
292        let mut tools_list = ListView::new();
293
294        // Get available tools grouped by server
295        if let Ok(available_tools) = self.state.available_tools.read() {
296            for (server_id, tools) in available_tools.iter() {
297                // Server header (clickable to collapse/expand)
298                let server_display = format!("▼ {} ({} tools)", server_id, tools.len());
299                let server_id_clone = server_id.clone();
300
301                tools_list.add_child(
302                    &format!("server_{}", server_id),
303                    Button::new_raw(server_display, move |s| {
304                        toggle_mcp_server_collapsed(s, server_id_clone.clone())
305                    })
306                    .full_width(),
307                );
308
309                // Tools under this server (visible when expanded)
310                for tool in tools.iter().take(10) {
311                    // Limit to first 10 tools
312                    let tool_name = tool.name.clone();
313                    let server_id_clone = server_id.clone();
314                    let tool_display = format!("  · {}", tool.name);
315
316                    tools_list.add_child(
317                        &format!("tool_{}_{}", server_id, tool.name),
318                        Button::new_raw(tool_display, move |s| {
319                            show_tool_details(s, server_id_clone.clone(), tool_name.clone())
320                        })
321                        .full_width(),
322                    );
323                }
324
325                if tools.len() > 10 {
326                    tools_list.add_child(
327                        &format!("more_{}", server_id),
328                        TextView::new(format!("  ... and {} more", tools.len() - 10)),
329                    );
330                }
331            }
332        } else {
333            tools_list.add_child("no_tools", TextView::new("No MCP tools available"));
334        }
335
336        ScrollView::new(tools_list.with_name("mcp_tools_list"))
337            .scroll_strategy(cursive::view::scroll::ScrollStrategy::KeepRow)
338    }
