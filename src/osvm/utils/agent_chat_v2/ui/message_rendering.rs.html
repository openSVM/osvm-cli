170    /// Render a message with all effects
171    pub fn render_message(&self, message: &ChatMessage, msg_type: MessageType) -> StyledString {
172        let mut result = StyledString::new();
173
174        // Add timestamp if enabled
175        if self.show_timestamps && !self.compact_mode {
176            let timestamp = ChatMessage::get_display_timestamp();
177            result.append_styled(timestamp, ColorStyle::secondary());
178            result.append_plain(" ");
179        }
180
181        // Add sender with semantic color
182        let sender_style = if self.semantic_colors {
183            msg_type.color()
184        } else {
185            ColorStyle::primary()
186        };
187
188        let icon = if self.compact_mode {
189            msg_type.ascii_icon()
190        } else {
191            msg_type.icon()
192        };
193
194        // Pre-allocate strings for cases that need owned values
195        let plan_display: String;
196
197        // Extract sender and content using explicit lifetime handling
198        let (sender, content): (&str, &str) = match message {
199            ChatMessage::User(text) => ("User", text.as_str()),
200            ChatMessage::Agent(text) => ("Agent", text.as_str()),
201            ChatMessage::System(text) => ("System", text.as_str()),
202            ChatMessage::ToolCall { tool_name, .. } => (tool_name.as_str(), "Calling tool"),
203            ChatMessage::ToolResult { tool_name, .. } => (tool_name.as_str(), "Tool completed"),
204            ChatMessage::Error(text) => ("Error", text.as_str()),
205            ChatMessage::AgentThinking(text) => ("Thinking", text.as_str()),
206            ChatMessage::AgentPlan(plan) => {
207                plan_display = format!("{} planned tools", plan.len());
208                ("Planning", plan_display.as_str())
209            }
210            ChatMessage::Processing { message: msg, .. } => ("Processing", msg.as_str()),
211        };
212
213        result.append_styled(format!("{} {}: ", icon, sender), sender_style);
214
215        // Add message content
216        let content_style = match msg_type {
217            MessageType::Code => msg_type.color(),
218            MessageType::Error => {
219                ColorStyle::new(Color::Rgb(254, 202, 202), Color::Rgb(30, 30, 30))
220            }
221            _ => ColorStyle::primary(),
222        };
223
224        result.append_styled(content, content_style);
225
226        result
227    }

229    /// Render message in a box with borders
230    pub fn render_boxed_message(
231        &self,
232        message: &ChatMessage,
233        msg_type: MessageType,
234        width: usize,
235    ) -> Vec<String> {
236        let border = msg_type.border_style();
237        let color = msg_type.color();
238
239        let mut lines = Vec::new();
240
241        // Top border
242        let top = format!(
243            "{}{}{}",
244            border.top_left,
245            border.horizontal.to_string().repeat(width - 2),
246            border.top_right
247        );
248        lines.push(top);
249
250        // Message content wrapped
251        let content = match message {
252            ChatMessage::User(text) => text,
253            ChatMessage::Agent(text) => text,
254            ChatMessage::System(text) => text,
255            ChatMessage::ToolCall {
256                tool_name,
257                description,
258                ..
259            } => &format!("{}: {}", tool_name, description),
260            ChatMessage::ToolResult {
261                tool_name, result, ..
262            } => &format!("{} result: {}", tool_name, result),
263            ChatMessage::Error(text) => text,
264            ChatMessage::AgentThinking(text) => text,
265            ChatMessage::AgentPlan(plan) => &format!("Plan with {} tools", plan.len()),
266            ChatMessage::Processing { message, .. } => message,
267        };
268
269        let wrapped = self.wrap_text(content, width - 4);
270        for line in wrapped {
271            lines.push(format!(
272                "{} {:<width$} {}",
273                border.vertical,
274                line,
275                border.vertical,
276                width = width - 4
277            ));
278        }
279
280        // Bottom border
281        let bottom = format!(
282            "{}{}{}",
283            border.bottom_left,
284            border.horizontal.to_string().repeat(width - 2),
285            border.bottom_right
286        );
287        lines.push(bottom);
288
289        lines
290    }

298    /// Wrap text to fit width
299    fn wrap_text(&self, text: &str, width: usize) -> Vec<String> {
300        let mut lines = Vec::new();
301        let mut current_line = String::new();
302
303        for word in text.split_whitespace() {
304            if current_line.len() + word.len() + 1 > width {
305                if !current_line.is_empty() {
306                    lines.push(current_line.clone());
307                    current_line.clear();
308                }
309            }
310
311            if !current_line.is_empty() {
312                current_line.push(' ');
313            }
314            current_line.push_str(word);
315        }
316
317        if !current_line.is_empty() {
318            lines.push(current_line);
319        }
320
321        if lines.is_empty() {
322            lines.push(String::new());
323        }
324
325        lines
326    }

328    /// Create gradient effect for important messages
329    pub fn render_gradient_message(&self, message: &str, colors: Vec<Color>) -> StyledString {
330        let mut result = StyledString::new();
331
332        if colors.len() < 2 {
333            result.append_plain(message);
334            return result;
335        }
336
337        let chars: Vec<char> = message.chars().collect();
338        let segment_size = chars.len() / (colors.len() - 1);
339
340        for (i, ch) in chars.iter().enumerate() {
341            let segment = i / segment_size.max(1);
342            let segment_progress = (i % segment_size.max(1)) as f32 / segment_size.max(1) as f32;
343
344            let color = if segment < colors.len() - 1 {
345                self.lerp_color(colors[segment], colors[segment + 1], segment_progress)
346            } else {
347                colors[colors.len() - 1]
348            };
349
350            result.append_styled(
351                ch.to_string(),
352                ColorStyle::new(color, Color::Rgb(30, 30, 30)),
353            );
354        }
355
356        result
357    }

359    fn lerp_color(&self, start: Color, end: Color, t: f32) -> Color {
360        match (start, end) {
361            (Color::Rgb(r1, g1, b1), Color::Rgb(r2, g2, b2)) => Color::Rgb(
362                (r1 as f32 + (r2 as f32 - r1 as f32) * t) as u8,
363                (g1 as f32 + (g2 as f32 - g1 as f32) * t) as u8,
364                (b1 as f32 + (b2 as f32 - b1 as f32) * t) as u8,
365            ),
366            _ => end,
367        }
368    }

390
391        // Code lines
392        for (i, line) in code.lines().enumerate() {
393            let line_num = if self.line_numbers {
394                format!("{:3} ‚îÇ ", i + 1)
395            } else {
396                String::new()
397            };
398
399            let highlight = if Some(i) == self.highlight_line {
400                "‚ñ∂ "
401            } else {
402                "  "
403            };
404
405            lines.push(format!("‚ïë{}{}{}", highlight, line_num, line));
406        }
407
408        // Footer
409        lines.push("‚ïö‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïê‚ïù".to_string());

411        lines
