1//! Non-blocking toast notification system
2//!
3//! Provides temporary notifications that appear at the top/bottom of the screen
4//! and automatically disappear after a timeout, without blocking user interaction.
5

14/// Type of toast notification
15#[derive(Debug, Clone, Copy, PartialEq)]
16pub enum ToastType {
17    Success,
18    Error,
19    Warning,
20    Info,
21}
22
23impl ToastType {
24    /// Get the color style for this toast type
25    fn color_style(&self) -> ColorStyle {
26        match self {
27            ToastType::Success => ColorStyle::new(
28                Color::Light(BaseColor::White),
29                Color::Dark(BaseColor::Green),
30            ),
31            ToastType::Error => ColorStyle::new(
32                Color::Light(BaseColor::White),
33                Color::Dark(BaseColor::Red),
34            ),
35            ToastType::Warning => ColorStyle::new(
36                Color::Dark(BaseColor::Black),
37                Color::Light(BaseColor::Yellow),
38            ),
39            ToastType::Info => ColorStyle::new(
40                Color::Light(BaseColor::White),
41                Color::Dark(BaseColor::Blue),
42            ),
43        }
44    }
45
46    /// Get the icon for this toast type
47    fn icon(&self) -> &'static str {
48        match self {
49            ToastType::Success => "✅",
50            ToastType::Error => "❌",
51            ToastType::Warning => "⚠️",
52            ToastType::Info => "ℹ️",
53        }
54    }
55}
56
57/// Toast notification
58#[derive(Clone)]
59pub struct Toast {
60    message: String,
61    toast_type: ToastType,
62    duration: Duration,
63    created_at: Instant,
64}
65
66impl Toast {
67    /// Create a new toast notification
68    pub fn new(message: impl Into<String>, toast_type: ToastType, duration: Duration) -> Self {
69        Toast {
70            message: message.into(),
71            toast_type,
72            duration,
73            created_at: Instant::now(),
74        }
75    }
76
77    /// Check if the toast has expired
78    pub fn is_expired(&self) -> bool {
79        self.created_at.elapsed() > self.duration
80    }
81
82    /// Create the view for this toast
83    pub fn create_view(&self) -> impl View {
84        let content = format!("{} {}", self.toast_type.icon(), self.message);
85        let mut text_view = TextView::new(content);
86        text_view.set_style(self.toast_type.color_style());
87
88        // Wrap in a panel for better visibility
89        Panel::new(text_view)
90            .title("")
91    }
92}
93
94/// Toast notification manager
95pub struct ToastManager {
96    toasts: Arc<Mutex<Vec<Toast>>>,
97}
98
99impl ToastManager {
100    /// Create a new toast manager
101    pub fn new() -> Self {
102        ToastManager {
103            toasts: Arc::new(Mutex::new(Vec::new())),
104        }
105    }
106
107    /// Add a new toast notification
108    pub fn add_toast(&self, message: impl Into<String>, toast_type: ToastType, duration: Duration) {
109        if let Ok(mut toasts) = self.toasts.lock() {
110            toasts.push(Toast::new(message, toast_type, duration));
111        }
112    }
113
114    /// Remove expired toasts
115    pub fn cleanup_expired(&self) {
116        if let Ok(mut toasts) = self.toasts.lock() {
117            toasts.retain(|toast| !toast.is_expired());
118        }
119    }
120
121    /// Get active toasts
122    pub fn get_active_toasts(&self) -> Vec<Toast> {
123        if let Ok(toasts) = self.toasts.lock() {
124            toasts.clone()
125        } else {
126            Vec::new()
127        }
128    }
129
130    /// Create a view showing all active toasts
131    pub fn create_toast_layer(&self) -> impl View {
132        self.cleanup_expired();
133
134        let mut layout = LinearLayout::vertical();
135
136        for toast in self.get_active_toasts() {
137            layout.add_child(toast.create_view());
138        }
139
140        // Position at top-right of screen
141        layout.fixed_width(40)
142    }
143}

167/// Show a toast notification
168pub fn show_toast(s: &mut Cursive, message: impl Into<String>, toast_type: ToastType, duration: Duration) {
169    // Create the toast view
170    let toast = Toast::new(message, toast_type, duration);
171    let toast_view = toast.create_view();
172
173    // Create a unique layer name
174    let layer_name = format!("toast_{}", uuid::Uuid::new_v4());
175
176    // Store the toast expiration time in user data
177    // This approach avoids spawning threads per toast
178    let manager = ToastManager::new();
179    manager.add_toast(toast.message.clone(), toast.toast_type, duration);
180
181    // Add as a dialog which is more suitable for temporary notifications
182    use cursive::views::Dialog;
183    let dialog = Dialog::around(toast_view)
184        .title("Notification")
185        .with_name(layer_name.clone());
186
187    s.add_layer(dialog);
188
189    // Use callback sink to schedule removal without spawning a thread
190    let cb_sink = s.cb_sink().clone();
191    let layer_id = layer_name.clone();
192
193    // Schedule removal using a timer callback instead of a thread
194    std::thread::spawn(move || {
195        std::thread::sleep(duration);
196        // Send callback to remove the layer
197        let _ = cb_sink.send(Box::new(move |s| {
198            // Safely pop the topmost layer (which should be our toast)
199            s.pop_layer();
200        }));
201    });
202}
