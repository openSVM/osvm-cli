8/// Format a cursive Event as a readable diagnostic string
9pub fn format_event(event: &Event) -> String {
10    match event {
11        Event::Char(c) => {
12            format!("Char('{}') - ASCII {}", c.escape_default(), *c as u32)
13        }
14        Event::CtrlChar(c) => {
15            format!("CtrlChar('{}') - ASCII {} - Ctrl+{}",
16                c.escape_default(),
17                *c as u32,
18                match *c {
19                    'a' => "A",
20                    'b' => "B",
21                    'c' => "C",
22                    'd' => "D",
23                    'e' => "E",
24                    'f' => "F",
25                    'g' => "G",
26                    'h' => "H",
27                    'i' => "I",
28                    'j' => "J",
29                    'k' => "K",
30                    'l' => "L",
31                    'm' => "M (Enter?)",
32                    'n' => "N",
33                    'o' => "O",
34                    'p' => "P",
35                    'q' => "Q",
36                    'r' => "R",
37                    's' => "S",
38                    't' => "T",
39                    'u' => "U",
40                    'v' => "V",
41                    'w' => "W",
42                    'x' => "X",
43                    'y' => "Y",
44                    'z' => "Z",
45                    '[' => "[ (Escape)",
46                    '\\' => "\\",
47                    ']' => "]",
48                    '^' => "^",
49                    '_' => "_ (Underscore)",
50                    _ => "Unknown"
51                }
52            )
53        }
54        Event::AltChar(c) => {
55            format!("AltChar('{}') - Alt+{}", c.escape_default(), c)
56        }
57        Event::Key(k) => {
58            format!("Key::{:?}", k)
59        }
60        Event::Shift(k) => {
61            format!("Shift({:?})", k)
62        }
63        Event::Alt(k) => {
64            format!("Alt({:?})", k)
65        }
66        Event::CtrlShift(k) => {
67            format!("CtrlShift({:?})", k)
68        }
69        Event::AltShift(k) => {
70            format!("AltShift({:?})", k)
71        }
72        Event::WindowResize => "WindowResize".to_string(),
73        Event::Refresh => "Refresh".to_string(),
74        _ => format!("{:?}", event),
75    }
76}

78/// Create a diagnostic overlay showing recent key presses
79pub fn show_key_diagnostics(siv: &mut Cursive) {
80    use cursive::views::{Dialog, TextView};
81
82    let diagnostics_text = vec![
83        "KEY DIAGNOSTICS - PRESS KEYS TO TEST".to_string(),
84        "".to_string(),
85        "This panel helps identify which cursive event codes correspond to".to_string(),
86        "physical key presses. Check the application logs for event details.".to_string(),
87        "".to_string(),
88        "Try pressing the keys listed below and watch the logs:".to_string(),
89        "  • Ctrl+Enter (what we're trying to fix)".to_string(),
90        "  • Ctrl+M".to_string(),
91        "  • Ctrl+J".to_string(),
92        "  • Regular Enter".to_string(),
93        "".to_string(),
94        "Watch the logs (run with RUST_LOG=info osvm chat --advanced)".to_string(),
95        "to see which event code each key press generates.".to_string(),
96        "".to_string(),
97        "Close this dialog (Escape or click Close) and press your test keys.".to_string(),
98    ].join("\n");
99
100    siv.add_layer(
101        Dialog::around(TextView::new(diagnostics_text))
102            .title("Key Event Diagnostics")
103            .button("Close", |s| { s.pop_layer(); })
104    );
105}

107/// Install a global key event logger for diagnostics
108pub fn install_key_logger(siv: &mut Cursive) {
109    // Log key presses by adding callbacks for various combinations
110
111    log::info!("╔════════════════════════════════════════════════════════════╗");
112    log::info!("║         KEY EVENT DIAGNOSTICS INSTALLED                   ║");
113    log::info!("║  Open diagnostics panel with: Ctrl+\\                      ║");
114    log::info!("║  Then press keys you want to test (e.g., Ctrl+Enter)      ║");
115    log::info!("║  Watch this log for captured events                       ║");
116    log::info!("╚════════════════════════════════════════════════════════════╝");
117
118    // Monitor common Ctrl combinations
119    siv.add_global_callback(Event::CtrlChar('m'), |_s| {
120        log::info!("🔍 EVENT: CtrlChar('m') = Ctrl+M detected");
121    });
122
123    siv.add_global_callback(Event::CtrlChar('j'), |_s| {
124        log::info!("🔍 EVENT: CtrlChar('j') = Ctrl+J detected");
125    });
126
127    siv.add_global_callback(Event::CtrlChar('l'), |_s| {
128        log::info!("🔍 EVENT: CtrlChar('l') = Ctrl+L detected");
129    });
130
131    siv.add_global_callback(Event::CtrlChar('n'), |_s| {
132        log::info!("🔍 EVENT: CtrlChar('n') = Ctrl+N detected");
133    });
134
135    siv.add_global_callback(Event::Key(Key::Enter), |_s| {
136        log::info!("🔍 EVENT: Key::Enter = Plain Enter detected");
137    });
138
139    log::info!("Key logger ready. Go to diagnostics panel (Ctrl+\\) and test your keys.");
140}
