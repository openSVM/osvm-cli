8/// Format a cursive Event as a readable diagnostic string
9pub fn format_event(event: &Event) -> String {
10    match event {
11        Event::Char(c) => {
12            format!("Char('{}') - ASCII {}", c.escape_default(), *c as u32)
13        }
14        Event::CtrlChar(c) => {
15            format!(
16                "CtrlChar('{}') - ASCII {} - Ctrl+{}",
17                c.escape_default(),
18                *c as u32,
19                match *c {
20                    'a' => "A",
21                    'b' => "B",
22                    'c' => "C",
23                    'd' => "D",
24                    'e' => "E",
25                    'f' => "F",
26                    'g' => "G",
27                    'h' => "H",
28                    'i' => "I",
29                    'j' => "J",
30                    'k' => "K",
31                    'l' => "L",
32                    'm' => "M (Enter?)",
33                    'n' => "N",
34                    'o' => "O",
35                    'p' => "P",
36                    'q' => "Q",
37                    'r' => "R",
38                    's' => "S",
39                    't' => "T",
40                    'u' => "U",
41                    'v' => "V",
42                    'w' => "W",
43                    'x' => "X",
44                    'y' => "Y",
45                    'z' => "Z",
46                    '[' => "[ (Escape)",
47                    '\\' => "\\",
48                    ']' => "]",
49                    '^' => "^",
50                    '_' => "_ (Underscore)",
51                    _ => "Unknown",
52                }
53            )
54        }
55        Event::AltChar(c) => {
56            format!("AltChar('{}') - Alt+{}", c.escape_default(), c)
57        }
58        Event::Key(k) => {
59            format!("Key::{:?}", k)
60        }
61        Event::Shift(k) => {
62            format!("Shift({:?})", k)
63        }
64        Event::Alt(k) => {
65            format!("Alt({:?})", k)
66        }
67        Event::CtrlShift(k) => {
68            format!("CtrlShift({:?})", k)
69        }
70        Event::AltShift(k) => {
71            format!("AltShift({:?})", k)
72        }
73        Event::WindowResize => "WindowResize".to_string(),
74        Event::Refresh => "Refresh".to_string(),
75        _ => format!("{:?}", event),
76    }
77}

79/// Create a diagnostic overlay showing recent key presses
80pub fn show_key_diagnostics(siv: &mut Cursive) {
81    use cursive::views::{Dialog, TextView};
82
83    let diagnostics_text = vec![
84        "KEY DIAGNOSTICS - PRESS KEYS TO TEST".to_string(),
85        "".to_string(),
86        "This panel helps identify which cursive event codes correspond to".to_string(),
87        "physical key presses. Check the application logs for event details.".to_string(),
88        "".to_string(),
89        "Try pressing the keys listed below and watch the logs:".to_string(),
90        "  â€¢ Ctrl+Enter (what we're trying to fix)".to_string(),
91        "  â€¢ Ctrl+M".to_string(),
92        "  â€¢ Ctrl+J".to_string(),
93        "  â€¢ Regular Enter".to_string(),
94        "".to_string(),
95        "Watch the logs (run with RUST_LOG=info osvm chat --advanced)".to_string(),
96        "to see which event code each key press generates.".to_string(),
97        "".to_string(),
98        "Close this dialog (Escape or click Close) and press your test keys.".to_string(),
99    ]
100    .join("\n");
101
102    siv.add_layer(
103        Dialog::around(TextView::new(diagnostics_text))
104            .title("Key Event Diagnostics")
105            .button("Close", |s| {
106                s.pop_layer();
107            }),
108    );
109}
