7/// Modern color palette for the advanced chat UI

10impl ModernTheme {
11    /// Get list of available theme names
12    pub fn available_themes() -> Vec<&'static str> {
13        vec!["dark", "light", "high_contrast"]
14    }
15
16    /// Get theme by name
17    pub fn by_name(name: &str) -> Theme {
18        match name {
19            "light" => Self::light(),
20            "high_contrast" => Self::high_contrast(),
21            _ => Self::dark(), // Default to dark
22        }
23    }

140/// Styled text builders for consistent formatting (VS Code colors)
141pub struct StyledText;
142
143impl StyledText {
144    /// Create gradient-like text (VS Code style)
145    pub fn gradient(text: &str) -> StyledString {
146        let mut styled = StyledString::new();
147        let colors = vec![
148            Color::Rgb(86, 156, 214), // #569cd6 - VS Code keyword blue
149            Color::Rgb(0, 122, 204),  // #007acc - VS Code bright blue
150            Color::Rgb(78, 201, 176), // #4ec9b0 - VS Code cyan
151        ];
152
153        for (i, ch) in text.chars().enumerate() {
154            let color_idx = i % colors.len();
155            styled.append_styled(
156                ch.to_string(),
157                ColorStyle::new(colors[color_idx], Color::Rgb(30, 30, 30)),
158            );
159        }
160
161        styled
162    }
163
164    /// Create success text (VS Code green)
165    pub fn success(text: &str) -> StyledString {
166        StyledString::styled(
167            text,
168            ColorStyle::new(Color::Rgb(106, 153, 85), Color::Rgb(30, 30, 30)), // #6a9955
169        )

213/// Progress bar builder
214pub struct ProgressBar;
215
216impl ProgressBar {
217    /// Create a styled progress bar string
218    pub fn render(progress: f32, width: usize) -> String {
219        let filled = (progress * width as f32) as usize;
220        let empty = width.saturating_sub(filled);
221
222        let filled_char = '█';
223        let empty_char = '░';
224
225        format!(
226            "{}{}{}",
227            filled_char.to_string().repeat(filled),
228            empty_char.to_string().repeat(empty),
229            format!(" {:.0}%", progress * 100.0)
230        )
231    }
232
233    /// Create a segmented progress bar
234    pub fn render_segmented(current: usize, total: usize, width: usize) -> String {
235        if total == 0 {
236            return "░".repeat(width);
237        }
238
239        let segment_width = width / total.max(1);
240        let mut bar = String::new();
241
242        for i in 0..total {
243            if i < current {
244                bar.push_str(&"█".repeat(segment_width));
245            } else {
246                bar.push_str(&"░".repeat(segment_width));
247            }
248        }
249
250        format!("{} [{}/{}]", bar, current, total)
251    }
252}
