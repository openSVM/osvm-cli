20    /// Apply theme styling to a message based on its type
21    fn format_message_with_theme(&self, message: &ChatMessage) -> String {
22        match message {
23            ChatMessage::User(text) => {
24                let sanitized = self.sanitize_text(text);
25                format!(
26                    "You: {}\n   [R]etry [C]opy [D]elete   (Alt+R/C/D)\n\n",
27                    sanitized
28                )
29            }
30            ChatMessage::Agent(text) => {
31                let rendered = self.render_markdown(text);
32                let sanitized = self.sanitize_text(&rendered);
33                format!(
34                    "Agent:\n{}\n   [F]ork [C]opy [R]etry [D]elete   (Alt+F/C/R/D)\n\n",
35                    sanitized
36                )
37            }
38            ChatMessage::System(text) => {
39                let sanitized = self.sanitize_text(text);
40                format!("System: {}\n\n", sanitized)
41            }
42            ChatMessage::Error(text) => {
43                let sanitized = self.sanitize_text(text);
44                format!("Error: {}\n\n", sanitized)
45            }
46            ChatMessage::ToolCall {
47                tool_name,
48                description,
49                args,
50                execution_id,
51            } => {
52                let sanitized_tool_name = self.sanitize_text(tool_name);
53                let sanitized_description = self.sanitize_text(description);
54
55                let mut result = format!(
56                    "Calling tool: {} - {}\n",
57                    sanitized_tool_name, sanitized_description
58                );
59
60                if let Some(args) = args {
61                    let sanitized_args = self.sanitize_json(args);
62                    result.push_str(&format!("   Args: {}\n", sanitized_args));
63                }
64                result.push_str(&format!("   ID: {}\n\n", execution_id));
65                result
66            }
67            ChatMessage::ToolResult {
68                tool_name,
69                result,
70                execution_id,
71            } => {
72                let sanitized_tool_name = self.sanitize_text(tool_name);
73                let sanitized_result = self.sanitize_json(result);
74                format!(
75                    "Tool {} result (ID: {}):\n{}\n\n",
76                    sanitized_tool_name, execution_id, sanitized_result
77                )
78            }
79            ChatMessage::AgentThinking(text) => {
80                let sanitized = self.sanitize_text(text);
81                format!("Agent (thinking): {}\n\n", sanitized)
82            }
83            ChatMessage::AgentPlan(plan) => {
84                let mut result = String::from("Agent Plan:\n");
85                for (i, step) in plan.iter().enumerate() {
86                    let step_text = format!("{}: {}", step.tool_name, step.reason);
87                    let sanitized = self.sanitize_text(&step_text);
88                    result.push_str(&format!("  {}. {}\n", i + 1, sanitized));
89                }
90                result.push_str("\n");
91                result
92            }
93            ChatMessage::Processing {
94                message,
95                spinner_index,
96            } => {
97                let spinner_chars = vec!["|", "/", "-", "\\", "|", "/", "-", "\\"];
98                let spinner_char = spinner_chars[spinner_index % spinner_chars.len()];
99                let sanitized = self.sanitize_text(message);
100                format!("{} {}\n\n", spinner_char, sanitized)
101            }
102        }
103    }

143    pub fn update_chat_list(&self, siv: &mut Cursive) {
144        let session_names = self.state.get_session_names();
145        let active_id = self.state.active_session_id.read().ok().and_then(|id| *id);
146
147        if let Some(mut chat_list) = siv.find_name::<ListView>("chat_list") {
148            // Smart in-place updates instead of clear() + rebuild
149            // Only rebuild if session count changed, otherwise just update existing buttons
150            let current_count = chat_list.len();
151            let new_count = session_names.len();
152
153            if current_count != new_count {
154                // Session count changed - need full rebuild
155                chat_list.clear();
156
157                for (id, name, agent_state) in session_names.iter() {
158                    let status_icon = match agent_state {
159                        AgentState::Idle => Icons::IDLE,
160                        AgentState::Thinking => Icons::THINKING,
161                        AgentState::Planning => Icons::PLANNING,
162                        AgentState::ExecutingTool(_) => Icons::EXECUTING,
163                        AgentState::Waiting => Icons::WAITING,
164                        AgentState::Paused => Icons::PAUSED,
165                        AgentState::Error(_) => Icons::ERROR,
166                    };
167
168                    let text_name = format!("");
169                    let display_name = format!("{} {}", status_icon, name);
170                    let is_active = Some(*id) == active_id;
171                    let button_text = if is_active {
172                        format!("{} {}", Icons::ARROW_RIGHT, display_name)
173                    } else {
174                        format!("  {}", display_name)
175                    };
176
177                    let session_id = *id;
178                    let mut button = Button::new(button_text.clone(), move |siv| {
179                        handle_chat_selection(siv, session_id);
180                    });
181
182                    chat_list.add_child(&text_name, button);
183                }
184            } else {
185                // Same session count - for now just do a full rebuild
186                // TODO: Implement true in-place updates when cursive API supports it
187                chat_list.clear();
188
189                for (id, name, agent_state) in session_names.iter() {
190                    let status_icon = match agent_state {
191                        AgentState::Idle => Icons::IDLE,
192                        AgentState::Thinking => Icons::THINKING,
193                        AgentState::Planning => Icons::PLANNING,
194                        AgentState::ExecutingTool(_) => Icons::EXECUTING,
195                        AgentState::Waiting => Icons::WAITING,
196                        AgentState::Paused => Icons::PAUSED,
197                        AgentState::Error(_) => Icons::ERROR,
198                    };
199
200                    let display_name = format!("{} {}", status_icon, name);
201                    let is_active = Some(*id) == active_id;
202                    let button_text = if is_active {
203                        format!("{} {}", Icons::ARROW_RIGHT, display_name)
204                    } else {
205                        format!("  {}", display_name)
206                    };
207
208                    let session_id = *id;
209                    let mut button = Button::new(button_text.clone(), move |siv| {
210                        handle_chat_selection(siv, session_id);
211                    });
212
213                    chat_list.add_child(&display_name, button);
214                }
215            }
216        }

219    pub fn update_mcp_tools_list(&self, siv: &mut Cursive) {
220        let tools = self.state.available_tools.read().unwrap();
221
222        if let Some(mut mcp_tools_view) =
223            siv.find_name::<SelectView<(String, String)>>("mcp_tools_list")
224        {
225            mcp_tools_view.clear();
226
227            if tools.is_empty() {
228                mcp_tools_view.add_item("No tools available", ("".to_string(), "".to_string()));
229            } else {
230                // Add each tool individually
231                for (server_id, tool_list) in tools.iter() {
232                    // Add server header with icon
233                    mcp_tools_view.add_item(
234                        format!(
235                            "{} {} ({} tools)",
236                            Icons::FOLDER,
237                            server_id,
238                            tool_list.len()
239                        ),
240                        (server_id.clone(), "".to_string()),
241                    );
242
243                    // Add individual tools with icons
244                    for tool in tool_list.iter() {
245                        let display_name = format!("  {} {}", Icons::TOOL, tool.name);
246                        mcp_tools_view
247                            .add_item(display_name, (server_id.clone(), tool.name.clone()));
248                    }
249                }
250            }
251        }
252    }

254    pub fn update_chat_display(&self, siv: &mut Cursive) {
255        if let Some(session) = self.state.get_active_session() {
256            let mut display_text = String::new();
257
258            for message in &session.messages {
259                display_text.push_str(&self.format_message_with_theme(message));
260            }
261
262            // Stable content update - only set if content actually changed
263            if let Some(mut chat_display) = siv.find_name::<TextView>("chat_display") {
264                let content_ref = chat_display.get_content();
265                let current_content = content_ref.source().to_string();
266                if current_content != display_text {
267                    chat_display.set_content(display_text);
268                }
269            }
270        } else {
271            let no_session_text =
272                "No active session. Use Tab to navigate to the session list and select a session.";
273            if let Some(mut chat_display) = siv.find_name::<TextView>("chat_display") {
274                let content_ref = chat_display.get_content();
275                let current_content = content_ref.source().to_string();
276                if current_content != no_session_text {
277                    chat_display.set_content(no_session_text);
278                }
279            }
280        }
281    }

283    pub fn update_agent_status(&self, siv: &mut Cursive) {
284        use chrono::{DateTime, Utc};
285        use std::sync::atomic::Ordering;
286
287        let spinner_index = self.state.spinner_state.load(Ordering::Relaxed);
288        let spinner_chars = Spinners::DOTS_WAVE;
289        let spinner_char = spinner_chars[spinner_index % spinner_chars.len()];
290
291        let timestamp = Utc::now().format("%H:%M:%S");
292
293        let status_text = if let Some(session) = self.state.get_active_session() {
294            match session.agent_state {
295                AgentState::Idle => format!(
296                    "{} Agent: Idle | Ready for tasks | {}",
297                    Icons::IDLE,
298                    timestamp
299                ),
300                AgentState::Thinking => format!(
301                    "{} Agent: Thinking... | Analyzing request | {}",
302                    spinner_char, timestamp
303                ),
304                AgentState::Planning => format!(
305                    "{} Agent: Planning... | Creating execution plan | {}",
306                    spinner_char, timestamp
307                ),
308                AgentState::ExecutingTool(ref tool_name) => format!(
309                    "{} Agent: Executing {} | Processing... | {}",
310                    spinner_char, tool_name, timestamp
311                ),
312                AgentState::Waiting => {
313                    format!("Agent: Waiting | Awaiting response | {}", timestamp)
314                }
315                AgentState::Paused => {
316                    format!("Agent: Paused | Operations suspended | {}", timestamp)
317                }
318                AgentState::Error(ref err) => format!("Agent: Error | {} | {}", err, timestamp),
319            }
320        } else {
321            format!(
322                "Agent: No active session | Select a chat session | {}",
323                timestamp
324            )
325        };
326
327        if let Some(mut status_display) = siv.find_name::<TextView>("agent_status") {
328            status_display.set_content(status_text);
329        }
330    }

332    pub fn update_status_bar(&self, siv: &mut Cursive) {
333        use std::time::{Duration, Instant};
334
335        // Check if we need to refresh the cached status (throttle to avoid blocking UI thread too often)
336        const REFRESH_INTERVAL_SECS: u64 = 2; // Refresh every 2 seconds
337
338        let should_refresh = {
339            if let Ok(last_update) = self.state.last_status_update.read() {
340                last_update
341                    .map(|instant| instant.elapsed() > Duration::from_secs(REFRESH_INTERVAL_SECS))
342                    .unwrap_or(true)
343            } else {
344                true
345            }
346        };
347
348        // Only fetch new status if the cache is stale
349        if should_refresh {
350            // Spawn status fetch in background to avoid blocking UI thread
351            // For now, we still need to block briefly, but we do it less frequently
352            let status_text = tokio::task::block_in_place(|| {
353                tokio::runtime::Handle::current().block_on(async {
354                    let status =
355                        crate::utils::agent_chat::system_status_bar::get_system_status().await;
356
357                    // Format status similar to the carousel format
358                    let mcp_info = if status.mcp_deployments.is_empty() {
359                        "no MCP".to_string()
360                    } else {
361                        status
362                            .mcp_deployments
363                            .iter()
364                            .map(|d| {
365                                let server_short = d.server_name.replace("-mcp", "");
366                                let microvm =
367                                    d.microvm_id.as_ref().map(|id| id.as_str()).unwrap_or("?");
368                                let extra_mounts = d.mounts.len().saturating_sub(1);
369                                let readonly_count = d.mounts.iter().filter(|m| m.readonly).count();
370
371                                if extra_mounts > 0 {
372                                    if readonly_count > 0 {
373                                        format!(
374                                            "{}({}:cfg+{}:{}ro)",
375                                            server_short, microvm, extra_mounts, readonly_count
376                                        )
377                                    } else {
378                                        format!(
379                                            "{}({}:cfg+{})",
380                                            server_short, microvm, extra_mounts
381                                        )
382                                    }
383                                } else {
384                                    format!("{}({}:cfg)", server_short, microvm)
385                                }
386                            })
387                            .collect::<Vec<_>>()
388                            .join(" ")
389                    };
390
391                    format!(
392                        "OSVM: {}μVMs {}uK • {} • net:{} • up:{}s",
393                        status.microvms_running,
394                        status.unikernels_running,
395                        mcp_info,
396                        status.current_network,
397                        status.uptime_seconds % 3600
398                    )
399                })
400            });
401
402            // Update the cache
403            if let Ok(mut cached_status) = self.state.cached_status_text.write() {
404                *cached_status = Some(status_text.clone());
405            }
406            if let Ok(mut last_update) = self.state.last_status_update.write() {
407                *last_update = Some(Instant::now());
408            }
409
410            // Update the view
411            if let Some(mut status_bar_view) = siv.find_name::<TextView>("system_status_bar") {
412                let content_ref = status_bar_view.get_content();
413                let current_content = content_ref.source().to_string();
414                if current_content != status_text {
415                    status_bar_view.set_content(status_text);
416                }
417            }
418        } else {
419            // Use cached status to avoid blocking
420            if let Ok(cached_status) = self.state.cached_status_text.read() {
421                if let Some(status_text) = cached_status.as_ref() {
422                    if let Some(mut status_bar_view) =
423                        siv.find_name::<TextView>("system_status_bar")
424                    {
425                        let content_ref = status_bar_view.get_content();
426                        let current_content = content_ref.source().to_string();
427                        if &current_content != status_text {
428                            status_bar_view.set_content(status_text.clone());
429                        }
430                    }
431                }
432            }
433        }
434    }

436    pub fn update_suggestions_display(&self, siv: &mut Cursive) {
437        let suggestions_visible = self
438            .state
439            .suggestions_visible
440            .read()
441            .map(|v| *v)
442            .unwrap_or(false);
443
444        if let Some(mut container) = siv.find_name::<LinearLayout>("suggestions_container") {
445            container.clear();
446
447            if suggestions_visible {
448                let suggestions = self
449                    .state
450                    .current_suggestions
451                    .read()
452                    .map(|s| s.clone())
453                    .unwrap_or_default();
454
455                if !suggestions.is_empty() {
456                    for (i, sugg) in suggestions.iter().enumerate() {
457                        let mut styled = StyledString::new();
458
459                        // Create blue background with white text style
460                        let blue_style = ColorStyle::new(
461                            Color::Light(BaseColor::White),
462                            Color::Dark(BaseColor::Blue),
463                        );
464
465                        // Format: "1. Suggestion text" with full blue background
466                        let display_text = format!(" {}. {} ", i + 1, sugg);
467                        styled.append_styled(display_text, blue_style);
468
469                        // Add as a styled TextView
470                        let suggestion_view = TextView::new(styled);
471                        container.add_child(suggestion_view);
472                    }
473
474                    // Add a small spacer after suggestions
475                    container.add_child(cursive::views::DummyView.fixed_height(1));
476                }
477            }
478        }
