25    pub fn update_chat_list(&self, siv: &mut Cursive) {
26        let session_names = self.state.get_session_names();
27        let active_id = self.state.active_session_id.read().ok().and_then(|id| *id);
28
29        if let Some(mut chat_list) = siv.find_name::<ListView>("chat_list") {
30            // Smart in-place updates instead of clear() + rebuild
31            // Only rebuild if session count changed, otherwise just update existing buttons
32            let current_count = chat_list.len();
33            let new_count = session_names.len();
34
35            if current_count != new_count {
36                // Session count changed - need full rebuild
37                chat_list.clear();
38
39                for (id, name, agent_state) in session_names.iter() {
40                    let status_icon = match agent_state {
41                        AgentState::Idle => "●",
42                        AgentState::Thinking => "◐",
43                        AgentState::Planning => "◑",
44                        AgentState::ExecutingTool(_) => "◒",
45                        AgentState::Waiting => "◯",
46                        AgentState::Paused => "⏸",
47                        AgentState::Error(_) => "⚠",
48                    };
49
50                    let display_name = format!("{} {}", status_icon, name);
51                    let is_active = Some(*id) == active_id;
52                    let button_text = if is_active {
53                        format!("► {}", display_name)
54                    } else {
55                        format!("  {}", display_name)
56                    };
57
58                    let session_id = *id;
59                    let mut button = Button::new(button_text.clone(), move |siv| {
60                        handle_chat_selection(siv, session_id);
61                    });
62
63                    chat_list.add_child(&display_name, button);
64                }
65            } else {
66                // Same session count - for now just do a full rebuild
67                // TODO: Implement true in-place updates when cursive API supports it
68                chat_list.clear();
69
70                for (id, name, agent_state) in session_names.iter() {
71                    let status_icon = match agent_state {
72                        AgentState::Idle => "●",
73                        AgentState::Thinking => "◐",
74                        AgentState::Planning => "◑",
75                        AgentState::ExecutingTool(_) => "◒",
76                        AgentState::Waiting => "◯",
77                        AgentState::Paused => "⏸",
78                        AgentState::Error(_) => "⚠",
79                    };
80
81                    let display_name = format!("{} {}", status_icon, name);
82                    let is_active = Some(*id) == active_id;
83                    let button_text = if is_active {
84                        format!("► {}", display_name)
85                    } else {
86                        format!("  {}", display_name)
87                    };
88
89                    let session_id = *id;
90                    let mut button = Button::new(button_text.clone(), move |siv| {
91                        handle_chat_selection(siv, session_id);
92                    });
93
94                    chat_list.add_child(&display_name, button);
95                }
96            }
97        }
98    }

100    pub fn update_mcp_tools_list(&self, siv: &mut Cursive) {
101        let tools = self.state.available_tools.read().unwrap();
102        let mut content = String::new();
103        if tools.is_empty() {
104            content.push_str("No tools available.");
105        } else {
106            for (server_id, tool_list) in tools.iter() {
107                content.push_str(&format!("- {} ({} tools)\n", server_id, tool_list.len()));
108            }
109        }
110
111        if let Some(mut mcp_tools_view) = siv.find_name::<TextView>("mcp_tools_list") {
112            mcp_tools_view.set_content(content);
113        }
114    }

116    pub fn update_chat_display(&self, siv: &mut Cursive) {
117        if let Some(session) = self.state.get_active_session() {
118            let mut display_text = String::new();
119
120            for message in &session.messages {
121                match message {
122                    ChatMessage::User(text) => {
123                        let sanitized = self.sanitize_text(text);
124                        display_text.push_str(&format!("You: {}\n", sanitized));
125                        display_text.push_str("   [R]etry [C]opy [D]elete   (Alt+R/C/D)\n\n");
126                    }
127                    ChatMessage::Agent(text) => {
128                        let rendered = self.render_markdown(text);
129                        let sanitized = self.sanitize_text(&rendered);
130                        display_text.push_str(&format!("Agent:\n{}", sanitized));
131                        display_text
132                            .push_str("\n   [F]ork [C]opy [R]etry [D]elete   (Alt+F/C/R/D)\n\n");
133                    }
134                    ChatMessage::System(text) => {
135                        let sanitized = self.sanitize_text(text);
136                        display_text.push_str(&format!("System: {}\n\n", sanitized));
137                    }
138                    ChatMessage::ToolCall {
139                        tool_name,
140                        description,
141                        args,
142                        execution_id,
143                    } => {
144                        let sanitized_tool_name = self.sanitize_text(tool_name);
145                        let sanitized_description = self.sanitize_text(description);
146                        display_text.push_str(&format!(
147                            "Calling tool: {} - {}\n",
148                            sanitized_tool_name, sanitized_description
149                        ));
150                        if let Some(args) = args {
151                            let sanitized_args = self.sanitize_json(args);
152                            display_text.push_str(&format!("   Args: {}\n", sanitized_args));
153                        }
154                        display_text.push_str(&format!("   ID: {}\n\n", execution_id));
155                    }
156                    ChatMessage::ToolResult {
157                        tool_name,
158                        result,
159                        execution_id,
160                    } => {
161                        let sanitized_tool_name = self.sanitize_text(tool_name);
162                        display_text.push_str(&format!(
163                            "Tool {} result (ID: {}):\n",
164                            sanitized_tool_name, execution_id
165                        ));
166                        let sanitized_result = self.sanitize_json(result);
167                        display_text.push_str(&format!("{}\n\n", sanitized_result));
168                    }
169                    ChatMessage::Error(text) => {
170                        let sanitized = self.sanitize_text(text);
171                        display_text.push_str(&format!("Error: {}\n\n", sanitized));
172                    }
173                    ChatMessage::AgentThinking(text) => {
174                        let sanitized = self.sanitize_text(text);
175                        display_text.push_str(&format!("Agent (thinking): {}\n\n", sanitized));
176                    }
177                    ChatMessage::AgentPlan(plan) => {
178                        display_text.push_str("Agent Plan:\n");
179                        for (i, step) in plan.iter().enumerate() {
180                            let step_text = format!("{}: {}", step.tool_name, step.reason);
181                            let sanitized = self.sanitize_text(&step_text);
182                            display_text.push_str(&format!("  {}. {}\n", i + 1, sanitized));
183                        }
184                        display_text.push_str("\n");
185                    }
186                    ChatMessage::Processing {
187                        message,
188                        spinner_index,
189                    } => {
190                        let spinner_chars = vec!["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
191                        let spinner_char = spinner_chars[spinner_index % spinner_chars.len()];
192                        let sanitized = self.sanitize_text(message);
193                        display_text.push_str(&format!("{} {}\n\n", spinner_char, sanitized));
194                    }
195                }
196            }
197
198            // Stable content update - only set if content actually changed
199            if let Some(mut chat_display) = siv.find_name::<TextView>("chat_display") {
200                let content_ref = chat_display.get_content();
201                let current_content = content_ref.source().to_string();
202                if current_content != display_text {
203                    chat_display.set_content(display_text);
204                }
205            }
206        } else {
207            let no_session_text =
208                "No active session. Use Tab to navigate to the session list and select a session.";
209            if let Some(mut chat_display) = siv.find_name::<TextView>("chat_display") {
210                let content_ref = chat_display.get_content();
211                let current_content = content_ref.source().to_string();
212                if current_content != no_session_text {
213                    chat_display.set_content(no_session_text);
214                }
215            }
216        }
217    }

219    pub fn update_agent_status(&self, siv: &mut Cursive) {
220        use chrono::{DateTime, Utc};
221        use std::sync::atomic::Ordering;
222
223        let spinner_index = self.state.spinner_state.load(Ordering::Relaxed);
224        let spinner_chars = vec!["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
225        let spinner_char = spinner_chars[spinner_index % spinner_chars.len()];
226
227        let timestamp = Utc::now().format("%H:%M:%S");
228
229        let status_text = if let Some(session) = self.state.get_active_session() {
230            match session.agent_state {
231                AgentState::Idle => format!("🤖 Agent: Idle | Ready for tasks | {}", timestamp),
232                AgentState::Thinking => format!(
233                    "{} Agent: Thinking... | Analyzing request | {}",
234                    spinner_char, timestamp
235                ),
236                AgentState::Planning => format!(
237                    "{} Agent: Planning... | Creating execution plan | {}",
238                    spinner_char, timestamp
239                ),
240                AgentState::ExecutingTool(ref tool_name) => format!(
241                    "{} Agent: Executing {} | Processing... | {}",
242                    spinner_char, tool_name, timestamp
243                ),
244                AgentState::Waiting => {
245                    format!("⏳ Agent: Waiting | Awaiting response | {}", timestamp)
246                }
247                AgentState::Paused => {
248                    format!("⏸️ Agent: Paused | Operations suspended | {}", timestamp)
249                }
250                AgentState::Error(ref err) => format!("⚠️ Agent: Error | {} | {}", err, timestamp),
251            }
252        } else {
253            format!(
254                "❓ Agent: No active session | Select a chat session | {}",
255                timestamp
256            )
257        };
258
259        if let Some(mut status_display) = siv.find_name::<TextView>("agent_status") {
260            status_display.set_content(status_text);
261        }
262    }

264    pub fn update_suggestions_display(&self, siv: &mut Cursive) {
265        let suggestions_visible = self
266            .state
267            .suggestions_visible
268            .read()
269            .map(|v| *v)
270            .unwrap_or(false);
271
272        if let Some(mut container) = siv.find_name::<LinearLayout>("suggestions_container") {
273            container.clear();
274
275            if suggestions_visible {
276                let suggestions = self
277                    .state
278                    .current_suggestions
279                    .read()
280                    .map(|s| s.clone())
281                    .unwrap_or_default();
282
283                if !suggestions.is_empty() {
284                    for (i, sugg) in suggestions.iter().enumerate() {
285                        let mut styled = StyledString::new();
286
287                        // Create blue background with white text style
288                        let blue_style = ColorStyle::new(
289                            Color::Light(BaseColor::White),
290                            Color::Dark(BaseColor::Blue),
291                        );
292
293                        // Format: "1. Suggestion text" with full blue background
294                        let display_text = format!(" {}. {} ", i + 1, sugg);
295                        styled.append_styled(display_text, blue_style);
296
297                        // Add as a styled TextView
298                        let suggestion_view = TextView::new(styled);
299                        container.add_child(suggestion_view);
300                    }
301
302                    // Add a small spacer after suggestions
303                    container.add_child(cursive::views::DummyView.fixed_height(1));
304                }
305            }
306        }
307    }
