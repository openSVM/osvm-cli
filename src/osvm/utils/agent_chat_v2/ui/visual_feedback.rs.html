6/// Visual feedback levels
7pub enum FeedbackLevel {
8    Success,
9    Info,
10    Warning,
11    Error,
12    Processing,
13    Neutral,
14}
15
16impl FeedbackLevel {
17    /// Get color for feedback level
18    pub fn color(&self) -> Color {
19        match self {
20            FeedbackLevel::Success => Color::Rgb(34, 197, 94),    // Green
21            FeedbackLevel::Info => Color::Rgb(59, 130, 246),       // Blue
22            FeedbackLevel::Warning => Color::Rgb(251, 146, 60),    // Orange
23            FeedbackLevel::Error => Color::Rgb(239, 68, 68),       // Red
24            FeedbackLevel::Processing => Color::Rgb(139, 92, 246), // Purple
25            FeedbackLevel::Neutral => Color::Rgb(156, 163, 175),   // Gray
26        }
27    }
28
29    /// Get emoji icon for feedback level
30    pub fn icon(&self) -> &'static str {
31        match self {
32            FeedbackLevel::Success => "✓",
33            FeedbackLevel::Info => "ℹ",
34            FeedbackLevel::Warning => "⚠",
35            FeedbackLevel::Error => "✖",
36            FeedbackLevel::Processing => "⟳",
37            FeedbackLevel::Neutral => "•",
38        }
39    }
40
41    /// Get background color for feedback level
42    pub fn background(&self) -> Color {
43        match self {
44            FeedbackLevel::Success => Color::Rgb(20, 83, 45),    // Dark green
45            FeedbackLevel::Info => Color::Rgb(30, 58, 138),       // Dark blue
46            FeedbackLevel::Warning => Color::Rgb(120, 53, 15),    // Dark orange
47            FeedbackLevel::Error => Color::Rgb(127, 29, 29),      // Dark red
48            FeedbackLevel::Processing => Color::Rgb(67, 56, 202), // Dark purple
49            FeedbackLevel::Neutral => Color::Rgb(55, 65, 81),     // Dark gray
50        }
51    }
52}

54/// Create styled text with feedback level
55pub fn styled_feedback(text: &str, level: FeedbackLevel) -> StyledString {
56    let mut styled = StyledString::new();
57    let style = ColorStyle::new(level.color(), Color::Rgb(24, 24, 36));
58
59    // Add icon and text
60    styled.append_styled(format!("{} {}", level.icon(), text), style);
61
62    styled
63}
64
65/// Create styled text with background
66pub fn styled_badge(text: &str, level: FeedbackLevel) -> StyledString {
67    let mut styled = StyledString::new();
68    let style = ColorStyle::new(Color::Rgb(255, 255, 255), level.background());
69
70    // Add padding and text
71    styled.append_styled(format!(" {} {} ", level.icon(), text), style);
72
73    styled
74}

76/// Visual state indicators for different components
77pub struct VisualStates;
78
79impl VisualStates {
80    /// Agent state with full visual treatment
81    pub fn agent_state_badge(state_name: &str, is_active: bool) -> StyledString {
82        let level = if is_active {
83            FeedbackLevel::Processing
84        } else {
85            FeedbackLevel::Neutral
86        };
87
88        styled_badge(state_name, level)
89    }
90
91    /// Message state indicator
92    pub fn message_state(state: MessageState) -> StyledString {
93        match state {
94            MessageState::Sent => styled_feedback("Sent", FeedbackLevel::Success),
95            MessageState::Sending => styled_feedback("Sending...", FeedbackLevel::Processing),
96            MessageState::Failed => styled_feedback("Failed", FeedbackLevel::Error),
97            MessageState::Queued => styled_feedback("Queued", FeedbackLevel::Info),
98        }
99    }
100
101    /// Connection state indicator
102    pub fn connection_state(connected: bool) -> StyledString {
103        if connected {
104            styled_badge("Connected", FeedbackLevel::Success)
105        } else {
106            styled_badge("Disconnected", FeedbackLevel::Error)
107        }
108    }
109
110    /// Operation progress indicator
111    pub fn operation_progress(current: usize, total: usize) -> StyledString {
112        let percentage = if total > 0 {
113            (current as f64 / total as f64 * 100.0) as usize
114        } else {
115            0
116        };
117
118        styled_feedback(
119            &format!("{}% ({}/{})", percentage, current, total),
120            FeedbackLevel::Processing,
121        )
122    }
123
124    /// History navigation indicator
125    pub fn history_indicator(position: usize, total: usize) -> StyledString {
126        styled_badge(
127            &format!("History {}/{}", position + 1, total),
128            FeedbackLevel::Info,
129        )
130    }
131
132    /// Recording indicator (animated)
133    pub fn recording_indicator(is_recording: bool) -> StyledString {
134        if is_recording {
135            styled_badge("⏺ REC", FeedbackLevel::Error)
136        } else {
137            styled_badge("○ Ready", FeedbackLevel::Neutral)
138        }
139    }
140
141    /// Validation state indicator
142    pub fn validation_state(is_valid: bool, message: &str) -> StyledString {
143        if is_valid {
144            styled_feedback(&format!("✓ {}", message), FeedbackLevel::Success)
145        } else {
146            styled_feedback(&format!("✖ {}", message), FeedbackLevel::Error)
147        }
148    }
149
150    /// MCP server status

198/// Animated indicators
199pub struct AnimatedIndicators;
200
201impl AnimatedIndicators {
202    /// Spinner frames for different states
203    pub const SPINNER_DOTS: &'static [&'static str] = &["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
204    pub const SPINNER_CIRCLE: &'static [&'static str] = &["◐", "◓", "◑", "◒"];
205    pub const SPINNER_PULSE: &'static [&'static str] = &["○", "◔", "◑", "◕", "●", "◕", "◑", "◔"];
206    pub const SPINNER_BOUNCE: &'static [&'static str] = &["⠁", "⠂", "⠄", "⡀", "⢀", "⠠", "⠐", "⠈"];
207
208    /// Get spinner frame
209    pub fn spinner_frame(frame_index: usize, style: SpinnerStyle) -> &'static str {
210        let frames = match style {
211            SpinnerStyle::Dots => Self::SPINNER_DOTS,
212            SpinnerStyle::Circle => Self::SPINNER_CIRCLE,
213            SpinnerStyle::Pulse => Self::SPINNER_PULSE,
214            SpinnerStyle::Bounce => Self::SPINNER_BOUNCE,
215        };
216        frames[frame_index % frames.len()]
217    }
218
219    /// Create animated spinner with message
220    pub fn spinner_with_message(frame_index: usize, message: &str, style: SpinnerStyle) -> StyledString {
221        let frame = Self::spinner_frame(frame_index, style);
222        styled_feedback(&format!("{} {}", frame, message), FeedbackLevel::Processing)
223    }
224}

235/// Accessibility helpers
236pub struct AccessibilityHelpers;
237
238impl AccessibilityHelpers {
239    /// Create high-contrast styled text
240    pub fn high_contrast_text(text: &str, level: FeedbackLevel) -> StyledString {
241        let mut styled = StyledString::new();
242        let style = match level {
243            FeedbackLevel::Success => ColorStyle::new(Color::Rgb(0, 255, 0), Color::Rgb(0, 0, 0)),
244            FeedbackLevel::Error => ColorStyle::new(Color::Rgb(255, 0, 0), Color::Rgb(0, 0, 0)),
245            FeedbackLevel::Warning => ColorStyle::new(Color::Rgb(255, 255, 0), Color::Rgb(0, 0, 0)),
246            FeedbackLevel::Info => ColorStyle::new(Color::Rgb(0, 255, 255), Color::Rgb(0, 0, 0)),
247            _ => ColorStyle::new(Color::Rgb(255, 255, 255), Color::Rgb(0, 0, 0)),
248        };
249
250        styled.append_styled(format!("{} {}", level.icon(), text), style);
251        styled
252    }
253
254    /// Add semantic prefix for screen readers
255    pub fn screen_reader_text(text: &str, level: FeedbackLevel) -> String {
256        let prefix = match level {
257            FeedbackLevel::Success => "[SUCCESS]",
258            FeedbackLevel::Error => "[ERROR]",
259            FeedbackLevel::Warning => "[WARNING]",
260            FeedbackLevel::Info => "[INFO]",
261            FeedbackLevel::Processing => "[PROCESSING]",
262            FeedbackLevel::Neutral => "",
263        };
264
265        if prefix.is_empty() {
266            text.to_string()
267        } else {
268            format!("{} {}", prefix, text)
269        }
270    }
271
272    /// Check if high contrast mode is needed (from environment)
273    pub fn is_high_contrast_mode() -> bool {
274        std::env::var("HIGH_CONTRAST").unwrap_or_default() == "1"
275            || std::env::var("OSVM_HIGH_CONTRAST").unwrap_or_default() == "1"
276    }
277}

279/// Create visual dividers and separators
280pub struct Dividers;
281
282impl Dividers {
283    /// Horizontal divider
284    pub fn horizontal(width: usize) -> String {
285        "─".repeat(width)
286    }
287
288    /// Thick horizontal divider
289    pub fn horizontal_thick(width: usize) -> String {
290        "═".repeat(width)
291    }
292
293    /// Section divider with label
294    pub fn section(label: &str, width: usize) -> String {
295        let label_len = label.len() + 2; // Include spaces
296        let side_len = (width.saturating_sub(label_len)) / 2;
297        format!(
298            "{}─ {} ─{}",
299            "─".repeat(side_len),
300            label,
301            "─".repeat(width.saturating_sub(side_len + label_len))
302        )
303    }
304
305    /// Box drawing characters
306    pub const BOX_TOP_LEFT: &'static str = "┌";
307    pub const BOX_TOP_RIGHT: &'static str = "┐";
308    pub const BOX_BOTTOM_LEFT: &'static str = "└";
309    pub const BOX_BOTTOM_RIGHT: &'static str = "┘";
310    pub const BOX_HORIZONTAL: &'static str = "─";
311    pub const BOX_VERTICAL: &'static str = "│";
312    pub const BOX_T_DOWN: &'static str = "┬";
313    pub const BOX_T_UP: &'static str = "┴";
314    pub const BOX_T_RIGHT: &'static str = "├";
315    pub const BOX_T_LEFT: &'static str = "┤";
316    pub const BOX_CROSS: &'static str = "┼";
317
318    /// Create a simple box around text
319    pub fn box_around(text: &str, width: usize) -> Vec<String> {
320        let mut lines = Vec::new();
321
322        // Top border
323        lines.push(format!(
324            "{}{}{}",
325            Self::BOX_TOP_LEFT,
326            Self::BOX_HORIZONTAL.repeat(width - 2),
327            Self::BOX_TOP_RIGHT
328        ));
329
330        // Content
331        for line in text.lines() {
332            let padding = width.saturating_sub(line.len() + 2);
333            lines.push(format!(
334                "{} {}{}{}",
335                Self::BOX_VERTICAL,
336                line,
337                " ".repeat(padding),
338                Self::BOX_VERTICAL
339            ));
340        }
341
342        // Bottom border
343        lines.push(format!(
344            "{}{}{}",
345            Self::BOX_BOTTOM_LEFT,
346            Self::BOX_HORIZONTAL.repeat(width - 2),
347            Self::BOX_BOTTOM_RIGHT
348        ));
349
350        lines
351    }
352}
