6/// Visual feedback levels
7pub enum FeedbackLevel {
8    Success,
9    Info,
10    Warning,
11    Error,
12    Processing,
13    Neutral,
14}
15
16impl FeedbackLevel {
17    /// Get color for feedback level
18    pub fn color(&self) -> Color {
19        match self {
20            FeedbackLevel::Success => Color::Rgb(34, 197, 94), // Green
21            FeedbackLevel::Info => Color::Rgb(59, 130, 246),   // Blue
22            FeedbackLevel::Warning => Color::Rgb(251, 146, 60), // Orange
23            FeedbackLevel::Error => Color::Rgb(239, 68, 68),   // Red
24            FeedbackLevel::Processing => Color::Rgb(139, 92, 246), // Purple
25            FeedbackLevel::Neutral => Color::Rgb(156, 163, 175), // Gray
26        }
27    }
28
29    /// Get emoji icon for feedback level
30    pub fn icon(&self) -> &'static str {
31        match self {
32            FeedbackLevel::Success => "✓",
33            FeedbackLevel::Info => "ℹ",
34            FeedbackLevel::Warning => "⚠",
35            FeedbackLevel::Error => "✖",
36            FeedbackLevel::Processing => "⟳",
37            FeedbackLevel::Neutral => "•",
38        }
39    }
40
41    /// Get background color for feedback level
42    pub fn background(&self) -> Color {
43        match self {
44            FeedbackLevel::Success => Color::Rgb(20, 83, 45), // Dark green
45            FeedbackLevel::Info => Color::Rgb(30, 58, 138),   // Dark blue
46            FeedbackLevel::Warning => Color::Rgb(120, 53, 15), // Dark orange
47            FeedbackLevel::Error => Color::Rgb(127, 29, 29),  // Dark red
48            FeedbackLevel::Processing => Color::Rgb(67, 56, 202), // Dark purple
49            FeedbackLevel::Neutral => Color::Rgb(55, 65, 81), // Dark gray
50        }
51    }
52}

54/// Create styled text with feedback level
55pub fn styled_feedback(text: &str, level: FeedbackLevel) -> StyledString {
56    let mut styled = StyledString::new();
57    let style = ColorStyle::new(level.color(), Color::Rgb(24, 24, 36));
58
59    // Add icon and text
60    styled.append_styled(format!("{} {}", level.icon(), text), style);
61
62    styled
63}
64
65/// Create styled text with background
66pub fn styled_badge(text: &str, level: FeedbackLevel) -> StyledString {
67    let mut styled = StyledString::new();
68    let style = ColorStyle::new(Color::Rgb(255, 255, 255), level.background());
69
70    // Add padding and text
71    styled.append_styled(format!(" {} {} ", level.icon(), text), style);
72
73    styled
74}

76/// Visual state indicators for different components
77pub struct VisualStates;
78
79impl VisualStates {
80    /// Agent state with full visual treatment
81    pub fn agent_state_badge(state_name: &str, is_active: bool) -> StyledString {
82        let level = if is_active {
83            FeedbackLevel::Processing
84        } else {
85            FeedbackLevel::Neutral
86        };
87
88        styled_badge(state_name, level)
89    }
90
91    /// Message state indicator
92    pub fn message_state(state: MessageState) -> StyledString {
93        match state {
94            MessageState::Sent => styled_feedback("Sent", FeedbackLevel::Success),
95            MessageState::Sending => styled_feedback("Sending...", FeedbackLevel::Processing),
96            MessageState::Failed => styled_feedback("Failed", FeedbackLevel::Error),
97            MessageState::Queued => styled_feedback("Queued", FeedbackLevel::Info),
98        }
99    }
100
101    /// Connection state indicator
102    pub fn connection_state(connected: bool) -> StyledString {
103        if connected {
104            styled_badge("Connected", FeedbackLevel::Success)
105        } else {
106            styled_badge("Disconnected", FeedbackLevel::Error)
107        }
108    }
109
110    /// Operation progress indicator
111    pub fn operation_progress(current: usize, total: usize) -> StyledString {
112        let percentage = if total > 0 {
113            (current as f64 / total as f64 * 100.0) as usize
114        } else {
115            0
116        };
117
118        styled_feedback(
119            &format!("{}% ({}/{})", percentage, current, total),
120            FeedbackLevel::Processing,
121        )
122    }
123
124    /// History navigation indicator
125    pub fn history_indicator(position: usize, total: usize) -> StyledString {
126        styled_badge(
127            &format!("History {}/{}", position + 1, total),
128            FeedbackLevel::Info,
129        )
130    }
131
132    /// Recording indicator (animated)
133    pub fn recording_indicator(is_recording: bool) -> StyledString {
134        if is_recording {
135            styled_badge("⏺ REC", FeedbackLevel::Error)
136        } else {
137            styled_badge("○ Ready", FeedbackLevel::Neutral)
138        }
139    }
140
141    /// Validation state indicator
142    pub fn validation_state(is_valid: bool, message: &str) -> StyledString {
143        if is_valid {
144            styled_feedback(&format!("✓ {}", message), FeedbackLevel::Success)
145        } else {
146            styled_feedback(&format!("✖ {}", message), FeedbackLevel::Error)
147        }
148    }
149
150    /// MCP server status

156            ServerStatus::Error => styled_badge("Error", FeedbackLevel::Error),

198/// Animated indicators
199pub struct AnimatedIndicators;
200
201impl AnimatedIndicators {
202    /// Spinner frames for different states
203    pub const SPINNER_DOTS: &'static [&'static str] =
204        &["⠋", "⠙", "⠹", "⠸", "⠼", "⠴", "⠦", "⠧", "⠇", "⠏"];
205    pub const SPINNER_CIRCLE: &'static [&'static str] = &["◐", "◓", "◑", "◒"];
206    pub const SPINNER_PULSE: &'static [&'static str] = &["○", "◔", "◑", "◕", "●", "◕", "◑", "◔"];
207    pub const SPINNER_BOUNCE: &'static [&'static str] = &["⠁", "⠂", "⠄", "⡀", "⢀", "⠠", "⠐", "⠈"];
208
209    /// Get spinner frame
210    pub fn spinner_frame(frame_index: usize, style: SpinnerStyle) -> &'static str {
211        let frames = match style {
212            SpinnerStyle::Dots => Self::SPINNER_DOTS,
213            SpinnerStyle::Circle => Self::SPINNER_CIRCLE,
214            SpinnerStyle::Pulse => Self::SPINNER_PULSE,
215            SpinnerStyle::Bounce => Self::SPINNER_BOUNCE,
216        };
217        frames[frame_index % frames.len()]
218    }
219
220    /// Create animated spinner with message
221    pub fn spinner_with_message(
222        frame_index: usize,
223        message: &str,
224        style: SpinnerStyle,
225    ) -> StyledString {
226        let frame = Self::spinner_frame(frame_index, style);
227        styled_feedback(&format!("{} {}", frame, message), FeedbackLevel::Processing)
228    }
229}

240/// Accessibility helpers
241pub struct AccessibilityHelpers;
242
243impl AccessibilityHelpers {
244    /// Create high-contrast styled text
245    pub fn high_contrast_text(text: &str, level: FeedbackLevel) -> StyledString {
246        let mut styled = StyledString::new();
247        let style = match level {
248            FeedbackLevel::Success => ColorStyle::new(Color::Rgb(0, 255, 0), Color::Rgb(0, 0, 0)),
249            FeedbackLevel::Error => ColorStyle::new(Color::Rgb(255, 0, 0), Color::Rgb(0, 0, 0)),
250            FeedbackLevel::Warning => ColorStyle::new(Color::Rgb(255, 255, 0), Color::Rgb(0, 0, 0)),
251            FeedbackLevel::Info => ColorStyle::new(Color::Rgb(0, 255, 255), Color::Rgb(0, 0, 0)),
252            _ => ColorStyle::new(Color::Rgb(255, 255, 255), Color::Rgb(0, 0, 0)),
253        };
254
255        styled.append_styled(format!("{} {}", level.icon(), text), style);
256        styled
257    }
258
259    /// Add semantic prefix for screen readers
260    pub fn screen_reader_text(text: &str, level: FeedbackLevel) -> String {
261        let prefix = match level {
262            FeedbackLevel::Success => "[SUCCESS]",
263            FeedbackLevel::Error => "[ERROR]",
264            FeedbackLevel::Warning => "[WARNING]",
265            FeedbackLevel::Info => "[INFO]",
266            FeedbackLevel::Processing => "[PROCESSING]",
267            FeedbackLevel::Neutral => "",
268        };
269
270        if prefix.is_empty() {
271            text.to_string()
272        } else {
273            format!("{} {}", prefix, text)
274        }
275    }
276
277    /// Check if high contrast mode is needed (from environment)
278    pub fn is_high_contrast_mode() -> bool {
279        std::env::var("HIGH_CONTRAST").unwrap_or_default() == "1"
280            || std::env::var("OSVM_HIGH_CONTRAST").unwrap_or_default() == "1"
281    }
282}

284/// Create visual dividers and separators
285pub struct Dividers;
286
287impl Dividers {
288    /// Horizontal divider
289    pub fn horizontal(width: usize) -> String {
290        "─".repeat(width)
291    }
292
293    /// Thick horizontal divider
294    pub fn horizontal_thick(width: usize) -> String {
295        "═".repeat(width)
296    }
297
298    /// Section divider with label
299    pub fn section(label: &str, width: usize) -> String {
300        let label_len = label.len() + 2; // Include spaces
301        let side_len = (width.saturating_sub(label_len)) / 2;
302        format!(
303            "{}─ {} ─{}",
304            "─".repeat(side_len),
305            label,
306            "─".repeat(width.saturating_sub(side_len + label_len))
307        )
308    }
309
310    /// Box drawing characters
311    pub const BOX_TOP_LEFT: &'static str = "┌";
312    pub const BOX_TOP_RIGHT: &'static str = "┐";
313    pub const BOX_BOTTOM_LEFT: &'static str = "└";
314    pub const BOX_BOTTOM_RIGHT: &'static str = "┘";
315    pub const BOX_HORIZONTAL: &'static str = "─";
316    pub const BOX_VERTICAL: &'static str = "│";
317    pub const BOX_T_DOWN: &'static str = "┬";
318    pub const BOX_T_UP: &'static str = "┴";
319    pub const BOX_T_RIGHT: &'static str = "├";
320    pub const BOX_T_LEFT: &'static str = "┤";
321    pub const BOX_CROSS: &'static str = "┼";
322
323    /// Create a simple box around text
324    pub fn box_around(text: &str, width: usize) -> Vec<String> {
325        let mut lines = Vec::new();
326
327        // Top border
328        lines.push(format!(
329            "{}{}{}",
330            Self::BOX_TOP_LEFT,
331            Self::BOX_HORIZONTAL.repeat(width - 2),
332            Self::BOX_TOP_RIGHT
333        ));
334
335        // Content
336        for line in text.lines() {
337            let padding = width.saturating_sub(line.len() + 2);
338            lines.push(format!(
339                "{} {}{}{}",
340                Self::BOX_VERTICAL,
341                line,
342                " ".repeat(padding),
343                Self::BOX_VERTICAL
344            ));
345        }
346
347        // Bottom border
348        lines.push(format!(
349            "{}{}{}",
350            Self::BOX_BOTTOM_LEFT,
351            Self::BOX_HORIZONTAL.repeat(width - 2),
352            Self::BOX_BOTTOM_RIGHT
353        ));
354
355        lines
356    }
357}
