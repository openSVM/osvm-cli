10/// Layout configuration based on terminal size
11#[derive(Debug, Clone)]
12pub struct LayoutConfig {
13    pub sidebar_width: usize,
14    pub input_height: usize,
15    pub status_bar_height: usize,
16    pub padding: usize,
17    pub layout_type: LayoutType,
18}
19
20#[derive(Debug, Clone, PartialEq)]
21pub enum LayoutType {
22    Compact,      // < 80 cols: Single column, collapsible sections
23    Standard,     // 80-120 cols: Two column
24    Wide,         // > 120 cols: Three column with tools panel
25    UltraWide,    // > 160 cols: Four column with multiple panels
26    GoldenRatio,  // Golden ratio proportions
27    Fibonacci,    // Fibonacci sequence proportions
28    Custom,       // User-defined proportions
29}
30
31/// Main layout manager that handles responsive design
32pub struct LayoutManager {
33    current_config: LayoutConfig,
34    terminal_size: Vec2,
35    animations_enabled: bool,
36}
37
38impl LayoutManager {
39    pub fn new(terminal_size: Vec2) -> Self {
40        let config = Self::calculate_config(terminal_size);
41        Self {
42            current_config: config,
43            terminal_size,
44            animations_enabled: true,
45        }
46    }
47
48    /// Calculate optimal layout based on terminal size
49    pub fn calculate_config(size: Vec2) -> LayoutConfig {
50        let (width, height) = (size.x, size.y);
51
52        // Determine layout type
53        let layout_type = match (width, height) {
54            (w, h) if w < 60 || h < 20 => LayoutType::Compact,
55            (w, _) if w < 80 => LayoutType::Standard,
56            (w, _) if w < 120 => LayoutType::Wide,
57            (w, h) if w >= 160 && h >= 40 => LayoutType::UltraWide,
58            (w, h) if w >= 100 && h >= 30 => LayoutType::GoldenRatio,
59            _ => LayoutType::Standard,
60        };
61
62        // Calculate dimensions based on layout type
63        let (sidebar_width, input_height, status_bar_height, padding) = match layout_type {
64            LayoutType::Compact => {
65                (0, 3, 0, 0) // No sidebar, minimal input, no status bar
66            }
67            LayoutType::Standard => {
68                (25, 5, 1, 1) // Narrow sidebar, standard input
69            }
70            LayoutType::Wide => {
71                (30, 6, 1, 2) // Normal sidebar, comfortable input
72            }
73            LayoutType::UltraWide => {
74                (35, 8, 2, 3) // Wide sidebar, spacious input, double status
75            }
76            LayoutType::GoldenRatio => {
77                // Use golden ratio for proportions
78                let sidebar = (width as f64 * 0.382) as usize; // 38.2%
79                let input = (height as f64 * 0.236) as usize;   // 23.6%
80                (sidebar, input, 1, 2)
81            }
82            LayoutType::Fibonacci => {
83                // Use Fibonacci sequence for nested proportions
84                let sidebar = Self::fibonacci_width(width, 5); // 5th number
85                let input = Self::fibonacci_height(height, 3); // 3rd number
86                (sidebar, input, 1, 1)
87            }
88            LayoutType::Custom => {
89                // Default custom layout
90                (30, 5, 1, 2)
91            }
92        };
93
94        LayoutConfig {
95            sidebar_width,
96            input_height,
97            status_bar_height,
98            padding,
99            layout_type,
100        }
101    }
102
103    /// Calculate width using Fibonacci sequence
104    fn fibonacci_width(total: usize, n: usize) -> usize {
105        let fib_sum: usize = 1 + 1 + 2 + 3 + 5 + 8 + 13 + 21; // Sum of first 8
106        let fib_n = Self::fibonacci(n);
107        (total as f64 * (fib_n as f64 / fib_sum as f64)) as usize
108    }
109
110    /// Calculate height using Fibonacci sequence
111    fn fibonacci_height(total: usize, n: usize) -> usize {
112        let fib_n = Self::fibonacci(n);
113        (total as f64 * (fib_n as f64 / 21.0)) as usize // 21 is reasonable max
114    }
115
116    /// Get nth Fibonacci number
117    fn fibonacci(n: usize) -> usize {
118        match n {
119            0 => 0,
120            1 => 1,
121            2 => 1,
122            3 => 2,
123            4 => 3,
124            5 => 5,
125            6 => 8,
126            7 => 13,
127            8 => 21,
128            _ => 21, // Cap at 21 for practical use
129        }
130    }
131
132    /// Update layout when terminal is resized
133    pub fn on_resize(&mut self, new_size: Vec2) -> bool {
134        let new_config = Self::calculate_config(new_size);
135
136        if new_config.layout_type != self.current_config.layout_type {
137            // Layout type changed, need full rebuild
138            self.terminal_size = new_size;
139            self.current_config = new_config;
140            true // Signal that UI needs rebuild
141        } else {
142            // Just dimension changes, can animate
143            self.animate_transition(new_config);
144            false
145        }
146    }
147
148    /// Animate transition between layouts
149    fn animate_transition(&mut self, new_config: LayoutConfig) {
150        if !self.animations_enabled {
151            self.current_config = new_config;
152            return;
153        }
154
155        // FUTURE: Implement smooth animation (requires animation framework integration)
156        // For now, just update directly
157        self.current_config = new_config;
158    }
159
160    /// Get current layout configuration
161    pub fn config(&self) -> &LayoutConfig {
162        &self.current_config
163    }
164
165    /// Check if animations are enabled
166    pub fn animations_enabled(&self) -> bool {
167        self.animations_enabled
168    }
169
170    /// Toggle animations
171    pub fn set_animations(&mut self, enabled: bool) {
172        self.animations_enabled = enabled;
173    }
174}

176/// Helper functions for creating responsive views
177pub mod responsive {
178    use super::*;
179    use cursive::views::{Panel, ScrollView, TextView};
180
181    /// Create a responsive sidebar that adapts to width
182    pub fn sidebar(width: usize, content: impl View + 'static) -> Box<dyn View> {
183        if width == 0 {
184            // Hidden sidebar
185            Box::new(TextView::new(""))
186        } else if width < 20 {
187            // Ultra-narrow: Icons only
188            Box::new(
189                ResizedView::with_fixed_width(
190                    width,
191                    ScrollView::new(content)
192                )
193            )
194        } else if width < 30 {
195            // Narrow: Abbreviated content
196            Box::new(
197                ResizedView::with_fixed_width(
198                    width,
199                    Panel::new(ScrollView::new(content))
200                        .title("◂") // Collapsed indicator
201                )
202            )
203        } else {
204            // Normal: Full content
205            Box::new(
206                ResizedView::with_fixed_width(
207                    width,
208                    Panel::new(ScrollView::new(content))
209                        .title("Sessions")
210                )
211            )
212        }
213    }
214
215    /// Create responsive input area that adapts to height
216    pub fn input_area(height: usize, content: impl View + 'static) -> Box<dyn View> {
217        if height <= 3 {
218            // Single line input
219            Box::new(ResizedView::with_fixed_height(height, content))
220        } else if height <= 5 {
221            // Multi-line without scroll
222            Box::new(ResizedView::with_fixed_height(height, content))
223        } else {
224            // Multi-line with scroll
225            Box::new(
226                ResizedView::with_fixed_height(
227                    height,
228                    ScrollView::new(content)
229                )
230            )
231        }
232    }
233
234    /// Create adaptive padding based on available space
235    pub fn padding(size: usize) -> Box<dyn View> {
236        use cursive::views::DummyView;
237
238        match size {
239            0 => Box::new(DummyView),
240            _ => Box::new(ResizedView::with_fixed_width(size, DummyView)),
241        }
242    }
243}
