1//! Adaptive layout that responds to terminal size changes
2
3use cursive::direction::Orientation;
4use cursive::views::{
5    BoxedView, Dialog, DummyView, LinearLayout, Panel, ResizedView, ScrollView, TextView,
6};
7use cursive::View;

10/// Breakpoints for responsive design
11pub struct Breakpoints;
12
13impl Breakpoints {
14    pub const MOBILE: usize = 60;     // < 60 cols
15    pub const TABLET: usize = 80;     // 60-80 cols
16    pub const DESKTOP: usize = 120;   // 80-120 cols
17    pub const WIDE: usize = 160;      // 120-160 cols
18    pub const ULTRAWIDE: usize = 200; // > 160 cols
19}
20
21/// Adaptive container that changes based on available space
22pub struct AdaptiveContainer {
23    content: Vec<Box<dyn View>>,
24    current_layout: AdaptiveLayout,
25}
26
27#[derive(Clone, Copy, PartialEq, Debug)]
28pub enum AdaptiveLayout {
29    Stacked,    // All elements vertical
30    TwoColumn,  // Side by side
31    ThreeColumn,
32    Grid,       // 2x2 or 3x3 grid
33    Masonry,    // Pinterest-style
34}
35
36impl AdaptiveContainer {
37    pub fn new() -> Self {
38        Self {
39            content: Vec::new(),
40            current_layout: AdaptiveLayout::Stacked,
41        }
42    }
43
44    pub fn add_child(&mut self, view: impl View + 'static) {
45        self.content.push(Box::new(view));
46    }
47
48    pub fn build_for_size(&self, size: Vec2) -> Box<dyn View> {
49        let layout = Self::determine_layout(size, self.content.len());
50
51        match layout {
52            AdaptiveLayout::Stacked => self.build_stacked(),
53            AdaptiveLayout::TwoColumn => self.build_two_column(),
54            AdaptiveLayout::ThreeColumn => self.build_three_column(),
55            AdaptiveLayout::Grid => self.build_grid(),
56            AdaptiveLayout::Masonry => self.build_masonry(),
57        }
58    }
59
60    fn determine_layout(size: Vec2, num_items: usize) -> AdaptiveLayout {
61        match size.x {
62            w if w < Breakpoints::MOBILE => AdaptiveLayout::Stacked,
63            w if w < Breakpoints::TABLET => {
64                if num_items <= 2 {
65                    AdaptiveLayout::Stacked
66                } else {
67                    AdaptiveLayout::TwoColumn
68                }
69            }
70            w if w < Breakpoints::DESKTOP => AdaptiveLayout::TwoColumn,
71            w if w < Breakpoints::WIDE => {
72                if num_items >= 6 {
73                    AdaptiveLayout::Grid
74                } else {
75                    AdaptiveLayout::ThreeColumn
76                }
77            }
78            _ => {
79                if num_items >= 9 {
80                    AdaptiveLayout::Masonry
81                } else {
82                    AdaptiveLayout::ThreeColumn
83                }
84            }
85        }
86    }
87
88    fn build_stacked(&self) -> Box<dyn View> {
89        let mut layout = LinearLayout::vertical();
90        for child in &self.content {
91            layout.add_child(DummyView); // Placeholder for actual view cloning
92        }
93        Box::new(layout)
94    }
95
96    fn build_two_column(&self) -> Box<dyn View> {
97        let mut layout = LinearLayout::horizontal();
98        let mut left = LinearLayout::vertical();
99        let mut right = LinearLayout::vertical();
100
101        for (i, _child) in self.content.iter().enumerate() {
102            if i % 2 == 0 {
103                left.add_child(DummyView);
104            } else {
105                right.add_child(DummyView);
106            }
107        }
108
109        layout.add_child(ResizedView::with_full_screen(left));
110        layout.add_child(ResizedView::with_full_screen(right));
111        Box::new(layout)
112    }
113
114    fn build_three_column(&self) -> Box<dyn View> {
115        let mut layout = LinearLayout::horizontal();
116        let mut columns = vec![
117            LinearLayout::vertical(),
118            LinearLayout::vertical(),
119            LinearLayout::vertical(),
120        ];
121
122        for (i, _child) in self.content.iter().enumerate() {
123            columns[i % 3].add_child(DummyView);
124        }
125
126        for col in columns {
127            layout.add_child(ResizedView::with_full_screen(col));
128        }
129        Box::new(layout)
130    }
131
132    fn build_grid(&self) -> Box<dyn View> {
133        // Build a 3x3 grid
134        let mut rows = LinearLayout::vertical();
135        let mut current_row = LinearLayout::horizontal();
136        let items_per_row = 3;
137
138        for (i, _child) in self.content.iter().enumerate() {
139            current_row.add_child(DummyView);
140
141            if (i + 1) % items_per_row == 0 {
142                rows.add_child(current_row);
143                current_row = LinearLayout::horizontal();
144            }
145        }
146
147        // BUG-1009 fix: Use idiomatic empty check
148        if !current_row.is_empty() {
149            rows.add_child(current_row);
150        }
151
152        Box::new(rows)
153    }
154
155    fn build_masonry(&self) -> Box<dyn View> {
156        // Pinterest-style layout with varying heights
157        let mut layout = LinearLayout::horizontal();
158        let num_columns = 4;
159        let mut columns: Vec<LinearLayout> = (0..num_columns)
160            .map(|_| LinearLayout::vertical())
161            .collect();
162
163        // Distribute items to columns with least height (simplified)
164        for (i, _child) in self.content.iter().enumerate() {
165            columns[i % num_columns].add_child(DummyView);
166        }
167
168        for col in columns {
169            layout.add_child(ResizedView::with_full_width(col));
170        }
171
172        Box::new(layout)
173    }
174}

176/// Fluid container that scales content proportionally
177pub struct FluidContainer {
178    base_size: Vec2,
179    min_size: Vec2,
180    max_size: Vec2,
181}
182
183impl FluidContainer {
184    pub fn new(base: Vec2) -> Self {
185        Self {
186            base_size: base,
187            min_size: Vec2::new(base.x / 2, base.y / 2),
188            max_size: Vec2::new(base.x * 2, base.y * 2),
189        }
190    }
191
192    pub fn scale_for_terminal(&self, terminal_size: Vec2) -> Vec2 {
193        let scale_x = terminal_size.x as f32 / self.base_size.x as f32;
194        let scale_y = terminal_size.y as f32 / self.base_size.y as f32;
195        let scale = scale_x.min(scale_y); // Maintain aspect ratio
196
197        let new_width = (self.base_size.x as f32 * scale) as usize;
198        let new_height = (self.base_size.y as f32 * scale) as usize;
199
200        Vec2::new(
201            new_width.max(self.min_size.x).min(self.max_size.x),
202            new_height.max(self.min_size.y).min(self.max_size.y),
203        )
204    }
205}

207/// Collapsible sidebar that can be toggled
208pub struct CollapsibleSidebar {
209    expanded: bool,
210    expanded_width: usize,
211    collapsed_width: usize,
212}
213
214impl CollapsibleSidebar {
215    pub fn new(expanded_width: usize) -> Self {
216        Self {
217            expanded: true,
218            expanded_width,
219            collapsed_width: 5, // Just show icons
220        }
221    }
222
223    pub fn toggle(&mut self) {
224        self.expanded = !self.expanded;
225    }
226
227    pub fn current_width(&self) -> usize {
228        if self.expanded {
229            self.expanded_width
230        } else {
231            self.collapsed_width
232        }
233    }
234
235    pub fn build(&self, content: impl View + 'static) -> Box<dyn View> {
236        let width = self.current_width();
237        let title = if self.expanded { "◂ Sessions" } else { "▸" };
238
239        Box::new(
240            ResizedView::with_fixed_width(
241                width,
242                Panel::new(content).title(title)
243            )
244        )
245    }
246}

248#[cfg(test)]
249mod tests {

252    #[test]
253    fn test_adaptive_layout_selection() {
254        let small = Vec2::new(50, 30);
255        let layout = AdaptiveContainer::determine_layout(small, 5);
256        assert_eq!(layout, AdaptiveLayout::Stacked);
257
258        let medium = Vec2::new(100, 40);
259        let layout = AdaptiveContainer::determine_layout(medium, 5);
260        assert_eq!(layout, AdaptiveLayout::TwoColumn);
261
262        let large = Vec2::new(180, 50);
263        let layout = AdaptiveContainer::determine_layout(large, 10);
264        assert_eq!(layout, AdaptiveLayout::Masonry);
265    }
266
267    #[test]
268    fn test_fluid_scaling() {
269        let fluid = FluidContainer::new(Vec2::new(100, 50));
270
271        let scaled = fluid.scale_for_terminal(Vec2::new(200, 100));
272        assert_eq!(scaled, Vec2::new(200, 100));
273
274        let scaled_small = fluid.scale_for_terminal(Vec2::new(50, 25));
275        assert_eq!(scaled_small, Vec2::new(50, 25));
276    }
