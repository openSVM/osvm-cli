8/// Golden ratio constant (phi)
9pub const GOLDEN_RATIO: f64 = 1.618033988749895;
10
11/// Calculate golden ratio proportions for a given dimension
12pub struct GoldenRatioCalculator;
13
14impl GoldenRatioCalculator {
15    /// Split a dimension into golden ratio proportions
16    /// Returns (smaller, larger) where larger/smaller â‰ˆ 1.618
17    pub fn split(total: usize) -> (usize, usize) {
18        let smaller = (total as f64 / GOLDEN_RATIO) as usize;
19        let larger = total - smaller;
20        (smaller, larger)
21    }
22
23    /// Calculate multiple golden ratio segments
24    pub fn split_multiple(total: usize, segments: usize) -> Vec<usize> {
25        if segments <= 1 {
26            return vec![total];
27        }
28
29        let mut sizes = Vec::new();
30        let mut remaining = total;
31
32        for i in 0..segments - 1 {
33            let (smaller, larger) = Self::split(remaining);
34            if i % 2 == 0 {
35                sizes.push(smaller);
36                remaining = larger;
37            } else {
38                sizes.push(larger);
39                remaining = smaller;
40            }
41        }
42        sizes.push(remaining);
43
44        sizes
45    }
46
47    /// Calculate golden rectangle dimensions
48    pub fn golden_rectangle(width: usize) -> usize {
49        (width as f64 / GOLDEN_RATIO) as usize
50    }
51
52    /// Create a golden spiral layout
53    pub fn spiral_layout(size: Vec2) -> Vec<(usize, usize, usize, usize)> {
54        let mut rectangles = Vec::new();
55        let mut x = 0;
56        let mut y = 0;
57        let mut width = size.x;
58        let mut height = size.y;
59
60        for i in 0..5 {
61            // Create 5 nested golden rectangles
62            rectangles.push((x, y, width, height));
63
64            let (smaller, larger) = Self::split(width.min(height));
65
66            match i % 4 {
67                0 => {
68                    // Split horizontally, take right
69                    x += smaller;
70                    width = larger;
71                }
72                1 => {
73                    // Split vertically, take bottom
74                    y += smaller;
75                    height = larger;
76                }
77                2 => {
78                    // Split horizontally, take left
79                    width = smaller;
80                }
81                3 => {
82                    // Split vertically, take top
83                    height = smaller;
84                }
85                _ => {}
86            }
87
88            if width < 10 || height < 10 {
89                break; // Too small to continue
90            }
91        }
92
93        rectangles
94    }
95}

97/// Create a golden ratio based layout
98pub fn create_golden_layout(
99    size: Vec2,
100    sidebar_content: impl View + 'static,
101    main_content: impl View + 'static,
102    input_content: impl View + 'static,
103) -> LinearLayout {
104    let (sidebar_width, main_width) = GoldenRatioCalculator::split(size.x);
105    let (main_height, input_height) = GoldenRatioCalculator::split(size.y);
106
107    let mut root = LinearLayout::horizontal();
108
109    // Sidebar (smaller golden ratio part)
110    root.add_child(
111        ResizedView::with_fixed_width(
112            sidebar_width,
113            Panel::new(sidebar_content).title("Ï† Sessions"),
114        ),
115    );
116
117    // Main area (larger golden ratio part)
118    let mut main_layout = LinearLayout::vertical();
119
120    // Chat area (larger part of vertical split)
121    main_layout.add_child(
122        ResizedView::with_fixed_height(main_height, Panel::new(main_content).title("Ï† Chat")),
123    );
124
125    // Input area (smaller part of vertical split)
126    main_layout.add_child(
127        ResizedView::with_fixed_height(input_height, Panel::new(input_content).title("Ï† Input")),
128    );
129
130    root.add_child(ResizedView::with_fixed_width(main_width, main_layout));
131
132    root
133}

135/// Create a Fibonacci spiral layout for artistic presentation
136pub fn create_spiral_layout(size: Vec2) -> LinearLayout {
137    let mut layout = LinearLayout::vertical();
138
139    let rectangles = GoldenRatioCalculator::spiral_layout(size);
140
141    // Create nested panels following the spiral
142    for (i, (x, y, w, h)) in rectangles.iter().enumerate() {
143        let content = format!("Section {} ({}, {}) {}x{}", i + 1, x, y, w, h);
144        layout.add_child(Panel::new(
145            ResizedView::with_fixed_size((*w, *h), cursive::views::TextView::new(content)),
146        ));
147    }
148
149    layout
150}

152#[cfg(test)]

156    #[test]
157    fn test_golden_split() {
158        let (smaller, larger) = GoldenRatioCalculator::split(100);
159        let ratio = larger as f64 / smaller as f64;
160        assert!((ratio - GOLDEN_RATIO).abs() < 0.1);
161    }

163    #[test]
164    fn test_multiple_splits() {
165        let segments = GoldenRatioCalculator::split_multiple(100, 3);
166        assert_eq!(segments.len(), 3);
