129    /// Update with available MCP tools
130    pub fn update_mcp_tools(&mut self, state: &AdvancedChatState) {
131        self.mcp_tools.clear();
132
133        if let Ok(tools) = state.available_tools.read() {
134            for (server_id, tool_list) in tools.iter() {
135                for tool in tool_list {
136                    self.mcp_tools.push(ToolSuggestion {
137                        name: tool.name.clone(),
138                        server: server_id.clone(),
139                        description: tool.description.clone().unwrap_or_default(),
140                    });
141                }
142            }
143        }
144    }

146    /// Process input and generate suggestions
147    pub fn process_input(&mut self, text: &str, cursor_pos: usize) -> Vec<String> {
148        self.current_text = text.to_string();
149        self.cursor_pos = cursor_pos;
150        self.suggestions.clear();
151
152        // Get the word at cursor position
153        let (word_start, current_word) = self.get_word_at_cursor();
154
155        // Check for command completion (starts with /)
156        if current_word.starts_with('/') {
157            for cmd in &self.commands {
158                if cmd.trigger.starts_with(&current_word) && cmd.trigger != current_word {
159                    self.suggestions
160                        .push(format!("{} - {}", cmd.trigger, cmd.description));
161                }
162            }
163        }
164        // Check for tool mention (starts with @)
165        else if current_word.starts_with('@') {
166            let search = if current_word.len() > 1 {
167                current_word[1..].to_lowercase()
168            } else {
169                String::new()
170            };
171            for tool in &self.mcp_tools {
172                if tool.name.to_lowercase().contains(&search) {
173                    self.suggestions
174                        .push(format!("@{} ({})", tool.name, tool.server));
175                }
176            }
177            // Limit to 5 suggestions
178            self.suggestions.truncate(5);
179        }
180        // Check for template triggers
181        else if !current_word.is_empty() {
182            for template in &self.templates {
183                if template.trigger.starts_with(&current_word.to_lowercase()) {
184                    self.suggestions.push(template.template.clone());
185                }
186            }
187        }
188
189        self.suggestions.clone()
190    }

192    /// Get the word at the current cursor position
193    fn get_word_at_cursor(&self) -> (usize, String) {
194        if self.current_text.is_empty() {
195            return (0, String::new());
196        }
197
198        let bytes = self.current_text.as_bytes();
199        let mut start = self.cursor_pos.min(bytes.len());
200
201        // Find word start
202        while start > 0 && !bytes[start - 1].is_ascii_whitespace() {
203            start -= 1;
204        }
205
206        let mut end = self.cursor_pos.min(bytes.len());
207        // Find word end
208        while end < bytes.len() && !bytes[end].is_ascii_whitespace() {
209            end += 1;
210        }
211
212        let word = self.current_text[start..end].to_string();
213        (start, word)
214    }

216    /// Apply the selected suggestion
217    pub fn apply_suggestion(&self, suggestion_index: usize) -> Option<String> {
218        if suggestion_index >= self.suggestions.len() {
219            return None;
220        }
221
222        let suggestion = &self.suggestions[suggestion_index];
223        let (word_start, current_word) = self.get_word_at_cursor();
224
225        // Extract the completion text
226        let completion = if suggestion.starts_with('/') {
227            // Command suggestion - take just the command part
228            suggestion
229                .split(" - ")
230                .next()
231                .unwrap_or(suggestion)
232                .to_string()
233        } else if suggestion.starts_with('@') {
234            // Tool suggestion - take just the tool name
235            suggestion
236                .split(" (")
237                .next()
238                .unwrap_or(suggestion)
239                .to_string()
240        } else {
241            // Template - use as is
242            suggestion.clone()
243        };
244
245        // Build the new text with the completion
246        let mut new_text = String::new();
247        new_text.push_str(&self.current_text[..word_start]);
248        new_text.push_str(&completion);
249
250        // Add space after completion if it's a command or tool
251        if completion.starts_with('/') || completion.starts_with('@') {
252            new_text.push(' ');
253        }
254
255        // Add any text after the current word
256        let word_end = word_start + current_word.len();
257        if word_end < self.current_text.len() {
258            new_text.push_str(&self.current_text[word_end..]);
259        }
260
261        Some(new_text)
262    }

265/// Create autocomplete suggestion view
266pub fn create_autocomplete_view(suggestions: &[String]) -> impl View {
267    let mut layout = LinearLayout::vertical();
268
269    // Create styled suggestions
270    for (i, suggestion) in suggestions.iter().enumerate() {
271        let mut styled = StyledString::new();
272
273        // Highlight style
274        let highlight_style = if i == 0 {
275            // First item is selected
276            ColorStyle::new(Color::Dark(BaseColor::Black), Color::Light(BaseColor::Cyan))
277        } else {
278            ColorStyle::new(Color::Light(BaseColor::White), Color::Dark(BaseColor::Blue))
279        };
280
281        styled.append_styled(format!(" {} ", suggestion), highlight_style);
282
283        layout.add_child(TextView::new(styled));
284    }
285
286    layout
287}

295/// Handle Tab key for autocomplete
296pub fn handle_autocomplete_tab(s: &mut Cursive) -> bool {
297    // Get current input text
298    if let Some(input) = s.find_name::<cursive::views::TextArea>("input") {
299        let content = input.get_content();
300        let cursor_pos = content.len(); // Simplified - would need real cursor position
301
302        // Create autocomplete engine
303        let mut engine = AutocompleteEngine::new();
304
305        // Update with MCP tools if available
306        if let Some(state) = s.user_data::<AdvancedChatState>() {
307            engine.update_mcp_tools(state);
308        }
309
310        // Get suggestions
311        let suggestions = engine.process_input(&content, cursor_pos);
312
313        if !suggestions.is_empty() {
314            // Apply first suggestion
315            if let Some(completed) = engine.apply_suggestion(0) {
316                if let Some(mut input) = s.find_name::<cursive::views::TextArea>("input") {
317                    input.set_content(completed);
318                }
319                return true; // Handled
320            }
321        }
322    }
323
324    false // Not handled
325}
