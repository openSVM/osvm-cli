30    pub fn setup_far_ui(&self, siv: &mut Cursive) {
31        let state = self.state.clone();
32
33        // Main horizontal layout: Chat List | Chat History
34        let mut main_layout = LinearLayout::horizontal();
35
36        // Left panel: Chat sessions list - responsive width based on terminal size
37        let terminal_size = siv.screen_size();
38        let left_width = if terminal_size.x > 120 {
39            35 // Wide terminal - more space for session list
40        } else if terminal_size.x > 80 {
41            30 // Medium terminal
42        } else {
43            25 // Narrow terminal - minimal session list width
44        };
45
46        let chat_list_panel = self.create_chat_list_panel();
47        main_layout.add_child(ResizedView::with_fixed_width(left_width, chat_list_panel));
48
49        // Right panel: Active chat and controls - takes remaining space
50        let chat_panel = self.create_chat_panel();
51        main_layout.add_child(chat_panel.full_width());
52
53        // Wrap in main dialog with dynamic title showing agent status
54        let title = if let Some(session) = self.state.get_active_session() {
55            match session.agent_state {
56                AgentState::Idle => "OSVM Agent - Idle".to_string(),
57                AgentState::Thinking => "OSVM Agent - Thinking...".to_string(),
58                AgentState::Planning => "OSVM Agent - Planning...".to_string(),
59                AgentState::ExecutingTool(ref tool) => format!("OSVM Agent - Executing {}", tool),
60                AgentState::Waiting => "OSVM Agent - Waiting".to_string(),
61                AgentState::Paused => "OSVM Agent - Paused".to_string(),
62                AgentState::Error(ref err) => format!("OSVM Agent - Error: {}", err),
63            }
64        } else {
65            "OSVM Agent".to_string()
66        };
67        let dialog = Dialog::around(main_layout)
68            .title(title)
69            .title_position(cursive::align::HAlign::Center);
70
71        siv.add_fullscreen_layer(dialog);
72
73        // Set focus to input field initially
74        siv.focus_name("input").ok();
75
76        // Add Tab navigation between key UI elements - but don't interfere with other keys
77        siv.add_global_callback(cursive::event::Key::Tab, |s| {
78            // Try to focus the chat list first, then input as fallback
79            if s.focus_name("chat_list").is_ok() {
80                // Successfully focused chat list
81            } else {
82                let _ = s.focus_name("input");
83            }
84        });
85
86        // Add Shift+Tab for reverse navigation
87        siv.add_global_callback(
88            cursive::event::Event::Shift(cursive::event::Key::Tab),
89            |s| {
90                // Try to focus input first, then chat list as fallback
91                if s.focus_name("input").is_ok() {
92                    // Successfully focused input
93                } else {
94                    let _ = s.focus_name("chat_list");
95                }
96            },
97        );
98
99        // Tab and Arrow keys will handle navigation naturally through the ListView buttons
100
101        // Add context menu via F10 key (standard UI convention)
102        siv.add_global_callback(cursive::event::Key::F10, |s| {
103            show_context_menu(s, cursive::Vec2::new(0, 0));
104        });
105
106        // Also add Shift+F10 for alternate context menu access
107        siv.add_global_callback(
108            cursive::event::Event::Shift(cursive::event::Key::F10),
109            |s| {
110                show_context_menu(s, cursive::Vec2::new(0, 0));
111            },
112        );
113
114        // Add resize handling to prevent crashes - gentle approach
115        siv.add_global_callback(cursive::event::Event::WindowResize, |s| {
116            // Gentle refresh - just update displays, don't recreate UI
117            super::display::update_ui_displays(s);
118        });
119    }

121    pub fn setup_action_hotkeys(&self, siv: &mut Cursive) {
122        // Alt+M: Switch back to standard mode
123        siv.add_global_callback(cursive::event::Event::AltChar('m'), |siv| {
124            siv.add_layer(
125                Dialog::text("Switching back to standard mode requires restarting.\nPlease restart with 'osvm chat' (without --advanced)")
126                    .title("Switch Mode")
127                    .button("OK", |s| { s.pop_layer(); })
128            );
129        });
130
131        let state = self.state.clone();
132        // Alt+R: Retry last message
133        siv.add_global_callback(cursive::event::Event::AltChar('r'), move |s| {
134            retry_last_message(s, state.clone());
135        });
136
137        let state = self.state.clone();
138        // Alt+C: Copy last message
139        siv.add_global_callback(cursive::event::Event::AltChar('c'), move |s| {
140            copy_last_message(s, state.clone());
141        });
142
143        let state = self.state.clone();
144        // Alt+D: Delete last message
145        siv.add_global_callback(cursive::event::Event::AltChar('d'), move |s| {
146            delete_last_message(s, state.clone());
147        });
148
149        let state = self.state.clone();
150        // Alt+F: Fork conversation
151        siv.add_global_callback(cursive::event::Event::AltChar('f'), move |s| {
152            fork_conversation(s, state.clone());
153        });
154    }

156    pub fn setup_suggestion_hotkeys(&self, siv: &mut Cursive) {
157        // Add Ctrl+number key handlers for suggestions - only when input has focus
158        for i in 1..=5 {
159            let state = self.state.clone();
160            let key_char = char::from_digit(i as u32, 10).unwrap();
161
162            siv.add_global_callback(cursive::event::Event::CtrlChar(key_char), move |s| {
163                // Only insert suggestion if suggestions are visible AND input has focus
164                let suggestions_visible = state
165                    .suggestions_visible
166                    .read()
167                    .map(|v| *v)
168                    .unwrap_or(false);
169
170                if suggestions_visible && s.find_name::<EditView>("input").is_some() {
171                    insert_suggestion_at_cursor(s, (i - 1) as usize, state.clone());
172                }
173            });
174        }
175
176        // Also add Alt+number for easier access
177        for i in 1..=5 {
178            let state = self.state.clone();
179
180            siv.add_global_callback(
181                cursive::event::Event::AltChar(char::from_digit(i as u32, 10).unwrap()),
182                move |s| {
183                    // Only insert suggestion if suggestions are visible AND input has focus
184                    let suggestions_visible = state
185                        .suggestions_visible
186                        .read()
187                        .map(|v| *v)
188                        .unwrap_or(false);
189
190                    if suggestions_visible && s.find_name::<EditView>("input").is_some() {
191                        insert_suggestion_at_cursor(s, (i - 1) as usize, state.clone());
192                    }
193                },
194            );
195        }
196
197        // Hide suggestions on Escape - but don't interfere with other Escape usage
198        let state = self.state.clone();
199        siv.add_global_callback(cursive::event::Key::Esc, move |_s| {
200            if let Ok(mut vis) = state.suggestions_visible.write() {
201                *vis = false;
202            }
203        });
204
205        // Emergency clear - Ctrl+Alt+X to clear stuck processing states (safe combination)
206        let state = self.state.clone();
207        siv.add_global_callback(cursive::event::Event::AltChar('x'), move |siv| {
208            // Clear all processing messages from active session
209            if let Some(session) = state.get_active_session() {
210                let session_id = session.id;
211                if let Err(e) = state.remove_last_processing_message(session_id) {
212                    eprintln!("Emergency clear failed: {}", e);
213                }
214                super::display::update_ui_displays(siv);
215            }
216        });
217    }
