1//! Session persistence for chat2
2//! Save and load chat sessions from disk
3
4use anyhow::{Context, Result};
5use serde::{Deserialize, Serialize};
6use std::collections::HashMap;
7use std::path::PathBuf;
8use uuid::Uuid;
9
10use super::session::ChatSession;
11
12/// Persisted state container
13#[derive(Serialize, Deserialize)]
14pub struct PersistedState {
15    pub sessions: HashMap<Uuid, ChatSession>,
16    pub active_session_id: Option<Uuid>,
17    pub last_updated: chrono::DateTime<chrono::Utc>,
18}
19
20impl PersistedState {
21    /// Get the path to the persistence file
22    pub fn get_path() -> Result<PathBuf> {
23        let home = std::env::var("HOME").context("HOME not set")?;
24        let path = PathBuf::from(home)
25            .join(".osvm")
26            .join("chat_sessions")
27            .join("state.json");
28
29        // Ensure directory exists
30        if let Some(parent) = path.parent() {
31            std::fs::create_dir_all(parent)?;
32        }
33
34        Ok(path)
35    }
36
37    /// Load state from disk
38    pub fn load() -> Result<Option<Self>> {
39        let path = Self::get_path()?;
40
41        if !path.exists() {
42            return Ok(None);
43        }
44
45        let content = std::fs::read_to_string(&path)?;
46        let state: PersistedState =
47            serde_json::from_str(&content).context("Failed to parse persisted state")?;
48
49        Ok(Some(state))
50    }
51
52    /// Save state to disk
53    pub fn save(
54        sessions: &HashMap<Uuid, ChatSession>,
55        active_session_id: Option<Uuid>,
56    ) -> Result<()> {
57        let state = PersistedState {
58            sessions: sessions.clone(),
59            active_session_id,
60            last_updated: chrono::Utc::now(),
61        };
62
63        let path = Self::get_path()?;
64        let json = serde_json::to_string_pretty(&state)?;
65        std::fs::write(&path, json)?;
66
67        Ok(())
68    }
69
70    /// Create default sessions (only if no saved state exists)
71    pub fn create_defaults() -> HashMap<Uuid, ChatSession> {
72        let mut sessions = HashMap::new();
73
74        let main_id = Uuid::new_v4();
75        sessions.insert(
76            main_id,
77            ChatSession {
78                id: main_id,
79                name: "Main Chat".to_string(),
80                created_at: chrono::Utc::now(),
81                messages: Vec::new(),
82                agent_state: super::types::AgentState::Idle,
83                recording: false,
84                recording_file: None,
85            },
86        );
87
88        sessions
89    }
90}
