1//! Core data structures and types for the advanced agent chat interface
2
3use crate::services::ai_service::PlannedTool;
4use chrono::{DateTime, Utc};
5use serde::{Deserialize, Serialize};
6use serde_json::Value;
7use uuid::Uuid;
8
9/// Message types in the advanced chat interface
10#[derive(Clone, Debug, Serialize, Deserialize)]
11pub enum ChatMessage {
12    User(String),
13    Agent(String),
14    System(String),
15    ToolCall {
16        tool_name: String,
17        description: String,
18        args: Option<Value>,
19        execution_id: String,
20    },
21    ToolResult {
22        tool_name: String,
23        result: Value,
24        execution_id: String,
25    },
26    Error(String),
27    AgentThinking(String),
28    AgentPlan(Vec<PlannedTool>),
29    Processing {
30        message: String,
31        spinner_index: usize,
32    }, // Shows processing with spinner
33}
34
35impl ChatMessage {
36    /// Get a human-readable timestamp for this message
37    /// Returns a formatted time string like "14:32:15"
38    pub fn get_display_timestamp() -> String {
39        Utc::now().format("%H:%M:%S").to_string()
40    }
41
42    /// Get the message type as a string for display
43    pub fn message_type(&self) -> &str {
44        match self {
45            ChatMessage::User(_) => "User",
46            ChatMessage::Agent(_) => "Agent",
47            ChatMessage::System(_) => "System",
48            ChatMessage::ToolCall { .. } => "Tool Call",
49            ChatMessage::ToolResult { .. } => "Tool Result",
50            ChatMessage::Error(_) => "Error",
51            ChatMessage::AgentThinking(_) => "Thinking",
52            ChatMessage::AgentPlan(_) => "Plan",
53            ChatMessage::Processing { .. } => "Processing",
54        }
55    }
56}
