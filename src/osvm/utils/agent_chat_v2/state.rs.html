195    pub fn get_active_session(&self) -> Option<ChatSession> {
196        // Use a timeout to prevent deadlocks
197        let active_id = match self.active_session_id.try_read() {
198            Ok(lock) => lock,
199            Err(_) => {
200                warn!("Failed to acquire read lock for active_session_id");
201                return None;
202            }
203        };
204
205        let session_id = (*active_id)?;
206        drop(active_id); // Release lock early
207
208        let sessions = match self.sessions.try_read() {
209            Ok(lock) => lock,
210            Err(_) => {
211                warn!("Failed to acquire read lock for sessions");
212                return None;
213            }
214        };
215
216        sessions.get(&session_id).cloned()
217    }

259    pub fn remove_last_processing_message(&self, session_id: Uuid) -> Result<()> {
260        // Blocking lock: very short critical section, avoids missing cleanup
261        let mut sessions = self
262            .sessions
263            .write()
264            .map_err(|e| anyhow!("Failed to acquire write lock for sessions: {}", e))?;
265
266        // Remove ALL Processing messages from the session
267        if let Some(session) = sessions.get_mut(&session_id) {
268            let initial_count = session.messages.len();
269            session
270                .messages
271                .retain(|msg| !matches!(msg, ChatMessage::Processing { .. }));
272            let removed_count = initial_count - session.messages.len();
273            if removed_count > 0 {
274                debug!(
275                    "Removed {} processing message(s) from session {}",
276                    removed_count, session_id
277                );
278            }
279        } else {
280            warn!(
281                "Session {} not found when removing processing message",
282                session_id
283            );
284        }
285
286        Ok(())
287    }

303    /// Synchronous version for sending commands from UI callbacks
304    pub fn send_agent_command_sync(&self, command: AgentCommand) {
305        let sender = self.agent_command_sender.clone();
306
307        // Use std::thread to avoid runtime conflicts
308        std::thread::spawn(move || {
309            // Create a small runtime just for this operation. If runtime creation fails,
310            // log the error and return early to avoid panics in the UI thread.
311            match tokio::runtime::Runtime::new() {
312                Ok(runtime) => {
313                    runtime.block_on(async {
314                        let sender_guard = sender.lock().await;
315                        if let Some(sender) = sender_guard.as_ref() {
316                            if let Err(e) = sender.send(command) {
317                                error!("Failed to send agent command: {}", e);
318                            }
319                        } else {
320                            error!("Agent worker not initialized");
321                        }
322                    });
323                }
324                Err(e) => {
325                    error!(
326                        "Failed to create temporary Tokio runtime for send_agent_command_sync: {}",
327                        e
328                    );
329                }
330            }
331        });
332    }

334    /// Create a new chat session
335    pub fn create_session(&self, name: String) -> Result<Uuid> {
336        let session_id = Uuid::new_v4();
337        let session = ChatSession {
338            id: session_id,
339            name,
340            created_at: chrono::Utc::now(),
341            messages: Vec::new(),
342            agent_state: AgentState::Idle,
343            recording: false,
344            recording_file: None,
345        };
346
347        // BUG-2005 fix: Make session creation and activation atomic to prevent race conditions
348        // Previously, the lock was released and another thread could create a session between
349        // the drop() and set_active_session() calls, causing wrong session to become active
350        let mut sessions = self
351            .sessions
352            .write()
353            .map_err(|e| anyhow!("Failed to lock sessions: {}", e))?;
354        sessions.insert(session_id, session);
355
356        // Check if this is the first session and set as active WHILE HOLDING THE LOCK
357        let is_first_session = sessions.len() == 1;
358
359        if is_first_session {
360            drop(sessions); // Release lock before set_active_session
361            self.set_active_session(session_id)?;
362        } else {
363            drop(sessions); // Release lock
364            // Still save state if not first session
365            self.save_state_async();
366        }
367
368        Ok(session_id)
369    }

371    /// Delete a chat session by ID
372    pub fn delete_session(&self, session_id: Uuid) -> Result<()> {
373        let mut sessions = self
374            .sessions
375            .write()
376            .map_err(|e| anyhow!("Failed to lock sessions for deletion: {}", e))?;
377
378        if sessions.remove(&session_id).is_none() {
379            warn!("Session {} not found for deletion", session_id);
380            return Ok(()); // Not an error if already deleted
381        }
382
383        debug!("Deleted session {}", session_id);
384        drop(sessions); // Release lock
385
386        // Ensure active_session_id doesn't point to deleted session
387        if let Ok(mut active_id) = self.active_session_id.write() {
388            if *active_id == Some(session_id) {
389                // Set to first available session or None
390                if let Ok(sessions) = self.sessions.read() {
391                    *active_id = sessions.keys().next().copied();
392                    if *active_id == Some(session_id) {
393                        // Shouldn't happen but be safe
394                        *active_id = None;
395                    }
396                } else {
397                    *active_id = None;
398                }
399                info!("Reset active session after deletion");
400            }
401        }
402
403        // Auto-save after session deletion
404        self.save_state_async();
405        Ok(())
406    }

431    pub async fn refresh_tools_from_mcp(&self) -> Result<()> {
432        let servers = {
433            let service = self.mcp_service.lock().await;
434            service
435                .list_servers()
436                .into_iter()
437                .map(|(id, config)| (id.clone(), config.clone()))
438                .collect::<Vec<(String, McpServerConfig)>>()
439        };
440
441        let mut new_tools: HashMap<String, Vec<McpTool>> = HashMap::new();
442
443        for server_data in servers {
444            let (server_id, config) = server_data;
445            if config.enabled {
446                // Try to get real tools from MCP server
447                let tools = {
448                    // Initialize server if needed
449                    {
450                        let mut service = self.mcp_service.lock().await;
451                        if let Err(e) = service.initialize_server(&server_id).await {
452                            warn!(
453                                "Failed to initialize server {} for tool refresh: {}",
454                                server_id, e
455                            );
456                        }
457                    }
458
459                    // Try to list tools
460                    let mut service = self.mcp_service.lock().await;
461                    match service.list_tools(&server_id).await {
462                        Ok(tools) => tools,
463                        Err(e) => {
464                            warn!("Failed to fetch tools from {}: {}", server_id, e);
465                            // Don't show fallback tools - let it fail so the issue is visible
466                            vec![]
467                        }
468                    }
469                };
470                new_tools.insert(server_id, tools);
471            }
472        }
473
474        if let Ok(mut tools) = self.available_tools.write() {
475            *tools = new_tools;
476        }
477
478        Ok(())
479    }

494                    // Update the existing processing message

496                        message,
497                        spinner_index,

499                }
500            }
501        }
502        Ok(())
503    }

612    /// Load theme from configuration file or environment
613    pub fn load_theme_from_config(&self) -> Result<()> {
614        let config_path = Self::get_config_path()?;
615
616        if config_path.exists() {
617            let content = std::fs::read_to_string(&config_path)?;
618            if let Ok(config) = serde_json::from_str::<ThemeConfig>(&content) {
619                if let Some(theme_name) = config.current_theme {
620                    self.switch_theme(&theme_name)?;
621                }
622            }
623        }
624
625        Ok(())
626    }

628    /// Save current theme to configuration file
629    pub fn save_theme_config(&self) -> Result<()> {
630        let config_path = Self::get_config_path()?;
631
632        if let Some(parent) = config_path.parent() {
633            std::fs::create_dir_all(parent)?;
634        }
635
636        let theme_name = self.get_current_theme_name()?;
637        let config = ThemeConfig {
638            current_theme: Some(theme_name),
639            auto_theme_switching: false, // Note: Auto theme switching feature planned for future release
640        };
641
642        let json = serde_json::to_string_pretty(&config)?;
643        std::fs::write(&config_path, json)?;
644
645        Ok(())
646    }

659    /// Add a message to input history (like bash history)
660    pub fn add_to_history(&self, input: String) {
661        if input.trim().is_empty() {
662            return; // Don't add empty messages
663        }
664
665        if let Ok(mut history) = self.input_history.write() {
666            // Don't add if it's the same as the last entry (avoid duplicates)
667            if let Some(last) = history.last() {
668                if last == &input {
669                    return;
670                }
671            }
672
673            history.push(input);
674
675            // Limit history to 100 entries (like bash HISTSIZE)
676            if history.len() > 100 {
677                history.remove(0);
678            }
679        }
680
681        // Reset position when adding new entry
682        if let Ok(mut pos) = self.history_position.write() {
683            *pos = None;
684        }
685    }
686
687    /// Navigate to previous entry in history (up arrow)
688    pub fn history_previous(&self) -> Option<String> {
689        let history = self.input_history.read().ok()?;
690
691        if history.is_empty() {
692            return None;
693        }
694
695        let mut pos = self.history_position.write().ok()?;
696
697        let new_pos = match *pos {
698            None => {
699                // First time pressing up, go to most recent entry
700                history.len() - 1
701            }
702            Some(current) => {
703                // Already in history, go further back if possible
704                if current > 0 {
705                    current - 1
706                } else {
707                    return Some(history[current].clone()); // Already at oldest
708                }
709            }
710        };
711
712        *pos = Some(new_pos);
713        Some(history[new_pos].clone())
714    }
715
716    /// Navigate to next entry in history (down arrow)
717    pub fn history_next(&self) -> Option<String> {
718        let history = self.input_history.read().ok()?;
719
720        if history.is_empty() {
721            return None;
722        }
723
724        let mut pos = self.history_position.write().ok()?;
725
726        let new_pos = match *pos {
727            None => {
728                // Not in history, stay at prompt
729                return Some(String::new());
730            }
731            Some(current) => {
732                // In history, move forward
733                if current < history.len() - 1 {
734                    current + 1
735                } else {
736                    // Reached the end, go back to empty prompt
737                    *pos = None;
738                    return Some(String::new());
739                }
740            }
741        };
742
743        *pos = Some(new_pos);
744        Some(history[new_pos].clone())
745    }
746
747    /// Check if currently navigating history
748    pub fn is_in_history(&self) -> bool {
749        self.history_position
750            .read()
751            .ok()
752            .and_then(|pos| *pos)
753            .is_some()
754    }
755
756    /// Get current history position for display
757    pub fn get_history_indicator(&self) -> String {
758        if let Ok(pos) = self.history_position.read() {
759            if let Some(idx) = *pos {
760                if let Ok(history) = self.input_history.read() {
761                    let total = history.len();
762                    return format!(" (history {}/{})", idx + 1, total);
763                }
764            }
765        }
766        String::new()
767    }
768
769    /// Save state to disk asynchronously (non-blocking)
770    fn save_state_async(&self) {
771        // Clone the data we need to save (quick operation)
772        let sessions = match self.sessions.try_read() {
773            Ok(s) => s.clone(),
774            Err(_) => {
775                warn!("Could not acquire lock to save state");
776                return;
777            }
778        };
779
780        let active_id = match self.active_session_id.try_read() {
781            Ok(id) => *id,
782            Err(_) => {
783                warn!("Could not acquire lock for active session ID");
784                return;
785            }
786        };
787
788        // Spawn thread to do the actual save (don't block UI)
789        std::thread::spawn(move || {
790            if let Err(e) = super::persistence::PersistedState::save(&sessions, active_id) {
791                // Log error but don't crash - save is nice-to-have
792                debug!("Failed to persist state: {}", e);
793            }
794        });
795    }
796}
