16/// Plugin metadata and configuration
17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub struct PluginManifest {
19    pub name: String,
20    pub version: String,
21    pub description: String,
22    pub author: String,
23    pub license: String,
24    pub homepage: Option<String>,
25    pub repository: Option<String>,
26
27    // Plugin behavior
28    pub plugin_type: PluginType,
29    pub entry_point: String,
30    pub dependencies: Vec<String>,
31    pub permissions: Vec<Permission>,
32
33    // Compatibility
34    pub min_osvm_version: String,
35    pub supported_platforms: Vec<String>,
36
37    // Configuration
38    pub config_schema: Option<serde_json::Value>,
39    pub default_config: Option<serde_json::Value>,
40}

42/// Types of plugins supported
43#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
44pub enum PluginType {
45    /// Command plugins add new slash commands
46    Command,
47    /// Tool plugins provide new MCP-style tools
48    Tool,
49    /// Theme plugins provide visual customizations
50    Theme,
51    /// Filter plugins process input/output
52    Filter,
53    /// Integration plugins connect to external services
54    Integration,
55    /// AI plugins provide custom AI capabilities
56    AI,
57    /// Composite plugins combine multiple functionalities
58    Composite(Vec<PluginType>),
59}

61/// Plugin permissions system
62#[derive(Debug, Clone, Serialize, Deserialize)]
63pub enum Permission {
64    /// Read files from specified directories
65    ReadFiles(Vec<String>),
66    /// Write files to specified directories
67    WriteFiles(Vec<String>),
68    /// Execute system commands
69    ExecuteCommands,
70    /// Network access to specified hosts
71    NetworkAccess(Vec<String>),
72    /// Access to environment variables
73    EnvironmentAccess,
74    /// Access to MCP servers
75    MCPAccess,
76    /// Access to AI services
77    AIAccess,
78    /// Access to user configuration
79    ConfigAccess,
80    /// All permissions (dangerous)
81    All,
82}

304/// External plugin that executes scripts/binaries
305pub struct ExternalPlugin {
306    manifest: PluginManifest,
307    executable_path: PathBuf,
308    config: HashMap<String, serde_json::Value>,
309}
310
311impl ExternalPlugin {
312    pub fn from_manifest(manifest_path: &PathBuf) -> Result<Self> {
313        let content = fs::read_to_string(manifest_path)
314            .map_err(|e| anyhow!("Failed to read plugin manifest: {}", e))?;
315
316        let manifest: PluginManifest = serde_json::from_str(&content)
317            .map_err(|e| anyhow!("Failed to parse plugin manifest: {}", e))?;
318
319        let plugin_dir = manifest_path
320            .parent()
321            .ok_or_else(|| anyhow!("Invalid manifest path"))?;
322
323        let executable_path = plugin_dir.join(&manifest.entry_point);
324
325        Ok(Self {
326            manifest,
327            executable_path,
328            config: HashMap::new(),
329        })
330    }
331}
332
333#[async_trait]
334impl Plugin for ExternalPlugin {
335    fn manifest(&self) -> &PluginManifest {
336        &self.manifest
337    }
338
339    async fn initialize(&mut self, config: HashMap<String, serde_json::Value>) -> Result<()> {
340        self.config = config;
341        debug!("External plugin '{}' initialized", self.manifest.name);
342        Ok(())
343    }
344
345    async fn execute(&self, context: PluginContext) -> Result<PluginResult> {
346        // Prepare execution environment
347        let input_json = serde_json::to_string(&context)
348            .map_err(|e| anyhow!("Failed to serialize context: {}", e))?;
349
350        // Execute plugin
351        let mut cmd = TokioCommand::new(&self.executable_path);
352        cmd.stdin(Stdio::piped())
353            .stdout(Stdio::piped())
354            .stderr(Stdio::piped());
355
356        // Add environment variables
357        for (key, value) in &context.environment {
358            cmd.env(key, value);
359        }
360
361        let mut child = cmd
362            .spawn()
363            .map_err(|e| anyhow!("Failed to spawn plugin process: {}", e))?;
364
365        // Send input to plugin
366        if let Some(stdin) = child.stdin.as_mut() {
367            use tokio::io::AsyncWriteExt;
368            stdin
369                .write_all(input_json.as_bytes())
370                .await
371                .map_err(|e| anyhow!("Failed to write to plugin stdin: {}", e))?;
372            stdin
373                .shutdown()
374                .await
375                .map_err(|e| anyhow!("Failed to close plugin stdin: {}", e))?;
376        }
377
378        // Wait for completion
379        let output = child
380            .wait_with_output()
381            .await
382            .map_err(|e| anyhow!("Plugin execution failed: {}", e))?;
383
384        // Parse result
385        if output.status.success() {
386            let stdout = String::from_utf8_lossy(&output.stdout);
387
388            // Try to parse as JSON first
389            if let Ok(result) = serde_json::from_str::<PluginResult>(&stdout) {
390                Ok(result)
391            } else {
392                // Fallback to plain text output
393                Ok(PluginResult {
394                    success: true,
395                    output: stdout.to_string(),
396                    error: None,
397                    data: None,
398                    modified_input: None,
399                    suggestions: vec![],
400                })
401            }
402        } else {
403            let stderr = String::from_utf8_lossy(&output.stderr);
404            Ok(PluginResult {
405                success: false,
406                output: String::new(),
407                error: Some(stderr.to_string()),
408                data: None,
409                modified_input: None,
410                suggestions: vec![],
411            })
412        }
413    }
414
415    fn can_handle(&self, input: &str) -> bool {
416        // Check if input matches plugin's command pattern
417        match &self.manifest.plugin_type {
418            PluginType::Command => input.starts_with(&format!("/{}", self.manifest.name)),
419            PluginType::Filter => true, // Filters can process any input
420            _ => false,
421        }
422    }
423
424    async fn health_check(&self) -> Result<bool> {
425        Ok(self.executable_path.exists() && self.executable_path.is_file())
426    }
427
428    async fn cleanup(&mut self) -> Result<()> {
429        debug!("External plugin '{}' cleaned up", self.manifest.name);
430        Ok(())
431    }
432}

434/// Plugin manager handles loading, lifecycle, and execution of plugins
435pub struct PluginManager {
436    plugins: HashMap<String, Box<dyn Plugin>>,
437    plugin_configs: HashMap<String, HashMap<String, serde_json::Value>>,
438    enabled_plugins: Vec<String>,
439}
440
441impl PluginManager {
442    /// Create new plugin manager
443    pub fn new() -> Self {
444        Self {
445            plugins: HashMap::new(),
446            plugin_configs: HashMap::new(),
447            enabled_plugins: Vec::new(),
448        }
449    }
450
451    /// Initialize with built-in plugins
452    pub async fn with_builtin_plugins() -> Result<Self> {
453        let mut manager = Self::new();
454
455        // Add built-in plugins
456        manager.register_plugin(Box::new(EchoPlugin::new())).await?;
457        manager.register_plugin(Box::new(TimePlugin::new())).await?;
458
459        // Enable built-in plugins by default
460        manager
461            .enabled_plugins
462            .extend(vec!["echo".to_string(), "time".to_string()]);
463
464        info!(
465            "Plugin manager initialized with {} built-in plugins",
466            manager.plugins.len()
467        );
468        Ok(manager)
469    }
470
471    /// Register a plugin
472    pub async fn register_plugin(&mut self, mut plugin: Box<dyn Plugin>) -> Result<()> {
473        let name = plugin.manifest().name.clone();
474
475        // Initialize plugin with default config
476        if let Some(default_config) = &plugin.manifest().default_config {
477            if let Ok(config) =
478                serde_json::from_value::<HashMap<String, serde_json::Value>>(default_config.clone())
479            {
480                plugin.initialize(config.clone()).await?;
481                self.plugin_configs.insert(name.clone(), config);
482            }
483        } else {
484            plugin.initialize(HashMap::new()).await?;
485        }
486
487        self.plugins.insert(name.clone(), plugin);
488        debug!("Registered plugin: {}", name);
489
490        Ok(())
491    }
492
493    /// Load plugins from directory
494    pub async fn load_plugins_from_directory(&mut self, dir: &PathBuf) -> Result<()> {
495        if !dir.exists() {
496            warn!("Plugin directory does not exist: {:?}", dir);
497            return Ok(());
498        }
499
500        for entry in fs::read_dir(dir)? {
501            let entry = entry?;
502            let path = entry.path();
503
504            if path.is_dir() {
505                let manifest_path = path.join("plugin.json");
506                if manifest_path.exists() {
507                    match ExternalPlugin::from_manifest(&manifest_path) {
508                        Ok(plugin) => {
509                            info!("Loading external plugin from {:?}", path);
510                            self.register_plugin(Box::new(plugin)).await?;
511                        }
512                        Err(e) => {
513                            warn!("Failed to load plugin from {:?}: {}", path, e);
514                        }
515                    }
516                }
517            }
518        }
519
520        Ok(())
521    }
522
523    /// Execute input through appropriate plugins
524    pub async fn execute(&self, input: &str, context: PluginContext) -> Result<Vec<PluginResult>> {
525        let mut results = Vec::new();
526
527        for plugin_name in &self.enabled_plugins {
528            if let Some(plugin) = self.plugins.get(plugin_name) {
529                if plugin.can_handle(input) {
530                    debug!("Executing plugin: {}", plugin_name);
531                    match plugin.execute(context.clone()).await {
532                        Ok(result) => {
533                            results.push(result);
534
535                            // For command plugins, stop after first successful execution
536                            if matches!(plugin.manifest().plugin_type, PluginType::Command)
537                                && results.last().unwrap().success
538                            {
539                                break;
540                            }
541                        }
542                        Err(e) => {
543                            error!("Plugin '{}' execution failed: {}", plugin_name, e);
544                            results.push(PluginResult {
545                                success: false,
546                                output: String::new(),
547                                error: Some(e.to_string()),
548                                data: None,
549                                modified_input: None,
550                                suggestions: vec![],
551                            });
552                        }
553                    }
554                }
555            }
556        }
557
558        Ok(results)
559    }
560
561    /// Get available commands from all plugins
562    pub fn get_available_commands(&self) -> Vec<String> {
563        let mut commands = Vec::new();
564
565        for (name, plugin) in &self.plugins {
566            if self.enabled_plugins.contains(name) {
567                match plugin.manifest().plugin_type {
568                    PluginType::Command => {
569                        commands.push(format!("/{}", plugin.manifest().name));
570                    }
571                    PluginType::Composite(ref types) => {
572                        if types.contains(&PluginType::Command) {
573                            commands.push(format!("/{}", plugin.manifest().name));
574                        }
575                    }
576                    _ => {}
577                }
578            }
579        }
580
581        commands.sort();
582        commands
583    }

585    /// Enable plugin
586    pub fn enable_plugin(&mut self, name: &str) -> Result<()> {
587        if self.plugins.contains_key(name) {
588            if !self.enabled_plugins.contains(&name.to_string()) {
589                self.enabled_plugins.push(name.to_string());
590                info!("Enabled plugin: {}", name);
591            }
592            Ok(())
593        } else {
594            Err(anyhow!("Plugin '{}' not found", name))
595        }
596    }

598    /// Disable plugin
599    pub fn disable_plugin(&mut self, name: &str) {
600        self.enabled_plugins.retain(|n| n != name);
601        info!("Disabled plugin: {}", name);
602    }

619            match plugin.health_check().await {

633    /// Save plugin configuration
634    pub async fn save_config(&self) -> Result<()> {
635        let config_path = Self::config_path()?;
636
637        if let Some(parent) = config_path.parent() {
638            fs::create_dir_all(parent)?;
639        }
640
641        let config = serde_json::json!({
642            "enabled_plugins": self.enabled_plugins,
643            "plugin_configs": self.plugin_configs
644        });
645
646        fs::write(&config_path, serde_json::to_string_pretty(&config)?)?;
647        debug!("Saved plugin configuration to {:?}", config_path);
648
649        Ok(())
650    }

652    /// Load plugin configuration
653    pub async fn load_config(&mut self) -> Result<()> {
654        let config_path = Self::config_path()?;
655
656        if !config_path.exists() {
657            debug!("No plugin configuration found, using defaults");
658            return Ok(());
659        }
660
661        let content = fs::read_to_string(&config_path)?;
662        let config: serde_json::Value = serde_json::from_str(&content)?;
663
664        if let Some(enabled) = config.get("enabled_plugins").and_then(|v| v.as_array()) {
665            self.enabled_plugins = enabled
666                .iter()
667                .filter_map(|v| v.as_str().map(|s| s.to_string()))
668                .collect();
669        }
670
671        if let Some(configs) = config.get("plugin_configs").and_then(|v| v.as_object()) {
672            for (name, plugin_config) in configs {
673                if let Ok(config_map) = serde_json::from_value::<HashMap<String, serde_json::Value>>(
674                    plugin_config.clone(),
675                ) {
676                    self.plugin_configs.insert(name.clone(), config_map);
677                }
678            }
679        }
680
681        debug!("Loaded plugin configuration from {:?}", config_path);
682        Ok(())
683    }

700/// Plugin installer for managing external plugins
701pub struct PluginInstaller {
702    plugins_dir: PathBuf,
703}
704
705impl PluginInstaller {
706    pub fn new() -> Result<Self> {
707        let plugins_dir = PluginManager::plugins_dir()?;
708        fs::create_dir_all(&plugins_dir)?;
709
710        Ok(Self { plugins_dir })
711    }

713    /// Install plugin from GitHub repository
714    pub async fn install_from_github(&self, repo_url: &str) -> Result<String> {
715        let repo_name = Self::extract_repo_name(repo_url)?;
716        let plugin_dir = self.plugins_dir.join(&repo_name);
717
718        if plugin_dir.exists() {
719            return Err(anyhow!("Plugin '{}' already exists", repo_name));
720        }
721
722        // Clone repository
723        let output = Command::new("git")
724            .args(&["clone", repo_url, &plugin_dir.to_string_lossy()])
725            .output()
726            .map_err(|e| anyhow!("Failed to clone repository: {}", e))?;
727
728        if !output.status.success() {
729            let stderr = String::from_utf8_lossy(&output.stderr);
730            return Err(anyhow!("Git clone failed: {}", stderr));
731        }
732
733        // Verify plugin manifest exists
734        let manifest_path = plugin_dir.join("plugin.json");
735        if !manifest_path.exists() {
736            fs::remove_dir_all(&plugin_dir)?;
737            return Err(anyhow!("No plugin.json found in repository"));
738        }
739
740        // Run plugin installation script if present
741        let install_script = plugin_dir.join("install.sh");
742        if install_script.exists() {
743            let output = Command::new("bash")
744                .arg(&install_script)
745                .current_dir(&plugin_dir)
746                .output()
747                .map_err(|e| anyhow!("Failed to run install script: {}", e))?;
748
749            if !output.status.success() {
750                let stderr = String::from_utf8_lossy(&output.stderr);
751                warn!("Install script failed: {}", stderr);
752            }
753        }
754
755        info!(
756            "Successfully installed plugin '{}' from {}",
757            repo_name, repo_url
758        );
759        Ok(repo_name)
760    }

762    /// Uninstall plugin
763    pub async fn uninstall(&self, plugin_name: &str) -> Result<()> {
764        let plugin_dir = self.plugins_dir.join(plugin_name);
765
766        if !plugin_dir.exists() {
767            return Err(anyhow!("Plugin '{}' not found", plugin_name));
768        }
769
770        // Run uninstall script if present
771        let uninstall_script = plugin_dir.join("uninstall.sh");
772        if uninstall_script.exists() {
773            let output = Command::new("bash")
774                .arg(&uninstall_script)
775                .current_dir(&plugin_dir)
776                .output()
777                .map_err(|e| anyhow!("Failed to run uninstall script: {}", e))?;
778
779            if !output.status.success() {
780                let stderr = String::from_utf8_lossy(&output.stderr);
781                warn!("Uninstall script failed: {}", stderr);
782            }
783        }
784
785        // Remove plugin directory
786        fs::remove_dir_all(&plugin_dir)?;
787
788        info!("Successfully uninstalled plugin '{}'", plugin_name);
789        Ok(())
790    }

792    /// List installed plugins
793    pub fn list_installed(&self) -> Result<Vec<String>> {
794        let mut plugins = Vec::new();
795
796        if !self.plugins_dir.exists() {
797            return Ok(plugins);
798        }
799
800        for entry in fs::read_dir(&self.plugins_dir)? {
801            let entry = entry?;
802            let path = entry.path();
803
804            if path.is_dir() && path.join("plugin.json").exists() {
805                if let Some(name) = path.file_name().and_then(|n| n.to_str()) {
806                    plugins.push(name.to_string());
807                }
808            }
809        }
810
811        plugins.sort();
812        Ok(plugins)
813    }

815    /// Extract repository name from URL (public utility method)
816    ///
817    /// # Examples
818    /// ```
819    /// # use anyhow::Result;
820    /// # fn main() -> Result<()> {
821    /// use osvm::utils::plugins::PluginInstaller;
822    ///
823    /// let name = PluginInstaller::extract_repo_name("https://github.com/user/repo.git")?;
824    /// assert_eq!(name, "repo");
825    /// # Ok(())
826    /// # }
827    /// ```
828    pub fn extract_repo_name(url: &str) -> Result<String> {
829        let url = url.trim_end_matches('/');
830        if let Some(name) = url.split('/').last() {
831            let name = name.trim_end_matches(".git");
832            if name.is_empty() {
833                Err(anyhow!("Invalid repository URL"))
834            } else {
835                Ok(name.to_string())
836            }
837        } else {
838            Err(anyhow!("Invalid repository URL"))
839        }
840    }
