62/// Global finding ID allocator to ensure unique IDs across sessions
63static FINDING_ID_COUNTER: AtomicUsize = AtomicUsize::new(1);

65/// Session ID for this audit run based on timestamp
66static SESSION_ID: Lazy<String> = Lazy::new(|| {
67    use std::time::{SystemTime, UNIX_EPOCH};
68    let timestamp = SystemTime::now()
69        .duration_since(UNIX_EPOCH)
70        .unwrap_or_else(|_| {
71            // Fallback timestamp if system time is before epoch
72            std::time::Duration::from_secs(1640995200) // Jan 1, 2022
73        })
74        .as_secs();
75    format!("{:08x}", timestamp & 0xFFFFFFFF) // Use lower 32 bits for shorter IDs
76});

78/// Helper function to create regex patterns safely at compile time
79fn create_regex(pattern: &str, name: &str) -> regex::Regex {
80    regex::Regex::new(pattern).unwrap_or_else(|e| {
81        panic!(
82            "Failed to compile regex pattern '{}' for {}: {}",
83            pattern, name, e
84        )
85    })
86}
87
88/// Cached regex patterns for performance with lazy loading and optimization
89static REGEX_CACHE: Lazy<HashMap<&'static str, regex::Regex>> = Lazy::new(|| {
90    debug_print!("Initializing regex cache with lazy loading");
91    HashMap::new() // Start with empty cache
92});
93
94/// Lazy-loaded regex getter with performance optimization
95static REGEX_CACHE_STORE: Lazy<RwLock<HashMap<&'static str, regex::Regex>>> =
96    Lazy::new(|| RwLock::new(HashMap::new()));
97
98/// Get regex pattern with lazy loading for better startup performance
99fn get_regex(pattern_name: &'static str) -> Option<regex::Regex> {
100    // First try to read from cache
101    {
102        let cache = REGEX_CACHE_STORE.read().unwrap();
103        if let Some(regex) = cache.get(pattern_name) {
104            return Some(regex.clone());
105        }
106    }
107
108    // If not in cache, create and store it
109    let mut cache = REGEX_CACHE_STORE.write().unwrap();
110
111    // Double-check pattern in case another thread added it
112    if let Some(regex) = cache.get(pattern_name) {
113        return Some(regex.clone());
114    }
115
116    // Create the regex based on pattern name with simplified patterns for performance
117    let regex = match pattern_name {
118        // Basic secret patterns (simplified for performance)
119        "password_pattern" => create_regex(r#"(?i)password\s*=\s*['"'][^'"']+['"']"#, pattern_name),
120        "api_key_pattern" => create_regex(r#"(?i)api_key\s*=\s*['"'][^'"']+['"']"#, pattern_name),
121        "secret_pattern" => create_regex(r#"(?i)secret\s*=\s*['"'][^'"']+['"']"#, pattern_name),
122        "hex_key_pattern" => create_regex(r#"['"'][0-9a-fA-F]{32,}['"']"#, pattern_name),
123        "base64_pattern" => create_regex(r#"['"'][A-Za-z0-9+/]{20,}={0,2}['"']"#, pattern_name),
124
125        // Simplified command injection patterns for better performance
126        "command_injection" => create_regex(
127            r#"Command::new.*(?:format!|concat!|user_input)"#,
128            pattern_name,
129        ),
130        "shell_command" => create_regex(r#"(?:shell|system|exec|cmd|spawn)\("#, pattern_name),
131        "unsafe_exec" => create_regex(r#"process::Command.*(?:format!|user_input)"#, pattern_name),
132
133        // Simplified path traversal patterns
134        "path_traversal" => create_regex(r#"\.\.[\\/]"#, pattern_name), // Simplified from complex encoding patterns
135        "dynamic_path" => create_regex(r#"Path::new.*(?:format!|user_input)"#, pattern_name),
136
137        // Network patterns
138        "http_insecure" => create_regex(r#"http://[^/]+"#, pattern_name),
139        "tls_bypass" => create_regex(r#"danger_accept_invalid_certs\(true\)"#, pattern_name),
140
141        // Solana-specific patterns
142        "solana_signer" => create_regex(r#"AccountInfo.*is_signer"#, pattern_name),
143        "solana_pda" => create_regex(r#"find_program_address"#, pattern_name),
144        "solana_owner" => create_regex(r#"AccountInfo.*owner"#, pattern_name),
145        "rent_exempt" => create_regex(r#"rent_exempt"#, pattern_name),
146
147        _ => {
148            debug_warn!("Unknown regex pattern requested: {}", pattern_name);
149            return None;
150        }
151    };
152
153    cache.insert(pattern_name, regex.clone());
154    debug_print!("Lazy-loaded regex pattern: {}", pattern_name);
155    Some(regex)
156}

158/// Finding ID allocator for unique ID generation
159pub struct FindingIdAllocator;
160
161impl FindingIdAllocator {
162    /// Generate a unique finding ID with session context (using UUID for better collision resistance)
163    pub fn next_id() -> String {
164        // Default to UUID-based ID for maximum uniqueness
165        Self::next_uuid_id()
166    }
167
168    /// Generate a legacy timestamp-based finding ID with session context
169    pub fn next_legacy_id() -> String {
170        let id = FINDING_ID_COUNTER.fetch_add(1, Ordering::SeqCst);
171        format!("OSVM-{}-{:03}", &*SESSION_ID, id)
172    }
173
174    /// Generate a category-specific finding ID with UUID for maximum uniqueness
175    pub fn next_category_id(category: &str) -> String {
176        // Use UUID for guaranteed uniqueness instead of hash-based approach
177        let uuid = uuid::Uuid::new_v4();
178        match category {
179            "solana" => format!("OSVM-SOL-{}-{}", &*SESSION_ID, uuid.simple()),
180            "crypto" => format!("OSVM-CRYPTO-{}-{}", &*SESSION_ID, uuid.simple()),
181            "network" => format!("OSVM-NET-{}-{}", &*SESSION_ID, uuid.simple()),
182            "auth" => format!("OSVM-AUTH-{}-{}", &*SESSION_ID, uuid.simple()),
183            _ => format!("OSVM-{}-{}", &*SESSION_ID, uuid.simple()),
184        }
185    }
186
187    /// Generate UUID-based finding ID for maximum uniqueness
188    pub fn next_uuid_id() -> String {
189        // Use proper UUID instead of DefaultHasher for guaranteed uniqueness
190        let uuid = uuid::Uuid::new_v4();
191        // Include session ID for session context while maintaining uniqueness
192        format!("OSVM-{}-{}", &*SESSION_ID, uuid.simple())
193    }
194
195    /// Reset counter (for testing)
196    #[cfg(test)]
197    pub fn reset() {
198        FINDING_ID_COUNTER.store(1, Ordering::SeqCst);
199    }
200
201    /// Get current session ID
202    pub fn get_session_id() -> &'static str {
203        &*SESSION_ID
204    }
205}

207/// Trait for audit check modules
208pub trait AuditCheck {
209    /// Name of the audit check
210    fn name(&self) -> &'static str;
211
212    /// Category of vulnerabilities this check covers
213    fn category(&self) -> &'static str;
214
215    /// Run the audit check on parsed code
216    fn check(&self, analysis: &ParsedCodeAnalysis, file_path: &str) -> Result<Vec<AuditFinding>>;
217
218    /// Run the audit check on raw content (for backward compatibility)
219    fn check_content(&self, content: &str, file_path: &str) -> Result<Vec<AuditFinding>> {
220        let analysis = RustCodeParser::parse_code(content)?;
221        self.check(&analysis, file_path)
222    }
223}

227
228impl AuditCheck for MemorySafetyCheck {

231    }

238        let mut findings = Vec::new();

254                recommendation: "Review unsafe code blocks carefully, ensure proper bounds checking and memory management".to_string(),
255                code_location: Some(file_path.to_string()),
256                references: vec![
257                    "https://doc.rust-lang.org/book/ch19-01-unsafe-rust.html".to_string(),
258                    "https://cwe.mitre.org/data/definitions/119.html".to_string(),
259                ],
260            });
261        }
262
263        // Check for excessive unwrap usage
264        let unwrap_count = analysis.unwrap_usages.len();

271                    file_path, unwrap_count
272                ),
273                severity: AuditSeverity::Medium,
274                category: "Error Handling".to_string(),
275                cwe_id: Some("CWE-248".to_string()),
276                cvss_score: Some(4.0),
277                impact: "Application crashes due to unhandled panics, potential denial of service".to_string(),
278                recommendation: "Replace unwrap/expect with proper error handling using match or if let patterns".to_string(),
279                code_location: Some(file_path.to_string()),
280                references: vec![
281                    "https://doc.rust-lang.org/book/ch09-00-error-handling.html".to_string(),
282                    "https://cwe.mitre.org/data/definitions/248.html".to_string(),
283                ],
284            });
285        }
286
287        Ok(findings)
288    }

479                    ],
480                });
481            }
482

486                    id: FindingIdAllocator::next_category_id("solana"),

496                    impact: "PDA collision attacks possible, unauthorized access to program accounts".to_string(),
497                    recommendation: "Use multiple unique seeds for PDA creation including user-specific identifiers".to_string(),
498                    code_location: Some(format!("{}:{}", file_path, solana_op.line)),
499                    references: vec![
500                        "https://book.anchor-lang.com/anchor_bts/security.html".to_string(),
501                        "https://github.com/coral-xyz/sealevel-attacks/tree/master/programs/3-pda-sharing".to_string(),
502                    ],
503                });
504            }
505
506            // Missing account data validation

514                        "File {} accesses account data without proper validation at line {}",
515                        file_path, solana_op.line
516                    ),
517                    severity: AuditSeverity::Medium,
518                    category: "Solana Security".to_string(),
519                    cwe_id: Some("CWE-20".to_string()),
520                    cvss_score: Some(6.0),
521                    impact: "Account data confusion vulnerabilities, type safety bypasses".to_string(),
522                    recommendation: "Always deserialize and validate account data before use, check discriminators".to_string(),
523                    code_location: Some(format!("{}:{}", file_path, solana_op.line)),
524                    references: vec![
525                        "https://book.anchor-lang.com/anchor_bts/security.html".to_string(),
526                    ],
527                });
528            }
529        }
530
531        // Additional Solana-specific pattern checks using cached regexes
532        self.check_solana_patterns(analysis, file_path, &mut findings);
533
534        // Enhanced Solana-specific security checks
535        self.check_mev_protection(analysis, file_path, &mut findings);
536        self.check_authority_transfer_patterns(analysis, file_path, &mut findings);
537        self.check_duplicate_mutable_accounts(analysis, file_path, &mut findings);
538        self.check_precision_arithmetic(analysis, file_path, &mut findings);
539        self.check_atomic_validations(analysis, file_path, &mut findings);
540
541        Ok(findings)
542    }

545impl SolanaSecurityCheck {
546    /// Check if a string is a valid base58 encoded Solana public key
547    fn is_valid_base58_pubkey(value: &str) -> bool {
548        // Solana public keys are 32 bytes, which when base58 encoded are typically 32-44 characters
549        if value.len() < constants::SOLANA_PUBKEY_MIN_LENGTH
550            || value.len() > constants::SOLANA_PUBKEY_MAX_LENGTH
551        {
552            return false;
553        }
554
555        // Check if it contains only valid base58 characters
556        if !value.chars().all(|c| constants::BASE58_CHARS.contains(c)) {
557            return false;
558        }
559
560        // Try to decode as base58 and check if it's 32 bytes (Solana pubkey size)
561        match bs58::decode(value).into_vec() {
562            Ok(decoded) => decoded.len() == constants::SOLANA_PUBKEY_SIZE_BYTES,
563            Err(_) => false,
564        }
565    }
566
567    /// Determine confidence level and likelihood of Solana key with enhanced analysis
568    fn analyze_solana_key_confidence(
569        &self,
570        value: &str,
571        context: &str,
572    ) -> (bool, f32, AuditSeverity) {
573        let context_lower = context.to_lowercase();
574
575        // Skip common false positives (but exclude generic contexts)
576        if context_lower != "string_literal" {
577            for &indicator in constants::FALSE_POSITIVE_INDICATORS {
578                if context_lower.contains(indicator) {
579                    return (false, 0.0, AuditSeverity::Info);
580                }
581            }
582        }
583
584        let mut confidence: f32 = 0.0;
585
586        // Check if it matches known Solana program IDs (highest confidence)
587        if self.is_known_solana_program_id(value) {
588            confidence += 0.9;
589        }
590
591        // Strong Solana-specific indicators
592        let strong_indicators = [
593            ("pubkey", 0.8),
594            ("program_id", 0.9),
595            ("account", 0.6),
596            ("signer", 0.7),
597            ("authority", 0.7),
598            ("mint", 0.8),
599            ("pda", 0.9),
600            ("system_program", 0.9),
601            ("spl_token", 0.8),
602            ("metaplex", 0.8),
603            ("anchor", 0.7),
604            ("solana_sdk", 0.9),
605            ("solana_program", 0.9),
606            ("anchor_lang", 0.8),
607        ];

614        }

618            ("_pubkey", 0.7),
619            ("_program", 0.6),
620            ("_account", 0.5),
621            ("pubkey_", 0.7),
622            ("program_", 0.6),
623            ("solana_", 0.6),
624        ];
625
626        for (pattern, weight) in &naming_patterns {
627            if context_lower.contains(pattern) {
628                confidence += weight * 0.8; // Slightly reduce naming pattern confidence
629                break;
630            }
631        }
632
633        // Crypto/blockchain context (lower confidence)
634        let crypto_indicators = [
635            ("crypto", 0.3),
636            ("blockchain", 0.3),
637            ("defi", 0.4),
638            ("web3", 0.4),
639            ("transaction", 0.2),
640            ("wallet", 0.3),
641        ];
642
643        for (indicator, weight) in &crypto_indicators {
644            if context_lower.contains(indicator) {
645                confidence += weight;
646                break;
647            }
648        }
649
650        // Determine if it's likely a key and appropriate severity
651        let is_likely = confidence > 0.4;
652        let severity = if confidence > 0.8 {
653            AuditSeverity::Medium
654        } else if confidence > 0.6 {
655            AuditSeverity::Low
656        } else if confidence > 0.4 {
657            AuditSeverity::Info
658        } else {
659            AuditSeverity::Info
660        };
661
662        // Cap confidence at 1.0
663        (is_likely, confidence.min(1.0), severity)
664    }

1202            Box::new(MemorySafetyCheck),

1210    }

1212    /// Add a custom audit check
1213    pub fn add_check(mut self, check: Box<dyn AuditCheck + Send + Sync>) -> Self {
1214        self.checks.push(check);
1215        self
1216    }

1218    /// Run all audit checks on a file
1219    pub fn audit_file(&self, content: &str, file_path: &str) -> Result<Vec<AuditFinding>> {
1220        let mut all_findings = Vec::new();
1221
1222        // Parse the code once for all checks
1223        let analysis = RustCodeParser::parse_code(content)?;
1224
1225        // Run each check
1226        for check in &self.checks {
1227            match check.check(&analysis, file_path) {
1228                Ok(findings) => all_findings.extend(findings),
1229                Err(e) => {
1230                    log::warn!(
1231                        "Check '{}' failed for file {}: {}",
1232                        check.name(),
1233                        file_path,
1234                        e
1235                    );
1236                    // Continue with other checks even if one fails
1237                }
1238            }
1239        }
1240
1241        Ok(all_findings)
1242    }

1244    /// Get list of available checks
1245    pub fn list_checks(&self) -> Vec<(&str, &str)> {
1246        self.checks
1247            .iter()
1248            .map(|check| (check.name(), check.category()))
1249            .collect()
1250    }
1251}
