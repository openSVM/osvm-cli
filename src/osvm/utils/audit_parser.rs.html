15/// Line tracking utility for correlating AST nodes with source lines
16#[derive(Debug, Clone)]
17pub struct LineTracker {
18    source_lines: Vec<String>,
19    line_content_map: HashMap<usize, String>,
20}
21
22impl LineTracker {
23    pub fn new(source_code: &str) -> Self {
24        let source_lines: Vec<String> = source_code.lines().map(|s| s.to_string()).collect();
25        let mut line_content_map = HashMap::new();
26
27        for (line_num, line_content) in source_lines.iter().enumerate() {
28            line_content_map.insert(line_num + 1, line_content.clone());
29        }
30
31        Self {
32            source_lines,
33            line_content_map,
34        }
35    }
36
37    /// Find the line number where a specific pattern appears
38    pub fn find_pattern_line(&self, pattern: &str, start_from: usize) -> usize {
39        for (line_num, line_content) in self.line_content_map.iter() {
40            if *line_num >= start_from && line_content.contains(pattern) {
41                return *line_num;
42            }
43        }
44        start_from.max(1)
45    }
46
47    /// Get all lines containing a pattern
48    pub fn find_all_pattern_lines(&self, pattern: &str) -> Vec<usize> {
49        let mut lines = Vec::new();
50        for (line_num, line_content) in self.line_content_map.iter() {
51            if line_content.contains(pattern) {
52                lines.push(*line_num);
53            }
54        }
55        lines.sort();
56        lines
57    }
58}

162impl RustCodeParser {
163    /// Parse Rust code and extract security-relevant information
164    pub fn parse_code(content: &str) -> Result<ParsedCodeAnalysis> {
165        let syntax_tree = syn::parse_file(content).context("Failed to parse Rust syntax")?;
166
167        let mut visitor = SecurityVisitor::new(content);
168        visitor.visit_file(&syntax_tree);
169
170        Ok(visitor.into_analysis())
171    }

173    /// Check if code contains specific patterns using structured analysis
174    pub fn contains_pattern(analysis: &ParsedCodeAnalysis, pattern_type: &str) -> bool {
175        match pattern_type {
176            "unsafe" => !analysis.unsafe_blocks.is_empty(),
177            "unwrap" => !analysis.unwrap_usages.is_empty(),
178            "secrets" => !analysis.hardcoded_secrets.is_empty(),
179            "command_injection" => analysis.command_executions.iter().any(|c| c.is_dynamic),
180            "path_traversal" => analysis.path_operations.iter().any(|p| p.is_dynamic),
181            "insecure_network" => analysis.network_operations.iter().any(|n| !n.uses_https),
182            "solana_security" => !analysis.solana_operations.is_empty(),
183            "missing_signer_check" => analysis.solana_operations.iter().any(|op| !op.signer_check),
184            "missing_program_id_check" => analysis
185                .solana_operations
186                .iter()
187                .any(|op| !op.program_id_check),
188            "missing_owner_check" => analysis.solana_operations.iter().any(|op| !op.owner_check),
189            "weak_pda_seeds" => analysis
190                .solana_operations
191                .iter()
192                .any(|op| op.pda_seeds.len() < 2),
193            _ => false,
194        }
195    }

197    /// Analyze Solana-specific security vulnerabilities
198    pub fn analyze_solana_security(analysis: &ParsedCodeAnalysis) -> Vec<String> {
199        let mut vulnerabilities = Vec::new();
200
201        // Check for missing signer validation
202        let missing_signer_ops: Vec<_> = analysis
203            .solana_operations
204            .iter()
205            .filter(|op| !op.signer_check && op.operation_type.contains("account"))
206            .collect();
207
208        if !missing_signer_ops.is_empty() {
209            vulnerabilities.push(format!(
210                "Missing signer validation detected in {} operations: consider adding is_signer checks",
211                missing_signer_ops.len()
212            ));
213        }
214
215        // Check for weak PDA seed uniqueness
216        let weak_pda_ops: Vec<_> = analysis
217            .solana_operations
218            .iter()
219            .filter(|op| !op.pda_seeds.is_empty() && op.pda_seeds.len() < 2)
220            .collect();
221
222        if !weak_pda_ops.is_empty() {
223            vulnerabilities.push(format!(
224                "Weak PDA seed uniqueness detected in {} operations: use multiple unique seeds",
225                weak_pda_ops.len()
226            ));
227        }
228
229        // Check for missing program ID verification
230        let missing_program_id_ops: Vec<_> = analysis
231            .solana_operations
232            .iter()
233            .filter(|op| op.operation_type.contains("invoke") && !op.program_id_check)
234            .collect();
235
236        if !missing_program_id_ops.is_empty() {
237            vulnerabilities.push(format!(
238                "Missing program ID verification detected in {} CPI calls: validate program ID before invoke",
239                missing_program_id_ops.len()
240            ));
241        }
242
243        // Check for missing account owner verification
244        let missing_owner_ops: Vec<_> = analysis
245            .solana_operations
246            .iter()
247            .filter(|op| op.operation_type.contains("account") && !op.owner_check)
248            .collect();
249
250        if !missing_owner_ops.is_empty() {
251            vulnerabilities.push(format!(
252                "Missing account owner verification detected in {} operations: validate account owner",
253                missing_owner_ops.len()
254            ));
255        }
256
257        vulnerabilities
258    }

324    fn analyze_field_access(&mut self, expr: &Expr, field_expr: &syn::ExprField) {
325        let field_name = field_expr.member.to_token_stream().to_string();
326        let receiver_str = quote::quote!(#expr).to_string();
327
328        // Look for Solana account field access patterns with improved detection
329        if self.is_likely_account_access(&receiver_str) {
330            let line_num = self.get_line_number_for_pattern(&format!(".{}", field_name));
331
332            match field_name.as_str() {
333                "owner" => {
334                    let mut solana_op = SolanaOperation {
335                        line: line_num,
336                        operation_type: "account_owner_access".to_string(),
337                        account_validation: false,
338                        signer_check: false,
339                        pda_seeds: Vec::new(),
340                        program_id_check: false,
341                        owner_check: self.check_for_owner_validation(&receiver_str),
342                        account_data_validation: false,
343                    };
344
345                    self.solana_operations.push(solana_op);
346                }
347                "data" | "lamports" | "key" => {
348                    let mut solana_op = SolanaOperation {
349                        line: line_num,
350                        operation_type: format!("account_{}_access", field_name),
351                        account_validation: false,
352                        signer_check: self.check_for_signer_validation(&receiver_str),
353                        pda_seeds: Vec::new(),
354                        program_id_check: false,
355                        owner_check: false,
356                        account_data_validation: field_name == "data",
357                    };
358
359                    self.solana_operations.push(solana_op);
360                }
361                _ => {}
362            }
363        }
364    }

366    /// Check if the expression is likely an account access
367    fn is_likely_account_access(&self, receiver_str: &str) -> bool {
368        receiver_str.contains("account")
369            || receiver_str.contains("ctx . accounts")
370            || receiver_str.contains("ctx.accounts")
371            || receiver_str.contains("AccountInfo")

374    /// Check for owner validation patterns in the surrounding context
375    fn check_for_owner_validation(&self, receiver_str: &str) -> bool {
376        // Enhanced owner validation detection with deeper context analysis
377        debug_print!(
378            VerbosityLevel::Verbose,
379            "Checking for owner validation in context: {}",
380            receiver_str
381        );
382
383        // Check for explicit owner validation patterns using line tracker
384        // Look for common owner validation patterns in the code
385        let has_owner_check = self.line_tracker.find_all_pattern_lines("owner ==").len() > 0
386            || self.line_tracker.find_all_pattern_lines("owner !=").len() > 0
387            || (self.line_tracker.find_all_pattern_lines("require!(").len() > 0
388                && self.line_tracker.find_all_pattern_lines(".owner").len() > 0)
389            || (self
390                .line_tracker
391                .find_all_pattern_lines("assert_eq!(")
392                .len()
393                > 0
394                && self.line_tracker.find_all_pattern_lines(".owner").len() > 0)
395            || (self.line_tracker.find_all_pattern_lines("program_id").len() > 0
396                && self.line_tracker.find_all_pattern_lines("==").len() > 0)
397            || self
398                .line_tracker
399                .find_all_pattern_lines("#[account(owner")
400                .len()
401                > 0
402            || self.line_tracker.find_all_pattern_lines("owner_id").len() > 0;
403
404        // Check for Anchor constraint-based validation
405        let has_anchor_owner_constraint = self
406            .line_tracker
407            .find_all_pattern_lines("#[account(owner =")
408            .len()
409            > 0
410            || self.line_tracker.find_all_pattern_lines("owner @ ").len() > 0
411            || self.line_tracker.find_all_pattern_lines("owner: ").len() > 0;
412
413        debug_print!(
414            VerbosityLevel::Detailed,
415            "Owner validation found: {}",
416            has_owner_check || has_anchor_owner_constraint
417        );
418        has_owner_check || has_anchor_owner_constraint
419    }

421    /// Check for signer validation patterns in the surrounding context  
422    fn check_for_signer_validation(&self, receiver_str: &str) -> bool {
423        // Enhanced signer validation detection with deeper context analysis
424        debug_print!(
425            VerbosityLevel::Verbose,
426            "Checking for signer validation in context: {}",
427            receiver_str
428        );
429
430        // Check for explicit signer validation patterns using line tracker
431        // Look for common signer validation patterns in the code
432        let has_signer_check = self.line_tracker.find_all_pattern_lines("is_signer").len() > 0
433            || (self.line_tracker.find_all_pattern_lines("require!(").len() > 0
434                && (self.line_tracker.find_all_pattern_lines("signer").len() > 0
435                    || self.line_tracker.find_all_pattern_lines("signed").len() > 0))
436            || (self.line_tracker.find_all_pattern_lines("assert!(").len() > 0
437                && self.line_tracker.find_all_pattern_lines("is_signer").len() > 0)
438            || self
439                .line_tracker
440                .find_all_pattern_lines("#[account(signer")
441                .len()
442                > 0
443            || self.line_tracker.find_all_pattern_lines("Signer<").len() > 0;
444
445        // Check for conditional signer validation patterns
446        let has_conditional_signer = (self.line_tracker.find_all_pattern_lines("if").len() > 0
447            && self.line_tracker.find_all_pattern_lines("is_signer").len() > 0)
448            || (self.line_tracker.find_all_pattern_lines("match").len() > 0
449                && self.line_tracker.find_all_pattern_lines("signer").len() > 0);
450
451        debug_print!(
452            VerbosityLevel::Detailed,
453            "Signer validation found: {}",
454            has_signer_check || has_conditional_signer
455        );
456        has_signer_check || has_conditional_signer
457    }

459    fn analyze_method_call(&mut self, expr: &Expr) {
460        if let Expr::MethodCall(method_call) = expr {
461            let method_name = method_call.method.to_string();
462            let receiver_str = quote::quote!(#method_call).to_string();
463            let line_num = self.get_line_number_for_pattern(&format!(".{}(", method_name));
464
465            match method_name.as_str() {
466                "unwrap" => {
467                    self.unwrap_usages.push(UnwrapUsage {
468                        line: line_num,
469                        method: "unwrap".to_string(),
470                        context: "method_call".to_string(),
471                    });
472                }
473                "expect" => {
474                    self.unwrap_usages.push(UnwrapUsage {
475                        line: line_num,
476                        method: "expect".to_string(),
477                        context: "method_call".to_string(),
478                    });
479                }
480                "borrow" | "borrow_mut" => {
481                    // Check if this is a Solana account data access
482                    if receiver_str.contains("account") && receiver_str.contains("data") {
483                        let mut solana_op = SolanaOperation {
484                            line: line_num,
485                            operation_type: "account_data_access".to_string(),
486                            account_validation: false,
487                            signer_check: false,
488                            pda_seeds: Vec::new(),
489                            program_id_check: false,
490                            owner_check: false,
491                            account_data_validation: false,
492                        };
493
494                        // Check if there's any signer validation in the context
495                        if receiver_str.contains("is_signer") {
496                            solana_op.signer_check = true;
497                        }
498
499                        self.solana_operations.push(solana_op);
500                    }
501                }
502                _ => {}
503            }
504        }
505    }

507    fn analyze_function_call(&mut self, expr: &Expr) {
508        if let Expr::Call(call) = expr {
509            if let Expr::Path(path) = &*call.func {
510                let path_string = path_to_string(&path.path);
511                let line_num = self.get_line_number_for_pattern(&path_string);
512
513                // Check for command execution patterns
514                if path_string.contains("Command::new")
515                    || path_string.contains("std::process::Command")
516                {
517                    let is_dynamic = call.args.iter().any(|arg| is_dynamic_expr(arg));
518                    self.command_executions.push(CommandExecution {
519                        line: line_num,
520                        command_type: "Command::new".to_string(),
521                        is_dynamic,
522                    });
523                }
524
525                // Check for path operations
526                if path_string.contains("Path::new") || path_string.contains("PathBuf::from") {
527                    let is_dynamic = call.args.iter().any(|arg| is_dynamic_expr(arg));
528                    self.path_operations.push(PathOperation {
529                        line: line_num,
530                        operation_type: path_string.clone(),
531                        is_dynamic,
532                    });
533                }
534
535                // Check for Solana operations with detailed analysis
536                if path_string.contains("solana")
537                    || path_string.contains("anchor")
538                    || path_string.contains("invoke")
539                    || path_string.contains("is_signer")
540                    || path_string.contains("program_id")
541                    || path_string.contains("account")
542                {
543                    let mut solana_op = SolanaOperation {
544                        line: line_num,
545                        operation_type: path_string.clone(),
546                        account_validation: false,
547                        signer_check: false,
548                        pda_seeds: Vec::new(),
549                        program_id_check: false,
550                        owner_check: false,
551                        account_data_validation: false,
552                    };
553
554                    // Analyze for specific Solana security patterns
555                    self.analyze_solana_security_patterns(&path_string, &mut solana_op, call);
556                    self.solana_operations.push(solana_op);
557                }
558            }
559        }
560    }

562    fn analyze_string_literal(&mut self, lit: &LitStr) {
563        let value = lit.value();
564        let line_num = self.get_line_number_for_pattern(&format!("\"{}\"", value));
565
566        self.string_literals.push(StringLiteral {
567            line: line_num,
568            value: value.clone(),
569            context: "string_literal".to_string(),
570        });
571
572        // Check for hardcoded secrets
573        if self.is_potential_secret(&value) {
574            self.hardcoded_secrets.push(HardcodedSecret {
575                line: line_num,
576                secret_type: self.classify_secret(&value),
577                value: value.clone(),
578            });
579        }
580
581        // Check for network operations
582        if value.starts_with("http://") || value.starts_with("https://") {
583            self.network_operations.push(NetworkOperation {
584                line: line_num,
585                operation_type: "url".to_string(),
586                uses_https: value.starts_with("https://"),
587                endpoint: Some(value),
588            });
589        }
590    }
591
592    fn is_potential_secret(&self, value: &str) -> bool {
593        // More sophisticated secret detection using patterns
594        let secret_patterns = [
595            r"^[A-Za-z0-9+/]{20,}={0,2}$", // Base64
596            r"^[0-9a-fA-F]{32,}$",         // Hex strings
597            r"^sk-[A-Za-z0-9]{48}$",       // OpenAI API key pattern
598        ];
599
600        for pattern in &secret_patterns {
601            if regex::Regex::new(pattern).unwrap().is_match(value) {
602                return true;
603            }
604        }
605
606        // Check for common secret keywords
607        let lower_value = value.to_lowercase();
608        lower_value.len() > 16
609            && (lower_value.contains("password")
610                || lower_value.contains("secret")
611                || lower_value.contains("key")
612                || lower_value.contains("token")
613                || lower_value.contains("api"))
614    }
615
616    fn classify_secret(&self, value: &str) -> String {
617        if value.starts_with("sk-") {
618            "openai_api_key".to_string()
619        } else if regex::Regex::new(r"^[A-Za-z0-9+/]{20,}={0,2}$")
620            .unwrap()
621            .is_match(value)
622        {
623            "base64_encoded".to_string()
624        } else if regex::Regex::new(r"^[0-9a-fA-F]{32,}$")
625            .unwrap()
626            .is_match(value)
627        {
628            "hex_encoded".to_string()
629        } else {
630            "generic_secret".to_string()
631        }
632    }

635impl<'ast> Visit<'ast> for SecurityVisitor {
636    fn visit_expr(&mut self, expr: &'ast Expr) {
637        match expr {
638            Expr::Unsafe(_unsafe_expr) => {
639                let line_num = self.get_line_number_for_pattern("unsafe");
640                self.unsafe_blocks.push(UnsafeBlock {
641                    line: line_num,
642                    context: "unsafe block".to_string(),
643                });
644            }
645            Expr::Lit(lit) => {
646                if let syn::Lit::Str(lit_str) = &lit.lit {
647                    self.analyze_string_literal(lit_str);
648                }
649            }
650            Expr::MethodCall(_) => {
651                self.analyze_method_call(expr);
652            }
653            Expr::Call(_) => {
654                self.analyze_function_call(expr);
655            }
656            Expr::Field(field_expr) => {
657                self.analyze_field_access(expr, field_expr);
658            }
659            _ => {}
660        }
661
662        syn::visit::visit_expr(self, expr);
663    }
664
665    fn visit_item_fn(&mut self, item_fn: &'ast ItemFn) {
666        let signature = FunctionSignature {
667            name: item_fn.sig.ident.to_string(),
668            is_public: matches!(item_fn.vis, syn::Visibility::Public(_)),
669            is_async: item_fn.sig.asyncness.is_some(),
670            parameters: item_fn
671                .sig
672                .inputs
673                .iter()
674                .map(|_| "param".to_string())
675                .collect(),
676            return_type: None, // Note: Return type inference requires full type analysis
677        };
678        self.function_signatures.push(signature);
679
680        syn::visit::visit_item_fn(self, item_fn);
681    }
682
683    fn visit_item_use(&mut self, item_use: &'ast syn::ItemUse) {
684        let import = ImportDeclaration {
685            path: "use_item".to_string(), // Simplified for now
686            items: vec![], // Note: Detailed import analysis requires AST traversal
687        };
688        self.imports.push(import);
689
690        syn::visit::visit_item_use(self, item_use);
691    }

694impl SecurityVisitor {
695    /// Analyze Solana-specific security patterns in function calls
696    fn analyze_solana_security_patterns(
697        &mut self,
698        path_string: &str,
699        solana_op: &mut SolanaOperation,
700        call: &syn::ExprCall,
701    ) {
702        // Check for signer validation patterns
703        if path_string.contains("is_signer") || path_string.contains("signer") {
704            solana_op.signer_check = self.analyze_signer_check(call);
705        }
706
707        // Check for PDA (Program Derived Address) operations
708        if path_string.contains("find_program_address")
709            || path_string.contains("create_program_address")
710        {
711            solana_op.pda_seeds = self.extract_pda_seeds(call);
712        }
713
714        // Check for program ID verification
715        if path_string.contains("program_id") || path_string.contains("invoke") {
716            solana_op.program_id_check = self.analyze_program_id_check(call);
717        }
718
719        // Check for account owner verification
720        if path_string.contains("owner") && path_string.contains("account") {
721            solana_op.owner_check = self.analyze_owner_check(call);
722        }
723
724        // Check for account data validation
725        if path_string.contains("AccountInfo") || path_string.contains("account_data") {
726            solana_op.account_data_validation = self.analyze_account_data_validation(call);
727            solana_op.account_validation = true;
728        }
729    }
730
731    /// Analyze if proper signer validation is performed
732    fn analyze_signer_check(&self, call: &syn::ExprCall) -> bool {
733        // Look for patterns like: account.is_signer or !account.is_signer
734        // This is a simplified analysis - in practice, we'd need more sophisticated checking
735        call.args.iter().any(|arg| {
736            if let Expr::Path(path) = arg {
737                path_to_string(&path.path).contains("is_signer")
738            } else {
739                false
740            }
741        })
742    }
743
744    /// Extract PDA seeds from find_program_address or create_program_address calls
745    fn extract_pda_seeds(&self, call: &syn::ExprCall) -> Vec<String> {
746        let mut seeds = Vec::new();
747
748        // Look for seed arrays in the arguments
749        for arg in &call.args {
750            if let Expr::Array(array) = arg {
751                for elem in &array.elems {
752                    if let Expr::Lit(syn::ExprLit {
753                        lit: syn::Lit::Str(lit_str),
754                        ..
755                    }) = elem
756                    {
757                        seeds.push(lit_str.value());
758                    } else if let Expr::Path(path) = elem {
759                        seeds.push(path_to_string(&path.path));
760                    }
761                }
762            }
763        }
764
765        seeds
766    }
767
768    /// Analyze if program ID is properly checked before CPI calls
769    fn analyze_program_id_check(&self, call: &syn::ExprCall) -> bool {
770        // Look for program ID verification patterns
771        if let Expr::Path(path) = &*call.func {
772            let path_str = path_to_string(&path.path);
773
774            // If it's an invoke call, look for program ID validation in the arguments
775            if path_str.contains("invoke") {
776                // Check if any argument contains program ID validation patterns
777                let has_program_id_validation = call
778                    .args
779                    .iter()
780                    .any(|arg| self.contains_program_id_validation(arg));
781
782                // Also check for common validation patterns in the call itself
783                let has_validation_call = path_str.contains("verify_program")
784                    || path_str.contains("check_program")
785                    || path_str.contains("validate_program");
786
787                return has_program_id_validation || has_validation_call;
788            }
789        }
790
791        // For non-invoke calls, assume they don't need program ID validation
792        true
793    }
794
795    /// Check if an expression contains program ID validation patterns
796    fn contains_program_id_validation(&self, expr: &Expr) -> bool {
797        match expr {
798            // Check for binary comparisons involving program_id
799            Expr::Binary(binary) => {
800                if matches!(binary.op, syn::BinOp::Eq(_) | syn::BinOp::Ne(_)) {
801                    let left_str = self.expr_to_string(&binary.left);
802                    let right_str = self.expr_to_string(&binary.right);
803
804                    (left_str.contains("program_id") || left_str.contains("program.key"))
805                        || (right_str.contains("program_id") || right_str.contains("program.key"))
806                } else {
807                    false
808                }
809            }
810            // Check for method calls that might validate program ID
811            Expr::MethodCall(method) => {
812                let method_name = method.method.to_string();
813                method_name.contains("verify")
814                    || method_name.contains("check")
815                    || method_name.contains("validate")
816                    || method_name == "key"
817            }
818            // Check for paths that reference program validation
819            Expr::Path(path) => {
820                let path_str = path_to_string(&path.path);
821                path_str.contains("program_id") || path_str.contains("PROGRAM_ID")
822            }
823            _ => false,
824        }
825    }
826
827    /// Convert expression to string for analysis
828    fn expr_to_string(&self, expr: &Expr) -> String {
829        match expr {
830            Expr::Path(path) => path_to_string(&path.path),
831            Expr::Field(field) => {
832                format!(
833                    "{}.{}",
834                    self.expr_to_string(&field.base),
835                    field.member.to_token_stream()
836                )
837            }
838            Expr::MethodCall(method) => {
839                format!(
840                    "{}.{}",
841                    self.expr_to_string(&method.receiver),
842                    method.method
843                )
844            }
845            _ => String::new(),
846        }
847    }
848
849    /// Analyze if account owner is properly verified
850    fn analyze_owner_check(&self, call: &syn::ExprCall) -> bool {
851        // Look for owner verification patterns like: account.owner == expected_program_id
852        call.args
853            .iter()
854            .any(|arg| self.contains_owner_validation(arg))
855            || self.function_call_validates_owner(call)
856    }
857
858    /// Check if an expression contains owner validation patterns
859    fn contains_owner_validation(&self, expr: &Expr) -> bool {
860        match expr {
861            // Check for binary comparisons involving owner
862            Expr::Binary(binary) => {
863                if matches!(binary.op, syn::BinOp::Eq(_) | syn::BinOp::Ne(_)) {
864                    let left_str = self.expr_to_string(&binary.left);
865                    let right_str = self.expr_to_string(&binary.right);
866
867                    // Check if either side references account owner
868                    (left_str.contains("owner")
869                        && (right_str.contains("program") || right_str.contains("PROGRAM_ID")))
870                        || (right_str.contains("owner")
871                            && (left_str.contains("program") || left_str.contains("PROGRAM_ID")))
872                        || (left_str.contains(".owner") || right_str.contains(".owner"))
873                } else {
874                    false
875                }
876            }
877            // Check for method calls that validate ownership
878            Expr::MethodCall(method) => {
879                let method_name = method.method.to_string();
880                let receiver_str = self.expr_to_string(&method.receiver);
881
882                (method_name.contains("owner") || method_name.contains("check_owner"))
883                    || (receiver_str.contains("owner")
884                        && (method_name == "eq" || method_name == "ne"))
885            }
886            // Recursively check nested expressions
887            Expr::Group(group) => self.contains_owner_validation(&group.expr),
888            Expr::Paren(paren) => self.contains_owner_validation(&paren.expr),
889            _ => false,
890        }
891    }
892
893    /// Check if the function call itself validates owner
894    fn function_call_validates_owner(&self, call: &syn::ExprCall) -> bool {
895        if let Expr::Path(path) = &*call.func {
896            let path_str = path_to_string(&path.path);
897            path_str.contains("check_owner")
898                || path_str.contains("verify_owner")
899                || path_str.contains("validate_owner")
900        } else {
901            false
902        }
903    }
904
905    /// Analyze if account data validation is performed
906    fn analyze_account_data_validation(&self, call: &syn::ExprCall) -> bool {
907        // Look for account data validation patterns
908        call.args.iter().any(|arg| {
909            if let Expr::Path(path) = arg {
910                let path_str = path_to_string(&path.path);
911                path_str.contains("data")
912                    || path_str.contains("try_from")
913                    || path_str.contains("deserialize")
914            } else {
915                false
916            }
917        })
918    }
919}

921/// Helper function to convert a Path to a string
922fn path_to_string(path: &Path) -> String {
923    path.segments
924        .iter()
925        .map(|seg| seg.ident.to_string())
926        .collect::<Vec<_>>()
927        .join("::")
928}

930/// Check if an expression involves dynamic/runtime values
931fn is_dynamic_expr(expr: &Expr) -> bool {
932    match expr {
933        Expr::Lit(_) => false,       // Literal values are static
934        Expr::Path(_) => true,       // Variables are dynamic
935        Expr::Call(_) => true,       // Function calls are dynamic
936        Expr::MethodCall(_) => true, // Method calls are dynamic
937        Expr::Macro(_) => true,      // Macros could be dynamic
938        _ => true,                   // Conservative assumption for other expressions
939    }
