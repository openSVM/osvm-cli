27    /// Create a new SSH client
28    ///
29    /// # Arguments
30    /// * `config` - Server configuration
31    ///
32    /// # Returns
33    /// * `Result<SshClient, DeploymentError>` - New SSH client
34    pub fn new(config: ServerConfig) -> Result<Self, DeploymentError> {
35        // Validate and sanitize host input
36        Self::validate_host(&config.host)?;
37        Self::validate_port(config.port)?;
38
39        // Set connection timeout to prevent hanging
40        let tcp = std::net::TcpStream::connect_timeout(
41            &format!("{}:{}", config.host, config.port)
42                .parse()
43                .map_err(|e| {
44                    DeploymentError::ValidationError(format!("Invalid address format: {}", e))
45                })?,
46            std::time::Duration::from_secs(30),
47        )
48        .map_err(|e| DeploymentError::ConnectionError(format!("Failed to connect: {}", e)))?;
49
50        let mut session = Session::new().map_err(|e| {
51            DeploymentError::ConnectionError(format!("Failed to create session: {}", e))
52        })?;
53
54        session.set_tcp_stream(tcp);
55        session.handshake().map_err(|e| {
56            DeploymentError::ConnectionError(format!("SSH handshake failed: {}", e))
57        })?;
58
59        Ok(SshClient {
60            session,
61            config: config.clone(),
62            connected: false,
63        })
64    }

66    /// Connect to the server
67    ///
68    /// # Returns
69    /// * `Result<(), DeploymentError>` - Success/failure
70    pub fn connect(&mut self) -> Result<(), DeploymentError> {
71        if self.connected {
72            return Ok(());
73        }
74
75        match &self.config.auth {
76            AuthMethod::Password { username, password } => {
77                self.session
78                    .userauth_password(username, password)
79                    .map_err(|e| {
80                        DeploymentError::AuthError(format!("Password authentication failed: {}", e))
81                    })?;
82            }
83            AuthMethod::Key {
84                username,
85                key_path,
86                passphrase,
87            } => {
88                let mut key_file = fs::File::open(key_path).map_err(|e| {
89                    DeploymentError::AuthError(format!(
90                        "Failed to open key file {}: {}",
91                        key_path, e
92                    ))
93                })?;
94
95                let mut key_contents = String::new();
96                key_file.read_to_string(&mut key_contents).map_err(|e| {
97                    DeploymentError::AuthError(format!("Failed to read key file: {}", e))
98                })?;
99
100                self.session
101                    .userauth_pubkey_memory(username, None, &key_contents, passphrase.as_deref())
102                    .map_err(|e| {
103                        DeploymentError::AuthError(format!("Key authentication failed: {}", e))
104                    })?;
105            }
106        }
107
108        self.connected = true;
109        Ok(())
110    }

112    /// Execute a command on the remote server
113    ///
114    /// # Arguments
115    /// * `command` - Command to execute
116    ///
117    /// # Returns
118    /// * `Result<String, DeploymentError>` - Command output
119    pub fn execute_command(&mut self, command: &str) -> Result<String, DeploymentError> {
120        if !self.connected {
121            self.connect()?;
122        }
123
124        let mut channel = self.session.channel_session().map_err(|e| {
125            DeploymentError::CommandError(format!("Failed to create channel: {}", e))
126        })?;
127
128        channel.exec(command).map_err(|e| {
129            DeploymentError::CommandError(format!("Failed to execute command: {}", e))
130        })?;
131
132        let mut output = String::new();
133        channel.read_to_string(&mut output).map_err(|e| {
134            DeploymentError::CommandError(format!("Failed to read command output: {}", e))
135        })?;
136
137        channel.wait_close().map_err(|e| {
138            DeploymentError::CommandError(format!("Failed to close channel: {}", e))
139        })?;
140
141        let exit_status = channel.exit_status().map_err(|e| {
142            DeploymentError::CommandError(format!("Failed to get exit status: {}", e))
143        })?;
144
145        if exit_status != 0 {
146            return Err(DeploymentError::CommandError(format!(
147                "Command exited with status {}: {}\nCommand: {}",
148                exit_status, output, command
149            )));
150        }
151
152        Ok(output)
153    }

155    /// Upload a file to the remote server (part 1: setup)
156    ///
157    /// # Arguments
158    /// * `local_path` - Local file path
159    /// * `remote_path` - Remote file path
160    ///
161    /// # Returns
162    /// * `Result<(), DeploymentError>` - Success/failure
163    pub fn upload_file<P: AsRef<Path>, Q: AsRef<Path>>(
164        &mut self,
165        local_path: P,
166        remote_path: Q,
167    ) -> Result<(), DeploymentError> {
168        if !self.connected {
169            self.connect()?;
170        }
171
172        let local_path = local_path.as_ref();
173        let remote_path = remote_path.as_ref();
174
175        let mut local_file = fs::File::open(local_path).map_err(|e| {
176            DeploymentError::FileTransferError(format!(
177                "Failed to open local file {}: {}",
178                local_path.display(),
179                e
180            ))
181        })?;
182
183        let file_size = local_file
184            .metadata()
185            .map_err(|e| {
186                DeploymentError::FileTransferError(format!("Failed to get file metadata: {}", e))
187            })?
188            .len();
189
190        let mut remote_file = self
191            .session
192            .scp_send(remote_path, 0o644, file_size, None)
193            .map_err(|e| {
194                DeploymentError::FileTransferError(format!(
195                    "Failed to initiate SCP transfer to {}: {}",
196                    remote_path.display(),
197                    e
198                ))
199            })?;
200
201        self.upload_file_content(&mut local_file, &mut remote_file)
202    }

204    /// Upload file content (part 2: content transfer)
205    ///
206    /// # Arguments
207    /// * `local_file` - Local file handle
208    /// * `remote_file` - Remote file handle
209    ///
210    /// # Returns
211    /// * `Result<(), DeploymentError>` - Success/failure
212    fn upload_file_content(
213        &mut self,
214        local_file: &mut fs::File,
215        remote_file: &mut ssh2::Channel,
216    ) -> Result<(), DeploymentError> {
217        // Create buffer and copy file contents
218        let mut buffer = [0; 16384];
219        loop {
220            match local_file.read(&mut buffer) {
221                Ok(0) => break, // EOF
222                Ok(n) => {
223                    remote_file.write_all(&buffer[..n]).map_err(|e| {
224                        DeploymentError::FileTransferError(format!(
225                            "Failed to write data to remote file: {}",
226                            e
227                        ))
228                    })?;
229                }
230                Err(e) => {
231                    return Err(DeploymentError::FileTransferError(format!(
232                        "Failed to read from local file: {}",
233                        e
234                    )));
235                }
236            }
237        }
238
239        remote_file.send_eof().map_err(|e| {
240            DeploymentError::FileTransferError(format!("Failed to signal EOF: {}", e))
241        })?;
242
243        remote_file.wait_eof().map_err(|e| {
244            DeploymentError::FileTransferError(format!(
245                "Failed to wait for EOF confirmation: {}",
246                e
247            ))
248        })?;
249
250        remote_file.close().map_err(|e| {
251            DeploymentError::FileTransferError(format!("Failed to close remote file: {}", e))
252        })?;
253
254        remote_file.wait_close().map_err(|e| {
255            DeploymentError::FileTransferError(format!(
256                "Failed to wait for confirmation of file close: {}",
257                e
258            ))
259        })?;
260
261        Ok(())
262    }

306    /// Check if a package is installed on the remote server
307    ///
308    /// # Arguments
309    /// * `package` - Package name
310    ///
311    /// # Returns
312    /// * `Result<bool, DeploymentError>` - Whether the package is installed
313    pub fn is_package_installed(&mut self, package: &str) -> Result<bool, DeploymentError> {
314        let cmd = format!(
315            "if command -v dpkg >/dev/null 2>&1; then \
316             dpkg -l | grep -q '{}' && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
317             elif command -v rpm >/dev/null 2>&1; then \
318             rpm -q '{}' >/dev/null 2>&1 && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
319             else command -v '{}' >/dev/null 2>&1 && \
320             echo 'INSTALLED' || echo 'NOT_INSTALLED'; fi",
321            package, package, package
322        );
323
324        let output = self.execute_command(&cmd)?;
325        Ok(output.trim() == "INSTALLED")
326    }

328    /// Get system information from the remote server
329    ///
330    /// # Returns
331    /// * `Result<std::collections::HashMap<String, String>, DeploymentError>` - System information
332    pub fn get_system_info(
333        &mut self,
334    ) -> Result<std::collections::HashMap<String, String>, DeploymentError> {
335        let mut info = std::collections::HashMap::new();
336
337        // Get OS information
338        self.get_os_info(&mut info)?;
339
340        // Get CPU information
341        let cpu_info = self.execute_command("cat /proc/cpuinfo | grep -c processor")?;
342        info.insert("cpu_cores".to_string(), cpu_info.trim().to_string());
343
344        // Get memory information
345        let mem_info = self.execute_command("free -g | grep Mem | awk '{print $2}'")?;
346        info.insert("memory_gb".to_string(), mem_info.trim().to_string());
347
348        // Get disk and kernel information
349        self.get_disk_and_kernel_info(&mut info)?;
350
351        Ok(info)
352    }

354    /// Get OS information
355    ///
356    /// # Arguments
357    /// * `info` - Information map to populate
358    ///
359    /// # Returns
360    /// * `Result<(), DeploymentError>` - Success/failure
361    fn get_os_info(
362        &mut self,
363        info: &mut std::collections::HashMap<String, String>,
364    ) -> Result<(), DeploymentError> {
365        let os_info = self.execute_command(
366            "cat /etc/os-release | grep -E '^(NAME|VERSION_ID)' | sed 's/.*=//' | tr -d '\"'",
367        )?;
368        let os_lines: Vec<&str> = os_info.trim().split('\n').collect();
369
370        if os_lines.len() >= 2 {
371            info.insert("os_name".to_string(), os_lines[0].to_string());
372            info.insert("os_version".to_string(), os_lines[1].to_string());
373        }
374
375        Ok(())
376    }

378    /// Get disk and kernel information
379    ///
380    /// # Arguments
381    /// * `info` - Information map to populate
382    ///
383    /// # Returns
384    /// * `Result<(), DeploymentError>` - Success/failure
385    fn get_disk_and_kernel_info(
386        &mut self,
387        info: &mut std::collections::HashMap<String, String>,
388    ) -> Result<(), DeploymentError> {
389        // Get disk information
390        let disk_info = self.execute_command("df -h / | tail -1 | awk '{print $2,$4}'")?;
391        let disk_parts: Vec<&str> = disk_info.split_whitespace().collect();
392
393        if disk_parts.len() >= 2 {
394            info.insert("disk_total".to_string(), disk_parts[0].to_string());
395            info.insert("disk_available".to_string(), disk_parts[1].to_string());
396        }
397
398        // Get kernel information
399        let kernel_info = self.execute_command("uname -r")?;
400        info.insert("kernel".to_string(), kernel_info.trim().to_string());
401
402        Ok(())
403    }

405    /// Validate host input to prevent injection attacks
406    fn validate_host(host: &str) -> Result<(), DeploymentError> {
407        // Check for empty or whitespace-only host
408        if host.trim().is_empty() {
409            return Err(DeploymentError::ValidationError(
410                "Host cannot be empty".to_string(),
411            ));
412        }
413
414        // Check length to prevent resource exhaustion
415        if host.len() > 253 {
416            return Err(DeploymentError::ValidationError(
417                "Host name too long (max 253 characters)".to_string(),
418            ));
419        }
420
421        // Check for dangerous characters that could enable injection
422        for c in host.chars() {
423            if matches!(
424                c,
425                ';' | '&'
426                    | '|'
427                    | '`'
428                    | '$'
429                    | '('
430                    | ')'
431                    | '{'
432                    | '}'
433                    | '<'
434                    | '>'
435                    | '\n'
436                    | '\r'
437                    | '\t'
438            ) {
439                return Err(DeploymentError::ValidationError(format!(
440                    "Invalid character in host: '{}'",
441                    c
442                )));
443            }
444        }
445
446        // Basic format validation (IP address or domain name)
447        if host.parse::<std::net::IpAddr>().is_err() {
448            // If not a valid IP, check if it's a reasonable domain name
449            if !host
450                .chars()
451                .all(|c| c.is_alphanumeric() || matches!(c, '.' | '-' | '_'))
452            {
453                return Err(DeploymentError::ValidationError(
454                    "Invalid host format".to_string(),
455                ));
456            }
457        }
458
459        Ok(())
460    }

462    /// Validate port number
463    fn validate_port(port: u16) -> Result<(), DeploymentError> {
464        // Check for privileged ports (might require root)
465        if port < 22 && port != 22 {
466            return Err(DeploymentError::ValidationError(
467                "Port number too low (use 22 or higher)".to_string(),
468            ));
469        }
470
471        // Check for valid port range
472        if port == 0 {
473            return Err(DeploymentError::ValidationError(
474                "Port number cannot be 0".to_string(),
475            ));
476        }
477
478        Ok(())
479    }
