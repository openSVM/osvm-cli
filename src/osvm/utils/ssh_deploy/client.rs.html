39    /// Create a new SSH client
40    ///
41    /// # Arguments
42    /// * `config` - Server configuration
43    ///
44    /// # Returns
45    /// * `Result<SshClient, DeploymentError>` - New SSH client
46    pub fn new(config: ServerConfig) -> Result<Self, DeploymentError> {
47        // Validate and sanitize host input
48        Self::validate_host(&config.host)?;
49        Self::validate_port(config.port)?;
50
51        // Set connection timeout to prevent hanging (configurable, defaults to 30 seconds)
52        let tcp = std::net::TcpStream::connect_timeout(
53            &format!("{}:{}", config.host, config.port)
54                .parse()
55                .map_err(|e| {
56                    DeploymentError::ValidationError(format!("Invalid address format: {}", e))
57                })?,
58            std::time::Duration::from_secs(config.connection_timeout_secs),
59        )
60        .map_err(|e| DeploymentError::ConnectionError(format!("Failed to connect: {}", e)))?;
61
62        let mut session = Session::new().map_err(|e| {
63            DeploymentError::ConnectionError(format!("Failed to create session: {}", e))
64        })?;
65
66        session.set_tcp_stream(tcp);
67        session.handshake().map_err(|e| {
68            DeploymentError::ConnectionError(format!("SSH handshake failed: {}", e))
69        })?;
70
71        Ok(SshClient {
72            session,
73            config,  // Move ownership directly (no clone needed)
74            connected: false,
75        })
76    }

78    /// Connect to the server
79    ///
80    /// # Returns
81    /// * `Result<(), DeploymentError>` - Success/failure
82    pub fn connect(&mut self) -> Result<(), DeploymentError> {
83        if self.connected {
84            return Ok(());
85        }
86
87        match &self.config.auth {
88            AuthMethod::Password { username, password } => {
89                self.session
90                    .userauth_password(username, password)
91                    .map_err(|e| {
92                        DeploymentError::AuthError(format!("Password authentication failed: {}", e))
93                    })?;
94            }
95            AuthMethod::Key {
96                username,
97                key_path,
98                passphrase,
99            } => {
100                let mut key_file = fs::File::open(key_path).map_err(|e| {
101                    DeploymentError::AuthError(format!(
102                        "Failed to open key file {}: {}",
103                        key_path, e
104                    ))
105                })?;
106
107                let mut key_contents = String::new();
108                key_file.read_to_string(&mut key_contents).map_err(|e| {
109                    DeploymentError::AuthError(format!("Failed to read key file: {}", e))
110                })?;
111
112                self.session
113                    .userauth_pubkey_memory(username, None, &key_contents, passphrase.as_deref())
114                    .map_err(|e| {
115                        DeploymentError::AuthError(format!("Key authentication failed: {}", e))
116                    })?;
117            }
118        }
119
120        self.connected = true;
121        Ok(())
122    }

124    /// Execute a command on the remote server
125    ///
126    /// # Arguments
127    /// * `command` - Command to execute
128    ///
129    /// # Returns
130    /// * `Result<String, DeploymentError>` - Command output
131    pub fn execute_command(&mut self, command: &str) -> Result<String, DeploymentError> {
132        if !self.connected {
133            self.connect()?;
134        }
135
136        let mut channel = self.session.channel_session().map_err(|e| {
137            DeploymentError::CommandError(format!("Failed to create channel: {}", e))
138        })?;
139
140        channel.exec(command).map_err(|e| {
141            DeploymentError::CommandError(format!("Failed to execute command: {}", e))
142        })?;
143
144        let mut output = String::new();
145        channel.read_to_string(&mut output).map_err(|e| {
146            DeploymentError::CommandError(format!("Failed to read command output: {}", e))
147        })?;
148
149        channel.wait_close().map_err(|e| {
150            DeploymentError::CommandError(format!("Failed to close channel: {}", e))
151        })?;
152
153        let exit_status = channel.exit_status().map_err(|e| {
154            DeploymentError::CommandError(format!("Failed to get exit status: {}", e))
155        })?;
156
157        if exit_status != 0 {
158            return Err(DeploymentError::CommandError(format!(
159                "Command exited with status {}: {}\nCommand: {}",
160                exit_status, output, command
161            )));
162        }
163
164        Ok(output)
165    }

167    /// Upload a file to the remote server (part 1: setup)
168    ///
169    /// # Arguments
170    /// * `local_path` - Local file path
171    /// * `remote_path` - Remote file path
172    ///
173    /// # Returns
174    /// * `Result<(), DeploymentError>` - Success/failure
175    pub fn upload_file<P: AsRef<Path>, Q: AsRef<Path>>(
176        &mut self,
177        local_path: P,
178        remote_path: Q,
179    ) -> Result<(), DeploymentError> {
180        if !self.connected {
181            self.connect()?;
182        }
183
184        let local_path = local_path.as_ref();
185        let remote_path = remote_path.as_ref();
186
187        let mut local_file = fs::File::open(local_path).map_err(|e| {
188            DeploymentError::FileTransferError(format!(
189                "Failed to open local file {}: {}",
190                local_path.display(),
191                e
192            ))
193        })?;
194
195        let file_size = local_file
196            .metadata()
197            .map_err(|e| {
198                DeploymentError::FileTransferError(format!("Failed to get file metadata: {}", e))
199            })?
200            .len();
201
202        let mut remote_file = self
203            .session
204            .scp_send(remote_path, 0o644, file_size, None)
205            .map_err(|e| {
206                DeploymentError::FileTransferError(format!(
207                    "Failed to initiate SCP transfer to {}: {}",
208                    remote_path.display(),
209                    e
210                ))
211            })?;
212
213        self.upload_file_content(&mut local_file, &mut remote_file)
214    }

216    /// Upload file content (part 2: content transfer)
217    ///
218    /// # Arguments
219    /// * `local_file` - Local file handle
220    /// * `remote_file` - Remote file handle
221    ///
222    /// # Returns
223    /// * `Result<(), DeploymentError>` - Success/failure
224    fn upload_file_content(
225        &mut self,
226        local_file: &mut fs::File,
227        remote_file: &mut ssh2::Channel,
228    ) -> Result<(), DeploymentError> {
229        // Create buffer and copy file contents
230        let mut buffer = [0; 16384];
231        loop {
232            match local_file.read(&mut buffer) {
233                Ok(0) => break, // EOF
234                Ok(n) => {
235                    remote_file.write_all(&buffer[..n]).map_err(|e| {
236                        DeploymentError::FileTransferError(format!(
237                            "Failed to write data to remote file: {}",
238                            e
239                        ))
240                    })?;
241                }
242                Err(e) => {
243                    return Err(DeploymentError::FileTransferError(format!(
244                        "Failed to read from local file: {}",
245                        e
246                    )));
247                }
248            }
249        }
250
251        remote_file.send_eof().map_err(|e| {
252            DeploymentError::FileTransferError(format!("Failed to signal EOF: {}", e))
253        })?;
254
255        remote_file.wait_eof().map_err(|e| {
256            DeploymentError::FileTransferError(format!(
257                "Failed to wait for EOF confirmation: {}",
258                e
259            ))
260        })?;
261
262        remote_file.close().map_err(|e| {
263            DeploymentError::FileTransferError(format!("Failed to close remote file: {}", e))
264        })?;
265
266        remote_file.wait_close().map_err(|e| {
267            DeploymentError::FileTransferError(format!(
268                "Failed to wait for confirmation of file close: {}",
269                e
270            ))
271        })?;
272
273        Ok(())
274    }

289    /// Check if a file exists on the remote server
290    ///
291    /// # Arguments
292    /// * `path` - File path
293    ///
294    /// # Returns
295    /// * `Result<bool, DeploymentError>` - Whether the file exists
296    ///
297    /// # Note: Race Condition
298    /// This function performs a Time-of-Check-Time-of-Use (TOCTOU) race condition check:
299    /// - Between the time `file_exists()` returns true and the caller uses the file,
300    ///   another process could delete it
301    /// - Similarly, if `file_exists()` returns false, the file could be created before use
302    ///
303    /// For safety-critical operations, always handle ENOENT errors when reading/writing
304    /// files after calling this function, or use atomic operations (e.g., O_CREAT|O_EXCL).
305    pub fn file_exists(&mut self, path: &str) -> Result<bool, DeploymentError> {
306        let escaped_path = Self::escape_arg(path);
307        let output = self.execute_command(&format!(
308            "test -f {} && echo 'EXISTS' || echo 'NOT_EXISTS'",
309            escaped_path
310        ))?;
311        Ok(output.trim() == "EXISTS")
312    }

339    /// Check if a package is installed on the remote server
340    ///
341    /// # Arguments
342    /// * `package` - Package name
343    ///
344    /// # Returns
345    /// * `Result<bool, DeploymentError>` - Whether the package is installed
346    pub fn is_package_installed(&mut self, package: &str) -> Result<bool, DeploymentError> {
347        let escaped_package = Self::escape_arg(package);
348        let cmd = format!(
349            "if command -v dpkg >/dev/null 2>&1; then \
350             dpkg -l | grep -q {} && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
351             elif command -v rpm >/dev/null 2>&1; then \
352             rpm -q {} >/dev/null 2>&1 && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
353             else command -v {} >/dev/null 2>&1 && \
354             echo 'INSTALLED' || echo 'NOT_INSTALLED'; fi",
355            escaped_package, escaped_package, escaped_package
356        );
357
358        let output = self.execute_command(&cmd)?;
359        Ok(output.trim() == "INSTALLED")
360    }

362    /// Get system information from the remote server
363    ///
364    /// # Returns
365    /// * `Result<std::collections::HashMap<String, String>, DeploymentError>` - System information
366    pub fn get_system_info(
367        &mut self,
368    ) -> Result<std::collections::HashMap<String, String>, DeploymentError> {
369        let mut info = std::collections::HashMap::new();
370
371        // Get OS information
372        self.get_os_info(&mut info)?;
373
374        // Get CPU information
375        let cpu_info = self.execute_command("cat /proc/cpuinfo | grep -c processor")?;
376        info.insert("cpu_cores".to_string(), cpu_info.trim().to_string());
377
378        // Get memory information
379        let mem_info = self.execute_command("free -g | grep Mem | awk '{print $2}'")?;
380        info.insert("memory_gb".to_string(), mem_info.trim().to_string());
381
382        // Get disk and kernel information
383        self.get_disk_and_kernel_info(&mut info)?;
384
385        Ok(info)
386    }

388    /// Get OS information
389    ///
390    /// # Arguments
391    /// * `info` - Information map to populate
392    ///
393    /// # Returns
394    /// * `Result<(), DeploymentError>` - Success/failure
395    fn get_os_info(
396        &mut self,
397        info: &mut std::collections::HashMap<String, String>,
398    ) -> Result<(), DeploymentError> {
399        let os_info = self.execute_command(
400            "cat /etc/os-release | grep -E '^(NAME|VERSION_ID)' | sed 's/.*=//' | tr -d '\"'",
401        )?;
402        let os_lines: Vec<&str> = os_info.trim().split('\n').collect();
403
404        if os_lines.len() >= 2 {
405            info.insert("os_name".to_string(), os_lines[0].to_string());
406            info.insert("os_version".to_string(), os_lines[1].to_string());
407        }
408
409        Ok(())
410    }

412    /// Get disk and kernel information
413    ///
414    /// # Arguments
415    /// * `info` - Information map to populate
416    ///
417    /// # Returns
418    /// * `Result<(), DeploymentError>` - Success/failure
419    fn get_disk_and_kernel_info(
420        &mut self,
421        info: &mut std::collections::HashMap<String, String>,
422    ) -> Result<(), DeploymentError> {
423        // Get disk information
424        let disk_info = self.execute_command("df -h / | tail -1 | awk '{print $2,$4}'")?;
425        let disk_parts: Vec<&str> = disk_info.split_whitespace().collect();
426
427        if disk_parts.len() >= 2 {
428            info.insert("disk_total".to_string(), disk_parts[0].to_string());
429            info.insert("disk_available".to_string(), disk_parts[1].to_string());
430        }
431
432        // Get kernel information
433        let kernel_info = self.execute_command("uname -r")?;
434        info.insert("kernel".to_string(), kernel_info.trim().to_string());
435
436        Ok(())
437    }

439    /// Validate host input to prevent injection attacks
440    fn validate_host(host: &str) -> Result<(), DeploymentError> {
441        // Check for empty or whitespace-only host
442        if host.trim().is_empty() {
443            return Err(DeploymentError::ValidationError(
444                "Host cannot be empty".to_string(),
445            ));
446        }
447
448        // Check length to prevent resource exhaustion
449        if host.len() > 253 {
450            return Err(DeploymentError::ValidationError(
451                "Host name too long (max 253 characters)".to_string(),
452            ));
453        }
454
455        // Check for dangerous characters that could enable injection
456        for c in host.chars() {
457            if matches!(
458                c,
459                ';' | '&'
460                    | '|'
461                    | '`'
462                    | '$'
463                    | '('
464                    | ')'
465                    | '{'
466                    | '}'
467                    | '<'
468                    | '>'
469                    | '\n'
470                    | '\r'
471                    | '\t'
472            ) {
473                return Err(DeploymentError::ValidationError(format!(
474                    "Invalid character in host: '{}'",
475                    c
476                )));
477            }
478        }
479
480        // Basic format validation (IP address or domain name)
481        if host.parse::<std::net::IpAddr>().is_err() {
482            // If not a valid IP, check if it's a reasonable domain name
483            if !host
484                .chars()
485                .all(|c| c.is_alphanumeric() || matches!(c, '.' | '-' | '_'))
486            {
487                return Err(DeploymentError::ValidationError(
488                    "Invalid host format".to_string(),
489                ));
490            }
491        }
492
493        Ok(())
494    }

496    /// Validate port number
497    fn validate_port(port: u16) -> Result<(), DeploymentError> {
498        // Check for privileged ports (might require root)
499        if port < 22 && port != 22 {
500            return Err(DeploymentError::ValidationError(
501                "Port number too low (use 22 or higher)".to_string(),
502            ));
503        }
504
505        // Check for valid port range
506        if port == 0 {
507            return Err(DeploymentError::ValidationError(
508                "Port number cannot be 0".to_string(),
509            ));
510        }
511
512        Ok(())
513    }
