16/// SSH client for interacting with remote servers
17pub struct SshClient {
18    /// SSH session
19    session: Session,
20    /// Server config
21    config: ServerConfig,
22    /// Whether the session is connected

26impl SshClient {
27    /// Create a new SSH client
28    ///
29    /// # Arguments
30    /// * `config` - Server configuration
31    ///
32    /// # Returns
33    /// * `Result<SshClient, DeploymentError>` - New SSH client
34    pub fn new(config: ServerConfig) -> Result<Self, DeploymentError> {
35        let tcp = std::net::TcpStream::connect(format!("{}:{}", config.host, config.port))
36            .map_err(|e| DeploymentError::ConnectionError(format!("Failed to connect: {}", e)))?;
37
38        let mut session = Session::new().map_err(|e| {
39            DeploymentError::ConnectionError(format!("Failed to create session: {}", e))
40        })?;
41
42        session.set_tcp_stream(tcp);
43        session.handshake().map_err(|e| {
44            DeploymentError::ConnectionError(format!("SSH handshake failed: {}", e))
45        })?;
46
47        Ok(SshClient {
48            session,
49            config: config.clone(),
50            connected: false,
51        })
52    }

54    /// Connect to the server
55    ///
56    /// # Returns
57    /// * `Result<(), DeploymentError>` - Success/failure
58    pub fn connect(&mut self) -> Result<(), DeploymentError> {
59        if self.connected {
60            return Ok(());
61        }
62
63        match &self.config.auth {
64            AuthMethod::Password { username, password } => {
65                self.session
66                    .userauth_password(username, password)
67                    .map_err(|e| {
68                        DeploymentError::AuthError(format!("Password authentication failed: {}", e))
69                    })?;
70            }
71            AuthMethod::Key {
72                username,
73                key_path,
74                passphrase,
75            } => {
76                let mut key_file = fs::File::open(key_path).map_err(|e| {
77                    DeploymentError::AuthError(format!(
78                        "Failed to open key file {}: {}",
79                        key_path, e
80                    ))
81                })?;
82
83                let mut key_contents = String::new();
84                key_file.read_to_string(&mut key_contents).map_err(|e| {
85                    DeploymentError::AuthError(format!("Failed to read key file: {}", e))
86                })?;
87
88                self.session
89                    .userauth_pubkey_memory(username, None, &key_contents, passphrase.as_deref())
90                    .map_err(|e| {
91                        DeploymentError::AuthError(format!("Key authentication failed: {}", e))
92                    })?;
93            }
94        }
95
96        self.connected = true;
97        Ok(())
98    }

100    /// Execute a command on the remote server
101    ///
102    /// # Arguments
103    /// * `command` - Command to execute
104    ///
105    /// # Returns
106    /// * `Result<String, DeploymentError>` - Command output
107    pub fn execute_command(&mut self, command: &str) -> Result<String, DeploymentError> {
108        if !self.connected {
109            self.connect()?;
110        }
111
112        let mut channel = self.session.channel_session().map_err(|e| {
113            DeploymentError::CommandError(format!("Failed to create channel: {}", e))
114        })?;
115
116        channel.exec(command).map_err(|e| {
117            DeploymentError::CommandError(format!("Failed to execute command: {}", e))
118        })?;
119
120        let mut output = String::new();
121        channel.read_to_string(&mut output).map_err(|e| {
122            DeploymentError::CommandError(format!("Failed to read command output: {}", e))
123        })?;
124
125        channel.wait_close().map_err(|e| {
126            DeploymentError::CommandError(format!("Failed to close channel: {}", e))
127        })?;
128
129        let exit_status = channel.exit_status().map_err(|e| {
130            DeploymentError::CommandError(format!("Failed to get exit status: {}", e))
131        })?;
132
133        if exit_status != 0 {
134            return Err(DeploymentError::CommandError(format!(
135                "Command exited with status {}: {}\nCommand: {}",
136                exit_status, output, command
137            )));
138        }
139
140        Ok(output)
141    }

143    /// Upload a file to the remote server (part 1: setup)
144    ///
145    /// # Arguments
146    /// * `local_path` - Local file path
147    /// * `remote_path` - Remote file path
148    ///
149    /// # Returns
150    /// * `Result<(), DeploymentError>` - Success/failure
151    pub fn upload_file<P: AsRef<Path>, Q: AsRef<Path>>(
152        &mut self,
153        local_path: P,
154        remote_path: Q,
155    ) -> Result<(), DeploymentError> {
156        if !self.connected {
157            self.connect()?;
158        }
159
160        let local_path = local_path.as_ref();
161        let remote_path = remote_path.as_ref();
162
163        let mut local_file = fs::File::open(local_path).map_err(|e| {
164            DeploymentError::FileTransferError(format!(
165                "Failed to open local file {}: {}",
166                local_path.display(),
167                e
168            ))
169        })?;
170
171        let file_size = local_file
172            .metadata()
173            .map_err(|e| {
174                DeploymentError::FileTransferError(format!("Failed to get file metadata: {}", e))
175            })?
176            .len();
177
178        let mut remote_file = self
179            .session
180            .scp_send(remote_path, 0o644, file_size, None)
181            .map_err(|e| {
182                DeploymentError::FileTransferError(format!(
183                    "Failed to initiate SCP transfer to {}: {}",
184                    remote_path.display(),
185                    e
186                ))
187            })?;
188
189        self.upload_file_content(&mut local_file, &mut remote_file)
190    }

192    /// Upload file content (part 2: content transfer)
193    ///
194    /// # Arguments
195    /// * `local_file` - Local file handle
196    /// * `remote_file` - Remote file handle
197    ///
198    /// # Returns
199    /// * `Result<(), DeploymentError>` - Success/failure
200    fn upload_file_content(
201        &mut self,
202        local_file: &mut fs::File,
203        remote_file: &mut ssh2::Channel,
204    ) -> Result<(), DeploymentError> {
205        // Create buffer and copy file contents
206        let mut buffer = [0; 16384];
207        loop {
208            match local_file.read(&mut buffer) {
209                Ok(0) => break, // EOF
210                Ok(n) => {
211                    remote_file.write_all(&buffer[..n]).map_err(|e| {
212                        DeploymentError::FileTransferError(format!(
213                            "Failed to write data to remote file: {}",
214                            e
215                        ))
216                    })?;
217                }
218                Err(e) => {
219                    return Err(DeploymentError::FileTransferError(format!(
220                        "Failed to read from local file: {}",
221                        e
222                    )));
223                }
224            }
225        }
226
227        remote_file.send_eof().map_err(|e| {
228            DeploymentError::FileTransferError(format!("Failed to signal EOF: {}", e))
229        })?;
230
231        remote_file.wait_eof().map_err(|e| {
232            DeploymentError::FileTransferError(format!(
233                "Failed to wait for EOF confirmation: {}",
234                e
235            ))
236        })?;
237
238        remote_file.close().map_err(|e| {
239            DeploymentError::FileTransferError(format!("Failed to close remote file: {}", e))
240        })?;
241
242        remote_file.wait_close().map_err(|e| {
243            DeploymentError::FileTransferError(format!(
244                "Failed to wait for confirmation of file close: {}",
245                e
246            ))
247        })?;
248
249        Ok(())
250    }

272        let output = self.execute_command(&format!(

294    /// Check if a package is installed on the remote server
295    ///
296    /// # Arguments
297    /// * `package` - Package name
298    ///
299    /// # Returns
300    /// * `Result<bool, DeploymentError>` - Whether the package is installed
301    pub fn is_package_installed(&mut self, package: &str) -> Result<bool, DeploymentError> {
302        let cmd = format!(
303            "if command -v dpkg >/dev/null 2>&1; then \
304             dpkg -l | grep -q '{}' && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
305             elif command -v rpm >/dev/null 2>&1; then \
306             rpm -q '{}' >/dev/null 2>&1 && echo 'INSTALLED' || echo 'NOT_INSTALLED'; \
307             else command -v '{}' >/dev/null 2>&1 && \
308             echo 'INSTALLED' || echo 'NOT_INSTALLED'; fi",
309            package, package, package
310        );
311
312        let output = self.execute_command(&cmd)?;
313        Ok(output.trim() == "INSTALLED")
314    }

316    /// Get system information from the remote server
317    ///
318    /// # Returns
319    /// * `Result<std::collections::HashMap<String, String>, DeploymentError>` - System information
320    pub fn get_system_info(
321        &mut self,
322    ) -> Result<std::collections::HashMap<String, String>, DeploymentError> {
323        let mut info = std::collections::HashMap::new();
324
325        // Get OS information
326        self.get_os_info(&mut info)?;
327
328        // Get CPU information
329        let cpu_info = self.execute_command("cat /proc/cpuinfo | grep -c processor")?;
330        info.insert("cpu_cores".to_string(), cpu_info.trim().to_string());
331
332        // Get memory information
333        let mem_info = self.execute_command("free -g | grep Mem | awk '{print $2}'")?;
334        info.insert("memory_gb".to_string(), mem_info.trim().to_string());
335
336        // Get disk and kernel information
337        self.get_disk_and_kernel_info(&mut info)?;
338
339        Ok(info)
340    }

342    /// Get OS information
343    ///
344    /// # Arguments
345    /// * `info` - Information map to populate
346    ///
347    /// # Returns
348    /// * `Result<(), DeploymentError>` - Success/failure
349    fn get_os_info(
350        &mut self,
351        info: &mut std::collections::HashMap<String, String>,
352    ) -> Result<(), DeploymentError> {
353        let os_info = self.execute_command(
354            "cat /etc/os-release | grep -E '^(NAME|VERSION_ID)' | sed 's/.*=//' | tr -d '\"'",
355        )?;
356        let os_lines: Vec<&str> = os_info.trim().split('\n').collect();
357
358        if os_lines.len() >= 2 {
359            info.insert("os_name".to_string(), os_lines[0].to_string());
360            info.insert("os_version".to_string(), os_lines[1].to_string());
361        }
362
363        Ok(())
364    }

366    /// Get disk and kernel information
367    ///
368    /// # Arguments
369    /// * `info` - Information map to populate
370    ///
371    /// # Returns
372    /// * `Result<(), DeploymentError>` - Success/failure
373    fn get_disk_and_kernel_info(
374        &mut self,
375        info: &mut std::collections::HashMap<String, String>,
376    ) -> Result<(), DeploymentError> {
377        // Get disk information
378        let disk_info = self.execute_command("df -h / | tail -1 | awk '{print $2,$4}'")?;
379        let disk_parts: Vec<&str> = disk_info.split_whitespace().collect();
380
381        if disk_parts.len() >= 2 {
382            info.insert("disk_total".to_string(), disk_parts[0].to_string());
383            info.insert("disk_available".to_string(), disk_parts[1].to_string());
384        }
385
386        // Get kernel information
387        let kernel_info = self.execute_command("uname -r")?;
388        info.insert("kernel".to_string(), kernel_info.trim().to_string());
389
390        Ok(())
391    }
