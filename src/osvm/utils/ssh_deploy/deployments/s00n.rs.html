13/// Deploy s00n node
14///
15/// # Arguments
16/// * `client` - SSH client
17/// * `server_config` - Server configuration
18/// * `deployment_config` - Deployment configuration
19/// * `progress_callback` - Callback function for progress updates
20///
21/// # Returns
22/// * `Result<(), DeploymentError>` - Success/failure
23pub async fn deploy_s00n(
24    client: &mut SshClient,
25    server_config: &ServerConfig,
26    deployment_config: &DeploymentConfig,
27    progress_callback: Option<&crate::prelude::ProgressCallback>,
28) -> Result<(), DeploymentError> {
29    // Clone s00n repository
30    if let Some(callback) = progress_callback {
31        callback(40, "Cloning s00n repository");
32    }
33
34    let s00n_dir = format!("{}/s00n", server_config.install_dir);
35    clone_s00n_repository(client, &s00n_dir)?;
36
37    // Build s00n binary
38    if let Some(callback) = progress_callback {
39        callback(50, "Building s00n binary");
40    }
41
42    build_s00n_binary(client, &s00n_dir)?;
43
44    // Create configuration
45    if let Some(callback) = progress_callback {
46        callback(70, "Creating configuration");
47    }
48
49    create_s00n_configuration(client, &s00n_dir, deployment_config.network)?;
50
51    // Create systemd service
52    if let Some(callback) = progress_callback {
53        callback(80, "Creating systemd service");
54    }
55
56    create_s00n_service(client, deployment_config, &s00n_dir).await?;
57
58    Ok(())
59}

61/// Clone the s00n repository
62///
63/// # Arguments
64/// * `client` - SSH client
65/// * `s00n_dir` - s00n directory
66///
67/// # Returns
68/// * `Result<(), DeploymentError>` - Success/failure
69fn clone_s00n_repository(client: &mut SshClient, s00n_dir: &str) -> Result<(), DeploymentError> {
70    if !client.directory_exists(s00n_dir)? {
71        client.execute_command(&format!(
72            "git clone https://github.com/s00n-protocol/s00n-svm.git {}",
73            s00n_dir
74        ))?;
75    }
76
77    Ok(())
78}

80/// Build s00n binary
81///
82/// # Arguments
83/// * `client` - SSH client
84/// * `s00n_dir` - s00n directory
85///
86/// # Returns
87/// * `Result<(), DeploymentError>` - Success/failure
88fn build_s00n_binary(client: &mut SshClient, s00n_dir: &str) -> Result<(), DeploymentError> {
89    client.execute_command(&format!("cd {} && cargo build --release", s00n_dir))?;
90
91    Ok(())
92}

94/// Create s00n configuration
95///
96/// # Arguments
97/// * `client` - SSH client
98/// * `s00n_dir` - s00n directory
99/// * `network` - Network type
100///
101/// # Returns
102/// * `Result<(), DeploymentError>` - Success/failure
103fn create_s00n_configuration(
104    client: &mut SshClient,
105    s00n_dir: &str,
106    network: NetworkType,
107) -> Result<(), DeploymentError> {
108    let config_dir = format!("{}/config", s00n_dir);
109    client.create_directory(&config_dir)?;
110
111    let network_name = match network {
112        NetworkType::Mainnet => "mainnet",
113        NetworkType::Testnet => "testnet",
114        NetworkType::Devnet => "devnet",
115    };
116
117    client.execute_command(&format!(
118        "cd {} && ./target/release/s00n-node setup --network {}",
119        s00n_dir, network_name
120    ))?;
121
122    Ok(())
123}

125/// Create s00n service
126///
127/// # Arguments
128/// * `client` - SSH client
129/// * `deployment_config` - Deployment configuration
130/// * `s00n_dir` - s00n directory
131///
132/// # Returns
133/// * `Result<(), DeploymentError>` - Success/failure
134async fn create_s00n_service(
135    client: &mut SshClient,
136    deployment_config: &DeploymentConfig,
137    s00n_dir: &str,
138) -> Result<(), DeploymentError> {
139    let service_name = format!(
140        "s00n-{}-{}",
141        deployment_config.node_type, deployment_config.network
142    );
143
144    // Get service arguments
145    let args = get_s00n_service_args(deployment_config, s00n_dir);
146
147    // Create service content
148    let args_ref: Vec<&str> = args.iter().map(|s| s.as_str()).collect();
149    let service_content = create_binary_service_content(
150        &format!("{}/target/release/s00n-node", s00n_dir),
151        &args_ref,
152        s00n_dir,
153        "s00n SVM Node",
154    );
155
156    // Create and start the service
157    create_systemd_service(client, &service_name, &service_content)?;
158    enable_and_start_service(client, &service_name)?;
159
160    // Wait for the service to start
161    await_service_startup(client, &service_name).await?;
162
163    Ok(())
164}

166/// Get s00n service arguments
167///
168/// # Arguments
169/// * `deployment_config` - Deployment configuration
170/// * `s00n_dir` - s00n directory
171///
172/// # Returns
173/// * `Vec<String>` - Service arguments
174fn get_s00n_service_args(deployment_config: &DeploymentConfig, s00n_dir: &str) -> Vec<String> {
175    let mut args = vec![
176        "start".to_string(),
177        format!("--config-path {}/config/node.yaml", s00n_dir),
178        format!("--network {}", deployment_config.network),
179    ];
180
181    if deployment_config.node_type == "validator" {
182        args.push("--validator".to_string());
183        args.push(format!("--stake-key {}/config/stake-key.json", s00n_dir));
184    } else if deployment_config.node_type == "rpc" {
185        args.push("--rpc-port 8899".to_string());
186        args.push("--public-rpc".to_string());
187    }
188
189    args
190}
