5/// Set up disk storage for Solana validator/RPC
6///
7/// # Arguments
8/// * `client` - SSH client
9/// * `ledger_disk` - Ledger disk device path (e.g., "/dev/nvme0n1")
10/// * `accounts_disk` - Accounts disk device path (e.g., "/dev/nvme1n1")
11///
12/// # Returns
13/// * `Result<(), DeploymentError>` - Success/failure
14pub fn setup_disk_storage(
15    client: &mut SshClient,
16    ledger_disk: &str,
17    accounts_disk: &str,
18) -> Result<(), DeploymentError> {
19    // Create mount directories
20    create_mount_directories(client)?;
21
22    // Format and mount ledger disk
23    format_and_mount_disk(client, ledger_disk, "/mnt/ledger")?;
24
25    // Format and mount accounts disk
26    format_and_mount_disk(client, accounts_disk, "/mnt/extras")?;
27
28    // Create subdirectories for accounts and snapshots
29    client.execute_command("mkdir -p /mnt/extras/snapshot /mnt/extras/accounts")?;
30
31    // Set appropriate permissions
32    client.execute_command("sudo chown -R $(whoami):$(whoami) /mnt/ledger /mnt/extras")?;
33
34    Ok(())
35}

37/// Create necessary mount directories
38///
39/// # Arguments
40/// * `client` - SSH client
41///
42/// # Returns
43/// * `Result<(), DeploymentError>` - Success/failure
44fn create_mount_directories(client: &mut SshClient) -> Result<(), DeploymentError> {
45    client.execute_command("sudo mkdir -p /mnt/ledger /mnt/extras")?;
46
47    Ok(())
48}

50/// Format and mount a disk
51///
52/// # Arguments
53/// * `client` - SSH client
54/// * `disk` - Disk device path
55/// * `mount_point` - Mount point path
56///
57/// # Returns
58/// * `Result<(), DeploymentError>` - Success/failure
59fn format_and_mount_disk(
60    client: &mut SshClient,
61    disk: &str,
62    mount_point: &str,
63) -> Result<(), DeploymentError> {
64    // Check if the disk is already formatted and mounted
65    let mount_check = client.execute_command(&format!("findmnt -S {} -o TARGET -n", disk))?;
66    if !mount_check.trim().is_empty() {
67        // Disk is already mounted somewhere
68        if mount_check.trim() == mount_point {
69            // Already mounted at the correct location
70            return Ok(());
71        }
72        // Mounted at a different location, return an error
73        return Err(DeploymentError::DeploymentError(format!(
74            "Disk {} is already mounted at {}, cannot mount at {}",
75            disk,
76            mount_check.trim(),
77            mount_point
78        )));
79    }
80
81    // Format the disk (with confirmation to avoid data loss)
82    let _confirmation =
83        client.execute_command(&format!("echo \"y\" | sudo mkfs -t ext4 {}", disk))?;
84
85    // Mount the disk
86    client.execute_command(&format!("sudo mount {} {}", disk, mount_point))?;
87
88    // Add to fstab for persistence across reboots
89    let fstab_entry = format!("{} {} ext4 defaults,noatime 0 2", disk, mount_point);
90    let _fstab_check = client.execute_command(&format!(
91        "grep -q '{}' /etc/fstab || echo '{}' | sudo tee -a /etc/fstab",
92        disk, fstab_entry
93    ))?;
94
95    Ok(())
96}

98/// Validate disk requirements for Solana validator/RPC
99///
100/// # Arguments
101/// * `client` - SSH client
102/// * `ledger_disk` - Ledger disk device path
103/// * `accounts_disk` - Accounts disk device path
104///
105/// # Returns
106/// * `Result<(), DeploymentError>` - Success/failure with detailed information about any issues
107pub fn validate_disk_requirements(
108    client: &mut SshClient,
109    ledger_disk: &str,
110    accounts_disk: &str,
111) -> Result<(), DeploymentError> {
112    // Check if disks exist
113    if !disk_exists(client, ledger_disk)? {
114        return Err(DeploymentError::DeploymentError(format!(
115            "Ledger disk {} does not exist",
116            ledger_disk
117        )));
118    }
119
120    if !disk_exists(client, accounts_disk)? {
121        return Err(DeploymentError::DeploymentError(format!(
122            "Accounts disk {} does not exist",
123            accounts_disk
124        )));
125    }
126
127    // Check disk sizes
128    let ledger_size = get_disk_size(client, ledger_disk)?;
129    let accounts_size = get_disk_size(client, accounts_disk)?;
130
131    // Minimum size requirements (in GB)
132    const MIN_LEDGER_SIZE: u64 = 2000; // 2TB
133    const MIN_ACCOUNTS_SIZE: u64 = 2000; // 2TB
134
135    if ledger_size < MIN_LEDGER_SIZE {
136        return Err(DeploymentError::DeploymentError(format!(
137            "Ledger disk size is {}GB, minimum required is {}GB",
138            ledger_size, MIN_LEDGER_SIZE
139        )));
140    }
141
142    if accounts_size < MIN_ACCOUNTS_SIZE {
143        return Err(DeploymentError::DeploymentError(format!(
144            "Accounts disk size is {}GB, minimum required is {}GB",
145            accounts_size, MIN_ACCOUNTS_SIZE
146        )));
147    }
148
149    Ok(())
150}

152/// Check if a disk exists
153///
154/// # Arguments
155/// * `client` - SSH client
156/// * `disk` - Disk device path
157///
158/// # Returns
159/// * `Result<bool, DeploymentError>` - Whether the disk exists
160fn disk_exists(client: &mut SshClient, disk: &str) -> Result<bool, DeploymentError> {
161    let result = client.execute_command(&format!("test -b {} && echo 'exists'", disk))?;
162    Ok(result.trim() == "exists")
163}

165/// Get disk size in GB
166///
167/// # Arguments
168/// * `client` - SSH client
169/// * `disk` - Disk device path
170///
171/// # Returns
172/// * `Result<u64, DeploymentError>` - Disk size in GB
173fn get_disk_size(client: &mut SshClient, disk: &str) -> Result<u64, DeploymentError> {
174    let size_bytes = client.execute_command(&format!("lsblk -b -n -o SIZE {}", disk))?;
175
176    let size_bytes = size_bytes.trim().parse::<u64>().map_err(|e| {
177        DeploymentError::DeploymentError(format!("Failed to parse disk size: {}", e))
178    })?;
179
180    // Convert bytes to GB
181    Ok(size_bytes / 1_000_000_000)
182}
