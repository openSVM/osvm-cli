5/// Set up monitoring for Solana validator/RPC
6///
7/// # Arguments
8/// * `client` - SSH client
9/// * `install_dir` - Installation directory
10/// * `metrics_config` - Metrics configuration string
11/// * `node_type` - Node type (validator or rpc)
12///
13/// # Returns
14/// * `Result<(), DeploymentError>` - Success/failure
15pub fn setup_monitoring(
16    client: &mut SshClient,
17    install_dir: &str,
18    metrics_config: &str,
19    node_type: &str,
20) -> Result<(), DeploymentError> {
21    // Create directory for monitoring scripts
22    let monitoring_dir = format!("{}/monitoring", install_dir);
23    client.execute_command(&format!("mkdir -p {}", monitoring_dir))?;
24
25    // Configure Solana metrics if provided
26    if !metrics_config.is_empty() {
27        configure_solana_metrics(client, metrics_config)?;
28    }
29
30    // Set up basic monitoring script
31    setup_basic_monitoring(client, &monitoring_dir, node_type)?;
32
33    // Set up alerts for critical issues
34    setup_alert_system(client, &monitoring_dir)?;
35
36    Ok(())
37}

39/// Configure Solana metrics environment
40///
41/// # Arguments
42/// * `client` - SSH client
43/// * `metrics_config` - Metrics configuration string
44///
45/// # Returns
46/// * `Result<(), DeploymentError>` - Success/failure
47fn configure_solana_metrics(
48    client: &mut SshClient,
49    metrics_config: &str,
50) -> Result<(), DeploymentError> {
51    // Add metrics config to .bashrc
52    let bashrc_entry = format!("export SOLANA_METRICS_CONFIG=\"{}\"", metrics_config);
53
54    // Add to .bashrc if not already present
55    client.execute_command(&format!(
56        r#"grep -q "SOLANA_METRICS_CONFIG" ~/.bashrc || echo '{}' >> ~/.bashrc"#,
57        bashrc_entry
58    ))?;
59
60    Ok(())
61}

63/// Set up basic monitoring script
64///
65/// # Arguments
66/// * `client` - SSH client
67/// * `monitoring_dir` - Monitoring directory
68/// * `node_type` - Node type (validator or rpc)
69///
70/// # Returns
71/// * `Result<(), DeploymentError>` - Success/failure
72fn setup_basic_monitoring(
73    client: &mut SshClient,
74    monitoring_dir: &str,
75    node_type: &str,
76) -> Result<(), DeploymentError> {
77    let monitor_script_content = format!(
78        r#"#!/bin/bash
79# Solana {} monitoring script
80
81SERVICE_NAME="solana-validator.service"
82LOG_FILE="/home/$(whoami)/solana-validator.log"
83MONITOR_LOG="{}/monitor.log"
84
85# Check service status
86SERVICE_STATUS=$(systemctl is-active "$SERVICE_NAME")
87echo "$(date): Service status: $SERVICE_STATUS" >> "$MONITOR_LOG"
88
89# Get validator stats
90if [ "$SERVICE_STATUS" = "active" ]; then
91    # Check validator progress and health
92    CATCHUP_INFO=$(solana catchup --our-localhost | tail -n2)
93    echo "$(date): $CATCHUP_INFO" >> "$MONITOR_LOG"
94
95    # Get system metrics
96    LOAD=$(uptime | awk -F'load average: ' '{{ print $2 }}')
97    DISK=$(df -h /mnt/ledger /mnt/extras | grep -v "Filesystem")
98    MEM=$(free -h | grep "Mem:" | awk '{{ print "Total: "$2", Used: "$3", Free: "$4 }}')
99
100    # Log system metrics
101    echo "$(date): Load average: $LOAD" >> "$MONITOR_LOG"
102    echo "$(date): Disk usage:" >> "$MONITOR_LOG"
103    echo "$DISK" >> "$MONITOR_LOG"
104    echo "$(date): Memory usage: $MEM" >> "$MONITOR_LOG"
105else
106    echo "$(date): Service is not running, attempting to restart..." >> "$MONITOR_LOG"
107    sudo systemctl restart "$SERVICE_NAME"
108fi
109
110# Clean up old monitor logs (keep last 7 days)
111find "$(dirname "$MONITOR_LOG")" -name "*.log" -type f -mtime +7 -delete
112"#,
113        node_type, monitoring_dir
114    );
115
116    // Create monitoring script
117    let monitor_script_path = format!("{}/monitor.sh", monitoring_dir);
118    client.execute_command(&format!(
119        "cat > {} << 'EOL'\n{}\nEOL",
120        monitor_script_path, monitor_script_content
121    ))?;
122    client.execute_command(&format!("chmod +x {}", monitor_script_path))?;
123
124    // Set up cron job to run monitoring every 15 minutes
125    let cron_entry = format!("*/15 * * * * {}", monitor_script_path);
126    client.execute_command(&format!(
127        "(crontab -l 2>/dev/null | grep -v '{}' ; echo '{}') | crontab -",
128        monitor_script_path, cron_entry
129    ))?;
130
131    Ok(())
132}

134/// Set up alert system for critical issues
135///
136/// # Arguments
137/// * `client` - SSH client
138/// * `monitoring_dir` - Monitoring directory
139///
140/// # Returns
141/// * `Result<(), DeploymentError>` - Success/failure
142fn setup_alert_system(client: &mut SshClient, monitoring_dir: &str) -> Result<(), DeploymentError> {
143    let alert_script_content = r#"#!/bin/bash
144# Alert script for Solana validator critical issues
145
146SERVICE_NAME="solana-validator.service"
147LOG_FILE="/home/$(whoami)/solana-validator.log"
148ALERT_LOG="$(dirname "$0")/alerts.log"
149EMAIL_TO=""  # Set this to your email address to receive alerts
150
151# Function to send alerts
152send_alert() {
153    local subject="$1"
154    local message="$2"
155
156    # Log the alert
157    echo "$(date): $subject - $message" >> "$ALERT_LOG"
158
159    # Send email if configured
160    if [ -n "$EMAIL_TO" ] && command -v mail >/dev/null 2>&1; then
161        echo "$message" | mail -s "Solana Validator Alert: $subject" "$EMAIL_TO"
162    fi
163}
164
165# Check if service is running
166if ! systemctl is-active --quiet "$SERVICE_NAME"; then
167    send_alert "Service Down" "Validator service is not running on $(hostname)"
168    exit 1
169fi
170
171# Check for excessive catching up
172if grep -q "behind by" "$LOG_FILE" | tail -n 100; then
173    SLOTS_BEHIND=$(grep "behind by" "$LOG_FILE" | tail -n 1 | grep -oP "behind by \K[0-9]+")
174
175    if [ -n "$SLOTS_BEHIND" ] && [ "$SLOTS_BEHIND" -gt 1000 ]; then
176        send_alert "Falling Behind" "Validator is $SLOTS_BEHIND slots behind"
177    fi
178fi
179
180# Check disk space
181LEDGER_DISK_USAGE=$(df -h /mnt/ledger | tail -n1 | awk '{print $5}' | sed 's/%//')
182ACCOUNTS_DISK_USAGE=$(df -h /mnt/extras | tail -n1 | awk '{print $5}' | sed 's/%//')
183
184if [ "$LEDGER_DISK_USAGE" -gt 90 ]; then
185    send_alert "Disk Space Critical" "Ledger disk usage is at ${LEDGER_DISK_USAGE}%"
186fi
187
188if [ "$ACCOUNTS_DISK_USAGE" -gt 90 ]; then
189    send_alert "Disk Space Critical" "Accounts disk usage is at ${ACCOUNTS_DISK_USAGE}%"
190fi
191
192# Check for any errors in the log
193ERROR_COUNT=$(grep -i "error" "$LOG_FILE" | tail -n 1000 | wc -l)
194if [ "$ERROR_COUNT" -gt 50 ]; then
195    RECENT_ERRORS=$(grep -i "error" "$LOG_FILE" | tail -n 10)
196    send_alert "Excessive Errors" "Found $ERROR_COUNT errors in recent logs. Latest: $RECENT_ERRORS"
197fi
198"#;
199
200    // Create alert script
201    let alert_script_path = format!("{}/alert.sh", monitoring_dir);
202    client.execute_command(&format!(
203        "cat > {} << 'EOL'\n{}\nEOL",
204        alert_script_path, alert_script_content
205    ))?;
206    client.execute_command(&format!("chmod +x {}", alert_script_path))?;
207
208    // Set up cron job to run alerts every 30 minutes
209    let cron_entry = format!("*/30 * * * * {}", alert_script_path);
210    client.execute_command(&format!(
211        "(crontab -l 2>/dev/null | grep -v '{}' ; echo '{}') | crontab -",
212        alert_script_path, cron_entry
213    ))?;
214
215    Ok(())
216}

218/// Install Grafana and InfluxDB monitoring stack (optional)
219///
220/// # Arguments
221/// * `client` - SSH client
222/// * `install_dir` - Installation directory
223///
224/// # Returns
225/// * `Result<(), DeploymentError>` - Success/failure
226pub fn install_monitoring_stack(
227    client: &mut SshClient,
228    install_dir: &str,
229) -> Result<(), DeploymentError> {
230    // Create directory for monitoring stack
231    let stack_dir = format!("{}/monitoring-stack", install_dir);
232    client.execute_command(&format!("mkdir -p {}", stack_dir))?;
233
234    // Create docker-compose file for monitoring stack
235    let docker_compose_content = r#"version: '3'
236services:
237  influxdb:
238    image: influxdb:1.8
239    container_name: influxdb
240    ports:
241      - "8086:8086"
242    volumes:
243      - influxdb-data:/var/lib/influxdb
244    environment:
245      - INFLUXDB_DB=solana
246      - INFLUXDB_ADMIN_USER=admin
247      - INFLUXDB_ADMIN_PASSWORD=admin
248
249  grafana:
250    image: grafana/grafana:latest
251    container_name: grafana
252    ports:
253      - "3000:3000"
254    volumes:
255      - grafana-data:/var/lib/grafana
256    environment:
257      - GF_SECURITY_ADMIN_USER=admin
258      - GF_SECURITY_ADMIN_PASSWORD=admin
259    depends_on:
260      - influxdb
261
262volumes:
263  influxdb-data:
264  grafana-data:
265"#;
266
267    // Write docker-compose file
268    let docker_compose_path = format!("{}/docker-compose.yml", stack_dir);
269    client.execute_command(&format!(
270        "cat > {} << 'EOL'\n{}\nEOL",
271        docker_compose_path, docker_compose_content
272    ))?;
273
274    // Create script to install Docker and Docker Compose if needed
275    let setup_script_content = r#"#!/bin/bash
276# Setup script for Grafana + InfluxDB monitoring stack
277
278# Install Docker if not installed
279if ! command -v docker &> /dev/null; then
280    echo "Installing Docker..."
281    curl -fsSL https://get.docker.com -o get-docker.sh
282    sudo sh get-docker.sh
283    sudo usermod -aG docker $(whoami)
284fi
285
286# Install Docker Compose if not installed
287if ! command -v docker-compose &> /dev/null; then
288    echo "Installing Docker Compose..."
289    sudo curl -L "https://github.com/docker/compose/releases/download/v2.17.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
290    sudo chmod +x /usr/local/bin/docker-compose
291fi
292
293# Start monitoring stack
294docker-compose up -d
295
296# Print access instructions
297echo "Monitoring stack is now running"
298echo "Grafana is available at http://localhost:3000"
299echo "Default login: admin/admin"
300echo ""
301echo "Remember to configure Grafana to use InfluxDB as a data source:"
302echo "URL: http://influxdb:8086"
303echo "Database: solana"
304echo "User: admin"
305echo "Password: admin"
306"#;
307
308    // Write setup script
309    let setup_script_path = format!("{}/setup.sh", stack_dir);
310    client.execute_command(&format!(
311        "cat > {} << 'EOL'\n{}\nEOL",
312        setup_script_path, setup_script_content
313    ))?;
314    client.execute_command(&format!("chmod +x {}", setup_script_path))?;
315
316    // Don't automatically install - just provide the scripts
317    client.execute_command(&format!(
318        "echo 'Monitoring stack setup script created at {}. Run this script to install Grafana + InfluxDB.'",
319        setup_script_path
320    ))?;
321
322    Ok(())
323}
