9/// Network type
10#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
11pub enum NetworkType {
12    /// Mainnet
13    Mainnet,
14    /// Testnet
15    Testnet,
16    /// Devnet
17    Devnet,
18}
19
20impl fmt::Display for NetworkType {
21    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
22        match self {
23            NetworkType::Mainnet => write!(f, "mainnet"),
24            NetworkType::Testnet => write!(f, "testnet"),
25            NetworkType::Devnet => write!(f, "devnet"),
26        }
27    }
28}
29
30impl FromStr for NetworkType {
31    type Err = String;
32
33    fn from_str(s: &str) -> Result<Self, Self::Err> {
34        match s.to_lowercase().as_str() {
35            "mainnet" => Ok(NetworkType::Mainnet),
36            "testnet" => Ok(NetworkType::Testnet),
37            "devnet" => Ok(NetworkType::Devnet),
38            _ => Err(format!("Invalid network type: {}", s)),
39        }
40    }
41}

108impl ServerConfig {
109    /// Create a ServerConfig from a connection string (user@host[:port])
110    ///
111    /// # Arguments
112    /// * `conn_str` - Connection string in the format user@host[:port]
113    ///
114    /// # Returns
115    /// * `Result<ServerConfig, DeploymentError>` - Server config or error
116    pub fn from_connection_string(conn_str: &str) -> Result<Self, DeploymentError> {
117        // Validate input is not empty
118        if conn_str.trim().is_empty() {
119            return Err(DeploymentError::InvalidConfig(
120                "Connection string cannot be empty".to_string(),
121            ));
122        }
123
124        let parts: Vec<&str> = conn_str.split('@').collect();
125        if parts.len() != 2 {
126            return Err(DeploymentError::InvalidConfig(format!(
127                "Invalid connection string format. Expected 'user@host[:port]', got: {}",
128                conn_str
129            )));
130        }
131
132        // Validate and sanitize username
133        let username = parts[0].trim().to_string();
134        Self::validate_username(&username)?;
135
136        let host_parts: Vec<&str> = parts[1].split(':').collect();
137        let host = host_parts[0].trim().to_string();
138
139        // Validate host before proceeding
140        if host.is_empty() {
141            return Err(DeploymentError::InvalidConfig(
142                "Host cannot be empty in connection string".to_string(),
143            ));
144        }
145
146        // Parse port with proper error handling
147        let port = if host_parts.len() > 1 {
148            host_parts[1].trim().parse::<u16>().map_err(|_| {
149                DeploymentError::InvalidConfig(format!(
150                    "Invalid port number '{}'. Port must be a number between 1 and 65535",
151                    host_parts[1]
152                ))
153            })?
154        } else {
155            22
156        };
157
158        Ok(ServerConfig {
159            host,
160            port,
161            auth: AuthMethod::Key {
162                username,
163                key_path: "~/.ssh/id_rsa".to_string(),
164                passphrase: None,
165            },
166            install_dir: "/opt/osvm".to_string(),
167            connection_timeout_secs: default_connection_timeout(),
168        })
169    }
170
171    /// Validate username to prevent injection attacks
172    fn validate_username(username: &str) -> Result<(), DeploymentError> {
173        // Check for empty username
174        if username.is_empty() {
175            return Err(DeploymentError::ValidationError(
176                "Username cannot be empty".to_string(),
177            ));
178        }
179
180        // Check length (Unix usernames typically max 32 chars)
181        if username.len() > 32 {
182            return Err(DeploymentError::ValidationError(
183                "Username too long (max 32 characters)".to_string(),
184            ));
185        }
186
187        // Check for dangerous characters
188        if username.contains(|c: char| {
189            matches!(
190                c,
191                ';' | '&' | '|' | '$' | '`' | '\n' | '\r' | '\0' | '(' | ')' | '<' | '>'
192            )
193        }) {
194            return Err(DeploymentError::ValidationError(
195                "Username contains invalid characters".to_string(),
196            ));
197        }
198
199        // Ensure username starts with letter or underscore (Unix convention)
200        if let Some(first_char) = username.chars().next() {
201            if !first_char.is_alphabetic() && first_char != '_' {
202                return Err(DeploymentError::ValidationError(
203                    "Username must start with a letter or underscore".to_string(),
204                ));
205            }
206        }
207
208        // Check that username only contains valid characters (alphanumeric, underscore, hyphen)
209        if !username
210            .chars()
211            .all(|c| c.is_alphanumeric() || matches!(c, '_' | '-'))
212        {
213            return Err(DeploymentError::ValidationError(
214                "Username can only contain letters, numbers, underscores, and hyphens".to_string(),
215            ));
216        }
217
218        Ok(())
219    }
220}
