43/// Execute an agent command from the CLI
44pub async fn execute_agent_command(
45    prompt: &str,
46    json_output: bool,
47    verbose: u8,
48    no_tools: bool,
49    timeout_seconds: u64,
50) -> Result<()> {
51    let start_time = std::time::Instant::now();
52
53    // Initialize services
54    let ai_service = AiService::new_with_debug(verbose > 1);
55    let mut mcp_service = McpService::new_with_debug(verbose > 1);
56
57    // Load MCP configurations
58    if let Err(e) = mcp_service.load_config() {
59        if verbose > 0 {
60            eprintln!("‚ö†Ô∏è  Warning: Failed to load MCP configurations: {}", e);
61        }
62    }
63
64    // Get available tools
65    let mut available_tools: HashMap<String, Vec<McpTool>> = HashMap::new();
66
67    if !no_tools {
68        // In a real implementation, we would get tools from active MCP servers
69        // For now, we'll simulate some basic tools
70        let mock_tools = vec![
71            McpTool {
72                name: "get_balance".to_string(),
73                description: Some("Get wallet balance from blockchain".to_string()),
74                input_schema: serde_json::json!({"type": "object", "properties": {}}),
75            },
76            McpTool {
77                name: "get_transactions".to_string(),
78                description: Some("Get recent transactions from wallet".to_string()),
79                input_schema: serde_json::json!({"type": "object", "properties": {"limit": {"type": "number", "default": 10}}}),
80            },
81            McpTool {
82                name: "deploy_validator".to_string(),
83                description: Some("Deploy a new validator node".to_string()),
84                input_schema: serde_json::json!({"type": "object", "properties": {"network": {"type": "string"}}}),
85            },
86            McpTool {
87                name: "get_network_status".to_string(),
88                description: Some("Get current network status and statistics".to_string()),
89                input_schema: serde_json::json!({"type": "object", "properties": {}}),
90            },
91        ];
92
93        if !mock_tools.is_empty() {
94            available_tools.insert("blockchain_tools".to_string(), mock_tools);
95        }
96    }
97
98    // Show initial status if not in JSON mode
99    if !json_output {
100        println!("ü§ñ OSVM Agent CLI");
101        println!("{}", "‚îÅ".repeat(50));
102        println!("üìù Prompt: {}", prompt);
103
104        if verbose > 0 {
105            println!("\n‚è≥ Analyzing request...");
106        }
107    }
108
109    // Try OSVM command planner first (for CLI commands)
110    let planner = OsvmCommandPlanner::new(verbose > 1);
111    if let Ok(osvm_plan) = planner.create_plan(prompt).await {
112        if !json_output {
113            println!("\nüí° Detected OSVM command intent");
114            println!("{}", "‚îÅ".repeat(50));
115            println!("üìã Execution Plan:");
116            println!("   üí≠ {}", osvm_plan.reasoning);
117            println!("   üéØ Confidence: {:.0}%", osvm_plan.confidence * 100.0);
118            println!();
119
120            for (i, step) in osvm_plan.steps.iter().enumerate() {
121                println!("   {}. {}", i + 1, step.full_command);
122                println!("      ‚Üí {}", step.explanation);
123            }
124
125            println!();
126            println!("   ‚ú® Expected: {}", osvm_plan.expected_outcome);
127            println!();
128
129            // Execute the OSVM commands
130            match planner.execute_plan(&osvm_plan, true).await {
131                Ok(results) => {
132                    println!("{}", "‚îÅ".repeat(50));
133                    println!("üìä Execution Results:");
134                    println!();
135
136                    for (i, cmd_result) in results.iter().enumerate() {
137                        let status = if cmd_result.success { "‚úÖ" } else { "‚ùå" };
138                        println!("   {}. {} {}", i + 1, status, cmd_result.command);
139
140                        if !cmd_result.stdout.is_empty() {
141                            // Indent the output
142                            for line in cmd_result.stdout.lines() {
143                                println!("      {}", line);
144                            }
145                        }
146
147                        if !cmd_result.success && !cmd_result.stderr.is_empty() {
148                            println!("      ‚ùå {}", cmd_result.stderr);
149                        }
150
151                        println!("      ‚è±Ô∏è  {}ms", cmd_result.execution_time_ms);
152                        println!();
153                    }
154
155                    let agent_result = AgentExecutionResult {
156                        prompt: prompt.to_string(),
157                        plan: None,
158                        tool_results: vec![],
159                        final_response: format!(
160                            "Executed {} OSVM command(s) successfully",
161                            results.len()
162                        ),
163                        execution_time_ms: start_time.elapsed().as_millis(),
164                        success: results.iter().all(|r| r.success),
165                        error: None,
166                    };
167
168                    if json_output {
169                        println!("{}", serde_json::to_string_pretty(&agent_result)?);
170                    }
171
172                    return Ok(());
173                }
174                Err(e) => {
175                    if verbose > 0 {
176                        println!("‚ö†Ô∏è  OSVM command execution failed: {}", e);
177                        println!("   Falling back to MCP tools...\n");
178                    }
179                }
180            }
181        }
182    }
183
184    // Create execution result
185    let mut result = AgentExecutionResult {
186        prompt: prompt.to_string(),
187        plan: None,
188        tool_results: Vec::new(),
189        final_response: String::new(),
190        execution_time_ms: 0,
191        success: false,
192        error: None,
193    };
194
195    // Apply timeout to the entire operation
196    let execution_future = execute_with_plan(
197        &ai_service,
198        &mut mcp_service,
199        prompt,
200        &available_tools,
201        verbose,
202        no_tools,
203        json_output,
204    );
205
206    match tokio::time::timeout(Duration::from_secs(timeout_seconds), execution_future).await {
207        Ok(Ok((plan, tool_results, final_response))) => {
208            result.plan = Some(plan);
209            result.tool_results = tool_results;
210            result.final_response = final_response;
211            result.success = true;
212        }
213        Ok(Err(e)) => {
214            result.error = Some(format!("Execution failed: {}", e));
215            result.final_response = format!(
216                "I encountered an error while processing your request: {}",
217                e
218            );
219        }
220        Err(_) => {
221            result.error = Some(format!(
222                "Execution timed out after {} seconds",
223                timeout_seconds
224            ));
225            result.final_response = "The operation timed out. Please try again with a simpler request or increase the timeout.".to_string();
226        }
227    }
228
229    result.execution_time_ms = start_time.elapsed().as_millis();
230
231    // Output results
232    if json_output {
233        println!("{}", serde_json::to_string_pretty(&result)?);
234    } else {
235        // Pretty print the results
236        println!("\n{}", "‚îÅ".repeat(50));
237
238        if let Some(plan) = &result.plan {
239            if verbose > 0 {
240                println!("\nüìã Plan: {}", plan.reasoning);
241
242                if !plan.osvm_tools_to_use.is_empty() {
243                    println!("\nüîß Tools to execute:");
244                    for tool in &plan.osvm_tools_to_use {
245                        println!("   ‚Ä¢ {} ({})", tool.tool_name, tool.server_id);
246                        if verbose > 1 {
247                            println!("     Reason: {}", tool.reason);
248                        }
249                    }
250                }
251            }
252        }
253
254        if !result.tool_results.is_empty() && verbose > 0 {
255            println!("\n‚öôÔ∏è  Tool Results:");
256            for tool_result in &result.tool_results {
257                if let Some(ref error) = tool_result.error {
258                    println!("   ‚ùå {}: {}", tool_result.tool_name, error);
259                } else if let Some(ref result_value) = tool_result.result {
260                    println!(
261                        "   ‚úÖ {}: {}",
262                        tool_result.tool_name,
263                        format_tool_result(result_value)
264                    );
265                }
266            }
267        }
268
269        println!("\nüí¨ Response: {}", result.final_response);
270
271        if verbose > 0 {
272            println!("\n‚è±Ô∏è  Execution time: {}ms", result.execution_time_ms);
273        }
274
275        if let Some(ref error) = result.error {
276            println!("\n‚ùå Error: {}", error);
277        }
278    }
279
280    Ok(())
281}

283/// Execute the agent plan
284async fn execute_with_plan(
285    ai_service: &AiService,
286    mcp_service: &mut McpService,
287    prompt: &str,
288    available_tools: &HashMap<String, Vec<McpTool>>,
289    verbose: u8,
290    no_tools: bool,
291    json_output: bool,
292) -> Result<(ToolPlan, Vec<ToolExecutionResult>, String)> {
293    // Create tool plan using AI
294    let tool_plan = if no_tools || available_tools.is_empty() {
295        // Create a simple plan without tools
296        ToolPlan {
297            reasoning: "No tools available or tool execution disabled. Providing direct response."
298                .to_string(),
299            osvm_tools_to_use: Vec::new(),
300            expected_outcome: "Direct AI response to the query".to_string(),
301            raw_ovsm_plan: None, // No OVSM plan when tools are disabled
302        }
303    } else {
304        // Try to get OSVM plan from AI with validation
305        match ai_service
306            .create_validated_tool_plan(prompt, available_tools, 3)
307            .await
308        {
309            Ok(plan) => {
310                if verbose > 1 {
311                    eprintln!(
312                        "üìù OSVM Plan received and validated with {} tools",
313                        plan.osvm_tools_to_use.len()
314                    );
315                    if plan.raw_ovsm_plan.is_some() {
316                        eprintln!("   ‚úÖ OVSM syntax validated successfully");
317                    }
318                }
319                plan
320            }
321            Err(e) => {
322                eprintln!(
323                    "‚ùå Failed to get validated OSVM plan from AI service: {}",
324                    e
325                );
326                eprintln!("   Please ensure the AI service is configured and accessible.");
327                // Return empty plan on error - no fallback demos allowed
328                return Err(anyhow::anyhow!(
329                    "Failed to create validated OSVM plan: {}",
330                    e
331                ));
332            }
333        }
334    };
335
336    if !json_output && verbose > 0 {
337        println!("üéØ Creating execution plan...");
338    }
339
340    // Execute tools if any
341    let mut tool_results = Vec::new();
342
343    for planned_tool in &tool_plan.osvm_tools_to_use {
344        if !json_output && verbose > 0 {
345            println!("üîß Executing {}...", planned_tool.tool_name);
346        }
347
348        let tool_start = std::time::Instant::now();
349
350        // Simulate tool execution (in real implementation, would call MCP)
351        let (result, error) = execute_mock_tool(&planned_tool).await;
352
353        tool_results.push(ToolExecutionResult {
354            tool_name: planned_tool.tool_name.clone(),
355            server_id: planned_tool.server_id.clone(),
356            args: planned_tool.args.clone(),
357            result,
358            error,
359            execution_time_ms: tool_start.elapsed().as_millis(),
360        });
361    }
362
363    // Generate final response
364    let tool_results_for_ai: Vec<(String, Value)> = tool_results
365        .iter()
366        .filter_map(|tr| {
367            tr.result
368                .as_ref()
369                .map(|r| (tr.tool_name.clone(), r.clone()))
370        })
371        .collect();
372
373    let final_response = if tool_results_for_ai.is_empty() && tool_plan.osvm_tools_to_use.is_empty()
374    {
375        // No tools were used, get direct AI response
376        ai_service
377            .query_with_debug(prompt, verbose > 1)
378            .await
379            .unwrap_or_else(|e| format!("I apologize, but I couldn't generate a response: {}", e))
380    } else {
381        ai_service.generate_contextual_response(
382            prompt,
383            &tool_results_for_ai,
384            &tool_plan.expected_outcome
385        ).await
386            .unwrap_or_else(|e| {
387                format!("I executed the tools but encountered an issue generating the final response: {}", e)
388            })
389    };
390
391    Ok((tool_plan, tool_results, final_response))
392}

510/// Format tool result for display
511fn format_tool_result(value: &Value) -> String {
512    // Try to format the result nicely
513    if let Some(obj) = value.as_object() {
514        if let Some(balance) = obj.get("balance") {
515            return format!("{}", balance);
516        }
517        if let Some(transactions) = obj.get("transactions") {
518            if let Some(arr) = transactions.as_array() {
519                return format!("{} transactions", arr.len());
520            }
521        }
522    }
523
524    // Fallback to compact JSON
525    serde_json::to_string(value).unwrap_or_else(|_| value.to_string())
526}
