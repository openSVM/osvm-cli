14/// Represents a single OSVM command to execute
15#[derive(Debug, Clone, Serialize, Deserialize)]
16pub struct OsvmCommandStep {
17    /// The OSVM command to execute (e.g., "balance", "svm list")
18    pub command: String,
19    /// Full command with arguments (e.g., "osvm balance")
20    pub full_command: String,
21    /// Arguments to pass to the command
22    pub args: Vec<String>,
23    /// Human-readable explanation of what this command does
24    pub explanation: String,
25    /// Whether this command requires user confirmation
26    pub requires_confirmation: bool,
27    /// Expected output description
28    pub expected_output: String,
29}

31/// Complete execution plan with multiple OSVM commands
32#[derive(Debug, Clone, Serialize, Deserialize)]
33pub struct OsvmExecutionPlan {
34    /// AI's reasoning for this plan
35    pub reasoning: String,
36    /// Confidence score (0.0 to 1.0)
37    pub confidence: f32,
38    /// Ordered list of commands to execute
39    pub steps: Vec<OsvmCommandStep>,
40    /// Expected final outcome
41    pub expected_outcome: String,
42    /// Whether the plan can be executed automatically
43    pub auto_executable: bool,
44}

72    /// Create an execution plan from a user query
73    pub async fn create_plan(&self, user_query: &str) -> Result<OsvmExecutionPlan> {
74        if self.debug {
75            println!("üß† Creating execution plan for: {}", user_query);
76        }
77
78        // Try AI-based planning first
79        match self.try_ai_planning(user_query).await {
80            Ok(plan) => Ok(plan),
81            Err(ai_error) => {
82                // Fall back to rule-based planning
83                if self.debug {
84                    eprintln!(
85                        "‚ö†Ô∏è  AI planning failed: {}, falling back to rule-based",
86                        ai_error
87                    );
88                }
89                self.create_rule_based_plan(user_query)
90            }
91        }
92    }

94    /// Try to create a plan using AI
95    async fn try_ai_planning(&self, user_query: &str) -> Result<OsvmExecutionPlan> {
96        // Build the planning prompt
97        let planning_prompt = self.build_planning_prompt(user_query);
98
99        // Call AI to generate plan
100        let ai_response = self
101            .ai_service
102            .query_with_debug(&planning_prompt, self.debug)
103            .await
104            .context("Failed to get AI planning response")?;
105
106        // Parse the AI response into a structured plan
107        self.parse_plan_response(&ai_response).or_else(|parse_error| {
108            // AI didn't return expected format - show helpful error
109            Err(anyhow::anyhow!(
110                "AI returned unexpected format. \n\nResponse preview: {}\n\n\
111                üí° This feature is experimental. If you see this error frequently, please report it at:\n   \
112                https://github.com/opensvm/osvm-cli/issues",
113                &ai_response.chars().take(200).collect::<String>()
114            ))
115        })
116    }

118    /// Create a rule-based plan when AI is not available
119    fn create_rule_based_plan(&self, user_query: &str) -> Result<OsvmExecutionPlan> {
120        let query_lower = user_query.to_lowercase();
121
122        // Simple keyword matching for common queries
123        let (command, args, explanation) = if query_lower.contains("svm")
124            && (query_lower.contains("list")
125                || query_lower.contains("show")
126                || query_lower.contains("all"))
127        {
128            (
129                "svm",
130                vec!["svm".to_string(), "list".to_string()],
131                "List all available SVMs",
132            )
133        } else if query_lower.contains("balance") || query_lower.contains("wallet") {
134            (
135                "balance",
136                vec!["balance".to_string()],
137                "Check your SOL balance",
138            )
139        } else if query_lower.contains("node") && query_lower.contains("list") {
140            (
141                "nodes",
142                vec!["nodes".to_string(), "list".to_string()],
143                "List all nodes",
144            )
145        } else if query_lower.contains("health")
146            || query_lower.contains("status")
147            || query_lower.contains("doctor")
148        {
149            ("doctor", vec!["doctor".to_string()], "Check system health")
150        } else if query_lower.contains("example") || query_lower.contains("help") {
151            (
152                "examples",
153                vec!["examples".to_string()],
154                "Show usage examples",
155            )
156        } else {
157            // Default to showing examples
158            (
159                "examples",
160                vec!["examples".to_string()],
161                "Show examples to help you get started",
162            )
163        };
164
165        Ok(OsvmExecutionPlan {
166            reasoning: format!(
167                "Matched query '{}' to {} command using keyword matching",
168                user_query, command
169            ),
170            confidence: 0.7,
171            steps: vec![OsvmCommandStep {
172                command: command.to_string(),
173                full_command: format!("osvm {}", args.join(" ")),
174                args,
175                explanation: explanation.to_string(),
176                requires_confirmation: false,
177                expected_output: format!("Output from {} command", command),
178            }],
179            expected_outcome: format!("Execute {} command based on your query", command),
180            auto_executable: true,
181        })
182    }

240    /// Parse AI response into execution plan
241    fn parse_plan_response(&self, ai_response: &str) -> Result<OsvmExecutionPlan> {
242        // Try to extract JSON from the response
243        let json_str = self.extract_json(ai_response)?;
244
245        // Parse as JSON
246        let plan: OsvmExecutionPlan =
247            serde_json::from_str(&json_str).context("Failed to parse execution plan JSON")?;
248
249        // Validate the plan
250        if plan.steps.is_empty() {
251            anyhow::bail!("Execution plan contains no steps");
252        }
253
254        Ok(plan)
255    }

257    /// Extract JSON from AI response (handles cases where AI adds extra text)
258    fn extract_json(&self, response: &str) -> Result<String> {
259        // First, try to parse the whole response as JSON
260        if let Ok(_) = serde_json::from_str::<serde_json::Value>(response) {
261            return Ok(response.to_string());
262        }
263
264        // Look for JSON between curly braces
265        if let Some(start) = response.find('{') {
266            if let Some(end) = response.rfind('}') {
267                if start < end {
268                    let json_candidate = &response[start..=end];
269                    if serde_json::from_str::<serde_json::Value>(json_candidate).is_ok() {
270                        return Ok(json_candidate.to_string());
271                    }
272                }
273            }
274        }
275
276        anyhow::bail!("Could not extract valid JSON from AI response")
277    }

279    /// Execute a single OSVM command
280    pub async fn execute_command(&self, step: &OsvmCommandStep) -> Result<OsvmCommandResult> {
281        let start_time = std::time::Instant::now();
282
283        if self.debug {
284            println!("‚öôÔ∏è  Executing: {}", step.full_command);
285        }
286
287        // Build the command
288        let mut cmd = Command::new("osvm");
289        cmd.args(&step.args);
290
291        // Execute the command
292        let output = cmd
293            .output()
294            .context(format!("Failed to execute command: {}", step.full_command))?;
295
296        let execution_time = start_time.elapsed().as_millis();
297
298        Ok(OsvmCommandResult {
299            command: step.full_command.clone(),
300            success: output.status.success(),
301            stdout: String::from_utf8_lossy(&output.stdout).to_string(),
302            stderr: String::from_utf8_lossy(&output.stderr).to_string(),
303            execution_time_ms: execution_time,
304            exit_code: output.status.code(),
305        })
306    }

308    /// Execute an entire execution plan
309    pub async fn execute_plan(
310        &self,
311        plan: &OsvmExecutionPlan,
312        skip_confirmation: bool,
313    ) -> Result<Vec<OsvmCommandResult>> {
314        let mut results = Vec::new();
315
316        for (i, step) in plan.steps.iter().enumerate() {
317            if self.debug {
318                println!(
319                    "üìç Step {}/{}: {}",
320                    i + 1,
321                    plan.steps.len(),
322                    step.explanation
323                );
324            }
325
326            // Check if confirmation is needed
327            if step.requires_confirmation && !skip_confirmation {
328                println!("\n‚ö†Ô∏è  This command requires confirmation:");
329                println!("   {}", step.full_command);
330                println!("   {}", step.explanation);
331                println!("\nThis step will be skipped. Use --yes to auto-confirm.");
332                continue;
333            }
334
335            // Execute the command
336            match self.execute_command(step).await {
337                Ok(result) => {
338                    if !result.success {
339                        if self.debug {
340                            eprintln!("‚ö†Ô∏è  Command failed with exit code: {:?}", result.exit_code);
341                            if !result.stderr.is_empty() {
342                                eprintln!("   Error: {}", result.stderr);
343                            }
344                        }
345                    }
346                    results.push(result);
347                }
348                Err(e) => {
349                    eprintln!("‚ùå Failed to execute step {}: {}", i + 1, e);
350                    // Create a failed result
351                    results.push(OsvmCommandResult {
352                        command: step.full_command.clone(),
353                        success: false,
354                        stdout: String::new(),
355                        stderr: format!("Execution error: {}", e),
356                        execution_time_ms: 0,
357                        exit_code: None,
358                    });
359                    // Continue with next command
360                }
361            }
362        }
363
364        Ok(results)
365    }

367    /// Format execution results for display
368    pub fn format_results(
369        &self,
370        plan: &OsvmExecutionPlan,
371        results: &[OsvmCommandResult],
372    ) -> String {
373        let mut output = String::new();
374
375        output.push_str(&format!("üìã Execution Plan Results\n"));
376        output.push_str(&format!("‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ‚îÅ\n\n"));
377        output.push_str(&format!("Reasoning: {}\n\n", plan.reasoning));
378
379        for (i, result) in results.iter().enumerate() {
380            let status = if result.success { "‚úÖ" } else { "‚ùå" };
381            output.push_str(&format!("{}. {} {}\n", i + 1, status, result.command));
382
383            if !result.stdout.is_empty() {
384                let stdout = result.stdout.trim();
385                if stdout.len() > 200 {
386                    output.push_str(&format!("   Output: {}...\n", &stdout[..200]));
387                } else {
388                    output.push_str(&format!("   Output: {}\n", stdout));
389                }
390            }
391
392            if !result.success && !result.stderr.is_empty() {
393                output.push_str(&format!("   Error: {}\n", result.stderr.trim()));
394            }
395
396            output.push_str(&format!("   Time: {}ms\n", result.execution_time_ms));
397            output.push_str("\n");
398        }
399
400        output.push_str(&format!("Expected outcome: {}\n", plan.expected_outcome));
401
402        output
403    }
