53    /// Detect available branches from remote repository
54    pub async fn detect_remote_branches(&self, repo_url: &str) -> Result<Vec<BranchInfo>> {
55        println!("üîç Probing remote branches for: {}", repo_url);
56
57        let output = Command::new("git")
58            .args(&["ls-remote", "--heads", repo_url])
59            .output()
60            .context("Failed to execute git ls-remote")?;
61
62        if !output.status.success() {
63            let stderr = String::from_utf8_lossy(&output.stderr);
64            anyhow::bail!("Failed to list remote branches: {}", stderr);
65        }
66
67        let stdout = String::from_utf8_lossy(&output.stdout);
68        let mut branches = Vec::new();
69
70        for line in stdout.lines() {
71            if let Some((commit_hash, ref_name)) = line.split_once('\t') {
72                if let Some(branch_name) = ref_name.strip_prefix("refs/heads/") {
73                    branches.push(BranchInfo {
74                        name: branch_name.to_string(),
75                        commit_hash: commit_hash.to_string(),
76                        is_default: false, // Will be determined later
77                        last_commit_date: "Unknown".to_string(),
78                    });
79                }
80            }
81        }
82
83        println!(
84            "üìã Found {} remote branches: {:?}",
85            branches.len(),
86            branches.iter().map(|b| &b.name).collect::<Vec<_>>()
87        );
88
89        Ok(branches)
90    }

92    /// Get the default branch of the remote repository
93    pub async fn get_default_branch(&self, repo_url: &str) -> Result<String> {
94        println!("üéØ Detecting default branch for: {}", repo_url);
95
96        let output = Command::new("git")
97            .args(&["ls-remote", "--symref", repo_url, "HEAD"])
98            .output()
99            .context("Failed to execute git ls-remote")?;
100
101        if !output.status.success() {
102            let stderr = String::from_utf8_lossy(&output.stderr);
103            println!("‚ö†Ô∏è  Could not detect default branch: {}", stderr);
104            return Ok("main".to_string()); // Fallback to main
105        }
106
107        let stdout = String::from_utf8_lossy(&output.stdout);
108
109        for line in stdout.lines() {
110            if line.starts_with("ref: refs/heads/") {
111                if let Some(branch_name) = line.strip_prefix("ref: refs/heads/") {
112                    if let Some(branch) = branch_name.split('\t').next() {
113                        println!("‚úÖ Detected default branch: {}", branch);
114                        return Ok(branch.to_string());
115                    }
116                }
117            }
118        }
119
120        println!("‚ö†Ô∏è  Could not parse default branch, using 'main'");
121        Ok("main".to_string())
122    }

124    /// Get comprehensive repository metadata
125    pub async fn get_repository_metadata(&self, repo_url: &str) -> Result<RepositoryMetadata> {
126        let default_branch = self.get_default_branch(repo_url).await?;
127        let mut available_branches = self.detect_remote_branches(repo_url).await?;
128
129        // Mark the default branch
130        for branch in &mut available_branches {
131            if branch.name == default_branch {
132                branch.is_default = true;
133            }
134        }
135
136        Ok(RepositoryMetadata {
137            default_branch,
138            available_branches,
139            clone_url: repo_url.to_string(),
140            last_activity: "Unknown".to_string(),
141        })
142    }

144    /// Generate intelligent branch priority list
145    pub fn generate_branch_priority(
146        &self,
147        repo_metadata: &RepositoryMetadata,
148        user_branch: Option<&str>,
149    ) -> Vec<String> {
150        let mut priority_branches = Vec::new();
151
152        // 1. If user specified a branch, try it first
153        if let Some(branch) = user_branch {
154            priority_branches.push(branch.to_string());
155        }
156
157        // 2. Add custom branches from configuration
158        priority_branches.extend(self.custom_branches.clone());
159
160        // 3. Add the detected default branch
161        if !priority_branches.contains(&repo_metadata.default_branch) {
162            priority_branches.push(repo_metadata.default_branch.clone());
163        }
164
165        // 4. Add common branch patterns if they exist in the repo
166        let common_branches = [
167            "main",
168            "master",
169            "develop",
170            "development",
171            "dev",
172            "trunk",
173            "stable",
174            "release",
175            "prod",
176            "production",
177        ];
178
179        for common_branch in &common_branches {
180            if repo_metadata
181                .available_branches
182                .iter()
183                .any(|b| b.name == *common_branch)
184            {
185                if !priority_branches.contains(&common_branch.to_string()) {
186                    priority_branches.push(common_branch.to_string());
187                }
188            }
189        }
190
191        // 5. Add any remaining branches from the repository
192        for branch in &repo_metadata.available_branches {
193            if !priority_branches.contains(&branch.name) {
194                priority_branches.push(branch.name.clone());
195            }
196        }
197
198        priority_branches
199    }

201    /// Clone repository with intelligent branch selection
202    pub async fn clone_with_branch_detection(
203        &self,
204        repo_url: &str,
205        target_dir: &Path,
206        preferred_branch: Option<&str>,
207    ) -> Result<String> {
208        // Get repository metadata first
209        let repo_metadata = if self.enable_dynamic_probing {
210            self.get_repository_metadata(repo_url).await?
211        } else {
212            // Fallback metadata for when dynamic probing is disabled
213            RepositoryMetadata {
214                default_branch: preferred_branch.unwrap_or("main").to_string(),
215                available_branches: Vec::new(),
216                clone_url: repo_url.to_string(),
217                last_activity: "Unknown".to_string(),
218            }
219        };
220
221        let priority_branches = self.generate_branch_priority(&repo_metadata, preferred_branch);
222
223        println!("üìã Branch priority order: {:?}", priority_branches);
224
225        let mut last_error = None;
226
227        for branch in &priority_branches {
228            println!("üîÑ Attempting to clone branch: {}", branch);
229
230            let result = self
231                .attempt_clone_branch(repo_url, target_dir, branch)
232                .await;
233
234            match result {
235                Ok(_) => {
236                    println!("‚úÖ Successfully cloned branch: {}", branch);
237                    return Ok(branch.clone());
238                }
239                Err(e) => {
240                    println!("‚ùå Failed to clone branch '{}': {}", branch, e);
241                    last_error = Some(e);
242
243                    // Clean up failed clone attempt
244                    if target_dir.exists() {
245                        let _ = std::fs::remove_dir_all(target_dir);
246                    }
247                }
248            }
249        }
250
251        // If all branches failed, return the last error
252        if let Some(error) = last_error {
253            anyhow::bail!("Failed to clone any branch. Last error: {}", error);
254        } else {
255            anyhow::bail!("No branches available to clone");
256        }
257    }

259    /// Attempt to clone a specific branch
260    async fn attempt_clone_branch(
261        &self,
262        repo_url: &str,
263        target_dir: &Path,
264        branch: &str,
265    ) -> Result<()> {
266        let output = Command::new("git")
267            .args(&[
268                "clone",
269                "--single-branch",
270                "--branch",
271                branch,
272                "--depth",
273                "1",
274                repo_url,
275                target_dir.to_str().unwrap(),
276            ])
277            .output()
278            .context("Failed to execute git clone")?;
279
280        if !output.status.success() {
281            let stderr = String::from_utf8_lossy(&output.stderr);
282            anyhow::bail!("Git clone failed: {}", stderr);
283        }
284
285        Ok(())
286    }

288    /// Add user-defined branch override flags
289    pub fn add_branch_override(&mut self, branch: String) {

291            self.custom_branches.insert(0, branch); // Add to front for highest priority
292        }

318    fn test_branch_priority_generation() {
319        let manager = GitRepositoryManager::new();
320
321        let repo_metadata = RepositoryMetadata {
322            default_branch: "develop".to_string(),
323            available_branches: vec![
324                BranchInfo {

326                    commit_hash: "abc123".to_string(),
327                    is_default: false,
328                    last_commit_date: "2024-01-01".to_string(),
329                },
330                BranchInfo {
331                    name: "develop".to_string(),
332                    commit_hash: "def456".to_string(),
333                    is_default: true,
334                    last_commit_date: "2024-01-02".to_string(),
335                },
336            ],
337            clone_url: "https://github.com/test/repo".to_string(),
338            last_activity: "2024-01-02".to_string(),
339        };
340
341        let priorities = manager.generate_branch_priority(&repo_metadata, Some("feature-branch"));
342
343        assert_eq!(priorities[0], "feature-branch"); // User branch first
344        assert!(priorities.contains(&"develop".to_string())); // Default branch included
345        assert!(priorities.contains(&"main".to_string())); // Common branch included
346    }
