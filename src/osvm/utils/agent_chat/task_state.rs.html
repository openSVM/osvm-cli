1//! Task state management for tracking chat operations and UI state

3use serde::{Deserialize, Serialize};

5/// Priority levels for todo items
6#[derive(Debug, Clone)]
7pub enum TodoPriority {
8    High,
9    Medium,
10    Low,
11}
12
13/// Input modes for the UI
14#[derive(Debug, Clone, PartialEq)]
15pub enum InputMode {
16    InputField,
17    TaskSelection,
18}
19
20/// Individual todo item with execution tracking
21#[derive(Debug, Clone)]
22pub struct TodoItem {
23    pub text: String,
24    pub completed: bool,
25    pub priority: TodoPriority,
26    pub reasoning: String,
27    pub tool_plan: Option<String>,
28    pub execution_results: Option<String>,
29}
30
31impl TodoItem {
32    /// Create a new high-priority todo item
33    pub fn new_high(text: String, reasoning: String) -> Self {
34        Self {
35            text,
36            completed: false,
37            priority: TodoPriority::High,
38            reasoning,
39            tool_plan: None,
40            execution_results: None,
41        }
42    }
43
44    /// Mark as completed with results
45    pub fn complete(&mut self, results: String) {
46        self.completed = true;
47        self.execution_results = Some(results);
48    }
49}

54/// Main task state for the chat application

58    pub todo_items: Vec<TodoItem>,

60    pub selected_todo_index: usize,

62    pub spinner_frame: usize,
63}

129    /// Update spinner animation frame
130    pub fn update_spinner(&mut self) {
131        self.spinner_frame = (self.spinner_frame + 1) % SPINNER_FRAMES.len();
132    }
133
134    /// Toggle todo item completion - safe bounds checking
135    pub fn toggle_todo(&mut self, index: usize) {
136        if let Some(item) = self.todo_items.get_mut(index) {
137            if !item.completed {
138                item.complete("✅ Task completed successfully".to_string());
139            }
140        }
141    }
142
143    /// Navigate to previous todo item
144    pub fn navigate_todo_up(&mut self) {
145        if !self.todo_items.is_empty() {
146            if self.selected_todo_index > 0 {
147                self.selected_todo_index -= 1;
148            } else {
149                self.selected_todo_index = self.todo_items.len().saturating_sub(1);
150            }
151        }
152    }
153
154    /// Navigate to next todo item
155    pub fn navigate_todo_down(&mut self) {
156        if !self.todo_items.is_empty() {
157            // Safe modulo operation - prevent divide by zero
158            let len = self.todo_items.len();
159            if len > 0 {
160                self.selected_todo_index = (self.selected_todo_index + 1) % len;
161            }
162        }
163    }
164
165    /// Get currently selected task details
166    pub fn get_selected_task_details(&self) -> Option<&TodoItem> {
167        self.todo_items.get(self.selected_todo_index)
168    }
169
170    /// Switch between input modes
171    pub fn toggle_input_mode(&mut self) {
172        self.input_mode = match self.input_mode {
173            InputMode::InputField => InputMode::TaskSelection,
174            InputMode::TaskSelection => InputMode::InputField,
175        };
176    }
