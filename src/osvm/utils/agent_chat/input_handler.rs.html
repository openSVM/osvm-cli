110    /// Insert character at cursor position (handles UTF-8 properly)
111    pub fn insert_char(&mut self, ch: char) {
112        // Convert cursor position from char index to byte index
113        let byte_pos = self
114            .input
115            .char_indices()
116            .nth(self.cursor_pos)
117            .map(|(i, _)| i)
118            .unwrap_or(self.input.len());
119
120        self.input.insert(byte_pos, ch);
121        self.cursor_pos += 1;
122        self.reset_selection();
123    }

125    /// Delete character before cursor (handles UTF-8 properly)
126    pub fn delete_before_cursor(&mut self) -> bool {
127        if self.cursor_pos > 0 {
128            // Find the byte position of the character to delete
129            let mut char_indices: Vec<(usize, char)> = self.input.char_indices().collect();
130
131            if self.cursor_pos <= char_indices.len() {
132                let byte_pos = if self.cursor_pos == char_indices.len() {
133                    // Deleting the last character
134                    char_indices.get(self.cursor_pos - 1).map(|(i, _)| *i)
135                } else {
136                    // Deleting a character in the middle
137                    char_indices.get(self.cursor_pos - 1).map(|(i, _)| *i)
138                };
139
140                if let Some(pos) = byte_pos {
141                    // Find the char to remove
142                    if let Some((_, ch)) = char_indices.get(self.cursor_pos - 1) {
143                        let char_len = ch.len_utf8();
144                        for _ in 0..char_len {
145                            if pos < self.input.len() {
146                                self.input.remove(pos);
147                            }
148                        }
149                        self.cursor_pos -= 1;
150                        self.reset_selection();
151                        return true;
152                    }
153                }
154            }
155        }
156        false
157    }

253    /// Generate smart auto-suggestions based on current input and context
254    pub fn generate_auto_suggestions(
255        &mut self,
256        context_history: &[String],
257    ) -> Vec<RealtimeSuggestion> {
258        if self.input.len() < 2 {
259            return Vec::new();
260        }
261
262        let mut suggestions = Vec::new();
263        let input_lower = self.input.to_lowercase();
264
265        // Command suggestions
266        let commands = vec![
267            ("/help", "Show available commands", "command"),
268            ("/clear", "Clear chat history", "command"),
269            ("/status", "Show system status", "command"),
270            ("/tools", "List available MCP tools", "command"),
271            ("/context", "Show conversation context", "command"),
272            ("/balance", "Check wallet balance", "crypto"),
273            ("/transactions", "Show recent transactions", "crypto"),
274            ("/stake", "Stake management", "crypto"),
275            ("/price", "Check token prices", "crypto"),
276            ("/network", "Network status", "system"),
277        ];
278
279        // Add command suggestions
280        for (cmd, desc, category) in commands {
281            if cmd.to_lowercase().contains(&input_lower) {
282                suggestions.push(RealtimeSuggestion::new(
283                    cmd.to_string(),
284                    desc.to_string(),
285                    category.to_string(),
286                ));
287            }
288        }
289
290        // MCP tool suggestions
291        let mcp_tools = vec![
292            (
293                "@solana/get_balance",
294                "Get wallet balance via Solana MCP",
295                "mcp",
296            ),
297            ("@solana/get_transactions", "Get transaction history", "mcp"),
298            (
299                "@solana/get_network_status",
300                "Check Solana network status",
301                "mcp",
302            ),
303            ("@solana/stake_account", "Manage stake accounts", "mcp"),
304        ];
305
306        for (tool, desc, category) in mcp_tools {
307            if tool.to_lowercase().contains(&input_lower) || input_lower.contains("@") {
308                suggestions.push(RealtimeSuggestion::new(
309                    tool.to_string(),
310                    desc.to_string(),
311                    category.to_string(),
312                ));
313            }
314        }
315
316        // History-based suggestions
317        let fuzzy_matcher = FuzzyMatcher::new(0.3);
318        for cmd in &self.command_history {
319            if cmd.to_lowercase().contains(&input_lower) && cmd != &self.input {
320                suggestions.push(RealtimeSuggestion::new(
321                    cmd.clone(),
322                    "From history".to_string(),
323                    "history".to_string(),
324                ));
325            }
326        }
327
328        // Context-aware suggestions based on recent conversation
329        if let Some(last_message) = context_history.last() {
330            if last_message.to_lowercase().contains("balance") && !input_lower.contains("balance") {
331                suggestions.push(RealtimeSuggestion::new(
332                    "/balance".to_string(),
333                    "Check your wallet balance".to_string(),
334                    "contextual".to_string(),
335                ));
336            }
337            if last_message.to_lowercase().contains("transaction") && !input_lower.contains("trans")
338            {
339                suggestions.push(RealtimeSuggestion::new(
340                    "/transactions".to_string(),
341                    "View recent transactions".to_string(),
342                    "contextual".to_string(),
343                ));
344            }
345        }
346
347        // Smart completions for partial commands
348        if input_lower.starts_with("/") && input_lower.len() > 1 {
349            let partial = &input_lower[1..];
350            if "balance".starts_with(partial) {
351                suggestions.push(RealtimeSuggestion::new(
352                    "/balance".to_string(),
353                    "Check wallet balance".to_string(),
354                    "completion".to_string(),
355                ));
356            }
357            if "transactions".starts_with(partial) {
358                suggestions.push(RealtimeSuggestion::new(
359                    "/transactions".to_string(),
360                    "Show transaction history".to_string(),
361                    "completion".to_string(),
362                ));
363            }
364        }
365
366        // Sort by relevance and limit
367        suggestions.sort_by(|a, b| {
368            // Prioritize exact matches and shorter suggestions
369            let a_exact = a.text.to_lowercase() == input_lower;
370            let b_exact = b.text.to_lowercase() == input_lower;
371
372            if a_exact && !b_exact {
373                std::cmp::Ordering::Less
374            } else if !a_exact && b_exact {
375                std::cmp::Ordering::Greater
376            } else {
377                a.text.len().cmp(&b.text.len())
378            }
379        });
380
381        suggestions.truncate(self.config.max_suggestions);
382        self.suggestions = suggestions.clone();
383        suggestions
384    }

386    /// Update suggestions without disturbing terminal state
387    pub fn update_suggestions_in_place(
388        &mut self,
389        renderer: &mut TerminalRenderer,
390        context_history: &[String],
391    ) -> Result<()> {
392        let suggestions = self.generate_auto_suggestions(context_history);
393
394        if suggestions.is_empty() {
395            // Clear suggestions area
396            let suggestion_area = renderer.areas().suggestions.clone();
397            renderer.render_component_in_place(&suggestion_area, |stdout, area| {
398                // Just clear the area
399                Ok(())
400            })?;
401            return Ok(());
402        }
403
404        let suggestion_area = renderer.areas().suggestions.clone();
405
406        renderer.render_component_in_place(&suggestion_area, |stdout, area| {
407            if area.height < 2 {
408                return Ok(()); // Not enough space
409            }
410
411            // Header
412            write!(
413                stdout,
414                "{}╭─ Suggestions (↑/↓ to navigate, Tab to select) ─╮{}",
415                Colors::DIM,
416                Colors::RESET
417            )?;
418
419            let max_suggestions = (area.height.saturating_sub(2) as usize).min(suggestions.len());
420
421            for (i, suggestion) in suggestions.iter().enumerate().take(max_suggestions) {
422                if i + 1 >= area.height as usize {
423                    break;
424                }
425
426                execute!(stdout, MoveTo(area.x, area.y + i as u16 + 1))?;
427
428                let selector = if i == self.selected_suggestion {
429                    "▶"
430                } else {
431                    " "
432                };
433                let icon = match suggestion.category.as_str() {
434                    "command" => "⌘",
435                    "mcp" => "⚙",
436                    "crypto" => "₿",
437                    "history" => "↺",
438                    "contextual" => "◉",
439                    "completion" => "→",
440                    _ => "•",
441                };
442
443                let color = if i == self.selected_suggestion {
444                    Color::Yellow
445                } else {
446                    Color::Grey
447                };
448
449                // Truncate text to fit terminal width - UTF-8 safe
450                let available_width = area.width.saturating_sub(8); // Account for prefix and margins
451                let display_text = if suggestion.text.chars().count() > available_width as usize {
452                    let truncate_at = available_width.saturating_sub(3) as usize;
453                    let truncated: String = suggestion.text.chars().take(truncate_at).collect();
454                    format!("{}...", truncated)
455                } else {
456                    suggestion.text.clone()
457                };
458
459                let display_desc = if suggestion.description.chars().count() > 30 {
460                    let truncated: String = suggestion.description.chars().take(27).collect();
461                    format!("{}...", truncated)
462                } else {
463                    suggestion.description.clone()
464                };
465
466                write!(
467                    stdout,
468                    "{}│{} {} {} - {}{}",
469                    Colors::DIM,
470                    selector,
471                    icon,
472                    format_colored_text(&display_text, color),
473                    format_colored_text(&display_desc, Color::White),
474                    Colors::RESET
475                )?;
476            }
477
478            // Footer
479            if max_suggestions < suggestions.len() {
480                let remaining = suggestions.len() - max_suggestions;
481                execute!(stdout, MoveTo(area.x, area.y + max_suggestions as u16 + 1))?;
482                write!(
483                    stdout,
484                    "{}│ ... {} more suggestions{}",
485                    Colors::DIM,
486                    remaining,
487                    Colors::RESET
488                )?;
489            }
490
491            // Bottom border (if space)
492            if area.height > max_suggestions as u16 + 2 {
493                execute!(stdout, MoveTo(area.x, area.y + area.height - 1))?;
494                write!(
495                    stdout,
496                    "{}╰{}─╯{}",
497                    Colors::DIM,
498                    "─".repeat(area.width.saturating_sub(3) as usize),
499                    Colors::RESET
500                )?;
501            }
502
503            Ok(())
504        })?;
505
506        Ok(())
507    }

510/// Input character classification

512pub enum InputChar {
513    Enter,
514    Backspace,
515    CtrlC,
516    CtrlT,
517    CtrlL,
518    Tab,

530    Down,
531    Left,

539    /// Read and classify a single character from input
540    pub fn read_character() -> Result<InputChar> {
541        let mut buffer = [0; 1];
542        if std::io::stdin().read(&mut buffer)? != 1 {
543            return Ok(InputChar::Unknown);
544        }
545
546        match buffer[0] {
547            b'\n' | b'\r' => Ok(InputChar::Enter),
548            0x7f | 0x08 => Ok(InputChar::Backspace),
549            0x03 => Ok(InputChar::CtrlC),
550            0x14 => Ok(InputChar::CtrlT),
551            0x0c => Ok(InputChar::CtrlL),
552            b'\t' => Ok(InputChar::Tab),
553            0x1b => Self::read_escape_sequence(),
554            ch if ch >= 0x20 && ch < 0x7f => Ok(InputChar::Regular(ch as char)),
555            // Filter out common mouse event byte sequences
556            0x00..=0x1f => Ok(InputChar::Unknown), // Control characters
557            0x80..=0xff => Ok(InputChar::Unknown), // Extended ASCII often used in mouse events
558            _ => Ok(InputChar::Unknown),
559        }
560    }
561
562    /// Read escape sequences (arrow keys, etc.)
563    fn read_escape_sequence() -> Result<InputChar> {
564        use std::io::{ErrorKind, Read};
565        use std::time::Duration;
566
567        let mut buffer = [0; 1];
568
569        // Try to read the next character with a timeout
570        // If it's not available quickly, it's just an ESC key
571        let mut stdin = std::io::stdin();
572
573        // First, check if there's a '[' following the ESC
574        match stdin.read_exact(&mut buffer) {
575            Ok(_) if buffer[0] == b'[' => {
576                // Now read the direction character
577                match stdin.read_exact(&mut buffer) {
578                    Ok(_) => match buffer[0] {
579                        b'A' => Ok(InputChar::Arrow(ArrowKey::Up)),
580                        b'B' => Ok(InputChar::Arrow(ArrowKey::Down)),
581                        b'C' => Ok(InputChar::Arrow(ArrowKey::Right)),
582                        b'D' => Ok(InputChar::Arrow(ArrowKey::Left)),
583                        b'M' | b'<' => {
584                            // Mouse event sequence detected, consume and ignore it
585                            // Read remaining mouse event bytes (typically 3 more bytes)
586                            let mut discard = [0; 10];
587                            let _ = stdin.read(&mut discard);
588                            Ok(InputChar::Mouse)
589                        }
590                        b'0'..=b'9' => {
591                            // Possible extended sequence (like mouse events)
592                            // Consume the rest of the sequence
593                            let mut discard = [0; 10];
594                            let _ = stdin.read(&mut discard);
595                            Ok(InputChar::Unknown)
596                        }
597                        _ => Ok(InputChar::Escape),
598                    },
599                    Err(_) => Ok(InputChar::Escape), // Incomplete sequence
600                }
601            }
602            Ok(_) => {
603                // We read a character that wasn't '[', put it back conceptually
604                // In practice, we'll just treat this as ESC + that character
605                Ok(InputChar::Escape)
606            }
607            Err(_) => Ok(InputChar::Escape), // Just ESC key
608        }
609    }

611    /// Process input character and update state with suggestion updates
612    pub async fn process_input_with_suggestions(
613        state: &mut InputState,
614        ch: InputChar,
615        renderer: &mut TerminalRenderer,
616        context_history: &[String],
617    ) -> Result<Option<String>> {
618        let mut should_update_suggestions = false;
619
620        let result = match ch {
621            InputChar::Enter => {
622                // If a suggestion is selected, apply it instead of submitting
623                if !state.suggestions.is_empty()
624                    && state.selected_suggestion < state.suggestions.len()
625                {
626                    state.apply_suggestion();
627                    should_update_suggestions = true;
628                    Ok(None)
629                } else {
630                    let result = state.input.clone();
631                    state.add_to_history(result.clone());
632                    state.clear();
633                    should_update_suggestions = true;
634                    Ok(Some(result))
635                }
636            }
637
638            InputChar::Backspace => {
639                if state.delete_before_cursor() {
640                    should_update_suggestions = true;
641                }
642                Ok(None)
643            }
644
645            InputChar::Tab => {
646                if state.apply_suggestion() {
647                    should_update_suggestions = true;
648                }
649                Ok(None)
650            }
651
652            InputChar::Arrow(ArrowKey::Up) => {
653                if state.suggestions.is_empty() {
654                    state.history_up();
655                    should_update_suggestions = true;
656                } else {
657                    state.suggestion_up();
658                    // Only update the suggestion display, not regenerate
659                    state.update_suggestions_in_place(renderer, context_history)?;
660                }
661                Ok(None)
662            }
663
664            InputChar::Arrow(ArrowKey::Down) => {
665                if state.suggestions.is_empty() {
666                    state.history_down();
667                    should_update_suggestions = true;
668                } else {
669                    state.suggestion_down();
670                    // Only update the suggestion display, not regenerate
671                    state.update_suggestions_in_place(renderer, context_history)?;
672                }
673                Ok(None)
674            }
675
676            InputChar::Arrow(ArrowKey::Left) => {
677                state.move_cursor_left();
678                Ok(None)
679            }
680
681            InputChar::Arrow(ArrowKey::Right) => {
682                state.move_cursor_right();
683                Ok(None)
684            }
685
686            InputChar::Regular(c) => {
687                state.insert_char(c);
688                should_update_suggestions = state.should_update_suggestions();
689                Ok(None)
690            }
691
692            InputChar::Escape => {
693                state.clear();
694                should_update_suggestions = true;
695                Ok(None)
696            }
697
698            InputChar::CtrlC => Err(anyhow!("User interrupted")),
699
700            InputChar::Mouse => {
701                // Silently ignore mouse events to preserve right-click functionality
702                Ok(None)
703            }
704
705            _ => Ok(None),
706        };
707
708        // Update suggestions if needed
709        if should_update_suggestions {
710            state.update_suggestions_in_place(renderer, context_history)?;
711        }
712
713        result
714    }

716    /// Legacy process input method for backward compatibility
717    pub async fn process_input(
718        state: &mut InputState,
719        ch: InputChar,
720        suggestion_tx: &mpsc::UnboundedSender<String>,
721    ) -> Result<Option<String>> {
722        match ch {
723            InputChar::Enter => {
724                let result = state.input.clone();
725                state.add_to_history(result.clone());
726                state.clear();
727                Ok(Some(result))
728            }
729
730            InputChar::Backspace => {
731                if state.delete_before_cursor() {
732                    let _ = suggestion_tx.send(state.input.clone());
733                }
734                Ok(None)
735            }
736
737            InputChar::Tab => {
738                state.apply_suggestion();
739                Ok(None)
740            }
741
742            InputChar::Arrow(ArrowKey::Up) => {
743                if state.suggestions.is_empty() {
744                    state.history_up();
745                } else {
746                    state.suggestion_up();
747                }
748                Ok(None)
749            }
750
751            InputChar::Arrow(ArrowKey::Down) => {
752                if state.suggestions.is_empty() {
753                    state.history_down();
754                } else {
755                    state.suggestion_down();
756                }
757                Ok(None)
758            }
759
760            InputChar::Arrow(ArrowKey::Left) => {
761                state.move_cursor_left();
762                Ok(None)
763            }
764
765            InputChar::Arrow(ArrowKey::Right) => {
766                state.move_cursor_right();
767                Ok(None)
768            }
769
770            InputChar::Regular(c) => {
771                state.insert_char(c);
772                if state.should_update_suggestions() {
773                    let _ = suggestion_tx.send(state.input.clone());
774                }
775                Ok(None)
776            }
777
778            InputChar::Escape => {
779                state.clear();
780                Ok(None)
781            }
782
783            InputChar::CtrlC => Err(anyhow!("User interrupted")),
784
785            InputChar::Mouse => {
786                // Silently ignore mouse events
787                Ok(None)
788            }
789
790            _ => Ok(None),
791        }
792    }
793}
