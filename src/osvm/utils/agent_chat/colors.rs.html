48    /// Apply gradient to text
49    pub fn gradient(text: &str, gradient_type: GradientType) -> String {
50        let colors = match gradient_type {
51            GradientType::Purple => Gradient::purple_gradient(),
52            GradientType::Fire => Gradient::fire_gradient(),
53            GradientType::Ocean => Gradient::ocean_gradient(),
54            GradientType::Sunset => Gradient::sunset_gradient(),
55            GradientType::Matrix => Gradient::matrix_gradient(),
56        };
57        Gradient::apply_to_text(text, &colors)
58    }
59
60    /// Create glowing text effect
61    pub fn glow(text: &str, base_color: &str) -> String {
62        format!(
63            "{}{}{}{}{}",
64            base_color,
65            Theme::BOLD,
66            text,
67            Theme::RESET,
68            "\x1b[0m" // Extra reset for safety
69        )
70    }
71
72    /// Create pulsing text (for animations)
73    pub fn pulse_text(text: &str, intensity: f32) -> String {
74        let alpha = (intensity * 255.0) as u8;
75        format!(
76            "\x1b[38;2;{};{};{}m{}{}",
77            alpha,
78            alpha,
79            alpha,
80            text,
81            Theme::RESET
82        )
83    }
84}

91    Sunset,

95/// Helper functions for color manipulation
96impl Colors {
97    /// Mix two RGB colors
98    pub fn mix_rgb(color1: (u8, u8, u8), color2: (u8, u8, u8), ratio: f32) -> String {
99        let r = (color1.0 as f32 * (1.0 - ratio) + color2.0 as f32 * ratio) as u8;
100        let g = (color1.1 as f32 * (1.0 - ratio) + color2.1 as f32 * ratio) as u8;
101        let b = (color1.2 as f32 * (1.0 - ratio) + color2.2 as f32 * ratio) as u8;
102        format!("\x1b[38;2;{};{};{}m", r, g, b)
103    }
104
105    /// Create color with transparency simulation (using dimming)
106    pub fn with_alpha(base_rgb: (u8, u8, u8), alpha: f32) -> String {
107        let r = (base_rgb.0 as f32 * alpha) as u8;
108        let g = (base_rgb.1 as f32 * alpha) as u8;
109        let b = (base_rgb.2 as f32 * alpha) as u8;
110        format!("\x1b[38;2;{};{};{}m", r, g, b)
111    }
