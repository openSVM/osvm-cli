1//! Simple sequential layout system for agent_chat with clean terminal rendering
2//!
3//! This module provides utilities for rendering TUI components sequentially in the terminal
4//! using simple println! statements that preserve text selection and normal terminal behavior.

9/// Screen areas for different components with absolute positioning
10#[derive(Debug, Clone)]
11pub struct ComponentAreas {
12    pub terminal_size: (u16, u16),
13    pub status_bar: ComponentArea,
14    pub task_panel: ComponentArea,
15    pub chat_area: ComponentArea,
16    pub input_field: ComponentArea,
17    pub suggestions: ComponentArea,
18}

20/// Individual component area with position and size
21#[derive(Debug, Clone)]
22pub struct ComponentArea {
23    pub x: u16,
24    pub y: u16,
25    pub width: u16,
26    pub height: u16,
27    pub visible: bool,
28}
29
30impl ComponentArea {
31    pub fn new(x: u16, y: u16, width: u16, height: u16) -> Self {
32        Self {
33            x,
34            y,
35            width,
36            height,
37            visible: true,
38        }
39    }
40
41    /// Check if area is valid for current terminal size
42    pub fn is_valid(&self, terminal_size: (u16, u16)) -> bool {
43        self.x + self.width <= terminal_size.0 && self.y + self.height <= terminal_size.1
44    }
45}
46
47impl ComponentAreas {
48    /// Calculate responsive component areas based on terminal size
49    pub fn calculate(terminal_size: (u16, u16)) -> Self {
50        let (cols, rows) = terminal_size;
51
52        // Don't use complex overlapping layout - use simple sequential layout
53        let mut current_y = 0u16;
54
55        // Status bar at top (1 row with 1 row spacing)
56        let status_bar = ComponentArea::new(0, current_y, cols, 1);
57        current_y += 2; // Status bar + spacing
58
59        // Task panel (fixed height, not overlapping)
60        let task_panel_height = if rows >= 20 {
61            6
62        } else if rows >= 15 {
63            4
64        } else {
65            0
66        };
67        let task_panel = if task_panel_height > 0 {
68            let area = ComponentArea::new(0, current_y, cols, task_panel_height);
69            current_y += task_panel_height + 1; // Task panel + spacing
70            area
71        } else {
72            ComponentArea::new(0, 0, 0, 0) // Hidden
73        };
74
75        // Chat area takes most remaining space
76        let remaining_rows = rows.saturating_sub(current_y).saturating_sub(4); // Reserve 4 for input/suggestions
77        let chat_height = remaining_rows.max(2); // At least 2 rows
78        let chat_area = ComponentArea::new(0, current_y, cols, chat_height);
79        current_y += chat_height;
80
81        // Suggestions (2 rows)
82        let suggestions_height = 2;
83        let suggestions = ComponentArea::new(0, current_y, cols, suggestions_height);
84        current_y += suggestions_height;
85
86        // Input field at bottom (1 row)
87        let input_field = ComponentArea::new(0, current_y, cols, 1);
88
89        Self {
90            terminal_size,
91            status_bar,
92            task_panel: ComponentArea {
93                visible: task_panel_height > 0,
94                ..task_panel
95            },
96            chat_area,
97            input_field,
98            suggestions: ComponentArea {
99                visible: false,
100                ..suggestions
101            }, // Hidden by default
102        }
103    }
104
105    /// Minimal layout for very small terminals
106    fn minimal_layout(terminal_size: (u16, u16)) -> Self {
107        let (cols, rows) = terminal_size;
108
109        // Only status bar and input field for tiny terminals
110        let status_bar = ComponentArea::new(0, 0, cols, 1);
111        let input_field = ComponentArea::new(0, rows.saturating_sub(2), cols, 2);
112
113        // Hide other components
114        let task_panel = ComponentArea::new(0, 0, 0, 0);
115        let chat_area = ComponentArea::new(0, 1, cols, rows.saturating_sub(3));
116        let suggestions = ComponentArea::new(0, 0, 0, 0);
117
118        Self {
119            terminal_size,
120            status_bar,
121            task_panel: ComponentArea {
122                visible: false,
123                ..task_panel
124            },
125            chat_area,
126            input_field,
127            suggestions: ComponentArea {
128                visible: false,
129                ..suggestions
130            },
131        }
132    }
133
134    /// Update layout for new terminal size
135    pub fn update_for_size(&mut self, new_size: (u16, u16)) {
136        *self = Self::calculate(new_size);
137    }
138}

140/// Simple sequential terminal renderer that preserves text selection
141pub struct TerminalRenderer {
142    areas: ComponentAreas,
143}
144
145impl TerminalRenderer {
146    /// Create new renderer - no complex positioning needed
147    pub fn new() -> Result<Self> {
148        Ok(Self {
149            areas: ComponentAreas::calculate((80, 24)), // Default size
150        })
151    }

153    /// Simple sequential rendering - just use println!
154    pub fn render_component_in_place<F>(
155        &mut self,
156        _area: &ComponentArea,
157        render_fn: F,
158    ) -> Result<()>
159    where
160        F: FnOnce(&mut std::io::Stdout, &ComponentArea) -> Result<()>,
161    {
162        // No positioning needed - just render sequentially
163        let dummy_area = ComponentArea::new(0, 0, 80, 10);
164        render_fn(&mut stdout(), &dummy_area)?;
165        stdout().flush()?;
166        Ok(())
167    }
168
169    /// Get areas (simplified)
170    pub fn areas(&self) -> &ComponentAreas {
171        &self.areas
172    }
173
174    /// Handle resize - not needed for sequential rendering
175    pub fn handle_resize(&mut self) -> Result<bool> {
176        Ok(false)
177    }

180/// Text formatting utilities for responsive display
181pub fn wrap_text_smart(text: &str, width: usize) -> Vec<String> {
182    if width < 10 {
183        // Very narrow - just split by characters
184        return text
185            .chars()
186            .collect::<Vec<char>>()
187            .chunks(width)
188            .map(|chunk| chunk.iter().collect())
189            .collect();
190    }
191
192    let mut lines = Vec::new();
193    let mut current_line = String::new();
194
195    for word in text.split_whitespace() {
196        // If word itself is too long, split it
197        if word.len() > width {
198            if !current_line.is_empty() {
199                lines.push(current_line.clone());
200                current_line.clear();
201            }
202
203            // Split long word
204            let chars: Vec<char> = word.chars().collect();
205            for chunk in chars.chunks(width) {
206                lines.push(chunk.iter().collect());
207            }
208            continue;
209        }
210
211        // Check if adding this word would exceed width
212        let space_needed = if current_line.is_empty() { 0 } else { 1 }; // Space before word
213        if current_line.len() + space_needed + word.len() > width {
214            if !current_line.is_empty() {
215                lines.push(current_line.clone());
216                current_line.clear();
217            }
218        }
219
220        if !current_line.is_empty() {
221            current_line.push(' ');
222        }
223        current_line.push_str(word);
224    }
225
226    if !current_line.is_empty() {
227        lines.push(current_line);
228    }
229
230    // Ensure we return at least one line (empty if needed)
231    if lines.is_empty() {
232        lines.push(String::new());
233    }
234
235    lines
236}

238/// Truncate text with ellipsis for narrow displays
239pub fn truncate_with_ellipsis(text: &str, max_width: usize) -> String {
240    if text.len() <= max_width {
241        return text.to_string();
242    }
243
244    if max_width <= 3 {
245        return "...".chars().take(max_width).collect();
246    }
247
248    let mut result = text.chars().take(max_width - 3).collect::<String>();
249    result.push_str("...");
250    result
251}

253/// Format text with color preservation for terminal compatibility
254pub fn format_colored_text(text: &str, color: crossterm::style::Color) -> String {
255    let color_code = match color {
256        crossterm::style::Color::Red => "31",
257        crossterm::style::Color::Green => "32",
258        crossterm::style::Color::Yellow => "33",
259        crossterm::style::Color::Blue => "34",
260        crossterm::style::Color::Magenta => "35",
261        crossterm::style::Color::Cyan => "36",
262        crossterm::style::Color::White => "37",
263        crossterm::style::Color::Grey => "90",
264        _ => "37", // Default to white
265    };
266    format!("\x1b[{}m{}\x1b[0m", color_code, text)
267}

269/// Simple rendering that respects terminal scroll buffer
270pub fn render_without_scroll<F>(render_fn: F) -> Result<()>
271where
272    F: FnOnce() -> Result<()>,
273{
274    // Just call the render function - no positioning needed
275    render_fn()?;
276    stdout().flush()?;
277    Ok(())
