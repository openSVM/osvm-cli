26pub struct App {

34impl App {
35    /// Create new app instance
36    pub fn new() -> Self {
37        let mut task_state = TaskState::new();
38
39        // Mark initial tasks as completed to simulate progress
40        if let Some(item) = task_state.todo_items.get_mut(0) {
41            item.completed = true;
42            item.execution_results = Some("✅ MCP service initialized successfully".to_string());
43        }
44        if let Some(item) = task_state.todo_items.get_mut(1) {
45            item.completed = true;
46            item.execution_results = Some("✅ AI service connected to osvm.ai".to_string());
47        }
48        if let Some(item) = task_state.todo_items.get_mut(2) {
49            item.completed = true;
50            item.execution_results = Some("✅ Chat interface ready".to_string());
51        }
52
53        task_state.current_reasoning = "OSVM Agent initialized and ready for user interaction. Press Ctrl+T to navigate tasks.".to_string();
54
55        Self {
56            task_state,
57            input_state: InputState::new(),
58            should_quit: false,
59            chat_history: Vec::new(),
60            suggestions: Vec::new(),
61        }
62    }
63
64    pub fn tick(&mut self) {
65        self.task_state.update_spinner();
66    }
67
68    pub fn handle_key_event(&mut self, key: crossterm::event::KeyEvent) {
69        use crossterm::event::{KeyCode, KeyModifiers};
70
71        match (key.code, key.modifiers) {
72            (KeyCode::Char('c'), KeyModifiers::CONTROL) => {
73                self.should_quit = true;
74            }
75            (KeyCode::Char('t'), KeyModifiers::CONTROL) => {
76                self.task_state.toggle_input_mode();
77            }
78            _ => {
79                // Handle other keys based on current mode
80                if self.task_state.input_mode == InputMode::TaskSelection {
81                    match key.code {
82                        KeyCode::Up => self.task_state.navigate_todo_up(),
83                        KeyCode::Down => self.task_state.navigate_todo_down(),
84                        KeyCode::Enter => {
85                            if let Some(index) = Some(self.task_state.selected_todo_index) {
86                                self.task_state.toggle_todo(index);
87                            }
88                        }
89                        _ => {}
90                    }
91                }
92            }
93        }
94    }
95}

379    io::stdout().flush()?;

385async fn get_enhanced_input_with_status(
386    suggestion_tx: &mpsc::UnboundedSender<String>,
387    task_state: &mut TaskState,
388    test_mode: bool,
389) -> Result<String> {
390    let mut input_state = InputState::new();

392

394    let (cols, rows) = terminal::size().unwrap_or((80, 24));
395    let input_start_row = rows.saturating_sub(3); // Reserve 3 rows for complete input box
396
397    // Position and draw input box at bottom
398    print!("\x1B[{};1H", input_start_row);
399    let input_border = "─".repeat(cols.saturating_sub(10) as usize); // Account for "┌─ Input " and "┐"
400    println!(
401        "{}┌─ Input {}┐{}",
402        Colors::GREEN,
403        input_border,
404        Colors::RESET
405    );
406
407    // Draw input content line
408    print!("\x1B[{};1H", input_start_row + 1);
409    print!("{}│{} > ", Colors::GREEN, Colors::RESET);
410    let content_width = cols.saturating_sub(6) as usize; // Account for "│ > " and " │"
411    print!(
412        "{:<width$}{} │{}",
413        "",
414        Colors::RESET,
415        Colors::GREEN,
416        width = content_width
417    );
418
419    // Draw bottom border
420    print!("\x1B[{};1H", input_start_row + 2);
421    let bottom_border = "─".repeat(cols.saturating_sub(2) as usize);
422    println!("{}└{}┘{}", Colors::GREEN, bottom_border, Colors::RESET);
423
424    // Position cursor in input field
425    print!("\x1B[{};4H", input_start_row + 1);
426    io::stdout()
427        .flush()
428        .map_err(|e| anyhow!("Failed to flush stdout: {}", e))?;
429
430    // Enable raw mode for character-by-character input (unless in test/fallback mode)
431    if test_mode {
432        // Test mode or fallback mode: Use line-buffered input
433        return read_line_buffered_input().await;
434    }
435
436    // Enable raw mode (we already checked it's available at startup)
437    super::input_handler::enable_raw_mode()?;
438
439    loop {
440        match read_single_character()? {
441            InputChar::Enter => {
442                if task_state.input_mode == InputMode::TaskSelection {
443                    // Toggle selected todo item
444                    if task_state.selected_todo_index < task_state.todo_items.len() {
445                        task_state.toggle_todo(task_state.selected_todo_index);
446                    }
447
448                    // Update reasoning based on completed tasks
449                    let completed_count =
450                        task_state.todo_items.iter().filter(|t| t.completed).count();
451                    task_state.current_reasoning = format!(
452                        "Progress: {}/{} tasks completed. Current focus: {}",
453                        completed_count,
454                        task_state.todo_items.len(),
455                        task_state
456                            .todo_items
457                            .get(task_state.selected_todo_index)
458                            .map(|t| &t.text)
459                            .unwrap_or(&"N/A".to_string())
460                    );
461
462                    // Refresh display
463                    print!("\x1B[2J\x1B[1;1H");
464                    show_enhanced_status_bar(task_state);
465                    continue;
466                } else {
467                    // In input mode - adopt suggestion if visible, then submit
468                    if !input_state.suggestions.is_empty()
469                        && input_state.selected_suggestion < input_state.suggestions.len()
470                    {
471                        let suggestion = &input_state.suggestions[input_state.selected_suggestion];
472                        input_state.input = suggestion.text.clone();
473                        input_state.cursor_pos = input_state.input.len();
474                        input_state.suggestions.clear();
475                        input_state.selected_suggestion = 0;
476                        input_state.original_before_sug = None;
477                        input_state.suggestions_suppressed = true; // Suppress until user requests again
478
479                        clear_suggestions_display();
480                        redraw_input_line(&input_state.input)?;
481                        continue; // Don't submit yet, let user confirm
482                    }
483
484                    // Submit input
485                    super::input_handler::disable_raw_mode()?;
486                    println!();
487                    // Draw bottom border with full width
488                    let (cols, _) = terminal::size().unwrap_or((80, 24));
489                    let bottom_border = "─".repeat(cols.saturating_sub(2) as usize);
490                    println!("{}└{}┘{}", Colors::GREEN, bottom_border, Colors::RESET);
491
492                    let result = input_state.input.clone();
493
494                    // Show task details after input is complete
495                    show_task_details_below_input(task_state);
496
497                    // Update task state with user input
498                    if !result.trim().is_empty() {
499                        task_state.current_task = format!(
500                            "Processing: {}",
501                            if result.len() > 30 {
502                                format!("{}...", &result[..27])
503                            } else {
504                                result.clone()
505                            }
506                        );
507                        task_state.current_reasoning = format!("User requested: {}", result);
508
509                        // Add new task to todo list
510                        task_state.todo_items.push(TodoItem {
511                            text: format!("Process: {}", result),
512                            completed: false,
513                            priority: TodoPriority::High,
514                            reasoning: format!("User requested to process: {}", result),
515                            tool_plan: Some("1. Parse user input\n2. Generate execution plan\n3. Execute required actions".to_string()),
516                            execution_results: None,
517                        });
518                    }
519
520                    return Ok(result);
521                }
522            }
523            InputChar::Backspace => {
524                if task_state.input_mode == InputMode::InputField {
525                    handle_backspace(&mut input_state, suggestion_tx).await?;
526                }
527            }
528            InputChar::CtrlC => {
529                super::input_handler::disable_raw_mode()?;
530                return Err(anyhow!("Interrupted"));
531            }
532            InputChar::CtrlT => {
533                // Toggle between input field and task selection
534                task_state.toggle_input_mode();
535
536                match task_state.input_mode {
537                    InputMode::TaskSelection => {
538                        clear_suggestions_display();
539                        task_state.current_reasoning =
540                            "Task selection mode active. Use ↑↓ to navigate, Enter to toggle."
541                                .to_string();
542                    }
543                    InputMode::InputField => {
544                        task_state.current_reasoning =
545                            "Input mode active. Type your command or query.".to_string();
546                    }
547                }
548
549                // Refresh display
550                print!("\x1B[2J\x1B[1;1H");
551                show_enhanced_status_bar(task_state);
552
553                if task_state.input_mode == InputMode::InputField {
554                    println!(
555                        "{}┌─ Input ──────────────────────────────────────────────┐{}",
556                        Colors::GREEN,
557                        Colors::RESET
558                    );
559                    print!(
560                        "{}│{} > {}",
561                        Colors::GREEN,
562                        Colors::RESET,
563                        input_state.input
564                    );
565                    io::stdout()
566                        .flush()
567                        .map_err(|e| anyhow!("Failed to flush stdout: {}", e))?;
568                }
569            }
570            InputChar::Escape => {
571                if task_state.input_mode == InputMode::TaskSelection {
572                    // Cancel task selection and return to input mode
573                    task_state.input_mode = InputMode::InputField;
574                    task_state.current_reasoning =
575                        "Cancelled task selection. Returned to input mode.".to_string();
576
577                    // Refresh display
578                    print!("\x1B[2J\x1B[1;1H");
579                    show_enhanced_status_bar(task_state);
580
581                    println!(
582                        "{}┌─ Input ──────────────────────────────────────────────┐{}",
583                        Colors::GREEN,
584                        Colors::RESET
585                    );
586                    print!(
587                        "{}│{} > {}",
588                        Colors::GREEN,
589                        Colors::RESET,
590                        input_state.input
591                    );
592                    io::stdout()
593                        .flush()
594                        .map_err(|e| anyhow!("Failed to flush stdout: {}", e))?;
595                } else {
596                    // In input mode, handle Esc based on suggestion state
597                    if !input_state.suggestions.is_empty() {
598                        // Suggestions are visible - restore original input and suppress
599                        if let Some(original) = input_state.original_before_sug.take() {
600                            input_state.input = original;
601                            input_state.cursor_pos = input_state.input.len();
602                        }
603                        input_state.suggestions.clear();
604                        input_state.selected_suggestion = 0;
605                        input_state.suggestions_suppressed = true;
606                        input_state.sug_win_start = 0;
607
608                        clear_suggestions_display();
609                        redraw_input_line(&input_state.input)?;
610                    } else {
611                        // No suggestions visible - clear input as before
612                        input_state.clear();
613                        clear_suggestions_display();
614                        redraw_input_line(&input_state.input)?;
615                    }
616                }
617            }
618            InputChar::Tab => {
619                if task_state.input_mode == InputMode::InputField {
620                    handle_tab_completion(&mut input_state).await?;
621                }
622            }
623            InputChar::Arrow(arrow_key) => match task_state.input_mode {
624                InputMode::InputField => {
625                    handle_arrow_key(&mut input_state, arrow_key).await?;
626                }
627                InputMode::TaskSelection => match arrow_key {
628                    ArrowKey::Up => {
629                        task_state.navigate_todo_up();
630                        if let Some(selected_task) = task_state.get_selected_task_details() {
631                            task_state.current_reasoning = selected_task.reasoning.clone();
632                        }
633                        print!("\x1B[2J\x1B[1;1H");
634                        show_enhanced_status_bar(task_state);
635                        show_task_details_below_input(task_state);
636                    }
637                    ArrowKey::Down => {
638                        task_state.navigate_todo_down();
639                        if let Some(selected_task) = task_state.get_selected_task_details() {
640                            task_state.current_reasoning = selected_task.reasoning.clone();
641                        }
642                        print!("\x1B[2J\x1B[1;1H");
643                        show_enhanced_status_bar(task_state);
644                        show_task_details_below_input(task_state);
645                    }
646                    _ => {}
647                },
648            },
649            InputChar::Regular(ch) => {
650                if task_state.input_mode == InputMode::InputField {
651                    handle_regular_character(&mut input_state, ch, suggestion_tx).await?;
652
653                    // Update reasoning based on input
654                    if input_state.input.len() > 0 {
655                        task_state.current_reasoning =
656                            format!("User typing: '{}'", input_state.input);
657                    }
658                }
659            }
660            InputChar::Unknown => {
661                debug!("Unknown character received in enhanced input");
662            }
663        }
664    }
665}

706/// Read and classify a single character from input
707fn read_single_character() -> Result<InputChar> {
708    let mut buffer = [0; 1];
709    if let Ok(1) = std::io::stdin().read(&mut buffer) {
710        match buffer[0] {
711            b'\n' | b'\r' => Ok(InputChar::Enter),
712            0x7f | 0x08 => Ok(InputChar::Backspace),
713            0x03 => Ok(InputChar::CtrlC),
714            0x14 => Ok(InputChar::CtrlT),
715            b'\t' => Ok(InputChar::Tab),
716            0x1b => {
717                // Handle escape sequences for arrow keys
718                let mut next_buffer = [0; 2];
719                if let Ok(2) = std::io::stdin().read(&mut next_buffer) {
720                    if next_buffer[0] == b'[' {
721                        match next_buffer[1] {
722                            b'A' => return Ok(InputChar::Arrow(ArrowKey::Up)),
723                            b'B' => return Ok(InputChar::Arrow(ArrowKey::Down)),
724                            b'C' => return Ok(InputChar::Arrow(ArrowKey::Right)),
725                            b'D' => return Ok(InputChar::Arrow(ArrowKey::Left)),
726                            _ => {}
727                        }
728                    }
729                }
730                Ok(InputChar::Escape)
731            }
732            ch if ch >= 0x20 && ch < 0x7f => Ok(InputChar::Regular(ch as char)),
733            _ => Ok(InputChar::Unknown),
734        }
735    } else {
736        Err(anyhow!("Failed to read character from stdin"))
737    }
738}

779/// Handle Tab completion - adopt suggestion and advance selector
780async fn handle_tab_completion(state: &mut InputState) -> Result<()> {
781    // Enable suggestions if they were suppressed
782    state.suggestions_suppressed = false;
783
784    if !state.suggestions.is_empty() && state.selected_suggestion < state.suggestions.len() {
785        // Snapshot original input if this is the first suggestion interaction
786        if state.original_before_sug.is_none() {
787            state.original_before_sug = Some(state.input.clone());
788        }
789
790        let suggestion = &state.suggestions[state.selected_suggestion];
791
792        // Update input with current suggestion
793        state.input = suggestion.text.clone();
794        state.cursor_pos = state.input.len();
795
796        // Advance selector to next suggestion (with wrap-around)
797        state.selected_suggestion = (state.selected_suggestion + 1) % state.suggestions.len();
798
799        // Update window if needed
800        adjust_suggestion_window(state);
801
802        // Redraw input line and suggestions
803        if let Err(e) = redraw_input_line(&state.input) {
804            error!("Failed to redraw input line: {}", e);
805        }
806
807        // Get new suggestions for the updated input
808        state.suggestions = get_instant_suggestions(&state.input).await;
809        if !state.suggestions.is_empty() {
810            show_navigable_suggestions_windowed(&state.suggestions, state);
811        }
812    }
813    Ok(())
814}

910/// Navigate command history down
911async fn navigate_history_down(state: &mut InputState) -> Result<()> {
912    if state.history_index < state.command_history.len() - 1 {
913        state.history_index += 1;
914        state.input = state.command_history[state.history_index].clone();
915        state.cursor_pos = state.input.len();
916
917        if let Err(e) = redraw_input_line(&state.input) {
918            error!("Failed to redraw input line: {}", e);
919        }
920
921        // Show suggestions for history item
922        state.suggestions = get_instant_suggestions(&state.input).await;
923        state.reset_selection();
924        if !state.suggestions.is_empty() {
925            show_navigable_suggestions_windowed(&state.suggestions, state);
926        }
927    } else if state.history_index == state.command_history.len() - 1 {
928        // Clear input when going past last history item
929        state.history_index = state.command_history.len();
930        state.input.clear();
931        state.cursor_pos = 0;

933        if let Err(e) = redraw_input_line(&state.input) {
934            error!("Failed to redraw input line: {}", e);
935        }
936
937        clear_suggestions_display();
938        state.suggestions.clear();
939    }
940    Ok(())

965/// Update suggestions for current input with suppression logic
966async fn update_suggestions_for_input(
967    state: &mut InputState,
968    suggestion_tx: &mpsc::UnboundedSender<String>,
969) -> Result<()> {
970    // Don't show suggestions if they are suppressed
971    if state.suggestions_suppressed {
972        return Ok(());
973    }
974
975    if state.input.len() > 0 {
976        // Send to background suggestion generation
977        if let Err(e) = suggestion_tx.send(state.input.clone()) {
978            warn!("Failed to send suggestion request: {}", e);
979        }
980
981        state.suggestions = get_instant_suggestions(&state.input).await;
982        if !state.suggestions.is_empty() {
983            show_navigable_suggestions_windowed(&state.suggestions, state);
984        }
985    } else {
986        // Clear suggestions when input is empty
987        clear_suggestions_display();
988        state.suggestions.clear();
989    }
990    Ok(())
991}

993/// Show navigable suggestions with windowed scrolling and reverse-video highlighting
994fn show_navigable_suggestions_windowed(suggestions: &[RealtimeSuggestion], state: &InputState) {
995    if suggestions.is_empty() {
996        return;
997    }
998
999    // Determine terminal dimensions
1000    let (cols, rows) = terminal::size().unwrap_or((80, 24));
1001    let inner_width = cols.saturating_sub(2) as usize;
1002    let win_height = state.win_height.min(suggestions.len()).min(6);
1003    let box_height = win_height + 3; // borders + hint
1004
1005    // Calculate window slice
1006    let win_start = state.sug_win_start;
1007    let win_end = (win_start + win_height).min(suggestions.len());
1008    let visible_suggestions = &suggestions[win_start..win_end];
1009
1010    // Calculate absolute position for suggestion box (above input)
1011    let input_row = rows.saturating_sub(3); // Fix: Use same calculation as input box
1012    let max_box_height = 9; // Always clear max possible height (6 suggestions + 3 borders)
1013    let suggestion_start_row = input_row.saturating_sub(max_box_height as u16);
1014
1015    // Clear maximum possible suggestion area to prevent artifacts
1016    for i in 0..max_box_height {
1017        print!("\x1B[{};1H\x1B[2K", suggestion_start_row + i as u16);
1018    }
1019
1020    // Recalculate actual start position for current box
1021    let actual_start_row = input_row.saturating_sub(box_height as u16);
1022
1023    // Draw top border at actual position
1024    print!("\x1B[{};1H", actual_start_row);
1025    println!(
1026        "{}┌─ Suggestions {}",
1027        Colors::DIM,
1028        "─".repeat(inner_width.saturating_sub(15))
1029    );
1030
1031    // Display windowed suggestions with reverse-video highlighting
1032    for (i, suggestion) in visible_suggestions.iter().enumerate() {
1033        let global_index = win_start + i;
1034        let is_selected = global_index == state.selected_suggestion;
1035
1036        let color = match suggestion.category.as_str() {
1037            "command" => Colors::CYAN,
1038            "query" => Colors::GREEN,
1039            "action" => Colors::MAGENTA,
1040            "address" => Colors::YELLOW,
1041            _ => Colors::BLUE,
1042        };
1043
1044        // Available width minus fixed elements
1045        let content_width = inner_width.saturating_sub(4);
1046        let max_text_width = content_width.min(25);
1047        let max_desc_width = content_width
1048            .saturating_sub(max_text_width)
1049            .saturating_sub(3);
1050
1051        let display_text = if suggestion.text.len() > max_text_width {
1052            format!(
1053                "{}...",
1054                &suggestion.text[..max_text_width.saturating_sub(3)]
1055            )
1056        } else {
1057            suggestion.text.clone()
1058        };
1059
1060        let display_desc = if suggestion.description.len() > max_desc_width {
1061            format!(
1062                "{}...",
1063                &suggestion.description[..max_desc_width.saturating_sub(3)]
1064            )
1065        } else {
1066            suggestion.description.clone()
1067        };
1068
1069        // Position cursor and draw suggestion line
1070        print!("\x1B[{};1H", actual_start_row + 1 + i as u16);
1071        if is_selected {
1072            // Reverse video for selected row
1073            println!(
1074                "{}│\x1B[7m {}{:<25}{} - {}{:<23}{}\x1B[27m │{}",
1075                Colors::DIM,
1076                color,
1077                display_text,
1078                Colors::RESET,
1079                Colors::DIM,
1080                display_desc,
1081                Colors::RESET,
1082                Colors::DIM
1083            );
1084        } else {
1085            println!(
1086                "{}│ {}{:<25}{} - {}{:<23}{} │{}",
1087                Colors::DIM,
1088                color,
1089                display_text,
1090                Colors::RESET,
1091                Colors::DIM,
1092                display_desc,
1093                Colors::RESET,
1094                Colors::DIM
1095            );
1096        }
1097    }
1098
1099    // Show navigation hint with scroll indicator
1100    let hint_row = actual_start_row + win_height as u16 + 1;
1101    print!("\x1B[{};1H", hint_row);
1102    let scroll_info = if suggestions.len() > win_height {
1103        format!(
1104            " ({}/{}) ",
1105            state.selected_suggestion + 1,
1106            suggestions.len()
1107        )
1108    } else {
1109        String::new()
1110    };
1111    let hint = format!("↑↓ Navigate • Tab Complete • Enter Submit{}", scroll_info);
1112    println!(
1113        "{}│{} {:<width$}{} │{}",
1114        Colors::DIM,
1115        Colors::DIM,
1116        hint,
1117        Colors::RESET,
1118        Colors::DIM,
1119        width = inner_width.saturating_sub(2)
1120    );
1121
1122    // Bottom border
1123    let bottom_row = actual_start_row + box_height as u16 - 1;
1124    print!("\x1B[{};1H", bottom_row);
1125    println!(
1126        "{}└{}┘{}",
1127        Colors::DIM,
1128        "─".repeat(inner_width),
1129        Colors::RESET
1130    );
1131
1132    // Return cursor to input line
1133    print!("\x1B[{};4H", input_row + 1); // Position after "> "
1134}

2537/// Wrap text to specified width
2538fn wrap_text(text: &str, width: usize) -> Vec<String> {
2539    let mut lines = Vec::new();
2540    let mut current_line = String::new();
2541
2542    for word in text.split_whitespace() {
2543        if current_line.is_empty() {
2544            current_line = word.to_string();
2545        } else if current_line.len() + 1 + word.len() <= width {
2546            current_line.push(' ');
2547            current_line.push_str(word);
2548        } else {
2549            lines.push(current_line);
2550            current_line = word.to_string();
2551        }
2552    }
2553
2554    if !current_line.is_empty() {
2555        lines.push(current_line);
2556    }
2557
2558    if lines.is_empty() {
2559        lines.push("No plan details available".to_string());
2560    }
2561
2562    lines
2563}

2584/// Redraw input line cleanly with error handling - keep it simple
2585fn redraw_input_line(input: &str) -> Result<()> {
2586    // Determine the terminal width dynamically
2587    let (cols, _rows) = terminal::size().unwrap_or((80, 24));
2588    // "│ > " prefix (4) + trailing " │" (2) = 6 chars for borders/prefix
2589    let max_input_width = cols.saturating_sub(6) as usize;
2590
2591    // Truncate if too long
2592    let display_input = if input.len() > max_input_width {
2593        // Leave space for "..."
2594        let slice_len = max_input_width.saturating_sub(3);
2595        format!("{}...", &input[..slice_len])
2596    } else {
2597        input.to_string()
2598    };
2599
2600    // Clear current line, return carriage, redraw
2601    print!(
2602        "\x1B[2K\r{}│{} > {:<width$}{} │{}",
2603        Colors::GREEN,
2604        Colors::RESET,
2605        display_input,
2606        Colors::RESET,
2607        Colors::GREEN,
2608        width = max_input_width
2609    );
2610
2611    if let Err(e) = io::stdout().flush() {
2612        error!("Failed to flush stdout during input redraw: {}", e);
