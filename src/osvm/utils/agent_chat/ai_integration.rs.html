85/// Show animated status message
86pub async fn show_animated_status(message: &str, chars: &str, duration_ms: u64) {
87    let frames: Vec<char> = chars.chars().collect();
88
89    for frame in frames.iter().take(10) {
90        print!(
91            "\r{}{}  {} {}{}",
92            Colors::YELLOW,
93            frame,
94            message,
95            Colors::DIM,
96            Colors::RESET
97        );
98        io::stdout().flush().unwrap_or(());
99        sleep(Duration::from_millis(duration_ms)).await;
100    }
101
102    print!("\r\x1b[K");
103    io::stdout().flush().unwrap_or(());
104}

162/// Wrap text to specified width
163pub fn wrap_text(text: &str, width: usize) -> Vec<String> {
164    let mut lines = Vec::new();
165    let mut current_line = String::new();
166
167    for word in text.split_whitespace() {
168        if current_line.len() + word.len() + 1 > width {
169            if !current_line.is_empty() {
170                lines.push(current_line.clone());
171                current_line.clear();
172            }
173        }
174
175        if !current_line.is_empty() {
176            current_line.push(' ');
177        }
178        current_line.push_str(word);
179    }
180
181    if !current_line.is_empty() {
182        lines.push(current_line);
183    }
184
185    lines
186}

188/// Get user choice (1 for yes, 0 for no)
189pub async fn get_user_choice() -> Result<u32> {
190    let mut buffer = String::new();
191    io::stdin().read_line(&mut buffer)?;
192
193    match buffer.trim().to_lowercase().as_str() {
194        "y" | "yes" => Ok(1),
195        _ => Ok(0),
196    }
197}

303/// Show context visualization
304pub async fn show_context_visualization(
305    chat_history: &[String],
306    ai_service: &Arc<AiService>,
307) -> Result<()> {
308    println!(
309        "\n{}╔════════════════════════════════════════════════╗",
310        Colors::BLUE
311    );
312    println!(
313        "║           {}CONVERSATION CONTEXT{}                 ║",
314        Colors::BOLD,
315        Colors::BLUE
316    );
317    println!(
318        "╚════════════════════════════════════════════════╝{}",
319        Colors::RESET
320    );
321
322    if chat_history.is_empty() {
323        println!(
324            "{}  No conversation history yet{}",
325            Colors::DIM,
326            Colors::RESET
327        );
328    } else {
329        println!("\n{}Recent Messages:{}", Colors::CYAN, Colors::RESET);
330        for (i, msg) in chat_history.iter().rev().take(5).enumerate() {
331            let truncated = if msg.chars().count() > 60 {
332                let truncated_chars: String = msg.chars().take(57).collect();
333                format!("{}...", truncated_chars)
334            } else {
335                msg.clone()
336            };
337            println!("  {}. {}", i + 1, truncated);
338        }
339
340        println!("\n{}Statistics:{}", Colors::YELLOW, Colors::RESET);
341        println!("  • Total messages: {}", chat_history.len());
342        println!("  • Current session: Active");
343    }
344
345    Ok(())
346}
