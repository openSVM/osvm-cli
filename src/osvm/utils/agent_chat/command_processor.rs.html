21fn screenshot_command(args: &[&str]) -> Result<CommandResult> {
22    // Parse arguments for --terminal-only or --fullscreen
23    let terminal_only = args.contains(&"--terminal-only");
24    let fullscreen = args.contains(&"--fullscreen");
25
26    // Default to terminal-only if no flag specified
27    let mode = if fullscreen {
28        false
29    } else {
30        // Default or explicitly specified
31        true
32    };
33
34    Ok(CommandResult::Screenshot(mode))
35}

87/// Command processor for handling various input types
88pub struct CommandProcessor {
89    commands: HashMap<String, CommandHandler>,
90}
91
92/// Handler function type for commands - simplified to avoid lifetime issues
93type CommandHandler = fn(&[&str]) -> Result<CommandResult>;
94
95/// Result of command execution
96pub enum CommandResult {
97    Success(String),
98    Error(String),
99    Exit,
100    Clear,
101    Continue,
102    Screenshot(bool), // bool = terminal_only (true) or fullscreen (false)
103}
104
105impl CommandProcessor {
106    /// Create new command processor with default handlers
107    pub fn new() -> Self {
108        let mut commands = HashMap::new();
109
110        // Register built-in commands
111        commands.insert("/help".to_string(), help_command as CommandHandler);
112        commands.insert("/clear".to_string(), clear_command as CommandHandler);
113        commands.insert("/exit".to_string(), exit_command as CommandHandler);
114        commands.insert("/quit".to_string(), exit_command as CommandHandler);
115        commands.insert("/screenshot".to_string(), screenshot_command as CommandHandler);
116
117        Self { commands }
118    }
119
120    /// Process input and determine action
121    pub fn process(&self, input: &str) -> Result<CommandResult> {
122        let input = input.trim();
123
124        // Check for empty input
125        if input.is_empty() {
126            return Ok(CommandResult::Continue);
127        }
128
129        // Parse command and arguments
130        let parts: Vec<&str> = input.split_whitespace().collect();
131        let command = parts.get(0).unwrap_or(&"");
132
133        // Check for registered commands
134        if let Some(handler) = self.commands.get(*command) {
135            return handler(&parts[1..]);
136        }
137
138        // Check for special command patterns
139        if command.starts_with('/') {
140            return Ok(CommandResult::Error(format!(
141                "Unknown command: {}",
142                command
143            )));
144        }
145
146        if command.starts_with('@') {
147            return self.process_tool_invocation(input);
148        }
149
150        // Default: process as regular message
151        Ok(CommandResult::Continue)
152    }
153
154    /// Process MCP tool invocation
155    fn process_tool_invocation(&self, input: &str) -> Result<CommandResult> {
156        // Parse @server/tool syntax with bounds checking
157        let parts: Vec<&str> = input[1..].split('/').collect();
158
159        if parts.len() != 2 {
160            return Ok(CommandResult::Error(
161                "Invalid tool syntax. Use @server/tool".to_string(),
162            ));
163        }
164
165        // Safe array access with bounds checking
166        let server = parts.get(0).ok_or_else(|| anyhow!("Missing server name"))?;
167        let tool = parts.get(1).ok_or_else(|| anyhow!("Missing tool name"))?;
168
169        if server.is_empty() || tool.is_empty() {
170            return Ok(CommandResult::Error(
171                "Server and tool names cannot be empty".to_string(),
172            ));
173        }
174
175        Ok(CommandResult::Success(format!(
176            "{}Executing tool: {}/{}{}",
177            Colors::YELLOW,
178            server,
179            tool,
180            Colors::RESET
181        )))
182    }
183
184    /// Register custom command handler
185    pub fn register_command(&mut self, command: &str, handler: CommandHandler) {
186        self.commands.insert(command.to_string(), handler);
187    }
188
189    /// Check if input is a command
190    pub fn is_command(input: &str) -> bool {
191        input.trim().starts_with('/') || input.trim().starts_with('@')
192    }
193
194    /// Parse tool invocation with safe array access
195    pub fn parse_tool_invocation(input: &str) -> Option<(String, String, Vec<String>)> {
196        if !input.starts_with('@') {
197            return None;
198        }
199
200        let parts: Vec<&str> = input[1..].split_whitespace().collect();
201        if parts.is_empty() {
202            return None;
203        }
204
205        let server_tool: Vec<&str> = parts.get(0)?.split('/').collect();
206        if server_tool.len() != 2 {
207            return None;
208        }
209
210        // Safe array access with bounds checking
211        let server = server_tool.get(0)?.to_string();
212        let tool = server_tool.get(1)?.to_string();
213
214        if server.is_empty() || tool.is_empty() {
215            return None;
216        }
217
218        let args: Vec<String> = parts
219            .get(1..)
220            .unwrap_or(&[])
221            .iter()
222            .map(|s| s.to_string())
223            .collect();
224
225        Some((server, tool, args))
226    }
227}

229/// Command context for advanced processing
230pub struct CommandContext {
231    pub chat_history: Vec<String>,
232    pub mcp_servers: Vec<(String, bool)>, // (server_id, enabled)
233    pub current_task: Option<String>,
234}
235
236impl CommandContext {
237    pub fn new() -> Self {
238        Self {
239            chat_history: Vec::new(),
240            mcp_servers: Vec::new(),
241            current_task: None,
242        }
243    }
244
245    /// Add message to history
246    pub fn add_to_history(&mut self, message: String) {
247        self.chat_history.push(message);
248
249        // Limit history size
250        if self.chat_history.len() > 1000 {
251            self.chat_history.remove(0);
252        }
253    }
254
255    /// Get recent history
256    pub fn get_recent_history(&self, count: usize) -> Vec<String> {
257        self.chat_history
258            .iter()
259            .rev()
260            .take(count)
261            .rev()
262            .cloned()
263            .collect()
264    }
265
266    /// Update MCP servers list
267    pub fn update_servers(&mut self, servers: Vec<(String, bool)>) {
268        self.mcp_servers = servers;
269    }
270
271    /// Get active servers
272    pub fn get_active_servers(&self) -> Vec<String> {
273        self.mcp_servers
274            .iter()
275            .filter(|(_, enabled)| *enabled)
276            .map(|(id, _)| id.clone())
277            .collect()
278    }
