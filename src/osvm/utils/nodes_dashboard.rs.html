10/// Generate HTML content for a monitoring dashboard

48    html.push_str("            margin: 0;\n");

50    html.push_str("        .last-update {\n");
51    html.push_str("            color: #666;\n");
52    html.push_str("            font-size: 0.9em;\n");
53    html.push_str("        }\n");
54    html.push_str("        .dashboard-grid {\n");
55    html.push_str("            display: grid;\n");
56    html.push_str("            grid-template-columns: repeat(auto-fit, minmax(400px, 1fr));\n");

60    html.push_str("        .card {\n");
61    html.push_str("            background: #fff;\n");
62    html.push_str("            border-radius: 8px;\n");

64    html.push_str("            padding: 20px;\n");
65    html.push_str("        }\n");
66    html.push_str("        .card h2 {\n");
67    html.push_str("            margin-top: 0;\n");
68    html.push_str("            color: #5f6368;\n");
69    html.push_str("            font-size: 1.2em;\n");
70    html.push_str("            border-bottom: 1px solid #eee;\n");
71    html.push_str("            padding-bottom: 10px;\n");
72    html.push_str("        }\n");
73    html.push_str("        .stat-grid {\n");
74    html.push_str("            display: grid;\n");
75    html.push_str("            grid-template-columns: repeat(auto-fit, minmax(120px, 1fr));\n");
76    html.push_str("            grid-gap: 15px;\n");
77    html.push_str("        }\n");
78    html.push_str("        .stat-box {\n");
79    html.push_str("            text-align: center;\n");
80    html.push_str("            padding: 15px;\n");
81    html.push_str("            border-radius: 6px;\n");
82    html.push_str("            background: #f8f9fa;\n");
83    html.push_str("        }\n");
84    html.push_str("        .stat-value {\n");
85    html.push_str("            font-size: 1.8em;\n");

87    html.push_str("            margin: 5px 0;\n");
88    html.push_str("            color: #4285f4;\n");
89    html.push_str("        }\n");
90    html.push_str("        .stat-label {\n");
91    html.push_str("            font-size: 0.9em;\n");
92    html.push_str("            color: #5f6368;\n");
93    html.push_str("        }\n");
94    html.push_str("        table {\n");
95    html.push_str("            width: 100%;\n");
96    html.push_str("            border-collapse: collapse;\n");
97    html.push_str("            margin-top: 15px;\n");
98    html.push_str("        }\n");
99    html.push_str("        th, td {\n");
100    html.push_str("            text-align: left;\n");
101    html.push_str("            padding: 12px;\n");
102    html.push_str("            border-bottom: 1px solid #eee;\n");
103    html.push_str("        }\n");
104    html.push_str("        th {\n");
105    html.push_str("            background-color: #f8f9fa;\n");
106    html.push_str("            color: #5f6368;\n");
107    html.push_str("        }\n");
108    html.push_str("        tr:hover {\n");
109    html.push_str("            background-color: #f8f9fa;\n");
110    html.push_str("        }\n");
111    html.push_str("        .status-running {\n");
112    html.push_str("            color: #34a853;\n");
113    html.push_str("            font-weight: bold;\n");
114    html.push_str("        }\n");
115    html.push_str("        .status-stopped {\n");
116    html.push_str("            color: #fbbc05;\n");
117    html.push_str("            font-weight: bold;\n");
118    html.push_str("        }\n");
119    html.push_str("        .status-error {\n");
120    html.push_str("            color: #ea4335;\n");
121    html.push_str("            font-weight: bold;\n");
122    html.push_str("        }\n");
123    html.push_str("        .status-unknown {\n");
124    html.push_str("            color: #9aa0a6;\n");
125    html.push_str("            font-weight: bold;\n");
126    html.push_str("        }\n");
127    html.push_str("        .actions button {\n");
128    html.push_str("            background: #4285f4;\n");
129    html.push_str("            color: white;\n");
130    html.push_str("            border: none;\n");
131    html.push_str("            padding: 6px 12px;\n");
132    html.push_str("            border-radius: 4px;\n");
133    html.push_str("            cursor: pointer;\n");
134    html.push_str("            font-size: 0.9em;\n");
135    html.push_str("            margin-right: 5px;\n");
136    html.push_str("        }\n");
137    html.push_str("        .actions button:hover {\n");
138    html.push_str("            background: #3b78e7;\n");
139    html.push_str("        }\n");
140    html.push_str("    </style>\n");
141    html.push_str("</head>\n<body>\n");
142
143    // Header
144    let current_time = Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
145    html.push_str("<div class=\"header\">\n");
146    html.push_str("    <h1>OSVM Node Monitoring Dashboard</h1>\n");
147    html.push_str(&format!(
148        "    <div class=\"last-update\">Last updated: {}</div>\n",
149        current_time
150    ));
151
152    // Add verbosity indicator if debug level
153    if verbosity >= 3 {
154        html.push_str("    <div style=\"margin-left: 20px; color: #888;\">\n");
155        html.push_str(&format!(
156            "        <small>Debug mode: Verbosity level {}</small>\n",
157            verbosity
158        ));
159        html.push_str("    </div>\n");
160    }
161    html.push_str("</div>\n");
162
163    // Summary statistics
164    let mut running = 0;
165    let mut stopped = 0;
166    let mut error = 0;
167    let mut unknown = 0;
168
169    for node in nodes {
170        match node.status {
171            NodeStatus::Running => running += 1,
172            NodeStatus::Stopped => stopped += 1,
173            NodeStatus::Error => error += 1,
174            NodeStatus::Unknown => unknown += 1,
175        }
176    }
177
178    html.push_str("<div class=\"dashboard-grid\">\n");
179    html.push_str("    <div class=\"card\">\n");
180    html.push_str("        <h2>Node Status Summary</h2>\n");
181    html.push_str("        <div class=\"stat-grid\">\n");
182
183    html.push_str("            <div class=\"stat-box\">\n");
184    html.push_str(&format!(
185        "                <div class=\"stat-value\">{}</div>\n",
186        nodes.len()
187    ));
188    html.push_str("                <div class=\"stat-label\">Total Nodes</div>\n");
189    html.push_str("            </div>\n");
190
191    html.push_str("            <div class=\"stat-box\" style=\"background-color: #e6f4ea;\">\n");
192    html.push_str(&format!(
193        "                <div class=\"stat-value\" style=\"color: #34a853;\">{}</div>\n",
194        running
195    ));
196    html.push_str("                <div class=\"stat-label\">Running</div>\n");
197    html.push_str("            </div>\n");
198
199    html.push_str("            <div class=\"stat-box\" style=\"background-color: #fef7e0;\">\n");
200    html.push_str(&format!(
201        "                <div class=\"stat-value\" style=\"color: #fbbc05;\">{}</div>\n",
202        stopped
203    ));
204    html.push_str("                <div class=\"stat-label\">Stopped</div>\n");
205    html.push_str("            </div>\n");
206
207    html.push_str("            <div class=\"stat-box\" style=\"background-color: #fce8e6;\">\n");
208    html.push_str(&format!(
209        "                <div class=\"stat-value\" style=\"color: #ea4335;\">{}</div>\n",
210        error
211    ));
212    html.push_str("                <div class=\"stat-label\">Error</div>\n");
213    html.push_str("            </div>\n");
214
215    html.push_str("            <div class=\"stat-box\" style=\"background-color: #f1f3f4;\">\n");
216    html.push_str(&format!(
217        "                <div class=\"stat-value\" style=\"color: #9aa0a6;\">{}</div>\n",
218        unknown
219    ));
220    html.push_str("                <div class=\"stat-label\">Unknown</div>\n");
221    html.push_str("            </div>\n");
222    html.push_str("        </div>\n");
223    html.push_str("    </div>\n");
224
225    // SVM distribution
226    if verbosity >= 1 {
227        let mut svm_counts = HashMap::new();
228        for node in nodes {
229            *svm_counts.entry(node.svm_type.clone()).or_insert(0) += 1;
230        }
231
232        html.push_str("    <div class=\"card\">\n");
233        html.push_str("        <h2>SVM Distribution</h2>\n");
234        html.push_str("        <table>\n");
235        html.push_str("            <thead>\n");
236        html.push_str("                <tr>\n");
237        html.push_str("                    <th>SVM Type</th>\n");
238        html.push_str("                    <th>Count</th>\n");
239        html.push_str("                    <th>Percentage</th>\n");
240        html.push_str("                </tr>\n");
241        html.push_str("            </thead>\n");
242        html.push_str("            <tbody>\n");
243
244        for (svm_type, count) in svm_counts {
245            let percentage = (count as f64 / nodes.len() as f64) * 100.0;
246            html.push_str("                <tr>\n");
247            html.push_str(&format!("                    <td>{}</td>\n", svm_type));
248            html.push_str(&format!("                    <td>{}</td>\n", count));
249            html.push_str(&format!(
250                "                    <td>{:.1}%</td>\n",
251                percentage
252            ));
253            html.push_str("                </tr>\n");
254        }
255
256        html.push_str("            </tbody>\n");
257        html.push_str("        </table>\n");
258        html.push_str("    </div>\n");
259    }
260    html.push_str("</div>\n");
261
262    // Node list table
263    html.push_str("<div class=\"card\">\n");
264    html.push_str("    <h2>Node List</h2>\n");
265    html.push_str("    <table>\n");
266    html.push_str("        <thead>\n");
267    html.push_str("            <tr>\n");
268    html.push_str("                <th>ID</th>\n");
269    html.push_str("                <th>Name</th>\n");
270    html.push_str("                <th>SVM Type</th>\n");
271    html.push_str("                <th>Node Type</th>\n");
272
273    // Only show network/host with verbosity >= 1
274    if verbosity >= 1 {
275        html.push_str("                <th>Network</th>\n");
276        html.push_str("                <th>Host</th>\n");
277    }
278
279    // Add more detail columns for higher verbosity
280    if verbosity >= 2 {
281        html.push_str("                <th>Last Updated</th>\n");
282        html.push_str("                <th>Version</th>\n");
283    }
284    html.push_str("                <th>Status</th>\n");
285    html.push_str("                <th>Actions</th>\n");
286    html.push_str("            </tr>\n");
287    html.push_str("        </thead>\n");
288    html.push_str("        <tbody>\n");
289
290    for node in nodes {
291        let status_class = match node.status {
292            NodeStatus::Running => "status-running",
293            NodeStatus::Stopped => "status-stopped",
294            NodeStatus::Error => "status-error",
295            NodeStatus::Unknown => "status-unknown",
296        };
297
298        let status_text = match node.status {
299            NodeStatus::Running => "● Running",
300            NodeStatus::Stopped => "○ Stopped",
301            NodeStatus::Error => "✕ Error",
302            NodeStatus::Unknown => "? Unknown",
303        };
304
305        html.push_str("            <tr>\n");
306        html.push_str(&format!("                <td>{}</td>\n", node.id));
307        html.push_str(&format!("                <td>{}</td>\n", node.name));
308        html.push_str(&format!("                <td>{}</td>\n", node.svm_type));
309        html.push_str(&format!("                <td>{}</td>\n", node.node_type));
310
311        // Only show network/host with verbosity >= 1
312        if verbosity >= 1 {
313            html.push_str(&format!("                <td>{}</td>\n", node.network));
314            html.push_str(&format!("                <td>{}</td>\n", node.host));
315        }
316
317        // Add extra details for higher verbosity
318        if verbosity >= 2 {
319            html.push_str(&format!(
320                "                <td>{}</td>\n",
321                Local::now().format("%Y-%m-%d %H:%M")
322            ));
323            html.push_str(&format!("                <td>v{}</td>\n", "1.0.0")); // Placeholder for version
324        }
325        html.push_str(&format!(
326            "                <td class=\"{}\">{}</td>\n",
327            status_class, status_text
328        ));
329        html.push_str("                <td class=\"actions\">\n");
330
331        // Add action buttons based on status
332        match node.status {
333            NodeStatus::Running => {
334                html.push_str(&format!("                    <button onclick=\"alert('Would restart node {}')\">Restart</button>\n", node.id));
335                html.push_str(&format!("                    <button onclick=\"alert('Would stop node {}')\">Stop</button>\n", node.id));
336            }
337            NodeStatus::Stopped => {
338                html.push_str(&format!("                    <button onclick=\"alert('Would start node {}')\">Start</button>\n", node.id));
339            }
340            NodeStatus::Error => {
341                html.push_str(&format!("                    <button onclick=\"alert('Would restart node {}')\">Restart</button>\n", node.id));
342                html.push_str(&format!("                    <button onclick=\"alert('Would view logs for node {}')\">View Logs</button>\n", node.id));
343            }
344            NodeStatus::Unknown => {
345                html.push_str(&format!("                    <button onclick=\"alert('Would check status of node {}')\">Check Status</button>\n", node.id));
346            }
347        }
348
349        html.push_str("                </td>\n");
350        html.push_str("            </tr>\n");
351    }
352
353    html.push_str("        </tbody>\n");
354    html.push_str("    </table>\n");
355    html.push_str("</div>\n");
356
357    // Add debug information section if verbosity is high
358    if verbosity >= 3 {
359        html.push_str("<div class=\"card\">\n");
360        html.push_str("    <h2>Debug Information</h2>\n");
361        html.push_str("    <pre style=\"background-color: #f1f3f4; padding: 15px; overflow: auto; font-size: 12px;\">\n");
362        html.push_str(&format!(
363            "Dashboard generated with verbosity level: {}\nTotal nodes: {}\nTimestamp: {}\n",
364            verbosity,
365            nodes.len(),
366            current_time
367        ));
368        html.push_str("    </pre>\n");
369        html.push_str("</div>\n");
370    }
371
372    // Close HTML
373    html.push_str("<script>\n");
374    html.push_str("    // In a real implementation, this would include JavaScript to update data in real-time\n");
375    html.push_str("    // and handle button clicks for node actions\n");
376
377    // Add different console logs based on verbosity
378    match verbosity {
379        0 => html.push_str("    console.log('Dashboard loaded');\n"),
380        1 => html.push_str("    console.log('Dashboard loaded with standard verbosity');\n"),
381        2 => html.push_str("    console.log('Dashboard loaded with detailed verbosity');\n"),
382        _ => {
383            html.push_str("    console.log('Dashboard loaded in debug mode');\n");
384            html.push_str("    console.log('Nodes:', ");
385            html.push_str(&format!(
386                "{}); // This would normally show node details\n",
387                nodes.len()
388            ));
389        }
390    }
391
392    html.push_str("</script>\n");
393    html.push_str("</body>\n");
394    html.push_str("</html>");
395
396    html
