1//! RPC Bridge for OVSM - allows OVSM scripts to call Solana RPC methods
2
3use ovsm::runtime::Value as OvsmValue;
4use ovsm::tools::{Tool, ToolRegistry};
5use ovsm::error::Result as OvsmResult;
6use serde_json::{json, Value};
7use std::sync::Arc;
8
9// Real Solana RPC endpoint
10const SOLANA_RPC_URL: &str = "https://osvm.ai/api/proxy/rpc";
11
12/// Generic RPC Bridge Tool - Dynamically calls ANY Solana RPC method
13pub struct RpcBridgeTool {
14    name: String,
15}
16
17impl RpcBridgeTool {
18    pub fn new(name: &str) -> Self {
19        RpcBridgeTool {
20            name: name.to_string(),
21        }
22    }
23}
24
25impl Tool for RpcBridgeTool {
26    fn name(&self) -> &str {
27        &self.name
28    }
29
30    fn description(&self) -> &str {
31        "Dynamic RPC bridge - calls Solana RPC methods"
32    }
33
34    fn execute(&self, args: &[OvsmValue]) -> OvsmResult<OvsmValue> {
35        let mut rpc_params = Vec::new();
36
37        // OVSM evaluator passes named parameters as positional arguments
38        // For RPC calls like getSignaturesForAddress(address: "...", limit: 5)
39        // We receive: [String("..."), Int(5)]
40        // We need to convert to: ["...", {"limit": 5}]
41
42        if args.is_empty() {
43            // No arguments - just call the method
44        } else if args.len() == 1 {
45            // Single argument - could be just primary param or an object
46            if let OvsmValue::Object(obj) = &args[0] {
47                // If it's an object, check if it's a config or has a primary param
48                let primary_param = obj.get("address")
49                    .or_else(|| obj.get("signature"))
50                    .or_else(|| obj.get("slot"))
51                    .or_else(|| obj.get("pubkey"));
52
53                if let Some(primary) = primary_param {
54                    rpc_params.push(ovsm_value_to_json(primary));
55
56                    // Build config object from remaining named parameters
57                    let mut config = serde_json::Map::new();
58                    for (key, val) in obj.iter() {
59                        if !matches!(key.as_str(), "address" | "signature" | "slot" | "pubkey") {
60                            config.insert(key.clone(), ovsm_value_to_json(val));
61                        }
62                    }
63
64                    // Only add config object if it has fields
65                    if !config.is_empty() {
66                        rpc_params.push(Value::Object(config));
67                    }
68                } else {
69                    // No primary param found, use the whole object as config
70                    rpc_params.push(ovsm_value_to_json(&args[0]));
71                }
72            } else {
73                // Single non-object argument (primary param only)
74                rpc_params.push(ovsm_value_to_json(&args[0]));
75            }
76        } else {
77            // Multiple arguments - first is primary param, rest form the config object
78            // This handles cases like: getSignaturesForAddress(address: "...", limit: 5, before: "...")
79            rpc_params.push(ovsm_value_to_json(&args[0]));
80
81            // Build config object from remaining arguments
82            // Check if second argument is already an object (named params)
83            if args.len() == 2 && matches!(&args[1], OvsmValue::Object(_)) {
84                // Second arg is a config object, use it directly
85                rpc_params.push(ovsm_value_to_json(&args[1]));
86            } else {
87                // Build config from positional args
88                let mut config = serde_json::Map::new();
89
90                // For getSignaturesForAddress specifically
91                if matches!(self.name.as_str(), "getSignaturesForAddress") {
92                    if args.len() >= 2 {
93                        config.insert("limit".to_string(), ovsm_value_to_json(&args[1]));
94                    }
95                    if args.len() >= 3 {
96                        config.insert("before".to_string(), ovsm_value_to_json(&args[2]));
97                    }
98                } else {
99                    // For other methods, try to infer config structure
100                    for (i, arg) in args.iter().enumerate().skip(1) {
101                        if let OvsmValue::Object(obj) = arg {
102                            for (k, v) in obj.iter() {
103                                config.insert(k.clone(), ovsm_value_to_json(v));
104                            }
105                        } else {
106                            config.insert(format!("param{}", i), ovsm_value_to_json(arg));
107                        }
108                    }
109                }
110
111                if !config.is_empty() {
112                    rpc_params.push(Value::Object(config));
113                }
114            }
115        }
116
117        // For getTransaction and getParsedTransaction, ensure we always include maxSupportedTransactionVersion
118        if matches!(self.name.as_str(), "getTransaction" | "getParsedTransaction") {
119            // If we have no config yet, add one
120            if rpc_params.len() == 1 {
121                let mut config = serde_json::Map::new();
122                config.insert("maxSupportedTransactionVersion".to_string(), json!(0));
123
124                // getParsedTransaction needs "jsonParsed" encoding
125                if self.name == "getParsedTransaction" {
126                    config.insert("encoding".to_string(), json!("jsonParsed"));
127                } else {
128                    config.insert("encoding".to_string(), json!("json"));
129                }
130
131                rpc_params.push(Value::Object(config));
132            } else if rpc_params.len() >= 2 {
133                // If we have a config, ensure it has maxSupportedTransactionVersion
134                if let Some(Value::Object(config)) = rpc_params.get_mut(1) {
135                    if !config.contains_key("maxSupportedTransactionVersion") {
136                        config.insert("maxSupportedTransactionVersion".to_string(), json!(0));
137                    }
138                    if !config.contains_key("encoding") {
139                        // getParsedTransaction needs "jsonParsed" encoding
140                        if self.name == "getParsedTransaction" {
141                            config.insert("encoding".to_string(), json!("jsonParsed"));
142                        } else {
143                            config.insert("encoding".to_string(), json!("json"));
144                        }
145                    }
146                }
147            }
148        }
149
150        let result = tokio::task::block_in_place(|| {
151            tokio::runtime::Handle::current().block_on(async {
152                call_solana_rpc(&self.name, rpc_params).await
153            })
154        });
155
156        match result {
157            Ok(json_val) => Ok(json_to_ovsm_value(&json_val)),
158            Err(e) => Err(ovsm::error::Error::RpcError {
159                message: format!("{} failed: {}", self.name, e),
160            }),
161        }
162    }
163}

165async fn call_solana_rpc(method: &str, params: Vec<Value>) -> anyhow::Result<Value> {
166    // Configure client with proper timeouts to prevent hanging
167    let client = reqwest::Client::builder()
168        .timeout(std::time::Duration::from_secs(30))       // Total request timeout
169        .connect_timeout(std::time::Duration::from_secs(10)) // Connection timeout
170        .build()?;
171
172    let request_body = json!({
173        "jsonrpc": "2.0",
174        "id": 1,
175        "method": method,
176        "params": params
177    });
178
179
180    let response = client
181        .post(SOLANA_RPC_URL)
182        .json(&request_body)
183        .send()
184        .await?;
185
186    let response_text = response.text().await?;
187    let response_json: Value = serde_json::from_str(&response_text)?;
188
189    if let Some(result) = response_json.get("result") {
190        Ok(result.clone())
191    } else if let Some(error) = response_json.get("error") {
192        Err(anyhow::anyhow!("RPC error: {}", error))
193    } else {
194        Err(anyhow::anyhow!("Invalid RPC response: {}", response_text))
195    }
196}

198fn ovsm_value_to_json(val: &OvsmValue) -> Value {
199    match val {
200        OvsmValue::Int(n) => json!(n),
201        OvsmValue::Float(f) => json!(f),
202        OvsmValue::String(s) => json!(s),
203        OvsmValue::Bool(b) => json!(b),
204        OvsmValue::Null => Value::Null,
205        OvsmValue::Array(arr) => {
206            let items: Vec<Value> = arr.iter().map(ovsm_value_to_json).collect();
207            json!(items)
208        }
209        OvsmValue::Object(obj) => {
210            let mut map = serde_json::Map::new();
211            for (k, v) in obj.iter() {
212                map.insert(k.clone(), ovsm_value_to_json(v));
213            }
214            Value::Object(map)
215        }
216        OvsmValue::Function { .. } => {
217            // Functions cannot be serialized to JSON for RPC calls
218            // This is an error - lambdas should not be passed to RPC
219            json!({"error": "Cannot serialize function to JSON for RPC call"})
220        }
221        OvsmValue::Range { start, end } => {
222            json!({"start": start, "end": end, "type": "range"})
223        }
224        OvsmValue::Multiple(vals) => {
225            // Serialize multiple values as array
226            let items: Vec<Value> = vals.iter().map(ovsm_value_to_json).collect();
227            json!({"type": "multiple-values", "values": items})
228        }
229        OvsmValue::Macro { params, .. } => {
230            // Macros cannot be serialized for RPC calls
231            json!({"error": "Cannot serialize macro to JSON for RPC call"})
232        }
233    }
234}

236fn json_to_ovsm_value(val: &Value) -> OvsmValue {
237    match val {
238        Value::Null => OvsmValue::Null,
239        Value::Bool(b) => OvsmValue::Bool(*b),
240        Value::Number(n) => {
241            if let Some(i) = n.as_i64() {
242                OvsmValue::Int(i)
243            } else if let Some(f) = n.as_f64() {
244                OvsmValue::Float(f)
245            } else {
246                OvsmValue::Null
247            }
248        }
249        Value::String(s) => OvsmValue::String(s.clone()),
250        Value::Array(arr) => {
251            let items: Vec<OvsmValue> = arr.iter().map(json_to_ovsm_value).collect();
252            OvsmValue::Array(Arc::new(items))
253        }
254        Value::Object(obj) => {
255            let mut map = std::collections::HashMap::new();
256            for (k, v) in obj.iter() {
257                map.insert(k.clone(), json_to_ovsm_value(v));
258            }
259            OvsmValue::Object(Arc::new(map))
260        }
261    }
262}

264/// Create a tool registry with RPC bridge tools
265pub fn create_rpc_registry() -> ToolRegistry {
266    let mut registry = ToolRegistry::new();
267
268    // Register common RPC methods
269    let methods = vec![
270        "getSignaturesForAddress",
271        "getParsedTransaction",
272        "getSlot",
273        "getBlock",
274        "getTransaction",
275        "getAccountInfo",
276        "getBalance",
277        "getBlockTime",
278        "getHealth",
279        "getVersion",
280    ];
281
282    for method in methods {
283        registry.register(RpcBridgeTool::new(method));
284    }
285
286    registry
287}
