3use ovsm::error::Result as OvsmResult;
4use ovsm::runtime::Value as OvsmValue;
5use ovsm::tools::{Tool, ToolRegistry};
6use serde_json::{json, Value};
7use std::sync::Arc;

9// Real Solana RPC endpoint
10const SOLANA_RPC_URL: &str = "https://osvm.ai/api/proxy/rpc";
11
12/// Generic RPC Bridge Tool - Dynamically calls ANY Solana RPC method
13pub struct RpcBridgeTool {
14    name: String,
15}
16
17impl RpcBridgeTool {
18    pub fn new(name: &str) -> Self {
19        RpcBridgeTool {
20            name: name.to_string(),
21        }
22    }
23}
24
25impl Tool for RpcBridgeTool {
26    fn name(&self) -> &str {
27        &self.name
28    }
29
30    fn description(&self) -> &str {
31        "Dynamic RPC bridge - calls Solana RPC methods"
32    }
33
34    fn execute(&self, args: &[OvsmValue]) -> OvsmResult<OvsmValue> {
35        let mut rpc_params = Vec::new();
36
37        // OVSM evaluator passes named parameters as positional arguments
38        // For RPC calls like getSignaturesForAddress(address: "...", limit: 5)
39        // We receive: [String("..."), Int(5)]
40        // We need to convert to: ["...", {"limit": 5}]
41
42        if args.is_empty() {
43            // No arguments - just call the method
44        } else if args.len() == 1 {
45            // Single argument - could be just primary param or an object
46            if let OvsmValue::Object(obj) = &args[0] {
47                // If it's an object, check if it's a config or has a primary param
48                let primary_param = obj
49                    .get("address")
50                    .or_else(|| obj.get("signature"))
51                    .or_else(|| obj.get("slot"))
52                    .or_else(|| obj.get("pubkey"));
53
54                if let Some(primary) = primary_param {
55                    rpc_params.push(ovsm_value_to_json(primary));
56
57                    // Build config object from remaining named parameters
58                    let mut config = serde_json::Map::new();
59                    for (key, val) in obj.iter() {
60                        if !matches!(key.as_str(), "address" | "signature" | "slot" | "pubkey") {
61                            config.insert(key.clone(), ovsm_value_to_json(val));
62                        }
63                    }
64
65                    // Only add config object if it has fields
66                    if !config.is_empty() {
67                        rpc_params.push(Value::Object(config));
68                    }
69                } else {
70                    // No primary param found, use the whole object as config
71                    rpc_params.push(ovsm_value_to_json(&args[0]));
72                }
73            } else {
74                // Single non-object argument (primary param only)
75                rpc_params.push(ovsm_value_to_json(&args[0]));
76            }
77        } else {
78            // Multiple arguments - first is primary param, rest form the config object
79            // This handles cases like: getSignaturesForAddress(address: "...", limit: 5, before: "...")
80            rpc_params.push(ovsm_value_to_json(&args[0]));
81
82            // Build config object from remaining arguments
83            // Check if second argument is already an object (named params)
84            if args.len() == 2 && matches!(&args[1], OvsmValue::Object(_)) {
85                // Second arg is a config object, use it directly
86                rpc_params.push(ovsm_value_to_json(&args[1]));
87            } else {
88                // Build config from positional args
89                let mut config = serde_json::Map::new();
90
91                // For getSignaturesForAddress specifically
92                if matches!(self.name.as_str(), "getSignaturesForAddress") {
93                    if args.len() >= 2 {
94                        config.insert("limit".to_string(), ovsm_value_to_json(&args[1]));
95                    }
96                    if args.len() >= 3 {
97                        config.insert("before".to_string(), ovsm_value_to_json(&args[2]));
98                    }
99                } else {
100                    // For other methods, try to infer config structure
101                    for (i, arg) in args.iter().enumerate().skip(1) {
102                        if let OvsmValue::Object(obj) = arg {
103                            for (k, v) in obj.iter() {
104                                config.insert(k.clone(), ovsm_value_to_json(v));
105                            }
106                        } else {
107                            config.insert(format!("param{}", i), ovsm_value_to_json(arg));
108                        }
109                    }
110                }
111
112                if !config.is_empty() {
113                    rpc_params.push(Value::Object(config));
114                }
115            }
116        }
117
118        // For getTransaction and getParsedTransaction, ensure we always include maxSupportedTransactionVersion
119        if matches!(
120            self.name.as_str(),
121            "getTransaction" | "getParsedTransaction"
122        ) {
123            // If we have no config yet, add one
124            if rpc_params.len() == 1 {
125                let mut config = serde_json::Map::new();
126                config.insert("maxSupportedTransactionVersion".to_string(), json!(0));
127
128                // getParsedTransaction needs "jsonParsed" encoding
129                if self.name == "getParsedTransaction" {
130                    config.insert("encoding".to_string(), json!("jsonParsed"));
131                } else {
132                    config.insert("encoding".to_string(), json!("json"));
133                }
134
135                rpc_params.push(Value::Object(config));
136            } else if rpc_params.len() >= 2 {
137                // If we have a config, ensure it has maxSupportedTransactionVersion
138                if let Some(Value::Object(config)) = rpc_params.get_mut(1) {
139                    if !config.contains_key("maxSupportedTransactionVersion") {
140                        config.insert("maxSupportedTransactionVersion".to_string(), json!(0));
141                    }
142                    if !config.contains_key("encoding") {
143                        // getParsedTransaction needs "jsonParsed" encoding
144                        if self.name == "getParsedTransaction" {
145                            config.insert("encoding".to_string(), json!("jsonParsed"));
146                        } else {
147                            config.insert("encoding".to_string(), json!("json"));
148                        }
149                    }
150                }
151            }
152        }
153
154        let result = tokio::task::block_in_place(|| {
155            tokio::runtime::Handle::current()
156                .block_on(async { call_solana_rpc(&self.name, rpc_params).await })
157        });
158
159        match result {
160            Ok(json_val) => Ok(json_to_ovsm_value(&json_val)),
161            Err(e) => Err(ovsm::error::Error::RpcError {
162                message: format!("{} failed: {}", self.name, e),
163            }),
164        }
165    }
166}

168async fn call_solana_rpc(method: &str, params: Vec<Value>) -> anyhow::Result<Value> {
169    // Configure client with proper timeouts to prevent hanging
170    let client = reqwest::Client::builder()
171        .timeout(std::time::Duration::from_secs(30)) // Total request timeout
172        .connect_timeout(std::time::Duration::from_secs(10)) // Connection timeout
173        .build()?;
174
175    let request_body = json!({
176        "jsonrpc": "2.0",
177        "id": 1,
178        "method": method,
179        "params": params
180    });
181
182    let response = client
183        .post(SOLANA_RPC_URL)
184        .json(&request_body)
185        .send()
186        .await?;
187
188    let response_text = response.text().await?;
189    let response_json: Value = serde_json::from_str(&response_text)?;
190
191    if let Some(result) = response_json.get("result") {
192        Ok(result.clone())
193    } else if let Some(error) = response_json.get("error") {
194        Err(anyhow::anyhow!("RPC error: {}", error))
195    } else {
196        Err(anyhow::anyhow!("Invalid RPC response: {}", response_text))
197    }
198}

200fn ovsm_value_to_json(val: &OvsmValue) -> Value {
201    match val {
202        OvsmValue::Int(n) => json!(n),
203        OvsmValue::Float(f) => json!(f),
204        OvsmValue::String(s) => json!(s),
205        OvsmValue::Bool(b) => json!(b),
206        OvsmValue::Null => Value::Null,
207        OvsmValue::Array(arr) => {
208            let items: Vec<Value> = arr.iter().map(ovsm_value_to_json).collect();
209            json!(items)
210        }
211        OvsmValue::Object(obj) => {
212            let mut map = serde_json::Map::new();
213            for (k, v) in obj.iter() {
214                map.insert(k.clone(), ovsm_value_to_json(v));
215            }
216            Value::Object(map)
217        }
218        OvsmValue::Function { .. } => {
219            // Functions cannot be serialized to JSON for RPC calls
220            // This is an error - lambdas should not be passed to RPC
221            json!({"error": "Cannot serialize function to JSON for RPC call"})
222        }
223        OvsmValue::Range { start, end } => {
224            json!({"start": start, "end": end, "type": "range"})
225        }
226        OvsmValue::Multiple(vals) => {
227            // Serialize multiple values as array
228            let items: Vec<Value> = vals.iter().map(ovsm_value_to_json).collect();
229            json!({"type": "multiple-values", "values": items})
230        }
231        OvsmValue::Macro { params, .. } => {
232            // Macros cannot be serialized for RPC calls
233            json!({"error": "Cannot serialize macro to JSON for RPC call"})
234        }
235    }
236}

238fn json_to_ovsm_value(val: &Value) -> OvsmValue {
239    match val {
240        Value::Null => OvsmValue::Null,
241        Value::Bool(b) => OvsmValue::Bool(*b),
242        Value::Number(n) => {
243            if let Some(i) = n.as_i64() {
244                OvsmValue::Int(i)
245            } else if let Some(f) = n.as_f64() {
246                OvsmValue::Float(f)
247            } else {
248                OvsmValue::Null
249            }
250        }
251        Value::String(s) => OvsmValue::String(s.clone()),
252        Value::Array(arr) => {
253            let items: Vec<OvsmValue> = arr.iter().map(json_to_ovsm_value).collect();
254            OvsmValue::Array(Arc::new(items))
255        }
256        Value::Object(obj) => {
257            let mut map = std::collections::HashMap::new();
258            for (k, v) in obj.iter() {
259                map.insert(k.clone(), json_to_ovsm_value(v));
260            }
261            OvsmValue::Object(Arc::new(map))
262        }
263    }
264}

266/// Create a tool registry with RPC bridge tools
267pub fn create_rpc_registry() -> ToolRegistry {
268    let mut registry = ToolRegistry::new();
269
270    // Register common RPC methods
271    let methods = vec![
272        "getSignaturesForAddress",
273        "getParsedTransaction",
274        "getSlot",
275        "getBlock",
276        "getTransaction",
277        "getAccountInfo",
278        "getBalance",
279        "getBlockTime",
280        "getHealth",
281        "getVersion",
282    ];
283
284    for method in methods {
285        registry.register(RpcBridgeTool::new(method));
286    }
287
288    registry
289}
