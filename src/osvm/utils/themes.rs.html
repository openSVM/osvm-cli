13/// Color definition with support for 256-color and RGB
14#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
15pub enum Color {
16    /// Named ANSI colors
17    Named(String),
18    /// 256-color palette (0-255)
19    Indexed(u8),
20    /// RGB color values
21    Rgb(u8, u8, u8),
22    /// HSL color values (hue: 0-360, saturation: 0-100, lightness: 0-100)
23    Hsl(u16, u8, u8),
24    /// Hex color code
25    Hex(String),
26}
27
28impl Color {
29    /// Convert color to ANSI escape sequence
30    pub fn to_ansi(&self) -> String {
31        match self {
32            Color::Named(name) => match name.as_str() {
33                "black" => "\x1b[30m".to_string(),
34                "red" => "\x1b[31m".to_string(),
35                "green" => "\x1b[32m".to_string(),
36                "yellow" => "\x1b[33m".to_string(),
37                "blue" => "\x1b[34m".to_string(),
38                "magenta" => "\x1b[35m".to_string(),
39                "cyan" => "\x1b[36m".to_string(),
40                "white" => "\x1b[37m".to_string(),
41                "bright_black" => "\x1b[90m".to_string(),
42                "bright_red" => "\x1b[91m".to_string(),
43                "bright_green" => "\x1b[92m".to_string(),
44                "bright_yellow" => "\x1b[93m".to_string(),
45                "bright_blue" => "\x1b[94m".to_string(),
46                "bright_magenta" => "\x1b[95m".to_string(),
47                "bright_cyan" => "\x1b[96m".to_string(),
48                "bright_white" => "\x1b[97m".to_string(),
49                _ => "\x1b[39m".to_string(), // Default
50            },
51            Color::Indexed(index) => format!("\x1b[38;5;{}m", index),
52            Color::Rgb(r, g, b) => format!("\x1b[38;2;{};{};{}m", r, g, b),
53            Color::Hsl(h, s, l) => {
54                // Convert HSL to RGB
55                let (r, g, b) = hsl_to_rgb(*h, *s, *l);
56                format!("\x1b[38;2;{};{};{}m", r, g, b)
57            }
58            Color::Hex(hex) => {
59                // Parse hex color (#RRGGBB or #RGB)
60                if let Ok((r, g, b)) = parse_hex_color(hex) {
61                    format!("\x1b[38;2;{};{};{}m", r, g, b)
62                } else {
63                    "\x1b[39m".to_string()
64                }
65            }
66        }
67    }
68
69    /// Convert color to background ANSI escape sequence
70    pub fn to_bg_ansi(&self) -> String {
71        match self {
72            Color::Named(name) => match name.as_str() {
73                "black" => "\x1b[40m".to_string(),
74                "red" => "\x1b[41m".to_string(),
75                "green" => "\x1b[42m".to_string(),
76                "yellow" => "\x1b[43m".to_string(),
77                "blue" => "\x1b[44m".to_string(),
78                "magenta" => "\x1b[45m".to_string(),
79                "cyan" => "\x1b[46m".to_string(),
80                "white" => "\x1b[47m".to_string(),
81                _ => "\x1b[49m".to_string(), // Default
82            },
83            Color::Indexed(index) => format!("\x1b[48;5;{}m", index),
84            Color::Rgb(r, g, b) => format!("\x1b[48;2;{};{};{}m", r, g, b),
85            Color::Hsl(h, s, l) => {
86                let (r, g, b) = hsl_to_rgb(*h, *s, *l);
87                format!("\x1b[48;2;{};{};{}m", r, g, b)
88            }
89            Color::Hex(hex) => {
90                if let Ok((r, g, b)) = parse_hex_color(hex) {
91                    format!("\x1b[48;2;{};{};{}m", r, g, b)
92                } else {
93                    "\x1b[49m".to_string()
94                }
95            }
96        }
97    }
98}

128impl TextStyle {
129    /// Convert style to ANSI escape sequence
130    pub fn to_ansi(&self) -> String {
131        let mut codes = vec![self.color.to_ansi()];
132
133        if let Some(bg) = &self.background {
134            codes.push(bg.to_bg_ansi());
135        }
136
137        if self.bold {
138            codes.push("\x1b[1m".to_string());
139        }
140        if self.italic {
141            codes.push("\x1b[3m".to_string());
142        }
143        if self.underline {
144            codes.push("\x1b[4m".to_string());
145        }
146        if self.strikethrough {
147            codes.push("\x1b[9m".to_string());
148        }
149        if self.dim {
150            codes.push("\x1b[2m".to_string());
151        }
152        if self.blink {
153            codes.push("\x1b[5m".to_string());
154        }
155
156        codes.join("")
157    }
158
159    /// Create a reset sequence
160    pub fn reset() -> String {
161        "\x1b[0m".to_string()
162    }
163
164    /// Apply style to text
165    pub fn apply(&self, text: &str) -> String {
166        format!("{}{}{}", self.to_ansi(), text, Self::reset())
167    }

491    /// Load theme from file
492    pub fn load(name: &str) -> Result<Self> {
493        let theme_path = Self::theme_path(name)?;
494
495        if !theme_path.exists() {
496            return Err(anyhow!("Theme '{}' not found", name));
497        }
498
499        let content = fs::read_to_string(&theme_path)
500            .map_err(|e| anyhow!("Failed to read theme file: {}", e))?;
501
502        let theme: Self =
503            serde_json::from_str(&content).map_err(|e| anyhow!("Failed to parse theme: {}", e))?;
504
505        debug!("Loaded theme '{}' from {:?}", name, theme_path);
506        Ok(theme)
507    }

509    /// Save theme to file
510    pub fn save(&self) -> Result<()> {
511        let theme_path = Self::theme_path(&self.name)?;
512
513        if let Some(parent) = theme_path.parent() {
514            fs::create_dir_all(parent)
515                .map_err(|e| anyhow!("Failed to create themes directory: {}", e))?;
516        }
517
518        let content = serde_json::to_string_pretty(self)
519            .map_err(|e| anyhow!("Failed to serialize theme: {}", e))?;
520
521        fs::write(&theme_path, content)
522            .map_err(|e| anyhow!("Failed to write theme file: {}", e))?;
523
524        debug!("Saved theme '{}' to {:?}", self.name, theme_path);
525        Ok(())
526    }

528    /// Get theme file path
529    fn theme_path(name: &str) -> Result<PathBuf> {
530        let home =
531            std::env::var("HOME").map_err(|_| anyhow!("HOME environment variable not set"))?;
532        let safe_name = name
533            .chars()
534            .map(|c| {
535                if c.is_alphanumeric() || c == '-' || c == '_' {
536                    c
537                } else {
538                    '_'
539                }
540            })
541            .collect::<String>();
542        Ok(PathBuf::from(home)
543            .join(".osvm")
544            .join("themes")
545            .join(format!("{}.json", safe_name)))
546    }

548    /// List available themes
549    pub fn list_available() -> Result<Vec<String>> {
550        let themes_dir = Self::theme_path("")?.parent().unwrap().to_path_buf();
551
552        if !themes_dir.exists() {
553            return Ok(vec!["default".to_string()]);
554        }
555
556        let mut themes = Vec::new();
557
558        for entry in fs::read_dir(&themes_dir)? {
559            let entry = entry?;
560            let path = entry.path();
561
562            if path.extension().and_then(|s| s.to_str()) == Some("json") {
563                if let Some(name) = path.file_stem().and_then(|s| s.to_str()) {
564                    themes.push(name.to_string());
565                }
566            }
567        }
568
569        if themes.is_empty() {
570            themes.push("default".to_string());
571        }
572
573        Ok(themes)
574    }

657/// Theme manager for runtime theme switching
658pub struct ThemeManager {
659    current_theme: Theme,
660    available_themes: Vec<String>,
661}
662
663impl ThemeManager {
664    /// Create new theme manager
665    pub fn new() -> Result<Self> {
666        let available_themes = Theme::list_available().unwrap_or_else(|e| {
667            warn!("Failed to list themes: {}, using default", e);
668            vec!["default".to_string()]
669        });
670
671        // Try to load saved theme preference, fallback to default
672        let current_theme =
673            Self::load_preferred_theme().unwrap_or_else(|_| Theme::default_dark_theme());
674
675        Ok(Self {
676            current_theme,
677            available_themes,
678        })
679    }
680
681    /// Get current theme
682    pub fn current_theme(&self) -> &Theme {
683        &self.current_theme
684    }
685
686    /// Switch to theme by name
687    pub fn switch_theme(&mut self, name: &str) -> Result<()> {
688        let theme = match name {
689            "default" | "dark" => Theme::default_dark_theme(),
690            "light" => Theme::default_light_theme(),
691            "high-contrast" => Theme::high_contrast_theme(),
692            "retro" => Theme::retro_theme(),
693            _ => Theme::load(name)?,
694        };
695
696        self.current_theme = theme;
697        self.save_theme_preference(name)?;
698
699        debug!("Switched to theme: {}", name);
700        Ok(())
701    }
702
703    /// Get available theme names
704    pub fn available_themes(&self) -> &[String] {
705        &self.available_themes
706    }
707
708    /// Refresh available themes list
709    pub fn refresh_themes(&mut self) -> Result<()> {
710        self.available_themes = Theme::list_available()?;
711        Ok(())
712    }
713
714    /// Load preferred theme from config
715    fn load_preferred_theme() -> Result<Theme> {
716        let config_path = Self::config_path()?;
717        if !config_path.exists() {
718            return Err(anyhow!("No theme preference saved"));
719        }
720
721        let theme_name = fs::read_to_string(&config_path)?;
722        match theme_name.trim() {
723            "default" | "dark" => Ok(Theme::default_dark_theme()),
724            "light" => Ok(Theme::default_light_theme()),
725            "high-contrast" => Ok(Theme::high_contrast_theme()),
726            "retro" => Ok(Theme::retro_theme()),
727            name => Theme::load(name),
728        }
729    }
730
731    /// Save theme preference
732    fn save_theme_preference(&self, name: &str) -> Result<()> {
733        let config_path = Self::config_path()?;
734        if let Some(parent) = config_path.parent() {
735            fs::create_dir_all(parent)?;
736        }
737        fs::write(&config_path, name)?;
738        Ok(())
739    }
740
741    /// Get theme preference config path
742    fn config_path() -> Result<PathBuf> {
743        let home =
744            std::env::var("HOME").map_err(|_| anyhow!("HOME environment variable not set"))?;
745        Ok(PathBuf::from(home).join(".osvm").join("current_theme"))
746    }
747}

749/// Utility functions for color conversion
750fn hsl_to_rgb(h: u16, s: u8, l: u8) -> (u8, u8, u8) {
751    let h = (h % 360) as f32 / 360.0;
752    let s = s as f32 / 100.0;
753    let l = l as f32 / 100.0;
754
755    let c = (1.0 - (2.0 * l - 1.0).abs()) * s;
756    let x = c * (1.0 - ((h * 6.0) % 2.0 - 1.0).abs());
757    let m = l - c / 2.0;
758
759    let (r, g, b) = if h < 1.0 / 6.0 {
760        (c, x, 0.0)
761    } else if h < 2.0 / 6.0 {
762        (x, c, 0.0)
763    } else if h < 3.0 / 6.0 {
764        (0.0, c, x)
765    } else if h < 4.0 / 6.0 {
766        (0.0, x, c)
767    } else if h < 5.0 / 6.0 {
768        (x, 0.0, c)
769    } else {
770        (c, 0.0, x)
771    };
772
773    (
774        ((r + m) * 255.0) as u8,
775        ((g + m) * 255.0) as u8,
776        ((b + m) * 255.0) as u8,
777    )
778}

780fn parse_hex_color(hex: &str) -> Result<(u8, u8, u8)> {
781    let hex = hex.trim_start_matches('#');
782
783    match hex.len() {
784        3 => {
785            // #RGB -> #RRGGBB
786            let r = u8::from_str_radix(&hex[0..1].repeat(2), 16)?;
787            let g = u8::from_str_radix(&hex[1..2].repeat(2), 16)?;
788            let b = u8::from_str_radix(&hex[2..3].repeat(2), 16)?;
789            Ok((r, g, b))
790        }
791        6 => {
792            // #RRGGBB
793            let r = u8::from_str_radix(&hex[0..2], 16)?;
794            let g = u8::from_str_radix(&hex[2..4], 16)?;
795            let b = u8::from_str_radix(&hex[4..6], 16)?;
796            Ok((r, g, b))
797        }
798        _ => Err(anyhow!("Invalid hex color format")),
799    }
800}
