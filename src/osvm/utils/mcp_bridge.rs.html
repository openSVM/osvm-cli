64        let _meta = tools

66            .find(|t| t.name == self.name)

68                message: format!(

74        // Convert OvsmValue args to JSON properly
75        fn ovsm_to_json(val: &OvsmValue) -> JsonValue {
76            match val {
77                OvsmValue::Null => JsonValue::Null,
78                OvsmValue::Bool(b) => JsonValue::Bool(*b),
79                OvsmValue::Int(i) => JsonValue::Number((*i).into()),
80                OvsmValue::Float(f) => serde_json::Number::from_f64(*f)
81                    .map(JsonValue::Number)
82                    .unwrap_or(JsonValue::Null),
83                OvsmValue::String(s) => JsonValue::String(s.clone()),
84                OvsmValue::Array(arr) => JsonValue::Array(arr.iter().map(ovsm_to_json).collect()),
85                OvsmValue::Object(obj) => {
86                    let mut map = serde_json::Map::new();
87                    for (k, v) in obj.iter() {
88                        map.insert(k.clone(), ovsm_to_json(v));
89                    }
90                    JsonValue::Object(map)
91                }
92                _ => {
93                    // Fallback for other types (Range, Function, etc.)
94                    serde_json::from_str(&val.to_string()).unwrap_or(JsonValue::Null)
95                }
96            }
97        }

144        // Convert JSON result back to OvsmValue
145        fn json_to_ovsm(json: &JsonValue) -> OvsmValue {
146            use std::collections::HashMap;
147            use std::sync::Arc;
148
149            match json {
150                JsonValue::Null => OvsmValue::Null,
151                JsonValue::Bool(b) => OvsmValue::Bool(*b),
152                JsonValue::Number(n) => {
153                    if let Some(i) = n.as_i64() {
154                        OvsmValue::Int(i)
155                    } else if let Some(f) = n.as_f64() {
156                        OvsmValue::Float(f)
157                    } else {
158                        OvsmValue::Null
159                    }
160                }
161                JsonValue::String(s) => OvsmValue::String(s.clone()),
162                JsonValue::Array(arr) => {
163                    let ovsm_arr: Vec<OvsmValue> = arr.iter().map(json_to_ovsm).collect();
164                    OvsmValue::Array(Arc::new(ovsm_arr))
165                }
166                JsonValue::Object(obj) => {
167                    // Check if this is an error object from the API
168                    if let Some(error_msg) = obj.get("error") {
169                        // Return null for errors - this allows the AI to check for null
170                        // and handle appropriately without breaking type expectations
171                        eprintln!("âš ï¸  API returned error: {}", error_msg);
172                        return OvsmValue::Null;
173                    }
174
175                    let mut ovsm_obj = HashMap::new();
176                    for (k, v) in obj.iter() {
177                        ovsm_obj.insert(k.clone(), json_to_ovsm(v));
178                    }
179                    OvsmValue::Object(Arc::new(ovsm_obj))
180                }
181            }
182        }
183
184        // Convert the result and log its type
185        let ovsm_result = json_to_ovsm(&result_json);
