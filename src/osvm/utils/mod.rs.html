84/// Loads a YAML configuration file and deserializes it into the specified type
85///
86/// # Arguments
87///
88/// * `config_file` - Path to the YAML configuration file
89///
90/// # Returns
91///
92/// * `Result<T, io::Error>` - The deserialized configuration or an error
93///
94/// # Examples
95///
96/// ```
97/// use osvm::utils::load_keys_config_file;
98/// use serde::Deserialize;
99///
100/// #[derive(Deserialize)]
101/// struct Config {
102///     key: String,
103/// }
104///
105/// let config: Config = load_keys_config_file("config.yml").unwrap();
106/// ```
107pub fn load_keys_config_file<T, P>(config_file: P) -> Result<T, io::Error>
108where
109    T: DeserializeOwned,
110    P: AsRef<Path>,
111{
112    let file = File::open(config_file)?;
113    let config =
114        serde_yaml::from_reader(file).map_err(|err| io::Error::other(format!("{:?}", err)))?;
115    Ok(config)
116}
