135/// Loads a YAML configuration file and deserializes it into the specified type
136///
137/// # Arguments
138///
139/// * `config_file` - Path to the YAML configuration file
140///
141/// # Returns
142///
143/// * `Result<T, io::Error>` - The deserialized configuration or an error
144///
145/// # Examples
146///
147/// ```
148/// use osvm::utils::load_keys_config_file;
149/// use serde::Deserialize;
150///
151/// #[derive(Deserialize)]
152/// struct Config {
153///     key: String,
154/// }
155///
156/// let config: Config = load_keys_config_file("config.yml").unwrap();
157/// ```
158pub fn load_keys_config_file<T, P>(config_file: P) -> Result<T, io::Error>
159where
160    T: DeserializeOwned,
161    P: AsRef<Path>,
162{
163    let file = File::open(config_file)?;
164    let config =
165        serde_yaml::from_reader(file).map_err(|err| io::Error::other(format!("{:?}", err)))?;
166    Ok(config)
167}
