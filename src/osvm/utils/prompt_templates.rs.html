42/// AI prompt template structure
43#[derive(Debug, Clone, Serialize, Deserialize)]
44pub struct PromptTemplate {
45    /// Template identifier
46    pub id: String,
47    /// Human-readable name
48    pub name: String,
49    /// Template description
50    pub description: String,
51    /// Template category
52    pub category: TemplateCategory,
53    /// Analysis vector (if applicable)
54    pub analysis_vector: Option<AnalysisVector>,
55    /// The actual prompt template with placeholders
56    pub template: String,
57    /// Variables used in the template
58    pub variables: Vec<TemplateVariable>,
59    /// Template version for compatibility
60    pub version: String,
61    /// Whether this template is enabled
62    pub enabled: bool,
63    /// Additional metadata
64    pub metadata: HashMap<String, String>,

141    pub fn load_from_directory_with_debug(
142        &mut self,
143        dir_path: &str,
144        debug_mode: bool,
145    ) -> Result<usize> {
146        self.template_dirs.push(dir_path.to_string());
147
148        let dir = Path::new(dir_path);
149        if !dir.exists() {
150            if debug_mode {
151                println!("üìÅ Creating template directory: {}", dir_path);
152            }
153            fs::create_dir_all(dir)?;
154            self.create_default_templates(dir)?;
155        }
156
157        let mut loaded_count = 0;
158
159        for entry in fs::read_dir(dir)? {
160            let entry = entry?;
161            let path = entry.path();
162
163            if path
164                .extension()
165                .map_or(false, |ext| ext == "yaml" || ext == "yml")
166            {
167                match self.load_template_file(&path) {
168                    Ok(count) => {
169                        loaded_count += count;
170                        if debug_mode {
171                            println!("‚úÖ Loaded {} templates from {}", count, path.display());
172                        }
173                    }
174                    Err(e) => {
175                        if debug_mode {
176                            println!("‚ö†Ô∏è  Failed to load template file {}: {}", path.display(), e);
177                        }
178                    }
179                }
180            }
181        }
182
183        self.rebuild_indices();
184        Ok(loaded_count)
185    }

187    /// Load a single template file
188    fn load_template_file(&mut self, file_path: &Path) -> Result<usize> {
189        let content = fs::read_to_string(file_path)
190            .with_context(|| format!("Failed to read template file: {}", file_path.display()))?;
191
192        // Try to load as a collection first
193        if let Ok(collection) = serde_yaml::from_str::<TemplateCollection>(&content) {
194            let count = collection.templates.len();
195            for template in collection.templates {
196                self.templates.insert(template.id.clone(), template);
197            }
198            return Ok(count);
199        }
200
201        // Try to load as a single template
202        if let Ok(template) = serde_yaml::from_str::<PromptTemplate>(&content) {
203            self.templates.insert(template.id.clone(), template);
204            return Ok(1);
205        }
206
207        anyhow::bail!("File is neither a valid template nor template collection");
208    }

210    /// Rebuild category and vector indices
211    fn rebuild_indices(&mut self) {
212        self.category_index.clear();
213        self.vector_index.clear();
214
215        for (id, template) in &self.templates {
216            // Index by category
217            self.category_index
218                .entry(template.category.clone())
219                .or_insert_with(Vec::new)
220                .push(id.clone());
221
222            // Index by analysis vector
223            if let Some(vector) = &template.analysis_vector {
224                self.vector_index
225                    .entry(vector.clone())
226                    .or_insert_with(Vec::new)
227                    .push(id.clone());
228            }
229        }
230    }

263    /// Render a template with provided variables
264    pub fn render_template(
265        &self,
266        template_id: &str,
267        variables: &HashMap<String, String>,
268    ) -> Result<String> {
269        let template = self
270            .get_template(template_id)
271            .ok_or_else(|| anyhow::anyhow!("Template not found: {}", template_id))?;
272
273        if !template.enabled {
274            anyhow::bail!("Template is disabled: {}", template_id);
275        }
276
277        let mut rendered = template.template.clone();
278
279        // Check required variables
280        for var in &template.variables {
281            if var.required && !variables.contains_key(&var.name) && var.default_value.is_none() {
282                anyhow::bail!(
283                    "Required variable '{}' not provided for template '{}'",
284                    var.name,
285                    template_id
286                );
287            }
288        }
289
290        // Replace variables
291        for var in &template.variables {
292            let value = if let Some(provided_value) = variables.get(&var.name) {
293                provided_value.clone()
294            } else if let Some(default_value) = &var.default_value {
295                default_value.clone()
296            } else {
297                continue; // Skip optional variables without defaults
298            };
299
300            let placeholder = format!("{{{}}}", var.name);
301            rendered = rendered.replace(&placeholder, &value);
302        }
303
304        Ok(rendered)
305    }

307    /// Get best template for a specific use case
308    pub fn get_best_template(
309        &self,
310        category: &TemplateCategory,
311        vector: Option<&AnalysisVector>,
312    ) -> Option<&PromptTemplate> {
313        // First try to find a template that matches both category and vector
314        if let Some(vector) = vector {
315            for template in self.get_templates_by_category(category) {
316                if template.analysis_vector.as_ref() == Some(vector) {
317                    return Some(template);
318                }
319            }
320        }
321
322        // Fallback to any template in the category
323        self.get_templates_by_category(category).first().copied()
324    }

504    /// Reload templates from all configured directories
505    pub fn reload_templates(&mut self) -> Result<usize> {
506        let dirs = self.template_dirs.clone();
507        self.templates.clear();
508        self.category_index.clear();
509        self.vector_index.clear();
510
511        let mut total_loaded = 0;
512        for dir in &dirs {
513            total_loaded += self.load_from_directory(dir)?;
514        }
515
516        Ok(total_loaded)
517    }

519    /// Validate template syntax
520    pub fn validate_template(&self, template: &PromptTemplate) -> Result<Vec<String>> {
521        let mut issues = Vec::new();
522
523        // Check for variable placeholders that don't have corresponding variable definitions
524        for var in &template.variables {
525            let placeholder = format!("{{{}}}", var.name);
526            if !template.template.contains(&placeholder) {
527                issues.push(format!(
528                    "Variable '{}' is defined but not used in template",
529                    var.name
530                ));
531            }
532        }
533
534        // Check for placeholders in template that don't have variable definitions
535        let mut in_placeholder = false;
536        let mut current_placeholder = String::new();
537
538        for char in template.template.chars() {
539            match char {
540                '{' => {
541                    in_placeholder = true;
542                    current_placeholder.clear();
543                }
544                '}' => {
545                    if in_placeholder {
546                        if !template
547                            .variables
548                            .iter()
549                            .any(|v| v.name == current_placeholder)
550                        {
551                            issues.push(format!(
552                                "Placeholder '{{{}}}' used but not defined",
553                                current_placeholder
554                            ));
555                        }
556                        in_placeholder = false;
557                    }
558                }
559                _ => {
560                    if in_placeholder {
561                        current_placeholder.push(char);
562                    }
563                }
564            }
565        }
566
567        Ok(issues)
568    }
