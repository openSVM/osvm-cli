1use crate::config::Config;

6pub fn handle_svm_command(
7    matches: &clap::ArgMatches,
8    rpc_client: &RpcClient,
9    config: &Config,
10) -> Result<(), Box<dyn std::error::Error>> {
11    let Some((svm_sub_command, svm_sub_matches)) = matches.subcommand() else {
12        eprintln!("No SVM subcommand provided");
13        exit(1);
14    };
15
16    match svm_sub_command {
17        "list" => {
18                    // List all SVMs
19                    let svms = svm_info::list_all_svms(&rpc_client, config.commitment_config)?;
20                    svm_info::display_svm_list(&svms);
21                }
22                "dashboard" => {
23                    // Launch the interactive dashboard
24                    match dashboard::run_dashboard(&rpc_client, config.commitment_config) {
25                        Ok(_) => println!("Dashboard closed"),
26                        Err(e) => {
27                            eprintln!("Error running dashboard: {}", e);
28                            exit(1);
29                        }
30                    }
31                }
32                "get" => {
33                    // Get details for a specific SVM
34                    let name = svm_sub_matches
35                        .get_one::<String>("name")
36                        .map(|s| s.as_str())
37                        .unwrap();
38                    match svm_info::get_svm_info(&rpc_client, name, config.commitment_config) {
39                        Ok(info) => svm_info::display_svm_info(&info),
40                        Err(e) => {
41                            eprintln!("Error: {}", e);
42                            exit(1);
43                        }
44                    }
45                }
46                "install" => {
47                    // Install an SVM on a remote host
48                    let svm_name = svm_sub_matches
49                        .get_one::<String>("name")
50                        .map(|s| s.as_str())
51                        .unwrap();
52                    let host = svm_sub_matches
53                        .get_one::<String>("host")
54                        .map(|s| s.as_str())
55                        .unwrap();
56
57                    println!("Installing SVM: {}", svm_name);
58                    println!("Host: {}", host);
59
60                    // First get SVM info to verify it exists and can be installed
61                    match svm_info::get_svm_info(&rpc_client, svm_name, config.commitment_config) {
62                        Ok(info) => {
63                            if !info.can_install_validator && !info.can_install_rpc {
64                                eprintln!("SVM '{}' cannot be installed", svm_name);
65                                exit(1);
66                            }
67
68                            // Default to installing as validator on mainnet
69                            match ssh_deploy::deploy_node(
70                                host,
71                                svm_name,
72                                "validator",
73                                ssh_deploy::NetworkType::Mainnet,
74                            ) {
75                                Ok(node_id) => {
76                                    println!("Installation complete");
77                                    println!(
78                                        "Successfully installed {} as node {}",
79                                        svm_name, node_id
80                                    );
81                                }
82                                Err(e) => eprintln!("Installation failed: {}", e),
83                            }
84                        }
85                        Err(e) => {
86                            eprintln!("Error: {}", e);
87                            exit(1);
88                        }
89                    }
90                }
91                _ => unreachable!(),
92    }
93
94    Ok(())
95}
