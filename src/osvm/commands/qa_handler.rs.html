3/// Handle QA command for automated testing and bug detection
4pub async fn handle_qa_command(
5    app_matches: &clap::ArgMatches,
6    matches: &clap::ArgMatches,
7) -> Result<(), Box<dyn std::error::Error>> {
8    let debug_mode = app_matches.get_flag("debug");
9    let verbose = app_matches.get_count("verbose");
10
11    // Create QA service
12    let qa_service = QaAgentService::new(debug_mode, verbose);
13
14    match matches.subcommand() {
15        Some(("run", sub_m)) => {
16            // Check if specific scenario or all
17            let category = if sub_m.get_flag("all") {
18                TestCategory::All
19            } else if let Some(scenario_name) = sub_m.get_one::<String>("scenario") {
20                // Filter by scenario name
21                match scenario_name.to_lowercase().as_str() {
22                    "chat" | "chat_ui" => TestCategory::Chat,
23                    "agent" | "agent_cli" => TestCategory::All, // Run all for agent scenarios
24                    _ => {
25                        eprintln!("Warning: Unknown scenario '{}', running all tests", scenario_name);
26                        TestCategory::All
27                    }
28                }
29            } else {
30                // Default to chat tests for testing
31                TestCategory::Chat
32            };
33
34            let create_issues = sub_m.get_flag("create_issues");
35            let github_repo = sub_m
36                .get_one::<String>("github_repo")
37                .map(|s| s.to_string())
38                .unwrap_or_else(|| "opensvm/osvm-cli".to_string());
39
40            println!("🤖 Running QA tests for category: {:?}", category);
41            if create_issues {
42                println!("📝 GitHub issues will be created for bugs found");
43                println!("📦 Repository: {}", github_repo);
44            }
45
46            // Run tests
47            let results = qa_service.run_tests(category).await?;
48
49            // Report results
50            let passed = results.iter().filter(|r| r.passed).count();
51            let failed = results.iter().filter(|r| !r.passed).count();
52            let bugs_found = results
53                .iter()
54                .map(|r| r.bugs_found.len())
55                .sum::<usize>();
56
57            println!("\n╔════════════════════════════════════════╗");
58            println!("║         QA Test Results                ║");
59            println!("╠════════════════════════════════════════╣");
60            println!("║ ✅ Passed:      {:>3}                    ║", passed);
61            println!("║ ❌ Failed:      {:>3}                    ║", failed);
62            println!("║ 🐛 Bugs Found:  {:>3}                    ║", bugs_found);
63            println!("╚════════════════════════════════════════╝");
64
65            // Create GitHub issues if requested
66            if create_issues && bugs_found > 0 {
67                println!("\n📝 Creating GitHub issues for bugs...");
68                for result in &results {
69                    for bug in &result.bugs_found {
70                        match qa_service.create_github_issue(bug).await {
71                            Ok(url) => println!("✅ Created issue: {}", url),
72                            Err(e) => eprintln!("❌ Failed to create issue: {}", e),
73                        }
74                    }
75                }
76            }
77
78            // Generate summary report
79            qa_service
80                .generate_summary_report(&results, category)
81                .await?;
82
83            // Exit with error code if any tests failed
84            if failed > 0 {
85                std::process::exit(1);
86            }
87        }
88        Some(("interactive", _)) => {
89            println!("🎮 Starting interactive QA session...");
90            qa_service.run_interactive().await?;
91        }
92        Some(("list", _)) => {
93            println!("📋 Available test scenarios:");
94            qa_service.list_scenarios();
95        }
96        Some(("reports", _)) => {
97            println!("📊 Recent QA reports:");
98            qa_service.show_reports().await?;
99        }
100        Some(("visual", sub_m)) => {
101            use crate::services::tui_test_agent::TuiTestAgent;
102
103            println!("🎨 Running visual TUI tests...");
104
105            // Parse arguments
106            let cols = sub_m
107                .get_one::<String>("cols")
108                .and_then(|s| s.parse::<u16>().ok())
109                .unwrap_or(120);
110            let rows = sub_m
111                .get_one::<String>("rows")
112                .and_then(|s| s.parse::<u16>().ok())
113                .unwrap_or(30);
114            let test_name = sub_m.get_one::<String>("test").map(|s| s.as_str());
115            let save_screenshots = sub_m.get_flag("screenshots");
116            let save_as_png = sub_m.get_flag("png");
117            let _headless = sub_m.get_flag("headless");
118
119            println!("🖥️  Terminal size: {}x{}", cols, rows);
120            if save_as_png && save_screenshots {
121                println!("📸 PNG screenshot mode enabled");
122            }
123            if let Some(name) = test_name {
124                println!("🔍 Running test: {}", name);
125            } else {
126                println!("🔍 Running all visual tests");
127            }
128
129            // Create TUI test agent
130            let agent = TuiTestAgent::new(cols, rows)?;
131
132            // Launch advanced chat
133            println!("🚀 Launching advanced chat TUI...");
134            agent.launch_advanced_chat().await?;
135
136            // Wait for UI to initialize
137            tokio::time::sleep(std::time::Duration::from_secs(2)).await;
138
139            // Get test scenarios
140            let scenarios = crate::services::tui_test_agent::create_default_tui_scenarios();
141
142            // Filter scenarios if specific test requested
143            let scenarios_to_run: Vec<_> = if let Some(name) = test_name {
144                scenarios
145                    .into_iter()
146                    .filter(|s| s.name.contains(name))
147                    .collect()
148            } else {
149                scenarios
150            };
151
152            if scenarios_to_run.is_empty() {
153                eprintln!("❌ No matching test scenarios found");
154                agent.stop().await?;
155                std::process::exit(1);
156            }
157
158            let mut all_passed = true;
159            let mut total_tests = 0;
160            let mut passed_tests = 0;
161
162            // Run scenarios
163            for scenario in &scenarios_to_run {
164                total_tests += 1;
165                println!("\n🧪 Running: {}", scenario.name);
166                println!("   Description: {}", scenario.description);
167
168                let result = agent.run_scenario(scenario, save_as_png && save_screenshots).await?;
169
170                if result.passed {
171                    passed_tests += 1;
172                    println!("   ✅ PASSED in {:.2}s", result.duration_seconds);
173                } else {
174                    all_passed = false;
175                    println!("   ❌ FAILED in {:.2}s", result.duration_seconds);
176                    for error in &result.errors {
177                        println!("      - {}", error);
178                    }
179                }
180
181                if save_screenshots && !result.screenshots.is_empty() {
182                    println!("   📸 Screenshots saved: {:?}", result.screenshots);
183                }
184            }
185
186            // Stop the agent
187            agent.stop().await?;
188
189            // Print summary
190            println!("\n╔════════════════════════════════════════╗");
191            println!("║     Visual TUI Test Results            ║");
192            println!("╠════════════════════════════════════════╣");
193            println!("║ ✅ Passed:      {:>3}/{:<3}                ║", passed_tests, total_tests);
194            println!("║ ❌ Failed:      {:>3}/{:<3}                ║", total_tests - passed_tests, total_tests);
195            println!("╚════════════════════════════════════════╝");
196
197            if save_screenshots {
198                println!("\n📸 Screenshots saved to: ~/.osvm/qa/screenshots/");
199            }
200
201            // Exit with error code if any tests failed
202            if !all_passed {
203                std::process::exit(1);
204            }
205        }
206        _ => {
207            eprintln!("❌ Unknown QA subcommand. Use 'osvm qa --help' for usage.");
208            std::process::exit(1);
209        }
210    }
211
212    Ok(())
213}
