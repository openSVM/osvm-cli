12/// Execute database command
13pub async fn execute_database_command(subcommand: &str, args: &DatabaseArgs) -> Result<()> {
14    match subcommand {
15        "init" => cmd_init(args).await,
16        "start" => cmd_start(args).await,
17        "stop" => cmd_stop(args).await,
18        "status" => cmd_status(args).await,
19        "query" => cmd_query(args).await,
20        "activity" => cmd_activity(args).await,
21        "sync" => cmd_sync(args).await,
22        _ => Err(anyhow::anyhow!(
23            "Unknown database subcommand: {}",
24            subcommand
25        )),
26    }
27}

68/// Initialize ClickHouse database
69async fn cmd_init(args: &DatabaseArgs) -> Result<()> {
70    println!("{}", "Initializing ClickHouse database...".cyan().bold());
71
72    let service = if let Some(data_dir) = &args.data_dir {
73        let config = crate::services::clickhouse_service::ClickHouseConfig {
74            data_dir: std::path::PathBuf::from(data_dir),
75            ..Default::default()
76        };
77        ClickHouseService::with_config(config)?
78    } else {
79        ClickHouseService::new()?
80    };
81
82    service
83        .init()
84        .await
85        .context("Failed to initialize ClickHouse")?;
86
87    println!("{}", "✓ ClickHouse initialized successfully".green().bold());
88    println!();
89    println!("Next steps:");
90    println!("  1. Start the database: {}", "osvm db start".yellow());
91    println!("  2. Check status: {}", "osvm db status".yellow());
92    println!("  3. Query data: {}", "osvm db query \"SELECT 1\"".yellow());
93
94    Ok(())
95}

172/// Execute a SQL query
173async fn cmd_query(args: &DatabaseArgs) -> Result<()> {
174    let service = ClickHouseService::new()?;
175
176    // Check if server is running
177    let status = service.status().await?;
178    if status != ClickHouseStatus::Running {
179        return Err(anyhow::anyhow!(
180            "ClickHouse server is not running. Start it with: osvm db start"
181        ));
182    }
183
184    let query = args
185        .query
186        .as_ref()
187        .ok_or_else(|| anyhow::anyhow!("No query provided. Use --query \"SELECT ...\""))?;
188
189    // SECURITY: Validate query to prevent SQL injection
190    validate_sql_query(query)?;
191
192    println!("{}", format!("Executing query: {}", query).cyan());
193    println!();
194
195    let result = service
196        .query_json(query)
197        .await
198        .context("Failed to execute query")?;
199
200    println!("{}", result);
201
202    Ok(())
203}

205/// Validate SQL query to prevent injection attacks
206fn validate_sql_query(query: &str) -> Result<()> {
207    let query_upper = query.trim().to_uppercase();
208
209    // Must start with SELECT
210    if !query_upper.starts_with("SELECT") {
211        anyhow::bail!(
212            "Only SELECT queries are allowed. Other operations are forbidden for security."
213        );
214    }
215
216    // Reject dangerous keywords
217    let dangerous_keywords = [
218        "DROP", "DELETE", "UPDATE", "ALTER", "INSERT", "CREATE", "TRUNCATE", "RENAME", "GRANT",
219        "REVOKE", "EXEC", "EXECUTE",
220    ];
221
222    for keyword in dangerous_keywords {
223        if query_upper.contains(keyword) {
224            anyhow::bail!(
225                "Query contains forbidden keyword '{}'. Only SELECT queries are allowed.",
226                keyword
227            );
228        }
229    }
230
231    // Reject stacked queries (semicolon anywhere except in quoted strings would be complex to parse)
232    // Simple check: reject if semicolon found (may have false positives but safer)
233    if query.contains(';') {
234        anyhow::bail!("Multiple statements (semicolons) are not allowed for security reasons.");
235    }
236
237    // Reject UNION attacks
238    if query_upper.contains("UNION") {
239        anyhow::bail!("UNION queries are not allowed for security reasons.");
240    }
241
242    // Whitelist allowed tables (osvm database only)
243    let allowed_tables = [
244        "osvm.cli_commands",
245        "osvm.blockchain_accounts",
246        "osvm.command_activity",
247        "osvm.chat_history",
248    ];
249    let has_allowed_table = allowed_tables
250        .iter()
251        .any(|table| query_upper.contains(&table.to_uppercase()));
252
253    if !has_allowed_table {
254        anyhow::bail!(
255            "Query must reference allowed tables: {}. Access to other tables is forbidden.",
256            allowed_tables.join(", ")
257        );
258    }
259
260    // Reject system table access
261    if query_upper.contains("SYSTEM.") {
262        anyhow::bail!("Access to system tables is forbidden.");
263    }
264
265    Ok(())
266}

268/// Show activity logs
269async fn cmd_activity(args: &DatabaseArgs) -> Result<()> {
270    let service = Arc::new(ClickHouseService::new()?);
271
272    // Check if server is running
273    let status = service.status().await?;
274    if status != ClickHouseStatus::Running {
275        return Err(anyhow::anyhow!(
276            "ClickHouse server is not running. Start it with: osvm db start"
277        ));
278    }
279
280    let logger = ActivityLogger::new(service);
281
282    // Show statistics if requested
283    if args.show_stats {
284        println!("{}", "Activity Statistics".cyan().bold());
285        println!();
286
287        let stats = logger
288            .get_activity_stats()
289            .await
290            .context("Failed to fetch activity statistics")?;
291        println!("{}", stats);
292        println!();
293    }
294
295    // Show command history if requested
296    if args.show_commands {
297        println!("{}", "CLI Command History".cyan().bold());
298        println!();
299
300        let history = logger
301            .query_command_history(args.limit, args.session_id.as_deref())
302            .await
303            .context("Failed to fetch command history")?;
304        println!("{}", history);
305        println!();
306    }
307
308    // Show chat history if requested
309    if args.show_chat {
310        println!("{}", "Chat Message History".cyan().bold());
311        println!();
312
313        let history = logger
314            .query_chat_history(args.limit, args.session_id.as_deref())
315            .await
316            .context("Failed to fetch chat history")?;
317        println!("{}", history);
318        println!();
319    }
320
321    // If nothing specific was requested, show summary
322    if !args.show_stats && !args.show_commands && !args.show_chat {
323        println!("{}", "Activity Summary".cyan().bold());
324        println!();
325        println!("Use the following flags to view specific data:");
326        println!("  {} - Show activity statistics", "--stats".yellow());
327        println!("  {} - Show CLI command history", "--commands".yellow());
328        println!("  {} - Show chat message history", "--chat".yellow());
329        println!();
330        println!("Options:");
331        println!(
332            "  {} <N> - Limit results to N entries (default: 100)",
333            "--limit".yellow()
334        );
335        println!("  {} <ID> - Filter by session ID", "--session-id".yellow());
336        println!();
337        println!("Examples:");
338        println!("  osvm db activity --stats");
339        println!("  osvm db activity --commands --limit 50");
340        println!("  osvm db activity --chat --session-id abc123");
341    }
342
343    Ok(())
344}

346/// Sync blockchain data to ClickHouse
347async fn cmd_sync(args: &DatabaseArgs) -> Result<()> {
348    use crate::services::blockchain_indexer::{
349        parse_sync_args, BlockchainIndexer, IndexingMode, SyncArgs,
350    };
351
352    let service = Arc::new(ClickHouseService::new()?);
353
354    // Check if server is running
355    let status = service.status().await?;
356    if status != ClickHouseStatus::Running {
357        return Err(anyhow::anyhow!(
358            "ClickHouse server is not running. Start it with: osvm db start"
359        ));
360    }
361
362    println!(
363        "{}",
364        "Syncing blockchain data to ClickHouse...".cyan().bold()
365    );
366    println!();
367
368    // Parse sync mode
369    let mode = match args.sync_mode.as_deref().unwrap_or("last-30-days") {
370        "last-30-days" => IndexingMode::LastNDays(30),
371        "full-historical" => IndexingMode::FullHistorical,
372        "realtime" => IndexingMode::RealtimeOnly,
373        _ => IndexingMode::LastNDays(30),
374    };
375
376    // Build sync arguments
377    let sync_args = SyncArgs {
378        mode,
379        programs: args
380            .programs
381            .as_ref()
382            .map(|p| p.split(',').map(|s| s.trim().to_string()).collect())
383            .unwrap_or_default(),
384        accounts: args
385            .accounts
386            .as_ref()
387            .map(|a| a.split(',').map(|s| s.trim().to_string()).collect())
388            .unwrap_or_default(),
389        patterns: args
390            .pattern
391            .as_ref()
392            .map(|p| vec![p.clone()])
393            .unwrap_or_default(),
394        ledger_path: args.ledger_path.clone(),
395        snapshot_path: args.snapshot_dir.clone(),
396    };
397
398    // Parse into indexing config
399    let config = parse_sync_args(&sync_args)?;
400
401    // Create indexer (clone config since we need it later)
402    let indexer = BlockchainIndexer::new(
403        args.ledger_path.clone(),
404        args.snapshot_dir.clone(),
405        service,
406        config.clone(),
407    )?;
408
409    // Sync accounts from snapshot
410    println!("📊 Syncing account data from snapshot...");
411    match indexer.sync_accounts(0).await {
412        Ok(count) => {
413            println!(
414                "{}",
415                format!("✓ Indexed {} accounts successfully", count)
416                    .green()
417                    .bold()
418            );
419        }
420        Err(e) => {
421            eprintln!("{}", format!("❌ Account sync failed: {}", e).red());
422            return Err(e);
423        }
424    }
425
426    println!();
427
428    // Sync transactions from ledger
429    println!("📊 Syncing transaction data from ledger...");
430
431    // Determine slot range based on mode
432    let current_slot = 1000000u64; // Placeholder - would query from ledger
433    let slot_range = match config.mode {
434        IndexingMode::LastNDays(days) => {
435            // Calculate slot range for last N days
436            // Assuming ~2 slots per second average
437            let slots_per_day = 2 * 60 * 60 * 24;
438            let start_slot = current_slot.saturating_sub(days as u64 * slots_per_day);
439            start_slot..current_slot
440        }
441        IndexingMode::FullHistorical => 0..u64::MAX,
442        IndexingMode::RealtimeOnly => current_slot..u64::MAX,
443        _ => 0..current_slot,
444    };
445
446    match indexer.sync_transactions(slot_range).await {
447        Ok(count) => {
448            println!(
449                "{}",
450                format!("✓ Indexed {} transactions successfully", count)
451                    .green()
452                    .bold()
453            );
454        }
455        Err(e) => {
456            eprintln!("{}", format!("⚠ Transaction sync failed: {}", e).yellow());
457            println!("Note: Transaction history requires ledger started with --enable-rpc-transaction-history");
458            // Don't return error, continue to show summary
459        }
460    }
461
462    println!();
463    println!("{}", "Sync completed successfully!".green().bold());
464    println!();
465    println!("Query your data with:");
466    println!(
467        "  Accounts: {}",
468        "osvm db query --query \"SELECT * FROM osvm.blockchain_accounts LIMIT 10\"".yellow()
469    );
470    println!(
471        "  Transactions: {}",
472        "osvm db query --query \"SELECT * FROM osvm.blockchain_transactions LIMIT 10\"".yellow()
473    );
474
475    Ok(())
476}
