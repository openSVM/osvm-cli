4/// Handle AI query with OVSM planning enabled (for p/plan/a/agent commands)
5pub async fn handle_ai_query_with_planning(
6    sub_command: &str,
7    sub_matches: &clap::ArgMatches,
8    app_matches: &clap::ArgMatches,
9) -> Result<(), Box<dyn std::error::Error>> {
10    // Extract the actual query from the planning subcommand args
11    let query_parts = if let Some(external_args) = sub_matches.get_many::<std::ffi::OsString>("") {
12        external_args.map(|os_str| os_str.to_string_lossy().to_string()).collect::<Vec<_>>()
13    } else {
14        // Fallback: parse from environment args
15        std::env::args().skip(2).collect()
16    };
17
18    if query_parts.is_empty() {
19        eprintln!("‚ùå No query provided for planning mode");
20        eprintln!();
21        eprintln!("üí° Usage:");
22        eprintln!("   osvm {} <your query>", sub_command);
23        eprintln!();
24        eprintln!("üìö Examples:");
25        eprintln!("   osvm p find top validators");
26        eprintln!("   osvm agent show me recent transactions");
27        return Ok(());
28    }
29
30    let query = sanitize_user_input(&query_parts.join(" "))?;
31    let debug_mode = app_matches.get_flag("debug");
32
33    if debug_mode {
34        println!("üìã Planning mode enabled - using OVSM agent");
35        println!("üîç Query: \"{}\"", query);
36    }
37
38    // Use streaming agent with planning enabled for OVSM integration
39    let verbose = app_matches.get_count("verbose");
40
41    // Route to streaming agent which includes OVSM planning capabilities
42    crate::utils::streaming_agent::execute_streaming_agent(&query, verbose)
43        .await
44        .map_err(|e| format!("OVSM planning agent failed: {}", e).into())
45}

47/// Handle regular AI query (without OVSM planning)
48pub async fn handle_ai_query(
49    sub_command: &str,
50    sub_matches: &clap::ArgMatches,
51    app_matches: &clap::ArgMatches,
52) -> Result<(), Box<dyn std::error::Error>> {
53    // For external subcommands, clap collects additional arguments in subcommand_value
54    // This is the proper way to handle external subcommands with clap
55    let mut query_parts = vec![sub_command.to_string()];
56
57    // Get additional arguments from clap's external subcommand handling
58    // External subcommands store arguments as OsString, not String
59    if let Some(external_args) = sub_matches.get_many::<std::ffi::OsString>("") {
60        query_parts.extend(external_args.map(|os_str| os_str.to_string_lossy().to_string()));
61    }
62
63    // If clap doesn't provide args (fallback), parse from environment
64    // This maintains compatibility while documenting the limitation
65    if query_parts.len() == 1 {
66        let args: Vec<String> = std::env::args().collect();
67
68        // Collect non-flag arguments starting from the subcommand
69        let mut found_subcommand = false;
70        for arg in args.iter().skip(1) {
71            if found_subcommand {
72                if !arg.starts_with('-') {
73                    query_parts.push(arg.clone());
74                }
75            } else if arg == sub_command {
76                found_subcommand = true;
77            }
78        }
79    }
80
81    let query = sanitize_user_input(&query_parts.join(" "))?;
82
83    // Get debug flag from global args
84    let debug_mode = app_matches.get_flag("debug");
85
86    // Check if this looks like a natural language query vs a typo/unknown command
87    if !looks_like_natural_language_query(&query) {
88        eprintln!("‚ùå Unknown command: '{}'", sub_command);
89        eprintln!();
90        eprintln!("üí° Did you mean one of these commands?");
91        eprintln!("   osvm balance <ADDRESS>     - Check SOL balance");
92        eprintln!("   osvm svm list              - List SVMs");
93        eprintln!("   osvm mcp list              - List MCP servers");
94        eprintln!("   osvm nodes list            - List nodes");
95        eprintln!("   osvm chat                  - Start interactive chat");
96        eprintln!();
97        eprintln!("üí¨ For AI-powered queries, use quotes:");
98        eprintln!("   osvm \"show me transactions for wallet ABC...\"");
99        eprintln!("   osvm \"what is the balance of...\"");
100        eprintln!();
101        eprintln!("üìö Use 'osvm --help' to see all available commands");
102        return Ok(());
103    }
104
105    // Make AI request
106    if debug_mode {
107        println!("üîç Interpreting as AI query: \"{}\"", query);
108    }
109
110    let ai_service = AiService::new_with_debug(debug_mode);
111    match ai_service.query_with_debug(&query, debug_mode).await {
112        Ok(response) => {
113            if debug_mode {
114                println!("ü§ñ AI Response:");
115            }
116            // Render the response as markdown for better formatting
117            let renderer = MarkdownRenderer::new();
118            renderer.render(&response);
119        }
120        Err(e) => {
121            eprintln!("‚ùå AI query failed: {}", e);
122            eprintln!("üí° Use 'osvm --help' to see available commands");
123        }
124    }
125
126    Ok(())
127}

129/// Check if input looks like a natural language query rather than a CLI command
130fn looks_like_natural_language_query(input: &str) -> bool {
131    // Natural language indicators:
132    // 1. Contains multiple words (at least 3)
133    // 2. Contains question words
134    // 3. Contains common natural language patterns
135    // 4. Has spaces between non-address-like tokens
136
137    let words: Vec<&str> = input.split_whitespace().collect();
138
139    // Single word commands are likely typos (e.g., "balanc")
140    if words.len() == 1 {
141        return false;
142    }
143
144    // Two words might be a command with argument (e.g., "balance ABC123")
145    // Only treat as NL if it has NL indicators
146    if words.len() == 2 {
147        let first_word = words[0].to_lowercase();
148        // Question words or verbs indicate natural language
149        let nl_indicators = [
150            "show", "what", "how", "why", "where", "when", "who",
151            "list", "find", "get", "check", "analyze", "explain",
152            "tell", "give", "display", "calculate"
153        ];
154        return nl_indicators.contains(&first_word.as_str());
155    }
156
157    // 3+ words are likely natural language
158    // But check for address-heavy queries (might be copy-paste errors)
159    let long_hex_count = words.iter()
160        .filter(|w| w.len() > 30 && w.chars().all(|c| c.is_alphanumeric()))
161        .count();
162
163    // If mostly long hex strings, might be malformed command
164    if long_hex_count > words.len() / 2 {
165        return false;
166    }
167
168    true
169}

171/// Sanitize user input to prevent command injection and ensure safe processing
172fn sanitize_user_input(input: &str) -> Result<String, Box<dyn std::error::Error>> {
173    // Remove potentially dangerous characters and sequences
174    let sanitized = input
175        .chars()
176        .filter(|c| {
177            // Allow alphanumeric, spaces, basic punctuation, but block command injection chars
178            c.is_alphanumeric()
179                || matches!(
180                    *c,
181                    ' ' | '.'
182                        | ','
183                        | '?'
184                        | '!'
185                        | ':'
186                        | ';'
187                        | '\''
188                        | '"'
189                        | '-'
190                        | '_'
191                        | '('
192                        | ')'
193                        | '['
194                        | ']'
195                        | '{'
196                        | '}'
197                )
198        })
199        .collect::<String>();
200
201    // Remove potentially dangerous sequences
202    let sanitized = sanitized
203        .replace("&&", " and ")
204        .replace("||", " or ")
205        .replace("$(", " ")
206        .replace("`", " ")
207        .replace("${", " ");
208
209    // Limit length to prevent resource exhaustion
210    if sanitized.len() > 2048 {
211        return Err("Input too long (max 2048 characters)".into());
212    }
213
214    // Ensure non-empty after sanitization
215    if sanitized.trim().is_empty() {
216        return Err("Input cannot be empty after sanitization".into());
217    }
218
219    Ok(sanitized.trim().to_string())
220}
