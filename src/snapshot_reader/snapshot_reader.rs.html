115fn read_account_file(path: &PathBuf) -> io::Result<AccountInfo> {
116    let mut file = fs::File::open(path)?;
117    let mut buffer = Vec::new();
118    file.read_to_end(&mut buffer)?;
119
120    // The file format appears to be:
121    // - 8 bytes: unknown (possibly padding or version)
122    // - 8 bytes: data length
123    // - 1 byte: version/type marker
124    // - 32 bytes: owner pubkey
125    // - 8 bytes: lamports
126    // - 8 bytes: rent_epoch (often 0xFFFFFFFFFFFFFFFF)
127    // - 32 bytes: another pubkey (possibly account address)
128    // - More data...
129
130    if buffer.len() < 100 {
131        return Err(io::Error::new(
132            io::ErrorKind::InvalidData,
133            "File too small to be a valid account",
134        ));
135    }
136
137    // Try to parse the structure
138    let data_len = u64::from_le_bytes(buffer[8..16].try_into().unwrap());
139
140    // Owner pubkey starts at offset 17
141    let owner_bytes: [u8; 32] = buffer[17..49].try_into().unwrap();
142    let owner = Pubkey::new_from_array(owner_bytes);
143
144    // Lamports at offset 49
145    let lamports = u64::from_le_bytes(buffer[49..57].try_into().unwrap());
146
147    // Rent epoch at offset 57
148    let rent_epoch = u64::from_le_bytes(buffer[57..65].try_into().unwrap());
149
150    // Executable flag (need to find it in the structure)
151    let executable = false; // Placeholder
152
153    // Extract account data - it appears to start after the header
154    // The actual data starts around offset 130-140 based on the structure
155    let data_start = 130; // Approximate offset where account data begins
156    let data = if buffer.len() > data_start {
157        buffer[data_start..].to_vec()
158    } else {
159        Vec::new()
160    };
161
162    Ok(AccountInfo {
163        lamports,
164        data_len,
165        owner,
166        executable,
167        rent_epoch,
168        data,
169    })
